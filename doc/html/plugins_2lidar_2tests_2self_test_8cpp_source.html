133DOCTEST_TEST_CASE("LiDAR Synthetic Almond Tree Test") {
134 Context context_4;
135 DOCTEST_CHECK_NOTHROW(context_4.loadOBJ("plugins/lidar/xml/AlmondWP.obj", make_vec3(0, 0, 0), 6., make_SphericalCoord(0, 0), RGB::red, true));
136
137 LiDARcloud synthetic_4;
138 synthetic_4.disableMessages();
139
140 DOCTEST_CHECK_NOTHROW(synthetic_4.loadXML("plugins/lidar/xml/almond.xml"));
141 DOCTEST_CHECK_NOTHROW(synthetic_4.syntheticScan(&context_4));
142 DOCTEST_CHECK_NOTHROW(synthetic_4.calculateSyntheticLeafArea(&context_4));
143 DOCTEST_CHECK_NOTHROW(synthetic_4.calculateSyntheticGtheta(&context_4));
144 DOCTEST_CHECK_NOTHROW(synthetic_4.triangulateHitPoints(0.05, 5));
145 DOCTEST_CHECK_NOTHROW(synthetic_4.calculateLeafAreaGPU());
146
147 // Calculate exact leaf area
148 uint Ncells = synthetic_4.getGridCellCount();
149
150 std::vector<float> total_area;
151 total_area.resize(Ncells);
152
153 std::vector<float> Gtheta;
154 Gtheta.resize(Ncells);
155
156 std::vector<float> area_sum;
157 area_sum.resize(Ncells, 0.f);
158 std::vector<float> sin_sum;
159 sin_sum.resize(Ncells, 0.f);
160 std::vector<uint> cell_tri_count;
161 cell_tri_count.resize(Ncells, 0);
162
163 std::vector<uint> UUIDs = context_4.getAllUUIDs();
164 for (int p = 0; p < UUIDs.size(); p++) {
165
166 uint UUID = UUIDs.at(p);
167
168 if (context_4.doesPrimitiveDataExist(UUID, "gridCell")) {
169
170 uint gridCell;
171 context_4.getPrimitiveData(UUID, "gridCell", gridCell);
172
173 if (gridCell >= 0 && gridCell < Ncells) {
174 total_area.at(gridCell) += context_4.getPrimitiveArea(UUID);
175 }
176
177 for (int s = 0; s < synthetic_4.getScanCount(); s++) {
178 vec3 origin = synthetic_4.getScanOrigin(s);
179 std::vector<vec3> vertices = context_4.getPrimitiveVertices(p);
180 float area = context_4.getPrimitiveArea(p);
181 vec3 normal = context_4.getPrimitiveNormal(p);
182 vec3 raydir = vertices.front() - origin;
183 raydir.normalize();
184 float theta = fabs(acos_safe(raydir.z));
185
186 if (area == area) { // in rare cases you can get area=NaN
187
188 Gtheta.at(gridCell) += fabs(normal * raydir) * area * fabs(sin(theta));
189
190 area_sum.at(gridCell) += area;
191 sin_sum.at(gridCell) += fabs(sin(theta));
192 cell_tri_count.at(gridCell) += 1;
193 }
194 }
195 }
196 }
197
198 for (uint v = 0; v < Ncells; v++) {
199 if (cell_tri_count[v] > 0) {
200 Gtheta[v] *= float(cell_tri_count[v]) / (area_sum[v] * sin_sum[v]);
201 }
202 }
203
204 float RMSE_LAD = 0.f;
205 float bias_LAD = 0.f;
206 float RMSE_Gtheta = 0.f;
207 for (uint i = 0; i < Ncells; i++) {
208 float LAD = synthetic_4.getCellLeafArea(i);
209 if (LAD == LAD && total_area.at(i) > 0 && total_area.at(i) == total_area.at(i)) {
210 RMSE_LAD += pow(LAD - total_area.at(i), 2) / float(Ncells);
211 bias_LAD += (LAD - total_area.at(i)) / float(Ncells);
212 }
213 float Gtheta_bar = synthetic_4.getCellGtheta(i);
214 if (Gtheta_bar == Gtheta_bar && Gtheta.at(i) > 0 && Gtheta.at(i) == Gtheta.at(i)) {
215 RMSE_Gtheta += pow(Gtheta_bar - Gtheta.at(i), 2) / float(Ncells);
216 }
217 }
218 RMSE_LAD = sqrt(RMSE_LAD);
219 RMSE_Gtheta = sqrt(RMSE_Gtheta);
220
221 DOCTEST_CHECK(RMSE_LAD <= 0.25f);
222 DOCTEST_CHECK(bias_LAD <= 0.0f);
223 DOCTEST_CHECK(RMSE_Gtheta <= 0.15f);
224 DOCTEST_CHECK(RMSE_LAD != 0.f);
