35void GeometryHandler::addGeometry(size_t UUID, const VisualizerGeometryType& geometry_type, const std::vector<helios::vec3>& vertices,
36 const helios::RGBAcolor &color, const std::vector<helios::vec2>& uvs, int textureID,
37 bool override_texture_color, bool has_glyph_texture, uint coordinate_system, bool visible_flag, bool iscontextgeometry, int size) {
38
39 char vertex_count = getVertexCount( geometry_type );
40
41#ifdef HELIOS_DEBUG
42 assert( vertices.size() == vertex_count );
43 assert( uvs.empty() || uvs.size() == vertex_count );
44#endif
45
46 std::vector<helios::vec3> vertices_copy = vertices; // make a copy so it can be modified
47
48 bool geometry_is_new = false;
49 if ( !doesGeometryExist(UUID) ) {
50 registerUUID( UUID, geometry_type );
51 geometry_is_new = true;
52 }
53
54 if( coordinate_system == 0 ){ //No vertex transformation (i.e., identity matrix)
55
56 //NOTE for vertex positions: OpenGL window coordinates range from -1 to 1, but our rectangle coordinates are from 0 to 1 ---- need to convert
57 for ( auto & vertex : vertices_copy ) {
58 vertex.x = 2.f * vertex.x - 1.f;
59 vertex.y = 2.f * vertex.y - 1.f;
60 }
61
62 }
63
64 size_t vertex_index = UUID_map.at(UUID).vertex_index;
65 size_t normal_index = UUID_map.at(UUID).normal_index;
66 size_t uv_index = UUID_map.at(UUID).uv_index;
67 size_t color_index = UUID_map.at(UUID).color_index;
68 size_t texture_flag_index = UUID_map.at(UUID).texture_flag_index;
69 size_t texture_ID_index = UUID_map.at(UUID).texture_ID_index;
70
71 for ( char i=0; i<vertex_count; i++ ) {
72
73 if ( geometry_is_new ) {
74 face_index_data[geometry_type].push_back( static_cast<int>(visible_flag_data[geometry_type].size()) );
75
76 vertex_data[geometry_type].push_back(vertices_copy.at(i).x);
77 vertex_data[geometry_type].push_back(vertices_copy.at(i).y);
78 vertex_data[geometry_type].push_back(vertices_copy.at(i).z);
79 }else {
80 //face index doesn't need to be updated
81
82 vertex_data[geometry_type].at( vertex_index) = vertices_copy.at(i).x;
83 vertex_data[geometry_type].at( vertex_index+1) = vertices_copy.at(i).y;
84 vertex_data[geometry_type].at( vertex_index+2) = vertices_copy.at(i).z;
85 vertex_index += 3;
86 }
87
88 if ( ( geometry_type == GEOMETRY_TYPE_TRIANGLE || geometry_type == GEOMETRY_TYPE_RECTANGLE ) && !uvs.empty() ) { //if (u, v)'s are provided, color triangle based on texture map image
89
90 if ( geometry_is_new ) {
91 uv_data[geometry_type].push_back(uvs.at(i).x);
92 uv_data[geometry_type].push_back(1.f - uvs.at(i).y);
93
94 if (i == 0) { //only need to do this one time
95 if (has_glyph_texture) {
96 texture_flag_data[geometry_type].push_back(3); // 3 means use RGB for color and red channel of texture for alpha
97 } else if (override_texture_color) { //if texture color is overridden, color primitive based on RGB color but use texture transparence for alpha mask
98 texture_flag_data[geometry_type].push_back(2); // 2 means use RGB for color and texture transparency for alpha
99 } else {
100 texture_flag_data[geometry_type].push_back(1); // 1 means use texture for color and transparency for alpha
101 }
102 texture_ID_data[geometry_type].push_back(textureID);
103 }
104 }else {
105 uv_data[geometry_type].at(uv_index) = uvs.at(i).x;
106 uv_data[geometry_type].at(uv_index+1) = 1.f - uvs.at(i).y;
107 uv_index += 2;
108
109 if ( i==0 ) { //only need to do this one time
110 if ( has_glyph_texture ) {
111 texture_flag_data[geometry_type].at(texture_flag_index) = 3; // 3 means use RGB for color and red channel of texture for alpha
112 }else if ( override_texture_color ) { //if texture color is overridden, color primitive based on RGB color but use texture transparence for alpha mask
113 texture_flag_data[geometry_type].at(texture_flag_index) = 2; // 2 means use RGB for color and texture transparency for alpha
114 }else {
115 texture_flag_data[geometry_type].at(texture_flag_index) = 1; // 1 means use texture for color and transparency for alpha
116 }
117 texture_flag_index++;
118 texture_ID_data[geometry_type].at(texture_ID_index) = textureID;
119 texture_ID_index++;
120
121 }
122 }
123 } else { //if (u,v)'s are not provided, color primitive based on RGB color
124
125 if ( geometry_is_new ) {
126 uv_data[geometry_type].push_back( 0.f );
127 uv_data[geometry_type].push_back( 0.f );
128
129 if ( i==0 ) { //only need to do this one time
130 if ( has_glyph_texture ) {
131 texture_flag_data[geometry_type].push_back( 3 ); // 3 means use RGB for color and red channel of texture for alpha
132 texture_ID_data[geometry_type].push_back( textureID );
133 }else {
134 texture_flag_data[geometry_type].push_back( 0 ); // 0 means use RGB color
135 texture_ID_data[geometry_type].push_back( 0 );
136 }
137 }
138 }else{
139 uv_data[geometry_type].at(uv_index) = 0.f;
140 uv_data[geometry_type].at(uv_index+1) = 0.f;
141 uv_index += 2;
142
143 if ( i==0 ) { //only need to do this one time
144 if ( has_glyph_texture ) {
145 texture_flag_data[geometry_type].at(texture_flag_index) = 3; // 3 means use RGB for color and red channel of texture for alpha
146 texture_ID_data[geometry_type].at(texture_ID_index) = textureID;
147 }else {
148 texture_flag_data[geometry_type].at(texture_flag_index) = 0; // 0 means use RGB color
149 texture_ID_data[geometry_type].at(texture_ID_index) = 0 ;
150 }
151 texture_flag_index++;
152 texture_ID_index++;
153 }
154 }
155 }
156 }
157
158 helios::vec3 normal;
159 if (geometry_type == GEOMETRY_TYPE_TRIANGLE || geometry_type == GEOMETRY_TYPE_RECTANGLE) {
160 normal = normalize(cross(vertices_copy.at(1) - vertices_copy.at(0), vertices_copy.at(2) - vertices_copy.at(0)));
161 }
162 if ( geometry_is_new ) {
163 normal_data[geometry_type].push_back(normal.x);
164 normal_data[geometry_type].push_back(normal.y);
165 normal_data[geometry_type].push_back(normal.z);
166
167 color_data[geometry_type].push_back(color.r);
168 color_data[geometry_type].push_back(color.g);
169 color_data[geometry_type].push_back(color.b);
170 color_data[geometry_type].push_back(color.a);
171
172 coordinate_flag_data[geometry_type].push_back(coordinate_system);
173
174 visible_flag_data[geometry_type].push_back(visible_flag);
175
176 delete_flag_data[geometry_type].push_back(false);
177
178 context_geometry_flag_data[geometry_type].push_back(iscontextgeometry);
179 }else {
180 normal_data[geometry_type].at(normal_index) = normal.x;
181 normal_data[geometry_type].at(normal_index+1) = normal.y;
182 normal_data[geometry_type].at(normal_index+2) = normal.z;
183
184 color_data[geometry_type].at(color_index) = color.r;
185 color_data[geometry_type].at(color_index+1) = color.g;
186 color_data[geometry_type].at(color_index+2) = color.b;
187 color_data[geometry_type].at(color_index+3) = color.a;
188 }

271void GeometryHandler::setVertices( size_t UUID, const std::vector<helios::vec3>& vertices ) {
272
273#ifdef HELIOS_DEBUG
274 assert( UUID_map.find(UUID) != UUID_map.end() );
275#endif
276
277 const PrimitiveIndexMap &index_map = UUID_map.at(UUID);
278
279 const char vertex_count = getVertexCount( index_map.geometry_type );
280
281#ifdef HELIOS_DEBUG
282 assert( vertices.size() == vertex_count );
283#endif
284
285 const size_t vertex_ind = index_map.vertex_index;
286
287 int ii=0;
288 for ( int i=0; i<vertex_count; i++ ) {
289 vertex_data[index_map.geometry_type].at( vertex_ind + ii+0 ) = vertices.at( i ).x;
290 vertex_data[index_map.geometry_type].at( vertex_ind + ii+1 ) = vertices.at( i ).y;
291 vertex_data[index_map.geometry_type].at( vertex_ind + ii+2 ) = vertices.at( i ).z;
292 ii+=3;
293 }
294
295 const size_t normal_ind = index_map.normal_index;
296
297 const helios::vec3 normal = normalize( cross( vertices.at(1) - vertices.at(0), vertices.at(2) - vertices.at(0) ) );
298 normal_data[index_map.geometry_type].at( normal_ind + ii+0 ) = normal.x;
299 normal_data[index_map.geometry_type].at( normal_ind + ii+1 ) = normal.y;
300 normal_data[index_map.geometry_type].at( normal_ind + ii+2 ) = normal.z;

683void GeometryHandler::getDomainBoundingBox(helios::vec2 &xbounds, helios::vec2 &ybounds, helios::vec2 &zbounds) const {
684
685 xbounds.x = 1e8;
686 xbounds.y = -1e8;
687 ybounds.x = 1e8;
688 ybounds.y = -1e8;
689 zbounds.x = 1e8;
690 zbounds.y = -1e8;
691
692 for (const auto &[UUID, index_map]: UUID_map) {
693 // Only primitives with Cartesian coordinates should be included in bounding box
694 if (coordinate_flag_data.at(index_map.geometry_type).at(index_map.coordinate_flag_index) == 0) {
695 continue;
696 }
697
698 const auto &vertices = getVertices(UUID);
699
700 for (const auto &vertex: vertices) {
701 if (vertex.x < xbounds.x) {
702 xbounds.x = vertex.x;
703 }
704 if (vertex.x > xbounds.y) {
705 xbounds.y = vertex.x;
706 }
707 if (vertex.y < ybounds.x) {
708 ybounds.x = vertex.y;
709 }
710 if (vertex.y > ybounds.y) {
711 ybounds.y = vertex.y;
712 }
713 if (vertex.z < zbounds.x) {
714 zbounds.x = vertex.z;
715 }
716 if (vertex.z > zbounds.y) {
717 zbounds.y = vertex.z;
718 }
719 }
720 }

752void GeometryHandler::defragmentBuffers() {
753
754 // If no primitives were deleted, nothing to do
755 if (deleted_primitive_count == 0) {
756 return;
757 }
758
759 // Iterate each geometry type
760
761 for ( const auto &geometry_type : all_geometry_types ) {
762 if ( vertex_data.find(geometry_type) == vertex_data.end() ) {
763 continue;
764 }
765
766 auto &oldFace = face_index_data.at(geometry_type);
767 auto &oldVertex = vertex_data.at(geometry_type);
768 auto &oldNormal = normal_data.at(geometry_type);
769 auto &oldUV = uv_data.at(geometry_type);
770 auto &oldColor = color_data.at(geometry_type);
771 auto &oldTexFlag = texture_flag_data.at(geometry_type);
772 auto &oldTexID = texture_ID_data.at(geometry_type);
773 auto &oldCoordFlag = coordinate_flag_data.at(geometry_type);
774 auto &oldVisible = visible_flag_data.at(geometry_type);
775 auto &oldContextFlag = context_geometry_flag_data.at(geometry_type);
776 auto &oldDeleteFlag = delete_flag_data.at(geometry_type);
777
778 // New buffers
779 std::vector<VisualizerGeometryType> newType;
780 std::vector<float> newVertex, newNormal, newUV, newColor;
781 std::vector<int> newFace, newTexFlag, newTexID, newCoordFlag;
782 std::vector<bool> newDeleteFlag, newContextFlag;
783 std::vector<char> newVisible;
784
785 // Collect UUIDs to drop
786 std::vector<size_t> toErase;
787
788 // Walk the UUID_map and rebuild entries
789 for (auto &[UUID, prim]: UUID_map) {
790 if (prim.geometry_type != geometry_type) {
791 continue; // this UUID is another geometry type
792 }
793
794 // If marked deleted => drop
795 if (oldDeleteFlag[prim.delete_flag_index]) {
796 toErase.push_back(UUID);
797 continue;
798 }
799
800 // Otherwise copy its slices into the new buffers
801 const char vcount = getVertexCount(prim.geometry_type);
802 const size_t v3 = static_cast<size_t>(vcount) * 3;
803 const size_t v2 = static_cast<size_t>(vcount) * 2;
804
805 // Record new base indices
806 const size_t fi = newFace.size();
807 const size_t vi = newVertex.size();
808 const size_t ni = newNormal.size();
809 const size_t ui = newUV.size();
810 const size_t ci = newColor.size();
811 const size_t tfi = newTexFlag.size();
812 const size_t tidi = newTexID.size();
813 const size_t cfi = newCoordFlag.size();
814 const size_t vi2 = newVisible.size();
815 const size_t cfi2 = newContextFlag.size();
816 const size_t dfi = newDeleteFlag.size();
817
818 // Copy raw data
819 newFace.insert( newFace.end(), vcount, scast<int>(newVisible.size()) ); //the new face index should be the new index not just copying the previous value. Note that we take the size of newVisible, but could be any per-face array size.
820
821 newVertex.insert(newVertex.end(),
822 oldVertex.begin() + prim.vertex_index,
823 oldVertex.begin() + prim.vertex_index + v3);
824
825 newNormal.insert(newNormal.end(),
826 oldNormal.begin() + prim.normal_index,
827 oldNormal.begin() + prim.normal_index + 3);
828
829 newUV.insert(newUV.end(),
830 oldUV.begin() + prim.uv_index,
831 oldUV.begin() + prim.uv_index + v2);
832
833 newColor.insert(newColor.end(),
834 oldColor.begin() + prim.color_index,
835 oldColor.begin() + prim.color_index + 4);
836
837 newTexFlag.push_back(oldTexFlag[prim.texture_flag_index]);
838 newTexID.push_back(oldTexID[prim.texture_ID_index]);
839 newCoordFlag.push_back(oldCoordFlag[prim.coordinate_flag_index]);
840 newVisible.push_back(oldVisible[prim.visible_index]);
841 newContextFlag.push_back(oldContextFlag[prim.context_geometry_flag_index]);
842 newDeleteFlag.push_back(oldDeleteFlag[prim.delete_flag_index]);
843
844 // Update the map entry to point at the new positions
845 prim.face_index_index = fi;
846 prim.vertex_index = vi;
847 prim.normal_index = ni;
848 prim.uv_index = ui;
849 prim.color_index = ci;
850 prim.texture_flag_index = tfi;
851 prim.texture_ID_index = tidi;
852 prim.coordinate_flag_index = cfi;
853 prim.visible_index = vi2;
854 prim.context_geometry_flag_index = cfi2;
855 prim.delete_flag_index = dfi;
856 }
857
858 // Erase deleted UUIDs
859 for (auto UUID: toErase) {
860 UUID_map.erase(UUID);
861 }
862
863 // Swap into place
864 oldFace.swap(newFace);
865 oldVertex.swap(newVertex);
866 oldNormal.swap(newNormal);
867 oldUV.swap(newUV);
868 oldColor.swap(newColor);
869 oldTexFlag.swap(newTexFlag);
870 oldTexID.swap(newTexID);
871 oldCoordFlag.swap(newCoordFlag);
872 oldVisible.swap(newVisible);
873 oldContextFlag.swap(newContextFlag);
874 oldDeleteFlag.swap(newDeleteFlag);
875 }
876
877 // Reset deleted count
878 deleted_primitive_count = 0;
879}
