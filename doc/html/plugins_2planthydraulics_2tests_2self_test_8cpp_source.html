172

174 DOCTEST_CHECK_NOTHROW(hydraulics.setModelCoefficients(phmc));
175
176 int plantID = hydraulics.getPlantID(leaves);
177 float soil_water_potential = -0.05;
178
179 DOCTEST_CHECK_NOTHROW(hydraulics.setSoilWaterPotentialOfPlant(plantID, soil_water_potential));

181
182 // Query the point-wise water potentials throughout the system

189
