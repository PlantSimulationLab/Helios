696 static_assert(std::is_same_v<T, uint> || std::is_same_v<T, int> || std::is_same_v<T, float> || std::is_same_v<T, double> || std::is_same_v<T, char> || std::is_same_v<T, size_t>, "helios::powi() was called with an unsupported type.");
697 T result = static_cast<T>(1);

699 // If the low bit is set, multiply result by current base
700 if (exp & 1) {
701 result *= base;
