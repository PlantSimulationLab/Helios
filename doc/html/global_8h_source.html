691 double toc(const char *message) const {
692 if (!running) {
693 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." <<
694 std::endl;
695 return 0;
696 }
697
698 auto timer_end = std::chrono::high_resolution_clock::now();;
699 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
700 if (strcmp(message, "mute") != 0) {
701 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;

704 }
