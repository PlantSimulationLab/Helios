690 double toc(const char *message) const {
691 if (!running) {
692 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." <<
693 std::endl;
694 return 0;
695 }
696
697 auto timer_end = std::chrono::high_resolution_clock::now();;
698 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
699 if (strcmp(message, "mute") != 0) {
700 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;

702 return duration;
703 }
