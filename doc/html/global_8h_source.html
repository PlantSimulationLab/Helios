492 };

495 double toc() const{
496 return toc("");
497 }
498
500
503 double toc( const char* message ) const {
504 if (!running) {
505 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." << std::endl;
506 return 0;
507 }
508
509 auto timer_end = std::chrono::high_resolution_clock::now();;
510 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
511 if (strcmp(message, "mute")!=0) {
512 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
513 }
514 return duration;
515 }
