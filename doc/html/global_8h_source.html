692 static_assert(std::is_same_v<T, uint> || std::is_same_v<T, int> || std::is_same_v<T, float> || std::is_same_v<T, double> || std::is_same_v<T, char> || std::is_same_v<T, size_t>, "helios::powi() was called with an unsupported type.");
693 T result = static_cast<T>(1);
694 while (exp > 0) {
695 // If the low bit is set, multiply result by current base
696 if (exp & 1) {
697 result *= base;
698 }
699 // Square the base for the next bit
700 base *= base;
701 // Shift off the processed bit
702 exp >>= 1;
703 }
