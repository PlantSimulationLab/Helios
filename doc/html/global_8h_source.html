490 Timer(){
491 running=false;
492 }
493
495 void tic(){
496 timer_start = std::chrono::high_resolution_clock::now();
497 running = true;
498 };
499
501 double toc() const{
502 return toc("");
503 }
504
506
509 double toc( const char* message ) const {
510 if (!running) {
511 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." << std::endl;
512 return 0;
513 }
514
515 auto timer_end = std::chrono::high_resolution_clock::now();;
516 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
517 if (strcmp(message, "mute")!=0) {
518 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
519 }
520 return duration;
521 }
