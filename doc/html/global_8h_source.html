542 double toc( const char* message ) const {
543 if (!running) {
544 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." << std::endl;
545 return 0;
546 }
547
548 auto timer_end = std::chrono::high_resolution_clock::now();;
549 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
550 if (strcmp(message, "mute")!=0) {
551 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
552 }
553 return duration;
554 }
