452 double toc( const char* message ) const {
453 if (!running) {
454 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..."
455 << std::endl;
456 return 0;
457 }
458
459 auto timer_end = std::chrono::high_resolution_clock::now();;
460 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
461 if (strcmp(message, "mute")!=0) {
462 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
463 }
464 return duration;
465 }
