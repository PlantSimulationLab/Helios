683 [[nodiscard]] double toc(const char *message) const {
684 if (!running) {
685 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." <<
686 std::endl;
687 return 0;
688 }
689
690 auto timer_end = std::chrono::high_resolution_clock::now();;
691 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
692 if (strcmp(message, "mute") != 0) {
693 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
694 }
695 return duration;
696 }
