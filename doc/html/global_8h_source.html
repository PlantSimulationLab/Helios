700 double toc(const char *message) const {
701 if (!running) {
702 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." <<
703 std::endl;
704 return 0;
705 }
706
707 auto timer_end = std::chrono::high_resolution_clock::now();;
708 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
709 if (strcmp(message, "mute") != 0) {
710 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
711 }
712 return duration;
713 }
