38std::vector<uint> generateSeparatedTriangles(Context* context, int count, float separation = 5.0f) {
39 std::vector<uint> uuids;
40 for(int i = 0; i < count; i++) {
41 float x = i * separation;
42 uint uuid = context->addTriangle(
43 make_vec3(x, -1, 0),
44 make_vec3(x + 1, -1, 0),
45 make_vec3(x + 0.5f, 1, 0)
46 );
47 uuids.push_back(uuid);
48 }
49 return uuids;
50}

55std::vector<uint> generateOverlappingCluster(Context* context, int count, vec3 center = make_vec3(0, 0, 0)) {
56 std::vector<uint> uuids;
57 for(int i = 0; i < count; i++) {
58 float angle = (2.0f * M_PI * i) / count;
59 float radius = 0.5f; // Overlapping radius
60 float x = center.x + radius * cos(angle);
61 float y = center.y + radius * sin(angle);
62
63 uint uuid = context->addTriangle(
64 make_vec3(x - 0.5f, y - 0.5f, center.z),
65 make_vec3(x + 0.5f, y - 0.5f, center.z),
66 make_vec3(x, y + 0.5f, center.z)
67 );
68 uuids.push_back(uuid);
69 }
70 return uuids;
71}
