9// TreeGraph-inspired graph-based distance calculation using Dijkstra's algorithm
10std::vector<float> LiDARcloud::buildDistanceGraph(const std::vector<helios::vec3>& points, const helios::vec3& base_point) {
11 if (points.empty()) return {};
12
13 std::vector<float> distances_from_base(points.size(), std::numeric_limits<float>::infinity());
14 std::vector<bool> visited(points.size(), false);
15
16 // Find base point index (nearest to provided base_point)
17 size_t base_idx = 0;
18 float min_dist_to_base = (points[0] - base_point).magnitude();
19 for (size_t i = 1; i < points.size(); i++) {
20 float dist = (points[i] - base_point).magnitude();
21 if (dist < min_dist_to_base) {
22 min_dist_to_base = dist;
23 base_idx = i;
24 }
25 }
26
27 // Initialize base distance
28 distances_from_base[base_idx] = 0.0f;
29
30 // Build k-NN graph with adaptive neighborhood size based on point density
31 const size_t k_neighbors = std::min(size_t(50), points.size() / 10); // Adaptive k
32 const float max_neighbor_dist = 0.15f; // Maximum connection distance (15cm)
33
34 // Simple priority queue implementation for Dijkstra
35 std::vector<std::pair<float, size_t>> queue; // {distance, point_index}
36 queue.push_back({0.0f, base_idx});
37
38 while (!queue.empty()) {
39 // Find minimum distance node in queue (simple linear search)
40 size_t min_idx = 0;
41 for (size_t i = 1; i < queue.size(); i++) {
42 if (queue[i].first < queue[min_idx].first) {
43 min_idx = i;
44 }
45 }
46
47 float current_dist = queue[min_idx].first;
48 size_t current_idx = queue[min_idx].second;
49 queue.erase(queue.begin() + min_idx);
50
51 if (visited[current_idx]) continue;
52 visited[current_idx] = true;
53
54 // Find k nearest neighbors
55 std::vector<std::pair<float, size_t>> neighbors;
56 for (size_t i = 0; i < points.size(); i++) {
57 if (i == current_idx || visited[i]) continue;
58
59 float dist = (points[current_idx] - points[i]).magnitude();
60 if (dist <= max_neighbor_dist) {
61 neighbors.push_back({dist, i});
62 }
63 }
64
65 // Keep only k nearest neighbors
66 if (neighbors.size() > k_neighbors) {
67 std::partial_sort(neighbors.begin(), neighbors.begin() + k_neighbors, neighbors.end());
68 neighbors.resize(k_neighbors);
69 }
70
71 // Update distances to neighbors
72 for (const auto& neighbor : neighbors) {
73 float edge_dist = neighbor.first;
74 size_t neighbor_idx = neighbor.second;
75
76 float new_dist = current_dist + edge_dist;
77 if (new_dist < distances_from_base[neighbor_idx]) {
78 distances_from_base[neighbor_idx] = new_dist;
79 queue.push_back({new_dist, neighbor_idx});
80 }
81 }
82 }
83
84 // Handle any unreachable points by using Euclidean distance as fallback
85 for (size_t i = 0; i < points.size(); i++) {
86 if (distances_from_base[i] == std::numeric_limits<float>::infinity()) {
87 distances_from_base[i] = (points[i] - base_point).magnitude();
88 }
89 }
90
91 return distances_from_base;
92}

94std::vector<uint> LiDARcloud::reconstructQSM(helios::Context *context, const std::string &filename,
95 uint radial_subdivisions) {
96 std::cout << "DEBUG: QSM reconstruction called with " << hits.size() << " hit points" << std::endl;
97
98 if (printmessages) {
99 std::cout << "Reconstructing QSM from LiDAR point cloud..." << std::flush;
100 }
101
102 // Validate input
103 if (hits.empty()) {
104 std::cout << "failed." << std::endl;
105 helios::helios_runtime_error(
106 "ERROR (LiDARcloud::reconstructQSM): No hit points loaded. Call addHitPoint() or synthetic scan first.");
107 }

109 // Data structures for QSM reconstruction
110 struct SkeletonNode {
111 helios::vec3 position;
112 float radius;
113 std::vector<uint> neighbors;
114 uint parent;
115 float distance_from_base;
116 uint node_id;
117 SkeletonNode() : position(0, 0, 0), radius(0.01f), parent(UINT_MAX), distance_from_base(0), node_id(UINT_MAX) {}
118 };
119
120 struct CylinderModel {
121 helios::vec3 start, end;
122 helios::vec3 axis;
123 float radius;
124 float length;
125 uint branch_id;
126 uint parent_cylinder;
127 uint branch_order;
128 uint position_in_branch;
129 CylinderModel() : start(0, 0, 0), end(0, 0, 0), axis(0, 0, 1), radius(0.01f), length(0),
130 branch_id(-1), parent_cylinder(-1), branch_order(1), position_in_branch(0) {}
131 };
132
133 std::vector<SkeletonNode> skeleton_nodes;
134 std::vector<CylinderModel> cylinders;
135
136 // Phase 1: Build distance graph and extract skeleton
137 if (printmessages) {
138 std::cout << "\n Building distance graph..." << std::flush;
139 }
140
141 // Create point cloud from hits
142 std::vector<helios::vec3> points;
143 std::vector<float> distances_from_base;
144 points.reserve(hits.size());
145 distances_from_base.reserve(hits.size());
146
147 // Find base point (lowest z-coordinate)
148 helios::vec3 base_point = hits[0].position;
149 for (const auto &hit: hits) {
150 if (hit.position.z < base_point.z) {
151 base_point = hit.position;
152 }
153 points.push_back(hit.position);
154 }
155
156 // Calculate distance from base using graph-based shortest paths (TreeGraph approach)
157 distances_from_base = buildDistanceGraph(points, base_point);
158
159 if (printmessages) {
160 std::cout << "done.\n Extracting skeleton..." << std::flush;
161 }
162
163 // Phase 2: 3D spatial clustering with DBSCAN to detect branching (TreeGraph approach)
164 float max_distance = *std::max_element(distances_from_base.begin(), distances_from_base.end());
165 uint num_bins = std::max(20, (int) (max_distance / 0.05f)); // 5cm bins
166 float bin_size = max_distance / num_bins;
167
168 if (printmessages) {
169 std::cout << "\n Debug: " << points.size() << " points, max_distance=" << max_distance
170 << ", num_bins=" << num_bins << ", bin_size=" << bin_size << std::flush;
171 }
172
173 // DBSCAN clustering parameters
174 const float eps = 0.08f; // 8cm neighborhood radius for clustering
175 const uint min_pts = 8; // Minimum points to form a cluster
176
177 uint bins_with_points = 0;
178 uint total_points_processed = 0;
179 uint total_clusters_found = 0;
180
181 for (uint bin = 0; bin < num_bins; bin++) {
182 float bin_start = bin * bin_size;
183 float bin_end = (bin + 1) * bin_size;
184
185 // Get points in this distance range
186 std::vector<helios::vec3> bin_points;
187 std::vector<size_t> bin_indices;
188 for (size_t i = 0; i < points.size(); i++) {
189 if (distances_from_base[i] >= bin_start && distances_from_base[i] < bin_end) {
190 bin_points.push_back(points[i]);
191 bin_indices.push_back(i);
192 }
193 }
194
195 if (bin_points.size() < min_pts) {
196 if (printmessages && bin_points.size() > 0) {
197 std::cout << "\n Bin " << bin << " has only " << bin_points.size() << " points (< " << min_pts << ")" << std::flush;
198 }
199 continue;
200 }
201
202 bins_with_points++;
203 total_points_processed += bin_points.size();
204
205 // DBSCAN clustering to detect multiple branches in this distance range
206 std::vector<int> cluster_labels(bin_points.size(), -1); // -1 = noise
207 int cluster_id = 0;
208
209 for (size_t i = 0; i < bin_points.size(); i++) {
210 if (cluster_labels[i] != -1) continue; // Already processed
211
212 // Find neighbors within eps distance
213 std::vector<size_t> neighbors;
214 for (size_t j = 0; j < bin_points.size(); j++) {
215 if (i != j && (bin_points[i] - bin_points[j]).magnitude() <= eps) {
216 neighbors.push_back(j);
217 }
218 }
219
220 if (neighbors.size() < min_pts - 1) continue; // Not enough neighbors (core point needs min_pts total)
221
222 // Create new cluster
223 cluster_labels[i] = cluster_id;
224 std::vector<size_t> seed_set = neighbors;
225
226 // Expand cluster
227 for (size_t k = 0; k < seed_set.size(); k++) {
228 size_t q = seed_set[k];
229
230 if (cluster_labels[q] == -1) { // Was noise, now border point
231 cluster_labels[q] = cluster_id;
232 }
233 if (cluster_labels[q] != -1) continue; // Already assigned
234
235 cluster_labels[q] = cluster_id;
236
237 // Find neighbors of q
238 std::vector<size_t> q_neighbors;
239 for (size_t j = 0; j < bin_points.size(); j++) {
240 if (q != j && (bin_points[q] - bin_points[j]).magnitude() <= eps) {
241 q_neighbors.push_back(j);
242 }
243 }
244
245 // If q is core point, add its neighbors to seed set
246 if (q_neighbors.size() >= min_pts - 1) {
247 for (size_t neighbor : q_neighbors) {
248 if (std::find(seed_set.begin(), seed_set.end(), neighbor) == seed_set.end()) {
249 seed_set.push_back(neighbor);
250 }
251 }
252 }
253 }
254 cluster_id++;
255 }
256
257 // Create skeleton nodes from clusters
258 for (int c = 0; c < cluster_id; c++) {
259 std::vector<helios::vec3> cluster_points;
260 for (size_t i = 0; i < bin_points.size(); i++) {
261 if (cluster_labels[i] == c) {
262 cluster_points.push_back(bin_points[i]);
263 }
264 }
265
266 if (cluster_points.size() < min_pts / 2) continue; // Skip very small clusters
267
268 // Calculate cluster centroid
269 helios::vec3 centroid(0, 0, 0);
270 for (const auto &point: cluster_points) {
271 centroid = centroid + point;
272 }
273 centroid = centroid / cluster_points.size();
274
275 // Improved radius estimation with branch tapering
276 float radius_sum = 0;
277 for (const auto &point: cluster_points) {
278 radius_sum += (point - centroid).magnitude();
279 }
280 float spread_radius = radius_sum / cluster_points.size();
281
282 // Apply tapering based on distance from base (further = smaller radius)
283 float taper_factor = std::max(0.2f, 1.0f - (bin_start + bin_end) * 0.5f / max_distance);
284
285 // Scale by cluster size (smaller clusters = smaller radius)
286 float density_factor = std::min(1.0f, (float)cluster_points.size() / 20.0f);
287
288 float radius = std::max(0.005f, spread_radius * taper_factor * density_factor);
289
290 SkeletonNode node;
291 node.position = centroid;
292 node.radius = radius;
293 node.distance_from_base = (bin_start + bin_end) * 0.5f;
294 node.node_id = skeleton_nodes.size();
295
296 skeleton_nodes.push_back(node);
297 total_clusters_found++;
298 }
299
300 if (printmessages && (bin < 5 || cluster_id > 1)) { // Debug first few bins or bins with multiple clusters
301 std::cout << "\n Bin " << bin << " (" << bin_start << "-" << bin_end
302 << "): " << bin_points.size() << " points → " << cluster_id << " clusters" << std::flush;
303 }
304 }
305
306 if (printmessages) {
307 std::cout << "\n Debug: " << bins_with_points << " bins with >=" << min_pts << " points, "
308 << total_points_processed << " total points processed, "
309 << total_clusters_found << " clusters found" << std::flush;
310 }
311
312 if (printmessages) {
313 std::cout << "extracted " << skeleton_nodes.size() << " skeleton nodes.\n Building connectivity..." <<
314 std::flush;
315 }
316
317 // Phase 3: Graph-based skeleton connectivity (TreeGraph approach)
318 if (skeleton_nodes.empty()) {
319 if (printmessages) {
320 std::cout << "failed - no skeleton nodes extracted." << std::endl;
321 }
322 return {};
323 }
324
325 // Build connectivity based on 3D proximity and graph-based distance ordering
326 const float max_connection_distance = 0.20f; // 20cm maximum connection distance
327
328 // Sort nodes by distance from base to establish hierarchy
329 std::vector<std::pair<float, size_t>> sorted_nodes;
330 for (size_t i = 0; i < skeleton_nodes.size(); i++) {
331 sorted_nodes.push_back({skeleton_nodes[i].distance_from_base, i});
332 }
333 std::sort(sorted_nodes.begin(), sorted_nodes.end());
334
335 // Connect each node to its nearest valid parent (closer to base)
336 for (size_t i = 1; i < sorted_nodes.size(); i++) {
337 size_t current_idx = sorted_nodes[i].second;
338 float current_dist = sorted_nodes[i].first;
339
340 float min_parent_distance = std::numeric_limits<float>::infinity();
341 size_t best_parent = 0;
342
343 // Find closest parent node (must be closer to base and within connection distance)
344 for (size_t j = 0; j < i; j++) {
345 size_t potential_parent_idx = sorted_nodes[j].second;
346 float parent_dist = sorted_nodes[j].first;
347
348 // Parent must be closer to base
349 if (parent_dist >= current_dist) continue;
350
351 // Check 3D distance
352 float spatial_dist = (skeleton_nodes[current_idx].position -
353 skeleton_nodes[potential_parent_idx].position).magnitude();
354
355 if (spatial_dist <= max_connection_distance && spatial_dist < min_parent_distance) {
356 min_parent_distance = spatial_dist;
357 best_parent = potential_parent_idx;
358 }
359 }
360
361 // Connect to best parent
362 skeleton_nodes[current_idx].parent = best_parent;
363 skeleton_nodes[best_parent].neighbors.push_back(current_idx);
364
365 if (printmessages && i < 5) { // Debug first few connections
366 std::cout << "\n Node " << current_idx << " -> parent " << best_parent
367 << " (dist=" << min_parent_distance << "m)" << std::flush;
368 }
369 }
370
371 if (printmessages) {
372 std::cout << "done.\n Phase 3: Reconstructing shoot topology..." << std::flush;
373 }
374
375 // Phase 4: Branch topology detection based on connectivity
376 struct Shoot {
377 std::vector<uint> node_path;
378 uint shoot_id;
379 uint parent_shoot;
380 uint branch_order;
381 Shoot() : shoot_id(0), parent_shoot(UINT_MAX), branch_order(1) {}
382 };
383
384 std::vector<Shoot> shoots;
385 std::vector<bool> node_visited(skeleton_nodes.size(), false);
386
387 // Find branch points (nodes with multiple children) and analyze connectivity
388 std::vector<uint> branch_points;
389 uint nodes_with_0_children = 0, nodes_with_1_child = 0, nodes_with_multiple_children = 0;
390
391 for (size_t i = 0; i < skeleton_nodes.size(); i++) {
392 uint num_children = skeleton_nodes[i].neighbors.size();
393 if (num_children == 0) nodes_with_0_children++;
394 else if (num_children == 1) nodes_with_1_child++;
395 else {
396 nodes_with_multiple_children++;
397 branch_points.push_back(i);
398 }
399 }
400
401 if (printmessages) {
402 std::cout << "\n Connectivity analysis: " << nodes_with_0_children << " leaves, "
403 << nodes_with_1_child << " linear nodes, " << nodes_with_multiple_children
404 << " branch points" << std::flush;
405 }
406
407 // Create shoots by following paths from base to branch points and leaves
408 uint shoot_id = 0;
409
410 // Start from base (node with distance_from_base closest to 0)
411 size_t base_node = 0;
412 float min_base_dist = skeleton_nodes[0].distance_from_base;
413 for (size_t i = 1; i < skeleton_nodes.size(); i++) {
414 if (skeleton_nodes[i].distance_from_base < min_base_dist) {
415 min_base_dist = skeleton_nodes[i].distance_from_base;
416 base_node = i;
417 }
418 }
419
420 // Recursive function to trace shoots
421 std::function<void(uint, uint, uint, std::vector<uint>&)> trace_shoot =
422 [&](uint start_node, uint parent_shoot_id, uint branch_order, std::vector<uint>& current_path) {
423
424 current_path.push_back(start_node);
425 node_visited[start_node] = true;
426
427 // If this is a leaf node or all children are visited, complete the shoot
428 bool has_unvisited_children = false;
429 for (uint child : skeleton_nodes[start_node].neighbors) {
430 if (!node_visited[child]) {
431 has_unvisited_children = true;
432 break;
433 }
434 }
435
436 if (!has_unvisited_children || skeleton_nodes[start_node].neighbors.empty()) {
437 // Complete current shoot
438 if (current_path.size() >= 2) { // Need at least 2 nodes for a shoot
439 Shoot shoot;
440 shoot.shoot_id = shoot_id++;
441 shoot.parent_shoot = parent_shoot_id;
442 shoot.branch_order = branch_order;
443 shoot.node_path = current_path;
444 shoots.push_back(shoot);
445 }
446 return;
447 }
448
449 // If multiple children, create branches
450 if (skeleton_nodes[start_node].neighbors.size() > 1) {
451 // Complete current shoot at branch point
452 if (current_path.size() >= 2) {
453 Shoot shoot;
454 shoot.shoot_id = shoot_id++;
455 shoot.parent_shoot = parent_shoot_id;
456 shoot.branch_order = branch_order;
457 shoot.node_path = current_path;
458 shoots.push_back(shoot);
459 parent_shoot_id = shoot.shoot_id;
460 }
461
462 // Create child shoots
463 for (uint child : skeleton_nodes[start_node].neighbors) {
464 if (!node_visited[child]) {
465 std::vector<uint> child_path = {start_node}; // Start child path with branch point
466 trace_shoot(child, parent_shoot_id, branch_order + 1, child_path);
467 }
468 }
469 } else {
470 // Continue current shoot
471 for (uint child : skeleton_nodes[start_node].neighbors) {
472 if (!node_visited[child]) {
473 trace_shoot(child, parent_shoot_id, branch_order, current_path);
474 break;
475 }
476 }
477 }
478 };
479
480 // Start tracing from base node
481 std::vector<uint> base_path;
482 trace_shoot(base_node, UINT_MAX, 1, base_path);
483
484 if (printmessages) {
485 std::cout << "\n Found " << shoots.size() << " shoot paths..." << std::flush;
486 }
487
488 // Phase 5: Generate cylinders
489 for (const auto& shoot : shoots) {
490 for (size_t i = 0; i < shoot.node_path.size() - 1; i++) {
491 uint node_idx = shoot.node_path[i];
492 uint next_node_idx = shoot.node_path[i + 1];
493
494 CylinderModel cylinder;
495 cylinder.start = skeleton_nodes[node_idx].position;
496 cylinder.end = skeleton_nodes[next_node_idx].position;
497 cylinder.axis = (cylinder.end - cylinder.start).normalize();
498 cylinder.length = (cylinder.end - cylinder.start).magnitude();
499 cylinder.radius = (skeleton_nodes[node_idx].radius + skeleton_nodes[next_node_idx].radius) * 0.5f;
500 cylinder.branch_id = shoot.shoot_id + 1;
501 cylinder.branch_order = shoot.branch_order;
502 cylinder.position_in_branch = i;
503
504 cylinders.push_back(cylinder);
505 }
506 }
507
508 if (printmessages) {
509 std::cout << "generated " << cylinders.size() << " cylinders." << std::endl;
510 }
511
512 // Write QSM file
513 if (printmessages) {
514 std::cout << " Writing QSM file..." << std::flush;
515 }
516
517 std::ofstream outfile(filename);
518 if (!outfile.is_open()) {
519 helios::helios_runtime_error("ERROR (LiDARcloud::reconstructQSM): Could not open output file: " + filename);
520 }
521
522 // Write header (tab-separated format expected by loadTreeQSM)
523 outfile << "radius (m)\tlength (m)\tstart_point\taxis_direction\tparent\textension\tbranch\tbranch_order\tposition_in_branch\tmad\tSurfCov\tadded\tUnmodRadius (m)" << std::endl;
524
525 // Write cylinder data with tab separators
526 for (size_t i = 0; i < cylinders.size(); i++) {
527 const auto& cyl = cylinders[i];
528 outfile << cyl.radius << "\t"
529 << cyl.length << "\t"
530 << cyl.start.x << "\t" << cyl.start.y << "\t" << cyl.start.z << "\t"
531 << cyl.axis.x << "\t" << cyl.axis.y << "\t" << cyl.axis.z << "\t"
532 << (i > 0 ? i-1 : 0) << "\t" // parent cylinder
533 << 0 << "\t" // extension
534 << cyl.branch_id << "\t"
535 << cyl.branch_order << "\t"
536 << i << "\t" // position_in_branch
537 << 0.0002 << "\t" // mad (using reference value)
538 << 1 << "\t" // surf_cov
539 << 0 << "\t" // added
540 << cyl.radius << std::endl; // unmod_radius
541 }
542
543 outfile.close();
544
545 if (printmessages) {
546 std::cout << "done.\n Creating visualization..." << std::flush;
547 }
548
549 // Create tube objects for visualization
550 std::vector<uint> tube_object_IDs;
551 tube_object_IDs.reserve(cylinders.size());
552
553 for (const auto& cylinder : cylinders) {
554 if (cylinder.length > 1e-6) {
555 // Create tube using vector of nodes and radii
556 std::vector<helios::vec3> nodes = {cylinder.start, cylinder.end};
557 std::vector<float> radii = {cylinder.radius, cylinder.radius};
558
559 uint objID = context->addTubeObject(radial_subdivisions, nodes, radii);
560
561 // Set brown color for branches
562 context->setObjectColor(objID, helios::make_RGBcolor(0.55f, 0.27f, 0.07f));
563 tube_object_IDs.push_back(objID);
564 }
565 }
566
567 if (printmessages) {
568 std::cout << "done." << std::endl;
569 std::cout << "QSM reconstruction complete: " << cylinders.size() << " cylinders, "
570 << tube_object_IDs.size() << " tube objects created." << std::endl;
571 }
572
573 return tube_object_IDs;
574}

576std::vector<uint> LiDARcloud::leafReconstructionTreeQSM(helios::Context *context, const std::vector<uint> &tube_object_IDs, const helios::vec2 &leaf_size, float proximity_distance, const char *mask_file) {
577
578 if (printmessages) {
579 cout << "Performing TreeQSM-based leaf reconstruction..." << flush;
580 }
581
582 if (triangles.size() == 0) {
583 std::cout << "failed." << std::endl;
584 helios_runtime_error("ERROR (LiDARcloud::leafReconstructionTreeQSM): There are no triangulated points. Either the triangulation failed or 'triangulateHitPoints()' was not called.");
585 }
586
587 if (tube_object_IDs.empty()) {
588 std::cout << "failed." << std::endl;
589 helios_runtime_error("ERROR (LiDARcloud::leafReconstructionTreeQSM): No tube object IDs provided. Call 'loadTreeQSM()' first to generate tube objects.");
590 }
591
592 // Validate mask file
593 std::string file = mask_file;
594 if (file.substr(file.find_last_of(".") + 1) != "png") {
595 std::cout << "failed." << std::endl;
596 helios_runtime_error("ERROR (LiDARcloud::leafReconstructionTreeQSM): Mask data file " + std::string(mask_file) + " must be PNG image format.");
597 }
598
599 reconstructed_alphamasks_maskfile = mask_file;
600
601 // Clear previous reconstruction data
602 reconstructed_alphamasks_center.clear();
603 reconstructed_alphamasks_size.clear();
604 reconstructed_alphamasks_rotation.clear();
605 reconstructed_alphamasks_gridcell.clear();
606 reconstructed_alphamasks_direct_flag.clear();
607
608 uint Ncells = getGridCellCount();
609
610 // Create a prototype leaf tile object with the alpha mask
611 uint prototype_objID = context->addTileObject(helios::make_vec3(0, 0, 0), leaf_size, helios::make_SphericalCoord(0, 0), helios::make_int2(1, 1), mask_file);
612
613 // Get the actual solid area of the prototype leaf
614 float prototype_leaf_area = context->getObjectArea(prototype_objID);
615
616 // Calculate leaf angle distributions for all voxels
617 const uint Nbins = 36; // 10-degree bins for angle distribution
618 std::vector<std::vector<float>> CDF_theta, CDF_phi;
619 calculateLeafAngleCDF(Nbins, CDF_theta, CDF_phi);
620
621 // Convert to proper angle ranges
622 const float dtheta = 0.5f * M_PI / float(Nbins); // Zenith angles: 0 to π/2
623 const float dphi = 2.0f * M_PI / float(Nbins); // Azimuth angles: 0 to 2π
624
625 if (printmessages) {
626 cout << "Calculated CDF for " << CDF_theta.size() << " voxels, " << Nbins << " bins each." << endl;
627
628 // Debug: Check how many voxels have valid CDF data
629 int valid_theta_count = 0, valid_phi_count = 0;
630 for (uint v = 0; v < CDF_theta.size() && v < 10; v++) {
631 bool has_theta = (!CDF_theta[v].empty() && CDF_theta[v].back() > 0.001f);
632 bool has_phi = (!CDF_phi[v].empty() && CDF_phi[v].back() > 0.001f);
633 if (has_theta)
634 valid_theta_count++;
635 if (has_phi)
636 valid_phi_count++;
637
638 if (v < 5) { // Debug first few voxels
639 cout << "Voxel " << v << ": theta_CDF_max=" << (CDF_theta[v].empty() ? 0.0f : CDF_theta[v].back()) << ", phi_CDF_max=" << (CDF_phi[v].empty() ? 0.0f : CDF_phi[v].back()) << endl;
640 }
641 }
642 cout << "Valid theta CDFs: " << valid_theta_count << "/" << std::min(10u, (uint) CDF_theta.size()) << ", Valid phi CDFs: " << valid_phi_count << "/" << std::min(10u, (uint) CDF_phi.size()) << endl;
643 }
644
645 uint total_leaves_placed = 0;
646 std::vector<uint> leaf_object_IDs; // Collect all created leaf object IDs
647
648 // Process each voxel
649 int voxels_with_leaf_area = 0;
650 int voxels_with_nearby_branches = 0;
651 for (uint v = 0; v < Ncells; v++) {
652 float voxel_leaf_area = getCellLeafArea(v);
653 if (voxel_leaf_area <= 0.0f) {
654 continue; // Skip voxels with no leaf area
655 }
656 voxels_with_leaf_area++;
657
658 if (printmessages && voxels_with_leaf_area <= 5) {
659 cout << "Voxel " << v << " has leaf area: " << voxel_leaf_area << endl;
660 }
661
662 helios::vec3 voxel_center = getCellCenter(v);
663 helios::vec3 voxel_size = getCellSize(v);
664
665 // Find nearby branches within proximity_distance
666 std::vector<uint> nearby_branches;
667 std::vector<float> branch_distances;
668
669 for (uint tube_id: tube_object_IDs) {
670 // Get tube object geometry from context
671 std::vector<helios::vec3> tube_nodes = context->getTubeObjectNodes(tube_id);
672
673 // Find minimum distance from voxel center to this tube
674 float min_distance = 1e6f;
675 for (size_t i = 0; i < tube_nodes.size() - 1; i++) {
676 helios::vec3 segment_start = tube_nodes[i];
677 helios::vec3 segment_end = tube_nodes[i + 1];
678
679 // Calculate distance from point to line segment
680 helios::vec3 segment_vec = segment_end - segment_start;
681 helios::vec3 point_vec = voxel_center - segment_start;
682
683 float segment_length = segment_vec.magnitude();
684 float segment_length_sq = segment_length * segment_length;
685 if (segment_length_sq < 1e-10f) {
686 // Degenerate segment, use point distance
687 min_distance = std::min(min_distance, (voxel_center - segment_start).magnitude());
688 } else {
689 float t = std::max(0.0f, std::min(1.0f, (point_vec * segment_vec) / segment_length_sq));
690 helios::vec3 closest_point = segment_start + t * segment_vec;
691 min_distance = std::min(min_distance, (voxel_center - closest_point).magnitude());
692 }
693 }
694
695 if (min_distance <= proximity_distance) {
696 nearby_branches.push_back(tube_id);
697 branch_distances.push_back(min_distance);
698 }
699 }
700
701 if (nearby_branches.empty()) {
702 if (printmessages && voxels_with_leaf_area <= 5) {
703 cout << "Voxel " << v << " has no nearby branches within distance " << proximity_distance << endl;
704 }
705 continue; // No nearby branches, skip this voxel
706 }
707 voxels_with_nearby_branches++;
708
709 // Calculate number of leaves needed for this voxel
710 int num_leaves = std::max(1, (int) round(voxel_leaf_area / prototype_leaf_area));
711
712 if (printmessages && v < 5) { // Debug first few voxels
713 cout << "Voxel " << v << ": leaf_area=" << voxel_leaf_area << ", prototype_area=" << prototype_leaf_area << ", num_leaves=" << num_leaves << endl;
714 }
715
716 // Place leaves near branches
717 for (int leaf_idx = 0; leaf_idx < num_leaves; leaf_idx++) {
718 // Select a nearby branch (weighted by proximity)
719 float total_weight = 0.0f;
720 std::vector<float> weights;
721 for (float dist: branch_distances) {
722 float weight = 1.0f / (1.0f + dist); // Closer branches have higher weight
723 weights.push_back(weight);
724 total_weight += weight;
725 }
726
727 // Random selection weighted by proximity
728 float rand_val = context->randu() * total_weight;
729 uint selected_branch_idx = 0;
730 float cumulative_weight = 0.0f;
731 for (size_t i = 0; i < weights.size(); i++) {
732 cumulative_weight += weights[i];
733 if (rand_val <= cumulative_weight) {
734 selected_branch_idx = i;
735 break;
736 }
737 }
738
739 uint selected_tube_id = nearby_branches[selected_branch_idx];
740
741 // Get a position near the selected branch
742 helios::vec3 leaf_position;
743
744 // Get tube nodes for the selected branch
745 std::vector<helios::vec3> tube_nodes = context->getTubeObjectNodes(selected_tube_id);
746 if (!tube_nodes.empty()) {
747 // Find the closest segment to voxel center
748 float min_dist = 1e6;
749 helios::vec3 closest_point = tube_nodes[0];
750
751 for (size_t i = 0; i < tube_nodes.size() - 1; i++) {
752 helios::vec3 segment_start = tube_nodes[i];
753 helios::vec3 segment_end = tube_nodes[i + 1];
754 helios::vec3 segment_vec = segment_end - segment_start;
755 helios::vec3 point_vec = voxel_center - segment_start;
756
757 float segment_length_sq = segment_vec.magnitude() * segment_vec.magnitude();
758 if (segment_length_sq > 0.0001f) {
759 float t = std::max(0.0f, std::min(1.0f, (point_vec * segment_vec) / segment_length_sq));
760 helios::vec3 point_on_segment = segment_start + t * segment_vec;
761 float dist = (voxel_center - point_on_segment).magnitude();
762 if (dist < min_dist) {
763 min_dist = dist;
764 closest_point = point_on_segment;
765 }
766 }
767 }
768
769 // Place leaf near the closest point on the branch, with some randomness
770 helios::vec3 random_offset;
771 random_offset.x = (context->randu() - 0.5f) * voxel_size.x * 0.3f;
772 random_offset.y = (context->randu() - 0.5f) * voxel_size.y * 0.3f;
773 random_offset.z = (context->randu() - 0.5f) * voxel_size.z * 0.3f;
774
775 leaf_position = closest_point + random_offset;
776
777 // Ensure leaf stays within voxel bounds
778 helios::vec3 voxel_min = voxel_center - 0.5f * voxel_size;
779 helios::vec3 voxel_max = voxel_center + 0.5f * voxel_size;
780 leaf_position.x = std::max(voxel_min.x, std::min(voxel_max.x, leaf_position.x));
781 leaf_position.y = std::max(voxel_min.y, std::min(voxel_max.y, leaf_position.y));
782 leaf_position.z = std::max(voxel_min.z, std::min(voxel_max.z, leaf_position.z));
783 } else {
784 // Fallback: random position within voxel
785 leaf_position = voxel_center;
786 leaf_position.x += (context->randu() - 0.5f) * voxel_size.x * 0.8f;
787 leaf_position.y += (context->randu() - 0.5f) * voxel_size.y * 0.8f;
788 leaf_position.z += (context->randu() - 0.5f) * voxel_size.z * 0.8f;
789 }
790
791 // Sample leaf orientation from the angle distribution for this voxel
792 float zenith_angle = 0.0f;
793 float azimuth_angle = 0.0f;
794
795 // Check if we have valid CDF data for this voxel
796 bool has_valid_theta_cdf = (v < CDF_theta.size() && !CDF_theta[v].empty() && CDF_theta[v].back() > 0.001f);
797 bool has_valid_phi_cdf = (v < CDF_phi.size() && !CDF_phi[v].empty() && CDF_phi[v].back() > 0.001f);
798
799 // Sample zenith angle from CDF only
800 if (has_valid_theta_cdf) {
801 float rand_theta = context->randu();
802 float normalized_rand = rand_theta * CDF_theta[v].back(); // Normalize by total CDF value
803 for (uint bin = 0; bin < CDF_theta[v].size() && bin < Nbins; bin++) {
804 if (normalized_rand <= CDF_theta[v][bin] || bin == CDF_theta[v].size() - 1) {
805 zenith_angle = (float(bin) + context->randu()) * dtheta;
806 break;
807 }
808 }
809 } else {
810 if (printmessages && total_leaves_placed < 5) {
811 cout << "Warning: No valid zenith CDF for voxel " << v << ", skipping leaf placement." << endl;
812 }
813 continue; // Skip this leaf if no valid CDF data
814 }
815
816 // Sample azimuth angle from CDF only
817 if (has_valid_phi_cdf) {
818 float rand_phi = context->randu();
819 float normalized_rand = rand_phi * CDF_phi[v].back(); // Normalize by total CDF value
820 for (uint bin = 0; bin < CDF_phi[v].size() && bin < Nbins; bin++) {
821 if (normalized_rand <= CDF_phi[v][bin] || bin == CDF_phi[v].size() - 1) {
822 azimuth_angle = (float(bin) + context->randu()) * dphi;
823 break;
824 }
825 }
826 } else {
827 if (printmessages && total_leaves_placed < 5) {
828 cout << "Warning: No valid azimuth CDF for voxel " << v << ", skipping leaf placement." << endl;
829 }
830 continue; // Skip this leaf if no valid CDF data
831 }
832
833 helios::SphericalCoord leaf_orientation = helios::make_SphericalCoord(1.0f, zenith_angle, azimuth_angle);
834
835 if (printmessages && total_leaves_placed < 3) { // Debug first few leaves
836 cout << "Leaf " << total_leaves_placed << ": zenith=" << zenith_angle << ", azimuth=" << azimuth_angle << endl;
837 }
838
839 // Copy the prototype and apply transformations
840 uint leaf_objID = context->copyObject(prototype_objID);
841
842 // Apply rotations - zenith rotation around y-axis, azimuth rotation around z-axis
843 if (zenith_angle > 0.001f) { // Only rotate if angle is meaningful
844 context->rotateObject(leaf_objID, zenith_angle, "y");
845 }
846 if (azimuth_angle > 0.001f) { // Only rotate if angle is meaningful
847 context->rotateObject(leaf_objID, azimuth_angle, "z");
848 }
849
850 context->translateObject(leaf_objID, leaf_position);
851
852 // Add object data for tracking
853 context->setObjectData(leaf_objID, "gridCell", (int) v);
854 context->setObjectData(leaf_objID, "directFlag", 1);
855
856 // Collect the leaf object ID
857 leaf_object_IDs.push_back(leaf_objID);
858
859 // Store in the reconstruction data structures for compatibility
860 reconstructed_alphamasks_center.push_back(leaf_position);
861 reconstructed_alphamasks_size.push_back(leaf_size);
862 reconstructed_alphamasks_rotation.push_back(leaf_orientation);
863 reconstructed_alphamasks_gridcell.push_back(v);
864 reconstructed_alphamasks_direct_flag.push_back(1);
865
866 total_leaves_placed++;
867 }
868 }
869
870 // Delete the prototype object
871 context->deleteObject(prototype_objID);
872
873 if (printmessages) {
874 cout << "done." << endl;
875 cout << "Found " << voxels_with_leaf_area << " voxels with leaf area, " << voxels_with_nearby_branches << " with nearby branches." << endl;
876 cout << "TreeQSM-based reconstruction placed " << total_leaves_placed << " leaves across " << Ncells << " voxels." << endl;
877 }
878
879 return leaf_object_IDs;
