117DOCTEST_TEST_CASE("AerialLiDARcloud Ground and Canopy Height Test") {
118 float err_tol = 0.1;
119
120 //------- sparse vegatation ground and canopy height estimation -------//
121
122 Context context_3;
123
124 vec3 boxsize(10, 10, 10);
125 int3 Nleaves = make_int3(25, 25, 35);
126 float L = 0.05;
127
128 for (int k = 0; k < Nleaves.z; k++) {
129 for (int j = 0; j < Nleaves.y; j++) {
130 for (int i = 0; i < Nleaves.x; i++) {
131
132 vec3 x(context_3.randu() * boxsize.x, context_3.randu() * boxsize.y, context_3.randu() * boxsize.z);
133
134 float theta = acos(1.f - context_3.randu());
135 float phi = context_3.randu() * 2.f * M_PI;
136
137 context_3.addPatch(x, make_vec2(L, L), make_SphericalCoord(theta, phi));
138 }
139 }
140 }
141
142 float zground = 0.2;
143
144 context_3.addPatch(make_vec3(0.5 * boxsize.x, 0.5 * boxsize.y, zground), make_vec2(boxsize.x, boxsize.y));
145
146 AerialLiDARcloud lidar_3;
147 lidar_3.disableMessages();
148
149 lidar_3.syntheticScan(&context_3, "plugins/aeriallidar/xml/synthetic_aerial_test.xml");
150
151 for (int r = 0; r < lidar_3.getHitCount(); r++) {
152 lidar_3.setHitData(r, "target_index", 1);
153 lidar_3.setHitData(r, "target_count", 1);
154 }
155
156 lidar_3.generateHeightModel(100, 0.5, 0.1, 0.5, 0.1);
157
158 for (int v = 0; v < 8; v++) {
159 int3 index = lidar_3.gridindex2ijk(v);
160 float zg = lidar_3.getCellGroundHeight(make_int2(index.x, index.y));
161 DOCTEST_CHECK(fabs(zg - zground) / fabs(zground) <= 1.5 * err_tol);
162 }
163
164 for (int r = 0; r < lidar_3.getHitCount(); r++) {
165 vec3 xyz = lidar_3.getHitXYZ(r);
166 if (fabs(xyz.z - zground) > 9) {
167 lidar_3.setHitData(r, "target_index", 1);
168 } else {
169 lidar_3.setHitData(r, "target_index", 2);
170 }
171 lidar_3.setHitData(r, "target_count", 2);
172 }
173
174 lidar_3.generateHeightModel(400, 0.5, 0.1, 1.0, 0.2);
175
176 for (int v = 0; v < 8; v++) {
177 int3 index = lidar_3.gridindex2ijk(v);
178 float zc = lidar_3.getCellVegetationHeight(make_int2(index.x, index.y));
179 float zm = lidar_3.getCellMaximumHitHeight(make_int2(index.x, index.y));
180
181 DOCTEST_CHECK(fabs(zc - (boxsize.z - 0.5)) / fabs(boxsize.z - 0.5) <= 1.5 * err_tol);
182 DOCTEST_CHECK(fabs(zm - boxsize.z) / fabs(boxsize.z) <= err_tol);
183 }
184}
