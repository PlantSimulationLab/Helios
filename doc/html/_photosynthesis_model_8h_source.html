21struct PhotosyntheticTemperatureResponseParameters {
22private:
23 static void validateOptimalTemperature(float optimum_temperature_in_C) {
24 if (optimum_temperature_in_C < 0.f) {
25 helios::helios_runtime_error("ERROR (PhotosyntheticTemperatureResponseParameters): Optimal temperature cannot be negative. Received Topt = " + std::to_string(optimum_temperature_in_C) +
26 " C. Please check that temperature is provided in units of Celsius, not Kelvin.");
27 }
28 if (optimum_temperature_in_C > 100.f) {
29 helios::helios_runtime_error("ERROR (PhotosyntheticTemperatureResponseParameters): Optimal temperature cannot exceed 100 C. Received Topt = " + std::to_string(optimum_temperature_in_C) +
30 " C. This value is biologically unrealistic and likely indicates temperature was provided in Kelvin instead of Celsius. Please convert to Celsius (subtract 273.15 from Kelvin value).");
31 }
32 }

52 if (rate_of_increase_dHa > 0.f) {
53 this->dHd = 10.f * rate_of_increase_dHa;
54 } else {
55 this->dHd = 600.f;
56 }
57 this->Topt = 10000.f;
58 }

60 PhotosyntheticTemperatureResponseParameters(float value_at_25C, float rate_of_increase_dHa, float optimum_temperature_in_C) {
61 validateOptimalTemperature(optimum_temperature_in_C);
62 this->value_at_25C = value_at_25C;
63 this->dHa = rate_of_increase_dHa;
64 if (rate_of_increase_dHa > 0.f) {
65 this->dHd = 10.f * rate_of_increase_dHa;
66 } else {
67 this->dHd = 600.f;
68 }
69 this->Topt = 273.15f + optimum_temperature_in_C;
70 }
71
72 PhotosyntheticTemperatureResponseParameters(float value_at_25C, float rate_of_increase_dHa, float optimum_temperature_in_C, float rate_of_decrease_dHd) {
73 validateOptimalTemperature(optimum_temperature_in_C);
74 this->value_at_25C = value_at_25C;
75 this->dHa = rate_of_increase_dHa;
76 this->dHd = rate_of_decrease_dHd;
77 this->Topt = 273.15f + optimum_temperature_in_C;
78 }
