175 context->getPrimitiveData(UUID,"twosided_flag",flag);

177 Nsides=1;
178 }
179 }

192float BLConductanceModel::calculateBoundaryLayerConductance( uint gH_model, float U, float L, char Nsides, float inclination, float TL, float Ta ){
193
194
195 assert( gH_model<4 );
196
197 float gH;
198
199 if( L==0 ){
200 return 0;
201 }
202
203 if( gH_model==0 ){ //Polhausen equation
204 //This comes from the correlation by Polhausen - see Eq. XX of Campbell and Norman (1998). It assumes a flat plate parallel to the direction of the flow, which extends infinitely in the cross-stream direction and "L" in the streamwise direction. It also assumes that the air is at standard temperature and pressure, and flow is laminar, forced convection.
205
206 gH = 0.135f*sqrt(U/L)*float(Nsides);
207
208 }else if( gH_model==1 ){ //Inclined Plate
209
210 float Pr=0.7f; //air Prandtl number
211
212 float nu=1.568e-5; //air viscosity (m^2/sec)
213 float alpha=nu/Pr; //air diffusivity (m^2/sec)
214
215 float Re=U*L/nu;
216
217 float Gr=9.81*fabs(TL-Ta)*pow(L,3)/((Ta)*nu*nu);
218
219 float F1=0.399*pow(Pr,1.f/3.f)*pow(1.f+pow(0.0468/Pr,2.f/3.f),-0.25);
220 float F2=0.75*pow(Pr,0.5)*pow(2.5*(1.f+2.f*pow(Pr,0.5)+2.f*Pr),-0.25);
221
222 //direction of free convection
223 float free_direction;
224 if(TL>=Ta){
225 free_direction=1;
226 }else{
227 free_direction=-1;
228 }
229
230 //free_direction=1;
231
232 if( inclination<75.f*M_PI/180.f || inclination>105.f*M_PI/180.f){
233
234 gH=41.4*alpha/L*2.f*F1*sqrtf(Re)*pow(1.f+free_direction*pow(2.f*F2*pow(Gr*fabs(cos(inclination))/(Re*Re),0.25)/(3.f*F1),3),1.f/3.f);
235
236 }else{
237
238 float C=0.07*sqrtf(fabs(cos(inclination)));
239 float F3=pow(Pr,0.5f)*pow(0.25+1.6*pow(Pr,0.5),-1)*pow(Pr/5.f,1.f/5.f+C);
240
241 gH=41.4*alpha/L*2.f*F1*sqrtf(Re)*pow(1.f+free_direction*pow((F3*pow(Gr*pow(Re,-5.f/2.f),1.f/5.f)*pow(Gr,C))/(6.f*(1.f/5.f+C)*F1),3),1.f/3.f);
242
243 }
244
245 }else if( gH_model==2 ){ //Sphere
246
247 //Laminar flow around a sphere (L is sphere diameter). From Eq. 4 of Smart and Sinclair (1976).
248
249 float Pr=0.7f; //air Prandtl number
250 float nu=1.568e-5; //air viscosity (m^2/sec)
251 float ka = 0.024; //air thermal conductivity (W/m-K)
252 float cp = 29.25; //air heat capacity (J/mol-K)
253
254 float Re=U*L/nu;
255
256 gH = (ka/cp)/L*(2.0 + 0.6*sqrtf(Re)*pow(Pr,1.f/3.f));
257
258 }else if( gH_model==3 ){ //Ground
259 //From Eq. A17 of Kustas and Norman (1999). For the soil-air interface.
260
261 gH = 0.004 + 0.012*U; //units in (m^3 air)/(m^2-sec.)
262
263 //assuming standard temperature and pressure, (m^3 air)*(1.2041 kg/m^3)/(0.02897 kg/mol) = 41.56 (mol air)/(m^2-sec.)
264
265 gH = gH*41.56;
266
267 }
268
269 return gH;
270
271}
