246 static float lin_to_srgb(float x) noexcept {
247 // Clamp negative values to 0, bright values > 1.0 to white (1.0)
248 if (x <= 0.0f) return 0.0f;
249 if (x >= 1.0f) return 1.0f; // Bright pixels clipped to white
