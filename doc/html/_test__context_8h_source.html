1592 SUBCASE("setTubeNodes maintains circular cross-sections after bending") {
1593 Context ctx;
1594 // Create a straight vertical tube
1595 std::vector<vec3> nodes = {make_vec3(0, 0, 0), make_vec3(0, 0, 1), make_vec3(0, 0, 2)};
1596 std::vector<float> radii = {0.5f, 0.5f, 0.5f};
1597 int subdiv = 16;
1598 uint tube = ctx.addTubeObject(subdiv, nodes, radii);
1599
1600 // Bend the tube 90 degrees at the tip
1601 std::vector<vec3> bent_nodes = {make_vec3(0, 0, 0), make_vec3(0, 0, 1), make_vec3(1, 0, 1)};
1602 ctx.setTubeNodes(tube, bent_nodes);
1603
1604 // Verify cross-sections are circular and perpendicular to the local axis:
1605 // For each triangle vertex, find the nearest node, check that the radial vector
1606 // (vertex - node) is perpendicular to the local axis and has the correct magnitude.
1607 std::vector<uint> uuids = ctx.getObjectPrimitiveUUIDs(tube);
1608
1609 // Compute axial directions at each node
1610 std::vector<vec3> axial(3);
1611 axial[0] = (bent_nodes[1] - bent_nodes[0]);
1612 axial[0].normalize();
1613 axial[1] = 0.5f * ((bent_nodes[1] - bent_nodes[0]) + (bent_nodes[2] - bent_nodes[1]));
1614 axial[1].normalize();
1615 axial[2] = (bent_nodes[2] - bent_nodes[1]);
1616 axial[2].normalize();
1617
1618 int perp_failures = 0;
1619 int radius_failures = 0;
1620
1621 for (uint uuid : uuids) {
1622 for (uint v = 0; v < 3; v++) {
1623 vec3 vert = ctx.getTriangleVertex(uuid, v);
1624
1625 // Find nearest node
1626 int nearest = 0;
1627 float min_dist = (vert - bent_nodes[0]).magnitude();
1628 for (int n = 1; n < 3; n++) {
1629 float d = (vert - bent_nodes[n]).magnitude();
1630 if (d < min_dist) {
1631 min_dist = d;
1632 nearest = n;
1633 }
1634 }
1635
1636 vec3 radial = vert - bent_nodes[nearest];
1637 float dot = fabs(radial * axial[nearest]);
1638 if (dot > 0.02f) {
1639 perp_failures++;
1640 }
1641 if (fabs(radial.magnitude() - 0.5f) > 0.02f) {
1642 radius_failures++;
1643 }
1644 }
1645 }
