107vec3 helios::rotatePoint(const vec3 &position, float theta, float phi) {
108 if (theta == 0.f && phi == 0.f) {
109 return position;
110 }
111
112 float Ry[3][3], Rz[3][3];
113
114 const float st = sin(theta);
115 const float ct = cos(theta);
116
117 const float sp = sin(phi);
118 const float cp = cos(phi);
119
120 // Setup the rotation matrix, this matrix is based off of the rotation matrix used in glRotatef.
121 Ry[0][0] = ct;
122 Ry[0][1] = 0.f;
123 Ry[0][2] = st;
124 Ry[1][0] = 0.f;
125 Ry[1][1] = 1.f;
126 Ry[1][2] = 0.f;
127 Ry[2][0] = -st;
128 Ry[2][1] = 0.f;
129 Ry[2][2] = ct;
130
131 Rz[0][0] = cp;
132 Rz[0][1] = -sp;
133 Rz[0][2] = 0.f;
134 Rz[1][0] = sp;
135 Rz[1][1] = cp;
136 Rz[1][2] = 0.f;
137 Rz[2][0] = 0.f;
138 Rz[2][1] = 0.f;
139 Rz[2][2] = 1.f;
140
141 // Multiply Ry*Rz
142
143 float rotMat[3][3] = {0.f};
144
145 for (int i = 0; i < 3; i++) {
146 for (int j = 0; j < 3; j++) {
147 for (int k = 0; k < 3; k++) {
148 rotMat[i][j] = rotMat[i][j] + Rz[i][k] * Ry[k][j];
149 }
150 }
151 }
152
153 // Multiply the rotation matrix with the position vector.
154 vec3 tmp;
155 tmp.x = rotMat[0][0] * position.x + rotMat[0][1] * position.y + rotMat[0][2] * position.z;
156 tmp.y = rotMat[1][0] * position.x + rotMat[1][1] * position.y + rotMat[1][2] * position.z;
157 tmp.z = rotMat[2][0] * position.x + rotMat[2][1] * position.y + rotMat[2][2] * position.z;
158
159 return tmp;
160}

162vec3 helios::rotatePointAboutLine(const vec3 &point, const vec3 &line_base, const vec3 &line_direction, float theta) {
163 if (theta == 0.f) {
164 return point;
165 }
166
167 //for reference this was taken from http://inside.mines.edu/fs_home/gmurray/ArbitraryAxisRotation/
168
169 vec3 position;
170
171 vec3 tmp = line_direction;
172 float mag = tmp.magnitude();
173 if (mag < 1e-6f) {
174 return point;
175 }
176 tmp = tmp / mag;
177 const float u = tmp.x;
178 const float v = tmp.y;
179 const float w = tmp.z;
180
181 const float a = line_base.x;
182 const float b = line_base.y;
183 const float c = line_base.z;
184
185 const float x = point.x;
186 const float y = point.y;
187 const float z = point.z;
188
189 const float st = sin(theta);
190 const float ct = cos(theta);
191
192 position.x = (a * (v * v + w * w) - u * (b * v + c * w - u * x - v * y - w * z)) * (1 - ct) + x * ct + (-c * v + b * w - w * y + v * z) * st;
193 position.y = (b * (u * u + w * w) - v * (a * u + c * w - u * x - v * y - w * z)) * (1 - ct) + y * ct + (c * u - a * w + w * x - u * z) * st;
194 position.z = (c * (u * u + v * v) - w * (a * u + b * v - u * x - v * y - w * z)) * (1 - ct) + z * ct + (-b * u + a * v - v * x + u * y) * st;
195
196 return position;
197}

223 // Compute “Julian day of year” for *this
224 const int jd = Calendar2Julian(*this);
225
226 // 2) Sanity-check jd
227 bool leap = isLeapYear();
228 const int maxJD = leap ? 366 : 365;

234 if (jd < maxJD) {
235 // still inside this year
236 const Date next = Julian2Calendar(jd + 1, year);
237 day = next.day;

284 T result = static_cast<T>(1);
285 while (exp > 0) {
286 // If the low bit is set, multiply result by current base
287 if (exp & 1) {
288 result *= base;

294 }

305bool helios::lineIntersection(const vec2 &p1, const vec2 &q1, const vec2 &p2, const vec2 &q2) {
306 float ax = q1.x - p1.x; // direction of line a
307 float ay = q1.y - p1.y; // ax and ay as above
308
309 float bx = p2.x - q2.x; // direction of line b, reversed
310 float by = p2.y - q2.y; // really -by and -by as above
311
312 float dx = p2.x - p1.x; // right-hand side
313 float dy = p2.y - p1.y;
314
315 float det = ax * by - ay * bx;
316
317 if (det == 0) return false;
318
319 float r = (dx * by - dy * bx) / det;
320 float s = (ax * dy - ay * dx) / det;
321
322 return !(r < 0 || r > 1 || s < 0 || s > 1);
323}

563

1298Date helios::CalendarDay(int Julian_day, int year) {
1299 // ----------------------------- input checks ----------------------------
1300 if (Julian_day < 1 || Julian_day > 366)
1301 helios_runtime_error("ERROR (CalendarDay): Julian day out of range [1–366].");
1302
1303 if (year < 1000)
1304 helios_runtime_error("ERROR (CalendarDay): Year must be given in YYYY format.");
1305
1306 const bool leap =
1307 (year % 4 == 0 && year % 100 != 0) || // divisible by 4 but not by 100
1308 (year % 400 == 0); // or divisible by 400
1309
1310 if (!leap && Julian_day == 366)
1311 helios_runtime_error("ERROR (CalendarDay): Day 366 occurs only in leap years.");
1312
1313 // ------------------- month lengths for the chosen year -----------------
1314 // Index 0 = January, …, 11 = December
1315 int month_lengths[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
1316 if (leap) // adjust February
1317 month_lengths[1] = 29;
1318
1319 // --------------------------- computation ------------------------------
1320 int d_remaining = Julian_day; // days still to account for
1321 int month = 1; // 1‑based calendar month
1322
1323 // subtract complete months until the remainder lies in the current month
1324 for (int i = 0; i < 12; ++i) {
1325 if (d_remaining > month_lengths[i]) {
1326 d_remaining -= month_lengths[i];
1327 ++month;
1328 } else {
1329 break;
1330 }
1331 }
1332
1333 // d_remaining is now the calendar day of the computed month
1334 return make_Date(d_remaining, month, year);
1335}

1587 png_structp png_ptr = nullptr;

1593 //

1604 throw std::runtime_error("Failed to read PNG header from " + filename);
