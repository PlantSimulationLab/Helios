275size_t findShootClosingBracket(const std::string &lstring) {
276
277 size_t pos_close = std::string::npos;
278 size_t pos_open = lstring.find_first_of('[', 0);
279 if (pos_open == std::string::npos) {
280 return pos_close;
281 }
282
283 size_t pos = pos_open;
284 int count = 1;
285 while (count > 0) {
286 pos++;
287 if (lstring[pos] == '[') {
288 count++;
289 } else if (lstring[pos] == ']') {
290 count--;
291 }
292 if (pos == lstring.length()) {
293 return pos_close;
294 }
295 }
296 if (count == 0) {
297 pos_close = pos;
298 }
299 return pos_close; // Return the position of the closing bracket
300}

489 output_xml << "\t\t\t\t\t\t<petiole_pitch>" << rad2deg(phytomer->petiole_pitch.at(petiole)) << "</petiole_pitch>" << std::endl;
490 output_xml << "\t\t\t\t\t\t<petiole_curvature>" << phytomer->petiole_curvature.at(petiole) << "</petiole_curvature>" << std::endl;
491 if( phytomer->leaf_rotation.at(petiole).size()==1 ){ //not compound leaf
492 output_xml << "\t\t\t\t\t\t<leaflet_scale>" << 1.0 << "</leaflet_scale>" << std::endl;
493 }else {
494 float tip_ind = floor(float(phytomer->leaf_rotation.at(petiole).size() - 1) / 2.f);
495 output_xml << "\t\t\t\t\t\t<leaflet_scale>" << phytomer->leaf_size_max.at(petiole).at(int(tip_ind-1)) / max(phytomer->leaf_size_max.at(petiole)) << "</leaflet_scale>" << std::endl;
496 }
497
498 for( uint leaf=0; leaf < phytomer->leaf_rotation.at(petiole).size(); leaf++ ){
499 output_xml << "\t\t\t\t\t\t<leaf>" << std::endl;
500 output_xml << "\t\t\t\t\t\t\t<leaf_scale>" << phytomer->leaf_size_max.at(petiole).at(leaf)*phytomer->current_leaf_scale_factor.at(petiole) << "</leaf_scale>" << std::endl;
501 output_xml << "\t\t\t\t\t\t\t<leaf_pitch>" << rad2deg(phytomer->leaf_rotation.at(petiole).at(leaf).pitch) << "</leaf_pitch>" << std::endl;
502 output_xml << "\t\t\t\t\t\t\t<leaf_yaw>" << rad2deg(phytomer->leaf_rotation.at(petiole).at(leaf).yaw) << "</leaf_yaw>" << std::endl;
503 output_xml << "\t\t\t\t\t\t\t<leaf_roll>" << rad2deg(phytomer->leaf_rotation.at(petiole).at(leaf).roll) << "</leaf_roll>" << std::endl;
504 output_xml << "\t\t\t\t\t\t</leaf>" << std::endl;
505 }
506
507 output_xml << "\t\t\t\t\t</petiole>" << std::endl;
508 }
509 output_xml << "\t\t\t\t</internode>" << std::endl;
510 output_xml << "\t\t\t</phytomer>" << std::endl;
511 }
512 output_xml << "\t\t</shoot>" << std::endl;
513 }
514 output_xml << "\t</plant_instance>" << std::endl;
515 output_xml << "</helios>" << std::endl;
516 output_xml.close();

521std::vector<uint> PlantArchitecture::readPlantStructureXML( const std::string &filename, bool quiet){
522
523 if( !quiet ) {
524 std::cout << "Loading plant architecture XML file: " << filename << "..." << std::flush;
525 }
526
527 std::string fn = filename;
528 std::string ext = getFileExtension(filename);
529 if( ext != ".xml" && ext != ".XML" ) {
530 helios_runtime_error("failed.\n File " + fn + " is not XML format.");
531 }
532
533 std::vector<uint> plantIDs;
534
535 // Using "pugixml" parser. See pugixml.org
536 pugi::xml_document xmldoc;
537
538 //load file
539 pugi::xml_parse_result load_result = xmldoc.load_file(filename.c_str());
540
541 //error checking
542 if (!load_result){
543 helios_runtime_error("ERROR (Context::readPlantStructureXML): Could not parse " + std::string(filename) + ":\nError description: " + load_result.description() );
544 }
545
546 pugi::xml_node helios = xmldoc.child("helios");
547
548 pugi::xml_node node;
549 std::string node_string;
550
551 if( helios.empty() ){
552 if( !quiet ) {
553 std::cout << "failed." << std::endl;
554 }
555 helios_runtime_error("ERROR (Context::readPlantStructureXML): XML file must have tag '<helios> ... </helios>' bounding all other tags.");
556 }
557
558 size_t phytomer_count = 0;
559
560 std::map<int,int> shoot_ID_mapping;
561
562 for (pugi::xml_node plant = helios.child("plant_instance"); plant; plant = plant.next_sibling("plant_instance")) {
563
564 int plantID = std::stoi(plant.attribute("ID").value());
565
566 // base position
567 node_string = "base_position";
568 vec3 base_position = parse_xml_tag_vec3(plant.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
569
570 // plant age
571 node_string = "plant_age";
572 float plant_age = parse_xml_tag_float(plant.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
573
574 plantID = addPlantInstance(base_position, plant_age);
575 plantIDs.push_back(plantID);
576
577 int current_shoot_ID;
578
579 for (pugi::xml_node shoot = plant.child("shoot"); shoot; shoot = shoot.next_sibling("shoot")) {
580
581 int shootID = std::stoi(shoot.attribute("ID").value());
582 bool base_shoot = true;
583
584 // shoot type
585 node_string = "shoot_type_label";
586 std::string shoot_type_label = parse_xml_tag_string(shoot.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
587
588 // parent shoot ID
589 node_string = "parent_shoot_ID";
590 int parent_shoot_ID = parse_xml_tag_int(shoot.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
591
592 // parent node index
593 node_string = "parent_node_index";
594 int parent_node_index = parse_xml_tag_int(shoot.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
595
596 // parent petiole index
597 node_string = "parent_petiole_index";
598 int parent_petiole_index = parse_xml_tag_int(shoot.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
599
600 // base rotation
601 node_string = "base_rotation";
602 vec3 base_rot = parse_xml_tag_vec3(shoot.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
603 AxisRotation base_rotation(deg2rad(base_rot.x), deg2rad(base_rot.y), deg2rad(base_rot.z));
604
605 for (pugi::xml_node phytomer = shoot.child("phytomer"); phytomer; phytomer = phytomer.next_sibling("phytomer")) {
606
607 pugi::xml_node internode = phytomer.child("internode");
608
609 // internode length
610 node_string = "internode_length";
611 float internode_length = parse_xml_tag_float(internode.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
612
613 // internode radius
614 node_string = "internode_radius";
615 float internode_radius = parse_xml_tag_float(internode.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
616
617 // internode pitch
618 node_string = "internode_pitch";
619 float internode_pitch = parse_xml_tag_float(internode.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
620
621 // internode phyllotactic angle
622 node_string = "internode_phyllotactic_angle";
623 float internode_phyllotactic_angle = parse_xml_tag_float(internode.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
624
625 float petiole_length;
626 float petiole_radius;
627 float petiole_pitch;
628 float petiole_curvature;
629 float leaflet_scale;
630 std::vector<std::vector<float>> leaf_scale; //first index is petiole within internode; second index is leaf within petiole
631 std::vector<std::vector<float>> leaf_pitch;
632 std::vector<std::vector<float>> leaf_yaw;
633 std::vector<std::vector<float>> leaf_roll;
634 for (pugi::xml_node petiole = internode.child("petiole"); petiole; petiole = petiole.next_sibling("petiole")) {
635
636 // petiole length
637 node_string = "petiole_length";
638 petiole_length = parse_xml_tag_float(petiole.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
639
640 // petiole radius
641 node_string = "petiole_radius";
642 petiole_radius = parse_xml_tag_float(petiole.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
643
644 // petiole pitch
645 node_string = "petiole_pitch";
646 petiole_pitch = parse_xml_tag_float(petiole.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
647
648 // petiole curvature
649 node_string = "petiole_curvature";
650 petiole_curvature = parse_xml_tag_float(petiole.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
651
652 // leaflet scale factor
653 node_string = "leaflet_scale";
654 leaflet_scale = parse_xml_tag_float(petiole.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML");
655
656 leaf_scale.resize(leaf_scale.size() + 1);
657 leaf_pitch.resize(leaf_pitch.size() + 1);
658 leaf_yaw.resize(leaf_yaw.size() + 1);
659 leaf_roll.resize(leaf_roll.size() + 1);
660 for (pugi::xml_node leaf = petiole.child("leaf"); leaf; leaf = leaf.next_sibling("leaf")) {
661
662 // leaf scale factor
663 node_string = "leaf_scale";
664 leaf_scale.back().push_back( parse_xml_tag_float(leaf.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML") );
665
666 // leaf pitch
667 node_string = "leaf_pitch";
668 leaf_pitch.back().push_back( parse_xml_tag_float(leaf.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML") );
669
670 // leaf yaw
671 node_string = "leaf_yaw";
672 leaf_yaw.back().push_back( parse_xml_tag_float(leaf.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML") );
673
674 // leaf roll
675 node_string = "leaf_roll";
676 leaf_roll.back().push_back( parse_xml_tag_float(leaf.child(node_string.c_str()), node_string, "PlantArchitecture::readPlantStructureXML") );
677
678 }
679 } //petioles
680
681 if( shoot_types.find(shoot_type_label)==shoot_types.end() ){
682 helios_runtime_error("ERROR (PlantArchitecture::readPlantStructureXML): Shoot type " + shoot_type_label + " not found in shoot types.");
683 }
684
685
686 ShootParameters shoot_parameters = getCurrentShootParameters(shoot_type_label);
687
688 shoot_parameters.phytomer_parameters.phytomer_creation_function = nullptr;
689
690 shoot_parameters.phytomer_parameters.internode.pitch = internode_pitch;
691 shoot_parameters.phytomer_parameters.internode.phyllotactic_angle = internode_phyllotactic_angle;
692
693 shoot_parameters.phytomer_parameters.petiole.length = petiole_length;
694 shoot_parameters.phytomer_parameters.petiole.radius = petiole_radius;
695 shoot_parameters.phytomer_parameters.petiole.pitch = petiole_pitch;
696 shoot_parameters.phytomer_parameters.petiole.curvature = petiole_curvature;
697
698 shoot_parameters.phytomer_parameters.leaf.prototype_scale = 1.f;//leaf_scale.front().at(tip_ind);
699 shoot_parameters.phytomer_parameters.leaf.pitch = 0;
700 shoot_parameters.phytomer_parameters.leaf.yaw = 0;
701 shoot_parameters.phytomer_parameters.leaf.roll = 0;
702 shoot_parameters.phytomer_parameters.leaf.leaflet_scale = leaflet_scale;
703
704 std::string shoot_label = "shoot_" + std::to_string(phytomer_count);
705 defineShootType( shoot_label, shoot_parameters);
706
707 if( base_shoot ){
708
709 if( parent_shoot_ID<0 ) { //this is the first shoot of the plant
710 current_shoot_ID = addBaseStemShoot(plantID, 1, base_rotation, internode_radius, internode_length, 1.f, 1.f, 0, shoot_label);
711 shoot_ID_mapping[shootID] = current_shoot_ID;
712 }else{ //this is a child of an existing shoot
713 current_shoot_ID = addChildShoot(plantID, shoot_ID_mapping.at(parent_shoot_ID), parent_node_index, 1, base_rotation, internode_radius, internode_length, 1.f, 1.f, 0, shoot_label, parent_petiole_index);
714 shoot_ID_mapping[shootID] = current_shoot_ID;
715 }
716
717 base_shoot = false;
718 }else{
719 appendPhytomerToShoot(plantID, current_shoot_ID, shoot_parameters.phytomer_parameters, internode_radius, internode_length, 1, 1);
720 }
721
722 //rotate and scale leaves
723 assert( leaf_scale.size() == leaf_pitch.size() );
724 auto phytomer_ptr = plant_instances.at(plantID).shoot_tree.at(current_shoot_ID)->phytomers.back();
725 for( int petiole = 0; petiole<leaf_scale.size(); petiole++ ){
726
727 float tip_ind = floor(float(leaf_scale.at(petiole).size() - 1) / 2.f);
728 phytomer_ptr->setLeafPrototypeScale( petiole, leaf_scale.at(petiole).at(tip_ind) );
729
730 phytomer_ptr->scaleLeafPrototypeScale( petiole, 5);
731
732 for( int leaf=0; leaf<phytomer_ptr->leaf_rotation.at(petiole).size(); leaf++ ){
733 phytomer_ptr->rotateLeaf(petiole, leaf, make_AxisRotation(deg2rad(leaf_pitch.at(petiole).at(leaf)), deg2rad(leaf_yaw.at(petiole).at(leaf)), deg2rad(-leaf_roll.at(petiole).at(leaf))));
734 }
735 }
736
737 phytomer_count++;
738 } //phytomers
739
740 } //shoots
741
742 } //plant instances
743
744 if( !quiet ) {
745 std::cout << "done." << std::endl;
746 }
747 return plantIDs;
