141SphericalCoord SolarPosition::calculateSunDirection( const helios::Time &time, const helios::Date &date ) const{
142
143 int solstice_day, LSTM;
144 float Gamma, delta, time_dec, B, EoT, TC, LST, h, theta, phi, rad;
145
146 rad=M_PI/180.f;
147
148 solstice_day=81;
149
150 //day angle (Iqbal Eq. 1.1.2)
151 Gamma = 2.f*M_PI*(float(date.JulianDay()-1))/365.f;
152
153 //solar declination angle (Iqbal Eq. 1.3.1 after Spencer)
154 delta = 0.006918f - 0.399912f*cos(Gamma) + 0.070257f*sin(Gamma)
155 - 0.006758f*cos(2.f*Gamma) + 0.000907f*sin(2.f*Gamma)
156 - 0.002697f*cos(3.f*Gamma) + 0.00148f*sin(3.f*Gamma);
157
158 //equation of time (Iqbal Eq. 1.4.1 after Spencer)
159 EoT = 229.18f*(0.000075f + 0.001868f*cos(Gamma) - 0.032077f*sin(Gamma)
160 - 0.014615f*cos(2.f*Gamma) - 0.04089f*sin(2.f*Gamma));
161
162 time_dec=time.hour+time.minute/60.f; //(hours)
163
164 LSTM=15.f*float(UTC); //degrees
165
166 TC=4.f*(LSTM-longitude)+EoT; //minutes
167 LST=time_dec+TC/60.f; //hours
168
169 h=(LST-12.f)*15.f*rad; //hour angle (rad)
170
171 //solar zentih angle
172 theta = asin_safe( sin(latitude*rad)*sin(delta) + cos(latitude*rad)*cos(delta)*cos(h) ); //(rad)
173
174 assert( theta>-0.5f*M_PI && theta<0.5f*M_PI );
175
176 //solar elevation angle
177 phi = acos_safe( (sin(delta) - sin(theta)*sin(latitude*rad))/(cos(theta)*cos(latitude*rad)));
178
179 if( LST>12.f ){
180 phi=2.f*M_PI-phi;
181 }
182
183 assert( phi>0 && phi<2.f*M_PI );
184
185 return make_SphericalCoord(theta,phi);
186

314void SolarPosition::GueymardSolarModel( float pressure, float temperature, float humidity, float turbidity, float& Eb_PAR, float& Eb_NIR, float &fdiff ) const{
315
316 float beta = turbidity;
317
318 uint DOY = context->getJulianDate();
319
320 float theta = getSunZenith();
321
322 if( theta>0.5*M_PI ){
323 Eb_PAR = 0.f;
324 Eb_NIR = 0.f;
325 fdiff = 1.f;
326 return;
327 }
328
329 float m=pow(cos(theta)+0.15*pow(93.885-theta*180/M_PI,-1.25),-1);
330
331 float E0_PAR=635.4;
332 float E0_NIR=709.7;
333
334 vec2 alpha(1.3,1.3);
335
336 //---- Rayleigh ----//
337 //NOTE: Rayleigh scattering dominates the atmospheric attenuation, and thus variations in the model predictions are almost entirely due to pressure (and theta)
338 float mR=1.f/(cos(theta)+0.48353*pow(theta*180/M_PI,0.095846)/pow(96.741-theta*180/M_PI,1.754));
339
340 float mR_p=mR*pressure/101325;
341
342 float TR_PAR=(1.f+1.8169*mR_p-0.033454*mR_p*mR_p)/(1.f+2.063*mR_p+0.31978*mR_p*mR_p);
343
344 float TR_NIR=(1.f-0.010394*mR_p)/(1.f-0.00011042*mR_p*mR_p);
345
346 //---- Uniform gasses ----//
347 mR=1.f/(cos(theta)+0.48353*pow(theta*180/M_PI,0.095846)/pow(96.741-theta*180/M_PI,1.754));
348
349 mR_p=mR*pressure/101325;
350
351 float Tg_PAR=(1.f+0.95885*mR_p+0.012871*mR_p*mR_p)/(1.f+0.96321*mR_p+0.015455*mR_p*mR_p);
352
353 float Tg_NIR=(1.f+0.27284*mR_p-0.00063699*mR_p*mR_p)/(1.f+0.30306*mR_p);
354
355 float BR_PAR=0.5f*(0.89013-0.0049558*mR+0.000045721*mR*mR);
356
357 float BR_NIR=0.5;
358
359 float Ba=1.f-exp(-0.6931-1.8326*cos(theta));
360
361 //---- Ozone -----//
362 float uo=(235+(150+40*sin(0.9856*(DOY-30)*M_PI/180.f)+20*sin(3*(longitude*M_PI/180.f+20)))*pow(sin(1.28*latitude*M_PI/180.f),2))*0.001f; //O3 atm-cm
363 //NOTE: uo model from van Heuklon (1979)
364 float mo=m;
365
366 float f1=uo*(10.979-8.5421*uo)/(1.f+2.0115*uo+40.189*uo*uo);
367 float f2=uo*(-0.027589-0.005138*uo)/(1.f-2.4857*uo+13.942*uo*uo);
368 float f3=uo*(10.995-5.5001*uo)/(1.f+1.678*uo+42.406*uo*uo);
369 float To_PAR=(1.f+f1*mo+f2*mo*mo)/(1.f+f3*mo);
370
371 float To_NIR=1.f;
372
373 //---- Nitrogen ---- //
374 float un=0.0002; //N atm-cm
375 float mw=1.f/(cos(theta)+1.065*pow(theta*180/M_PI,1.6132)/pow(111.55-theta*180/M_PI,3.2629));
376
377 float g1=(0.17499+41.654*un-2146.4*un*un)/(1+22295*un*un);
378 float g2=un*(-1.2134+59.324*un)/(1.f+8847.8*un*un);
379 float g3=(0.17499+61.658*un+9196.4*un*un)/(1.f+74109*un*un);
380 float Tn_PAR=fmin(1.f,(1.f+g1*mw+g2*mw*mw)/(1.f+g3*mw));
381 float Tn_PAR_p=fmin(1.f,(1.f+g1*1.66+g2*1.66*1.66)/(1.f+g3*1.66));
382
383 float Tn_NIR=1.f;
384 float Tn_NIR_p=1.f;
385
386 //---- Water -----//
387 float gamma=log(humidity)+17.67*(temperature-273)/(243+25);
388 float tdp=243*gamma/(17.67-gamma)*9/5+32;//dewpoint temperature in Fahrenheit
389 float w=exp((0.1133-log(4.0+1))+0.0393*tdp); //cm of precipitable water
390 //NOTE: precipitable water model from Viswanadham (1981), Eq. 5
391 mw=1.f/(cos(theta)+1.1212*pow(theta*180/M_PI,0.6379)/pow(93.781-theta*180/M_PI,1.9203));
392
393 float h1=w*(0.065445+0.00029901*w)/(1.f+1.2728*w);
394 float h2=w*(0.065687+0.0013218*w)/(1.f+1.2008*w);
395 float Tw_PAR=(1.f+h1*mw)/(1.f+h2*mw);
396 float Tw_PAR_p=(1.f+h1*1.66)/(1.f+h2*1.66);
397
398 float c1=w*(19.566-1.6506*w+1.0672*w*w)/(1.f+5.4248*w+1.6005*w*w);
399 float c2=w*(0.50158-0.14732*w+0.047584*w*w)/(1.f+1.1811*w+1.0699*w*w);
400 float c3=w*(21.286-0.39232*w+1.2692*w*w)/(1.f+4.8318*w+1.412*w*w);
401 float c4=w*(0.70992-0.23155*w+0.096541*w*w)/(1.f+0.44907*w+0.75425*w*w);
402 float Tw_NIR=(1.f+c1*mw+c2*mw*mw)/(1.f+c3*mw+c4*mw*mw);
403 float Tw_NIR_p=(1.f+c1*1.66+c2*1.66*1.66)/(1.f+c3*1.66+c4*1.66*1.66);
404
405 //---- Aerosol ----//
406 float ma=1.f/(cos(theta)+0.16851*pow(theta*180/M_PI,0.18198)/pow(95.318-theta*180/M_PI,1.9542));
407 float ua=log(1.f+ma*beta);
408
409 float d0=0.57664-0.024743*alpha.x;
410 float d1=(0.093942-0.2269*alpha.x+0.12848*alpha.x*alpha.x)/(1.f+0.6418*alpha.x);
411 float d2=(-0.093819+0.36668*alpha.x-0.12775*alpha.x*alpha.x)/(1.f-0.11651*alpha.x);
412 float d3=alpha.x*(0.15232-0.08721*alpha.x+0.012664*alpha.x*alpha.x)/(1.f-0.90454*alpha.x+0.26167*alpha.x*alpha.x);
413 float lambdae_PAR=(d0+d1*ua+d2*ua*ua)/(1.f+d3*ua*ua);
414 float Ta_PAR=exp(-ma*beta*pow(lambdae_PAR,-alpha.x));
415
416 float e0 = (1.183-0.022989*alpha.y+0.020829*alpha.y*alpha.y)/(1.f+0.11133*alpha.y);
417 float e1 = (-0.50003-0.18329*alpha.y+0.23835*alpha.y*alpha.y)/(1.f+1.6756*alpha.y);
418 float e2 = (-0.50001+1.1414*alpha.y+0.0083589*alpha.y*alpha.y)/(1.f+11.168*alpha.y);
419 float e3 = (-0.70003-0.73587*alpha.y+0.51509*alpha.y*alpha.y)/(1.f+4.7665*alpha.y);
420 float lambdae_NIR = (e0+e1*ua+e2*ua*ua)/(1.f+e3*ua);
421 float Ta_NIR=exp(-ma*beta*pow(lambdae_NIR,-alpha.y));
422
423 float omega_PAR=1.0;
424 float omega_NIR=1.0;
425
426 float Tas_PAR=exp(-ma*omega_PAR*beta*pow(lambdae_PAR,-alpha.x));
427
428 float Tas_NIR=exp(-ma*omega_NIR*beta*pow(lambdae_NIR,-alpha.y));
429
430 //direct irradiation
431 Eb_PAR=TR_PAR*Tg_PAR*To_PAR*Tn_PAR*Tw_PAR*Ta_PAR*E0_PAR;
432 Eb_NIR=TR_NIR*Tg_NIR*To_NIR*Tn_NIR*Tw_NIR*Ta_NIR*E0_NIR;
433 float Eb=Eb_PAR+Eb_NIR;
434
435 //diffuse irradiation
436 float Edp_PAR=To_PAR*Tg_PAR*Tn_PAR_p*Tw_PAR_p*(BR_PAR*(1.f-TR_PAR)*pow(Ta_PAR,0.25)+Ba*TR_PAR*(1.f-pow(Tas_PAR,0.25)))*E0_PAR;
437 float Edp_NIR=To_NIR*Tg_NIR*Tn_NIR_p*Tw_NIR_p*(BR_NIR*(1.f-TR_NIR)*pow(Ta_NIR,0.25)+Ba*TR_NIR*(1.f-pow(Tas_NIR,0.25)))*E0_NIR;
438 float Edp=Edp_PAR+Edp_NIR;
439
440 //diffuse fraction
441 fdiff = Edp/(Eb+Edp);
442
443 //fraction can't be greater than 1.0
444 if( fdiff>1.0 ){
445 fdiff=1;
446 }
447
448 //fraction can't be less than 0
449 if( fdiff<0 ){
450 fdiff=0;
451 }
452

456float SolarPosition::getAmbientLongwaveFlux(float temperature_K, float humidity_rel ) const{
457
458 //Model from Prata (1996) Q. J. R. Meteorol. Soc.
459
460 float e0 = 611.f * exp(17.502f * (temperature_K - 273.f) / ((temperature_K - 273.f) + 240.9f)) * humidity_rel; //Pascals
461
462 float K = 0.465f; //cm-K/Pa
463
464 float xi = e0 / temperature_K * K;
465 float eps = 1.f-(1.f+xi)*exp(-sqrt(1.2f+3.f*xi));
466
467 return eps*5.67e-8*pow(temperature_K, 4);
468
469}

495 for( int t=0; t<length; t++ ){
496 float flux = context->queryTimeseriesData(timeseries_shortwave_flux_label_Wm2.c_str(), t );
497 if( flux<min_flux ){
498 min_flux = flux;
499 }
500 if( flux>max_flux ){
501 max_flux = flux;
502 max_flux_index = t;
503 }
504 }

506 if( max_flux<750 || max_flux>1200 ){
507 helios_runtime_error("ERROR (SolarPosition::calibrateTurbidityFromTimeseries): The maximum flux for the timeseries data is not within the expected range. Either it is not solar flux data, or there are no clear sky days in the dataset");
508 }else if( min_flux<0 ){
509 helios_runtime_error("ERROR (SolarPosition::calibrateTurbidityFromTimeseries): The minimum flux for the timeseries data is negative. Solar fluxes cannot be negative.");
510 }
511
512 std::vector<float> parameters{ 101325, 300, 0.5, max_flux };
513
514 SolarPosition solarposition_copy( UTC, latitude, longitude, context );
515 Date date_max = context->queryTimeseriesDate(timeseries_shortwave_flux_label_Wm2.c_str(), max_flux_index );

519
520 float turbidity = fzero( turbidityResidualFunction, parameters, &solarposition_copy, 0.01 );
