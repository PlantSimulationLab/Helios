51SphericalCoord SolarPosition::calculateSunDirection( const helios::Time &time, const helios::Date &date ) const{
52
53 int solstice_day, LSTM;
54 float Gamma, delta, time_dec, B, EoT, TC, LST, h, theta, phi, rad;
55
56 rad=M_PI/180.f;
57
58 solstice_day=81;
59
60 //day angle (Iqbal Eq. 1.1.2)
61 Gamma = 2.f*M_PI*(float(date.JulianDay()-1))/365.f;
62
63 //solar declination angle (Iqbal Eq. 1.3.1 after Spencer)
64 delta = 0.006918f - 0.399912f*cos(Gamma) + 0.070257f*sin(Gamma)
65 - 0.006758f*cos(2.f*Gamma) + 0.000907f*sin(2.f*Gamma)
66 - 0.002697f*cos(3.f*Gamma) + 0.00148f*sin(3.f*Gamma);
67
68 //equation of time (Iqbal Eq. 1.4.1 after Spencer)
69 EoT = 229.18f*(0.000075f + 0.001868f*cos(Gamma) - 0.032077f*sin(Gamma)
70 - 0.014615f*cos(2.f*Gamma) - 0.04089f*sin(2.f*Gamma));
71
72 time_dec=time.hour+time.minute/60.f; //(hours)
73
74 LSTM=15.f*float(UTC); //degrees
75
76 TC=4.f*(LSTM-longitude)+EoT; //minutes
77 LST=time_dec+TC/60.f; //hours
78
79 h=(LST-12.f)*15.f*rad; //hour angle (rad)
80
81 //solar zentih angle
82 theta = asin_safe( sin(latitude*rad)*sin(delta) + cos(latitude*rad)*cos(delta)*cos(h) ); //(rad)
83
84 assert( theta>-0.5f*M_PI && theta<0.5f*M_PI );
85
86 //solar elevation angle
87 phi = acos_safe( (sin(delta) - sin(theta)*sin(latitude*rad))/(cos(theta)*cos(latitude*rad)));
88
89 if( LST>12.f ){
90 phi=2.f*M_PI-phi;
91 }
92
93 assert( phi>0 && phi<2.f*M_PI );
94
95 return make_SphericalCoord(theta,phi);
96
97}

224void SolarPosition::GueymardSolarModel( float pressure, float temperature, float humidity, float turbidity, float& Eb_PAR, float& Eb_NIR, float &fdiff ) const{
225
226 float beta = turbidity;
227
228 uint DOY = context->getJulianDate();
229
230 float theta = getSunZenith();
231
232 if( theta>0.5*M_PI ){
233 Eb_PAR = 0.f;
234 Eb_NIR = 0.f;
235 fdiff = 1.f;
236 return;
237 }
238
239 float m=pow(cos(theta)+0.15*pow(93.885-theta*180/M_PI,-1.25),-1);
240
241 float E0_PAR=635.4;
242 float E0_NIR=709.7;
243
244 vec2 alpha(1.3,1.3);
245
246 //---- Rayleigh ----//
247 //NOTE: Rayleigh scattering dominates the atmospheric attenuation, and thus variations in the model predictions are almost entirely due to pressure (and theta)
248 float mR=1.f/(cos(theta)+0.48353*pow(theta*180/M_PI,0.095846)/pow(96.741-theta*180/M_PI,1.754));
249
250 float mR_p=mR*pressure/101325;
251
252 float TR_PAR=(1.f+1.8169*mR_p-0.033454*mR_p*mR_p)/(1.f+2.063*mR_p+0.31978*mR_p*mR_p);
253
254 float TR_NIR=(1.f-0.010394*mR_p)/(1.f-0.00011042*mR_p*mR_p);
255
256 //---- Uniform gasses ----//
257 mR=1.f/(cos(theta)+0.48353*pow(theta*180/M_PI,0.095846)/pow(96.741-theta*180/M_PI,1.754));
258
259 mR_p=mR*pressure/101325;
260
261 float Tg_PAR=(1.f+0.95885*mR_p+0.012871*mR_p*mR_p)/(1.f+0.96321*mR_p+0.015455*mR_p*mR_p);
262
263 float Tg_NIR=(1.f+0.27284*mR_p-0.00063699*mR_p*mR_p)/(1.f+0.30306*mR_p);
264
265 float BR_PAR=0.5f*(0.89013-0.0049558*mR+0.000045721*mR*mR);
266
267 float BR_NIR=0.5;
268
269 float Ba=1.f-exp(-0.6931-1.8326*cos(theta));
270
271 //---- Ozone -----//
272 float uo=(235+(150+40*sin(0.9856*(DOY-30)*M_PI/180.f)+20*sin(3*(longitude*M_PI/180.f+20)))*pow(sin(1.28*latitude*M_PI/180.f),2))*0.001f; //O3 atm-cm
273 //NOTE: uo model from van Heuklon (1979)
274 float mo=m;
275
276 float f1=uo*(10.979-8.5421*uo)/(1.f+2.0115*uo+40.189*uo*uo);
277 float f2=uo*(-0.027589-0.005138*uo)/(1.f-2.4857*uo+13.942*uo*uo);
278 float f3=uo*(10.995-5.5001*uo)/(1.f+1.678*uo+42.406*uo*uo);
279 float To_PAR=(1.f+f1*mo+f2*mo*mo)/(1.f+f3*mo);
280
281 float To_NIR=1.f;
282
283 //---- Nitrogen ---- //
284 float un=0.0002; //N atm-cm
285 float mw=1.f/(cos(theta)+1.065*pow(theta*180/M_PI,1.6132)/pow(111.55-theta*180/M_PI,3.2629));
286
287 float g1=(0.17499+41.654*un-2146.4*un*un)/(1+22295*un*un);
288 float g2=un*(-1.2134+59.324*un)/(1.f+8847.8*un*un);
289 float g3=(0.17499+61.658*un+9196.4*un*un)/(1.f+74109*un*un);
290 float Tn_PAR=fmin(1.f,(1.f+g1*mw+g2*mw*mw)/(1.f+g3*mw));
291 float Tn_PAR_p=fmin(1.f,(1.f+g1*1.66+g2*1.66*1.66)/(1.f+g3*1.66));
292
293 float Tn_NIR=1.f;
294 float Tn_NIR_p=1.f;
295
296 //---- Water -----//
297 float gamma=log(humidity)+17.67*(temperature-273)/(243+25);
298 float tdp=243*gamma/(17.67-gamma)*9/5+32;//dewpoint temperature in Fahrenheit
299 float w=exp((0.1133-log(4.0+1))+0.0393*tdp); //cm of precipitable water
300 //NOTE: precipitable water model from Viswanadham (1981), Eq. 5
301 mw=1.f/(cos(theta)+1.1212*pow(theta*180/M_PI,0.6379)/pow(93.781-theta*180/M_PI,1.9203));
302
303 float h1=w*(0.065445+0.00029901*w)/(1.f+1.2728*w);
304 float h2=w*(0.065687+0.0013218*w)/(1.f+1.2008*w);
305 float Tw_PAR=(1.f+h1*mw)/(1.f+h2*mw);
306 float Tw_PAR_p=(1.f+h1*1.66)/(1.f+h2*1.66);
307
308 float c1=w*(19.566-1.6506*w+1.0672*w*w)/(1.f+5.4248*w+1.6005*w*w);
309 float c2=w*(0.50158-0.14732*w+0.047584*w*w)/(1.f+1.1811*w+1.0699*w*w);
310 float c3=w*(21.286-0.39232*w+1.2692*w*w)/(1.f+4.8318*w+1.412*w*w);
311 float c4=w*(0.70992-0.23155*w+0.096541*w*w)/(1.f+0.44907*w+0.75425*w*w);
312 float Tw_NIR=(1.f+c1*mw+c2*mw*mw)/(1.f+c3*mw+c4*mw*mw);
313 float Tw_NIR_p=(1.f+c1*1.66+c2*1.66*1.66)/(1.f+c3*1.66+c4*1.66*1.66);
314
315 //---- Aerosol ----//
316 float ma=1.f/(cos(theta)+0.16851*pow(theta*180/M_PI,0.18198)/pow(95.318-theta*180/M_PI,1.9542));
317 float ua=log(1.f+ma*beta);
318
319 float d0=0.57664-0.024743*alpha.x;
320 float d1=(0.093942-0.2269*alpha.x+0.12848*alpha.x*alpha.x)/(1.f+0.6418*alpha.x);
321 float d2=(-0.093819+0.36668*alpha.x-0.12775*alpha.x*alpha.x)/(1.f-0.11651*alpha.x);
322 float d3=alpha.x*(0.15232-0.08721*alpha.x+0.012664*alpha.x*alpha.x)/(1.f-0.90454*alpha.x+0.26167*alpha.x*alpha.x);
323 float lambdae_PAR=(d0+d1*ua+d2*ua*ua)/(1.f+d3*ua*ua);
324 float Ta_PAR=exp(-ma*beta*pow(lambdae_PAR,-alpha.x));
325
326 float e0 = (1.183-0.022989*alpha.y+0.020829*alpha.y*alpha.y)/(1.f+0.11133*alpha.y);
327 float e1 = (-0.50003-0.18329*alpha.y+0.23835*alpha.y*alpha.y)/(1.f+1.6756*alpha.y);
328 float e2 = (-0.50001+1.1414*alpha.y+0.0083589*alpha.y*alpha.y)/(1.f+11.168*alpha.y);
329 float e3 = (-0.70003-0.73587*alpha.y+0.51509*alpha.y*alpha.y)/(1.f+4.7665*alpha.y);
330 float lambdae_NIR = (e0+e1*ua+e2*ua*ua)/(1.f+e3*ua);
331 float Ta_NIR=exp(-ma*beta*pow(lambdae_NIR,-alpha.y));
332
333 float omega_PAR=1.0;
334 float omega_NIR=1.0;
335
336 float Tas_PAR=exp(-ma*omega_PAR*beta*pow(lambdae_PAR,-alpha.x));
337
338 float Tas_NIR=exp(-ma*omega_NIR*beta*pow(lambdae_NIR,-alpha.y));
339
340 //direct irradiation
341 Eb_PAR=TR_PAR*Tg_PAR*To_PAR*Tn_PAR*Tw_PAR*Ta_PAR*E0_PAR;
342 Eb_NIR=TR_NIR*Tg_NIR*To_NIR*Tn_NIR*Tw_NIR*Ta_NIR*E0_NIR;
343 float Eb=Eb_PAR+Eb_NIR;
344
345 //diffuse irradiation
346 float Edp_PAR=To_PAR*Tg_PAR*Tn_PAR_p*Tw_PAR_p*(BR_PAR*(1.f-TR_PAR)*pow(Ta_PAR,0.25)+Ba*TR_PAR*(1.f-pow(Tas_PAR,0.25)))*E0_PAR;
347 float Edp_NIR=To_NIR*Tg_NIR*Tn_NIR_p*Tw_NIR_p*(BR_NIR*(1.f-TR_NIR)*pow(Ta_NIR,0.25)+Ba*TR_NIR*(1.f-pow(Tas_NIR,0.25)))*E0_NIR;
348 float Edp=Edp_PAR+Edp_NIR;
349
350 //diffuse fraction
351 fdiff = Edp/(Eb+Edp);
352
353 //fraction can't be greater than 1.0
354 if( fdiff>1.0 ){
355 fdiff=1;
356 }
357
358 //fraction can't be less than 0
359 if( fdiff<0 ){
360 fdiff=0;
361 }
362
363
364}

366float SolarPosition::getAmbientLongwaveFlux(float temperature_K, float humidity_rel ) const{
367
368 //Model from Prata (1996) Q. J. R. Meteorol. Soc.
369
370 float e0 = 611.f * exp(17.502f * (temperature_K - 273.f) / ((temperature_K - 273.f) + 240.9f)) * humidity_rel; //Pascals
371
372 float K = 0.465f; //cm-K/Pa
373
374 float xi = e0 / temperature_K * K;
375 float eps = 1.f-(1.f+xi)*exp(-sqrt(1.2f+3.f*xi));
376
377 return eps*5.67e-8*pow(temperature_K, 4);
378
379}

394float SolarPosition::calibrateTurbidityFromTimeseries( const std::string &timeseries_shortwave_flux_label_Wm2 ) const{
395
396 if( !context->doesTimeseriesVariableExist(timeseries_shortwave_flux_label_Wm2.c_str()) ){
397 helios_runtime_error("ERROR (SolarPosition::calibrateTurbidityFromTimeseries): Timeseries variable " + timeseries_shortwave_flux_label_Wm2 + " does not exist.");
398 }
399
400 uint length = context->getTimeseriesLength(timeseries_shortwave_flux_label_Wm2.c_str() );
401
402 float min_flux = 1e6;
403 float max_flux = 0;
404 int max_flux_index = 0;
405 for( int t=0; t<length; t++ ){
406 float flux = context->queryTimeseriesData(timeseries_shortwave_flux_label_Wm2.c_str(), t );
407 if( flux<min_flux ){
408 min_flux = flux;
409 }
410 if( flux>max_flux ){
411 max_flux = flux;
412 max_flux_index = t;
413 }
414 }
415
416 if( max_flux<750 || max_flux>1200 ){
417 helios_runtime_error("ERROR (SolarPosition::calibrateTurbidityFromTimeseries): The maximum flux for the timeseries data is not within the expected range. Either it is not solar flux data, or there are no clear sky days in the dataset");
418 }else if( min_flux<0 ){
419 helios_runtime_error("ERROR (SolarPosition::calibrateTurbidityFromTimeseries): The minimum flux for the timeseries data is negative. Solar fluxes cannot be negative.");
420 }
421
422 std::vector<float> parameters{ 101325, 300, 0.5, max_flux };
423
424 SolarPosition solarposition_copy( UTC, latitude, longitude, context );
425 Date date_max = context->queryTimeseriesDate(timeseries_shortwave_flux_label_Wm2.c_str(), max_flux_index );
426 Time time_max = context->queryTimeseriesTime(timeseries_shortwave_flux_label_Wm2.c_str(), max_flux_index );
427
428 solarposition_copy.setSunDirection( solarposition_copy.calculateSunDirection( time_max, date_max) );
429
430 float turbidity = fzero( turbidityResidualFunction, parameters, &solarposition_copy, 0.01 );
431
432 return turbidity;
433
434}

450void SolarPosition::applyCloudCalibration(float &R_calc_Wm2, float &fdiff_calc) const{
451
452 assert( context->doesTimeseriesVariableExist( cloudcalibrationlabel.c_str()) );
453
454 float R_meas = context->queryTimeseriesData( cloudcalibrationlabel.c_str() );
455 float R_calc_horiz = R_calc_Wm2*cosf(getSunZenith());
456
457 float fdiff = fmin(fmax(0, 1.f-(R_meas - R_calc_horiz) / (R_calc_horiz)), 1);
458 float R = R_calc_Wm2 * R_meas / R_calc_horiz;
459
460 if( fdiff>0.001 && R_calc_horiz>1.f ){
461 R_calc_Wm2 = R;
462 fdiff_calc = fdiff;
463 }
