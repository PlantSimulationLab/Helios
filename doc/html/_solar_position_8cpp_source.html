142SphericalCoord SolarPosition::calculateSunDirection( const helios::Time &time, const helios::Date &date ) const{
143
144 int solstice_day, LSTM;
145 float Gamma, delta, time_dec, B, EoT, TC, LST, h, theta, phi, rad;
146
147 rad=M_PI/180.f;
148
149 solstice_day=81;
150
151 //day angle (Iqbal Eq. 1.1.2)
152 Gamma = 2.f*M_PI*(float(date.JulianDay()-1))/365.f;
153
154 //solar declination angle (Iqbal Eq. 1.3.1 after Spencer)
155 delta = 0.006918f - 0.399912f*cos(Gamma) + 0.070257f*sin(Gamma)
156 - 0.006758f*cos(2.f*Gamma) + 0.000907f*sin(2.f*Gamma)
157 - 0.002697f*cos(3.f*Gamma) + 0.00148f*sin(3.f*Gamma);
158
159 //equation of time (Iqbal Eq. 1.4.1 after Spencer)
160 EoT = 229.18f*(0.000075f + 0.001868f*cos(Gamma) - 0.032077f*sin(Gamma)
161 - 0.014615f*cos(2.f*Gamma) - 0.04089f*sin(2.f*Gamma));
162
163 time_dec=time.hour+time.minute/60.f; //(hours)
164
165 LSTM=15.f*float(UTC); //degrees
166
167 TC=4.f*(LSTM-longitude)+EoT; //minutes
168 LST=time_dec+TC/60.f; //hours
169
170 h=(LST-12.f)*15.f*rad; //hour angle (rad)
171
172 //solar zentih angle
173 theta = asin_safe( sin(latitude*rad)*sin(delta) + cos(latitude*rad)*cos(delta)*cos(h) ); //(rad)
174
175 assert( theta>-0.5f*M_PI && theta<0.5f*M_PI );
176
177 //solar elevation angle
178 phi = acos_safe( (sin(delta) - sin(theta)*sin(latitude*rad))/(cos(theta)*cos(latitude*rad)));
179
180 if( LST>12.f ){
181 phi=2.f*M_PI-phi;
182 }
183
184 assert( phi>0 && phi<2.f*M_PI );
185
186 return make_SphericalCoord(theta,phi);

302void SolarPosition::GueymardSolarModel( float pressure, float temperature, float humidity, float turbidity, float& Eb_PAR, float& Eb_NIR, float &fdiff ){
303
304 float beta = turbidity;
305
306 uint DOY = context->getJulianDate();
307
308 float theta = getSunZenith();
309
310 if( theta>0.5*M_PI ){
311 Eb_PAR = 0.f;
312 Eb_NIR = 0.f;
313 fdiff = 1.f;
314 return;
315 }
316
317 float m=pow(cos(theta)+0.15*pow(93.885-theta*180/M_PI,-1.25),-1);
318
319 float E0_PAR=635.4;
320 float E0_NIR=709.7;
321
322 vec2 alpha(1.3,1.3);
323
324 //---- Rayleigh ----//
325 //NOTE: Rayleigh scattering dominates the atmospheric attenuation, and thus variations in the model predictions are almost entirely due to pressure (and theta)
326 float mR=1.f/(cos(theta)+0.48353*pow(theta*180/M_PI,0.095846)/pow(96.741-theta*180/M_PI,1.754));
327
328 float mR_p=mR*pressure/101325;
329
330 float TR_PAR=(1.f+1.8169*mR_p-0.033454*mR_p*mR_p)/(1.f+2.063*mR_p+0.31978*mR_p*mR_p);
331
332 float TR_NIR=(1.f-0.010394*mR_p)/(1.f-0.00011042*mR_p*mR_p);
333
334 //---- Uniform gasses ----//
335 mR=1.f/(cos(theta)+0.48353*pow(theta*180/M_PI,0.095846)/pow(96.741-theta*180/M_PI,1.754));
336
337 mR_p=mR*pressure/101325;
338
339 float Tg_PAR=(1.f+0.95885*mR_p+0.012871*mR_p*mR_p)/(1.f+0.96321*mR_p+0.015455*mR_p*mR_p);
340
341 float Tg_NIR=(1+0.27284*mR_p-0.00063699*mR_p*mR_p)/(1.f+0.30306*mR_p);
342
343 float BR_PAR=0.5*(0.89013-0.0049558*mR+0.000045721*mR*mR);
344
345 float BR_NIR=0.5;
346
347 float Ba=1-exp(-0.6931-1.8326*cos(theta));
348
349 //---- Ozone -----//
350 float uo=(235+(150+40*sin(0.9856*(DOY-30)*M_PI/180.f)+20*sin(3*(longitude*M_PI/180.f+20)))*pow(sin(1.28*latitude*M_PI/180.f),2))*0.001f; //O3 atm-cm
351 //NOTE: uo model from van Heuklon (1979)
352 float mo=m;
353
354 float f1=uo*(10.979-8.5421*uo)/(1.f+2.0115*uo+40.189*uo*uo);
355 float f2=uo*(-0.027589-0.005138*uo)/(1.f-2.4857*uo+13.942*uo*uo);
356 float f3=uo*(10.995-5.5001*uo)/(1.f+1.678*uo+42.406*uo*uo);
357 float To_PAR=(1.f+f1*mo+f2*mo*mo)/(1.f+f3*mo);
358
359 float To_NIR=1.f;
360
361 //---- Nitrogen ---- //
362 float un=0.0002; //N atm-cm
363 float mw=1.f/(cos(theta)+1.065*pow(theta*180/M_PI,1.6132)/pow(111.55-theta*180/M_PI,3.2629));
364
365 float g1=(0.17499+41.654*un-2146.4*un*un)/(1+22295*un*un);
366 float g2=un*(-1.2134+59.324*un)/(1.f+8847.8*un*un);
367 float g3=(0.17499+61.658*un+9196.4*un*un)/(1.f+74109*un*un);
368 float Tn_PAR=fmin(1.f,(1.f+g1*mw+g2*mw*mw)/(1.f+g3*mw));
369 float Tn_PAR_p=fmin(1.f,(1.f+g1*1.66+g2*1.66*1.66)/(1.f+g3*1.66));
370
371 float Tn_NIR=1.f;
372 float Tn_NIR_p=1.f;
373
374 //---- Water -----//
375 float gamma=log(humidity)+17.67*(temperature-273)/(243+25);
376 float tdp=243*gamma/(17.67-gamma)*9/5+32;//dewpoint temperature in Fahrenheit
377 float w=exp((0.1133-log(4.0+1))+0.0393*tdp); //cm of precipitable water
378 //NOTE: precipitable water model from Viswanadham (1981), Eq. 5
379 mw=1.f/(cos(theta)+1.1212*pow(theta*180/M_PI,0.6379)/pow(93.781-theta*180/M_PI,1.9203));
380
381 float h1=w*(0.065445+0.00029901*w)/(1.f+1.2728*w);
382 float h2=w*(0.065687+0.0013218*w)/(1.f+1.2008*w);
383 float Tw_PAR=(1.f+h1*mw)/(1.f+h2*mw);
384 float Tw_PAR_p=(1.f+h1*1.66)/(1.f+h2*1.66);
385
386 float c1=w*(19.566-1.6506*w+1.0672*w*w)/(1.f+5.4248*w+1.6005*w*w);
387 float c2=w*(0.50158-0.14732*w+0.047584*w*w)/(1.f+1.1811*w+1.0699*w*w);
388 float c3=w*(21.286-0.39232*w+1.2692*w*w)/(1.f+4.8318*w+1.412*w*w);
389 float c4=w*(0.70992-0.23155*w+0.096541*w*w)/(1.f+0.44907*w+0.75425*w*w);
390 float Tw_NIR=(1.f+c1*mw+c2*mw*mw)/(1.f+c3*mw+c4*mw*mw);
391 float Tw_NIR_p=(1.f+c1*1.66+c2*1.66*1.66)/(1.f+c3*1.66+c4*1.66*1.66);
392
393 //---- Aerosol ----//
394 float ma=1.f/(cos(theta)+0.16851*pow(theta*180/M_PI,0.18198)/pow(95.318-theta*180/M_PI,1.9542));
395 float ua=log(1.f+ma*beta);
396
397 float d0=0.57664-0.024743*alpha.x;
398 float d1=(0.093942-0.2269*alpha.x+0.12848*alpha.x*alpha.x)/(1.f+0.6418*alpha.x);
399 float d2=(-0.093819+0.36668*alpha.x-0.12775*alpha.x*alpha.x)/(1.f-0.11651*alpha.x);
400 float d3=alpha.x*(0.15232-0.08721*alpha.x+0.012664*alpha.x*alpha.x)/(1.f-0.90454*alpha.x+0.26167*alpha.x*alpha.x);
401 float lambdae_PAR=(d0+d1*ua+d2*ua*ua)/(1.f+d3*ua*ua);
402 float Ta_PAR=exp(-ma*beta*pow(lambdae_PAR,-alpha.x));
403
404 float e0 = (1.183-0.022989*alpha.y+0.020829*alpha.y*alpha.y)/(1.f+0.11133*alpha.y);
405 float e1 = (-0.50003-0.18329*alpha.y+0.23835*alpha.y*alpha.y)/(1.f+1.6756*alpha.y);
406 float e2 = (-0.50001+1.1414*alpha.y+0.0083589*alpha.y*alpha.y)/(1.f+11.168*alpha.y);
407 float e3 = (-0.70003-0.73587*alpha.y+0.51509*alpha.y*alpha.y)/(1.f+4.7665*alpha.y);
408 float lambdae_NIR = (e0+e1*ua+e2*ua*ua)/(1.f+e3*ua);
409 float Ta_NIR=exp(-ma*beta*pow(lambdae_NIR,-alpha.y));
410
411 float omega_PAR=1.0;
412 float omega_NIR=1.0;
413
414 float Tas_PAR=exp(-ma*omega_PAR*beta*pow(lambdae_PAR,-alpha.x));
415
416 float Tas_NIR=exp(-ma*omega_NIR*beta*pow(lambdae_NIR,-alpha.y));
417
418 //direct irradiation
419 Eb_PAR=TR_PAR*Tg_PAR*To_PAR*Tn_PAR*Tw_PAR*Ta_PAR*E0_PAR;
420 Eb_NIR=TR_NIR*Tg_NIR*To_NIR*Tn_NIR*Tw_NIR*Ta_NIR*E0_NIR;
421 float Eb=Eb_PAR+Eb_NIR;
422
423 //diffuse irradiation
424 float Edp_PAR=To_PAR*Tg_PAR*Tn_PAR_p*Tw_PAR_p*(BR_PAR*(1.f-TR_PAR)*pow(Ta_PAR,0.25)+Ba*TR_PAR*(1.f-pow(Tas_PAR,0.25)))*E0_PAR;
425 float Edp_NIR=To_NIR*Tg_NIR*Tn_NIR_p*Tw_NIR_p*(BR_NIR*(1.f-TR_NIR)*pow(Ta_NIR,0.25)+Ba*TR_NIR*(1.f-pow(Tas_NIR,0.25)))*E0_NIR;
426 float Edp=Edp_PAR+Edp_NIR;
427
428 //diffuse fraction
429 fdiff = Edp/(Eb+Edp);
430
431 //fraction can't be greater than 1.0
432 if( fdiff>1.0 ){
433 fdiff=1;
434 }
435
436 //fraction can't be less than 0
437 if( fdiff<0 ){
438 fdiff=0;
439 }
440
441

444float SolarPosition::getAmbientLongwaveFlux( float temperature, float humidity ){
445
446 //Model from Prata (1996) Q. J. R. Meteorol. Soc.
447
448 float e0 = 611.f*exp(17.502f*(temperature-273.f)/((temperature-273.f)+240.9f))*humidity; //Pascals
449
450 float K = 0.465f; //cm-K/Pa
451
452 float xi = e0/temperature*K;
453 float eps = 1.f-(1.f+xi)*exp(-sqrt(1.2f+3.f*xi));
454
455 return eps*5.67e-8*pow(temperature,4);
