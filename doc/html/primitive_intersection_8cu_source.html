173RT_PROGRAM void triangle_intersect(int objID )
174{
175
176 if( prd.origin_UUID == triangle_UUID[objID] ){ //the ray should not intersect the primitive from which it was launched
177 return;
178 }else if( twosided_flag[objID]==2 ){ //if twosided_flag=2, ignore intersection (transparent)
179 return;

181
182 float3 v0 = triangle_vertices[ make_uint2(0, objID) ];
183 float3 v1 = triangle_vertices[ make_uint2(1, objID) ];
184 float3 v2 = triangle_vertices[ make_uint2(2, objID) ];
185
186 float a = v0.x - v1.x, b = v0.x - v2.x, c = ray.direction.x, d = v0.x - ray.origin.x;
187 float e = v0.y - v1.y, f = v0.y - v2.y, g = ray.direction.y, h = v0.y - ray.origin.y;
188 float i = v0.z - v1.z, j = v0.z - v2.z, k = ray.direction.z, l = v0.z - ray.origin.z;
189
190 float m = f * k - g * j, n = h * k - g * l, p = f * l - h * j;
191 float q = g * i - e * k, s = e * j - f * i;
192
193 float inv_denom = 1.f / (a * m + b * q + c * s);
194
195 float e1 = d * m - b * n - c * p;
196 float beta = e1 * inv_denom;
197
198 if (beta > 0.0){
199
200 float r = r = e * l - h * i;
201 float e2 = a * n + d * q + c * r;
202 float gamma = e2 * inv_denom;
203
204 if (gamma > 0.0 && beta + gamma < 1.0 ){
205
206 float e3 = a * p - b * r + d * s;
207 float t = e3 * inv_denom;
208
209 if( t>1e-8 ){
210
211 uint U = triangle_UUID[objID];
212
213 uint ID = objectID[U];
214
215 if( maskID[ID]==-1 ){ //no texture transparency
216 if( rtPotentialIntersection( t ) ) {
217 UUID = triangle_UUID[objID];
218 rtReportIntersection(0);
219 }
220 }else{ //has texture transparency
221
222 int2 sz = masksize[ maskID[ID] ];
223
224 float2 uv0 = uvdata[ make_uint2(0,uvID[ID]) ];
225 float2 uv1 = uvdata[ make_uint2(1,uvID[ID]) ];
226 float2 uv2 = uvdata[ make_uint2(2,uvID[ID]) ];
227
228 float2 uv = uv0 + beta*(uv1-uv0) + gamma*(uv2-uv0);
229// uv.x = 1.f-uv.x;

233 rtPrintf("ERROR: texture out of bounds.\n");
234 }

318RT_PROGRAM void voxel_intersect(int objID )
319{
320
321 if( prd.origin_UUID == voxel_UUID[objID] ){ //the ray should not intersect the primitive from which it was launched
322 return;
323 }else if( twosided_flag[objID]==2 ){ //if twosided_flag=2, ignore intersection (transparent)
324 return;
325 }
326
327 float x0 = voxel_vertices[ make_uint2(0, objID) ].x;
328 float y0 = voxel_vertices[ make_uint2(0, objID) ].y;
329 float z0 = voxel_vertices[ make_uint2(0, objID) ].z;
330
331 float x1 = voxel_vertices[ make_uint2(1, objID) ].x;
332 float y1 = voxel_vertices[ make_uint2(1, objID) ].y;
333 float z1 = voxel_vertices[ make_uint2(1, objID) ].z;
334
335 float ox = ray.origin.x; float oy = ray.origin.y; float oz = ray.origin.z;
336 float dx = ray.direction.x; float dy = ray.direction.y; float dz = ray.direction.z;
337
338 float tx_min, ty_min, tz_min;
339 float tx_max, ty_max, tz_max;
340
341 float a = 1.0 / dx;
342 if (a >= 0) {
343 tx_min = (x0 - ox) * a;
344 tx_max = (x1 - ox) * a;
345 }
346 else {
347 tx_min = (x1 - ox) * a;
348 tx_max = (x0 - ox) * a;
349 }
350
351 float b = 1.0 / dy;
352 if (b >= 0) {
353 ty_min = (y0 - oy) * b;
354 ty_max = (y1 - oy) * b;
355 }
356 else {
357 ty_min = (y1 - oy) * b;
358 ty_max = (y0 - oy) * b;
359 }
360
361 float c = 1.0 / dz;
362 if (c >= 0) {
363 tz_min = (z0 - oz) * c;
364 tz_max = (z1 - oz) * c;
365 }
366 else {
367 tz_min = (z1 - oz) * c;
368 tz_max = (z0 - oz) * c;
369 }
370
371 float t0, t1;
372
373 // find largest entering t value
374
375 if (tx_min > ty_min)
376 t0 = tx_min;
377 else
378 t0 = ty_min;
379
380 if (tz_min > t0)
381 t0 = tz_min;
382
383 // find smallest exiting t value
384
385 if (tx_max < ty_max)
386 t1 = tx_max;
387 else
388 t1 = ty_max;
389
390 if (tz_max < t1)
391 t1 = tz_max;
392
393 if (t0 < t1 && t0 > 1e-5){//note: if ray originated inside voxel, no intersection occurs.
394 if( rtPotentialIntersection( t0 ) ) {
395 prd.area = fabs(t1-t0);
396 UUID = voxel_UUID[objID];
397 rtReportIntersection(0);
398 }
399 }
