166 void set(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float minval_, float maxval_ ){
167
168 cmapsize = size;
169 minval = minval_;
170 maxval = maxval_;
171
172 size_t Ncolors=ctable.size();
173
174 assert( clocs.size() == Ncolors && minval_<maxval_ );
175
176 cmap.resize(Ncolors);
177
178 float cmin,cmax;
179 std::vector<float> cinds;
180 cinds.resize(Ncolors);
181
182 for( int i=0; i<Ncolors; i++ ){
183 cinds.at(i)=clocs.at(i)*float(cmapsize-1);
184 }
185
186 cmap.resize(cmapsize);
187 for( int c=0; c<Ncolors-1; c++ ){
188
189 cmin=cinds.at(c);
190 cmax=cinds.at(c+1);
191
192 for( int i=0; i<cmapsize; i++){
193
194 if( float(i)>=cmin && float(i)<=cmax ){
195
196 cmap.at(i).r=ctable.at(c).r+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).r-ctable.at(c).r);
197 cmap.at(i).g=ctable.at(c).g+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).g-ctable.at(c).g);
198 cmap.at(i).b=ctable.at(c).b+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).b-ctable.at(c).b);
199
200 }
201 }
202 }
203
204 }

206 helios::RGBcolor query( float x ) const{
207
208 assert( cmapsize>0 && !cmap.empty() );
209
210 helios::RGBcolor color;
211
212 int color_ind;
213 if( minval==maxval ){
214 color_ind=0;
215 }else{
216 color_ind=std::round( (x-minval)/(maxval-minval) * float(cmapsize-1) );
217 }
218
219 if(color_ind<0){color_ind=0;}
220 if(color_ind>cmapsize-1){color_ind=int(cmapsize-1);}
221 color.r=cmap.at(color_ind).r;
222 color.g=cmap.at(color_ind).g;
223 color.b=cmap.at(color_ind).b;
224
225 return color;
