161 Colormap(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float minval_, float maxval_) : cmapsize(size), minval(minval_), maxval(maxval_) {
162 set(ctable, clocs, size, minval_, maxval_);
163 }
164
165 void set(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float a_minval, float a_maxval) {
166 cmapsize = size;
167 minval = a_minval;
168 maxval = a_maxval;
169
170 size_t Ncolors = ctable.size();
171
172 assert(clocs.size() == Ncolors && minval < maxval );
173
174 cmap.resize(Ncolors);
175
176 std::vector<float> cinds;
177 cinds.resize(Ncolors);
178
179 for (uint i = 0; i < Ncolors; i++) {
180 cinds.at(i) = clocs.at(i) * static_cast<float>(cmapsize - 1);
181 }
182
183 cmap.resize(cmapsize);
184 for (uint c = 0; c < Ncolors - 1; c++) {
185 float cmin = cinds.at(c);
186 float cmax = cinds.at(c + 1);
187
188 for (uint i = 0; i < cmapsize; i++) {
189 auto i_f = static_cast<float>(i);
190
191 if (i_f >= cmin && i_f <= cmax) {
192 cmap.at(i).r = ctable.at(c).r + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).r - ctable.at(c).r);
193 cmap.at(i).g = ctable.at(c).g + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).g - ctable.at(c).g);
194 cmap.at(i).b = ctable.at(c).b + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).b - ctable.at(c).b);
195 }
196 }
197 }
198 }
199
200 [[nodiscard]] helios::RGBcolor query(float x) const {
201 assert(cmapsize>0 && !cmap.empty());
202
203 helios::RGBcolor color;
204
205 uint color_ind;
206 if (minval == maxval) {
207 color_ind = 0;
208 } else {
209 color_ind = std::round((x - minval) / (maxval - minval) * float(cmapsize - 1));
210 }
211
212 if (color_ind > cmapsize - 1) { color_ind = int(cmapsize - 1); }
213 color.r = cmap.at(color_ind).r;
214 color.g = cmap.at(color_ind).g;
215 color.b = cmap.at(color_ind).b;
216
217 return color;
218 }
219
220 void setRange(float min, float max) {
221 minval = min;
222 maxval = max;
223 }

225 [[nodiscard]] helios::vec2 getRange() const {
226 return {minval, maxval};

230 return minval;

234 return maxval;
