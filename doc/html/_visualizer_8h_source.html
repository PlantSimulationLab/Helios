170 void set(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float minval_, float maxval_ ){
171
172 cmapsize = size;
173 minval = minval_;
174 maxval = maxval_;
175
176 size_t Ncolors=ctable.size();
177
178 assert( clocs.size() == Ncolors && minval_<maxval_ );
179
180 cmap.resize(Ncolors);
181
182 float cmin,cmax;
183 std::vector<float> cinds;
184 cinds.resize(Ncolors);
185
186 for( int i=0; i<Ncolors; i++ ){
187 cinds.at(i)=clocs.at(i)*float(cmapsize-1);
188 }
189
190 cmap.resize(cmapsize);
191 for( int c=0; c<Ncolors-1; c++ ){
192
193 cmin=cinds.at(c);
194 cmax=cinds.at(c+1);
195
196 for( int i=0; i<cmapsize; i++){
197
198 if( float(i)>=cmin && float(i)<=cmax ){
199
200 cmap.at(i).r=ctable.at(c).r+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).r-ctable.at(c).r);
201 cmap.at(i).g=ctable.at(c).g+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).g-ctable.at(c).g);
202 cmap.at(i).b=ctable.at(c).b+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).b-ctable.at(c).b);
203
204 }
205 }
206 }
207
208 }

210 helios::RGBcolor query( float x ) const{
211
212 assert( cmapsize>0 && !cmap.empty() );
213
214 helios::RGBcolor color;
215
216 int color_ind;
217 if( minval==maxval ){
218 color_ind=0;
219 }else{
220 color_ind=std::round( (x-minval)/(maxval-minval) * float(cmapsize-1) );
221 }
222
223 if(color_ind<0){color_ind=0;}
224 if(color_ind>cmapsize-1){color_ind=int(cmapsize-1);}
225 color.r=cmap.at(color_ind).r;
226 color.g=cmap.at(color_ind).g;
227 color.b=cmap.at(color_ind).b;
228
229 return color;
