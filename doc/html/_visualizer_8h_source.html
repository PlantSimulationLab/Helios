127 Colormap(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float minval_, float maxval_) : cmapsize(size), minval(minval_), maxval(maxval_) {
128 set(ctable, clocs, size, minval_, maxval_);

130
131 void set(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float a_minval, float a_maxval) {
132 cmapsize = size;
133 minval = a_minval;
134 maxval = a_maxval;
135
136 size_t Ncolors = ctable.size();
137
138 assert(clocs.size() == Ncolors && minval < maxval);
139
140 cmap.resize(Ncolors);
141
142 std::vector<float> cinds;
143 cinds.resize(Ncolors);
144
145 for (uint i = 0; i < Ncolors; i++) {
146 cinds.at(i) = clocs.at(i) * static_cast<float>(cmapsize - 1);
147 }
148
149 cmap.resize(cmapsize);
150 for (uint c = 0; c < Ncolors - 1; c++) {
151 float cmin = cinds.at(c);
152 float cmax = cinds.at(c + 1);
153
154 for (uint i = 0; i < cmapsize; i++) {
155 auto i_f = static_cast<float>(i);
156
157 if (i_f >= cmin && i_f <= cmax) {
158 cmap.at(i).r = ctable.at(c).r + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).r - ctable.at(c).r);
159 cmap.at(i).g = ctable.at(c).g + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).g - ctable.at(c).g);
160 cmap.at(i).b = ctable.at(c).b + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).b - ctable.at(c).b);
161 }
162 }
163 }
164 }

166 [[nodiscard]] helios::RGBcolor query(float x) const {
167 assert(cmapsize > 0 && !cmap.empty());
168
169 helios::RGBcolor color;
170
171 uint color_ind;
172 if (minval == maxval) {
173 color_ind = 0;
174 } else {
175 color_ind = std::round((x - minval) / (maxval - minval) * float(cmapsize - 1));
176 }
177
178 if (color_ind > cmapsize - 1) {
179 color_ind = int(cmapsize - 1);
180 }
181 color.r = cmap.at(color_ind).r;
182 color.g = cmap.at(color_ind).g;
183 color.b = cmap.at(color_ind).b;
184
185 return color;
186 }

188 void setRange(float min, float max) {
189 minval = min;
190 maxval = max;
191 }
