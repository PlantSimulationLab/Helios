156 void set( std::vector<helios::RGBcolor> ctable, std::vector<float> clocs, int size, float minval_, float maxval_ ){
157
158 cmapsize = size;
159 minval=minval_;
160 maxval=maxval_;
161
162 int Ncolors=ctable.size();
163
164 if( clocs.size() != Ncolors ){
165 std::cerr << "ERROR: colormap - sizes of ctable and clocs must match." << std::endl;
166 exit(EXIT_FAILURE);
167 }else if( minval_==maxval_ || minval_>maxval_ ){
168 throw( std::runtime_error("ERROR(Colormap::set): Cannot initialize colormap because minimum colormap value must be less than the maximum value."));
169 }
170
171 cmap.resize(Ncolors);
172
173 float cmin,cmax;
174 std::vector<float> cinds;
175 cinds.resize(Ncolors);
176
177 for( int i=0; i<Ncolors; i++ ){
178 cinds.at(i)=clocs.at(i)*(cmapsize-1);
179 }
180
181 cmap.resize(cmapsize);
182 for( int c=0; c<Ncolors-1; c++ ){
183
184 cmin=cinds.at(c);
185 cmax=cinds.at(c+1);
186
187 for( int i=0; i<cmapsize; i++){
188
189 if( i>=cmin && i<=cmax ){
190
191 cmap.at(i).r=ctable.at(c).r+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).r-ctable.at(c).r);
192 cmap.at(i).g=ctable.at(c).g+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).g-ctable.at(c).g);
193 cmap.at(i).b=ctable.at(c).b+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).b-ctable.at(c).b);
194
195 }
196 }
197 }
198
199 }

201 helios::RGBcolor query( float x ) const{
202
203 if( cmapsize==0 || cmap.size()==0 ){
204 std::cerr << "ERROR: colormap was not initialized properly." << std::endl;
205 std::cout << cmapsize << " " << cmap.size() << std::endl;
206 exit(EXIT_FAILURE);
207 }
208
209 helios::RGBcolor color;
210
211 int color_ind;
212 if( minval==maxval ){
213 color_ind=0;
214 }else{
215 color_ind=round( (x-minval)/(maxval-minval) * (cmapsize-1) );
216 }
217
218 if(color_ind<0){color_ind=0;}
219 if(color_ind>cmapsize-1){color_ind=cmapsize-1;}
220 color.r=cmap.at(color_ind).r;
221 color.g=cmap.at(color_ind).g;
222 color.b=cmap.at(color_ind).b;
223
224 return color;
