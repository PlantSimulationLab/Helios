160 Colormap(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float minval_, float maxval_) : cmapsize(size), minval(minval_), maxval(maxval_) {
161 set(ctable, clocs, size, minval_, maxval_);
162 }
163
164 void set(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float a_minval, float a_maxval) {
165 cmapsize = size;
166 minval = a_minval;
167 maxval = a_maxval;
168
169 size_t Ncolors = ctable.size();
170
171 assert(clocs.size() == Ncolors && minval < maxval );
172
173 cmap.resize(Ncolors);
174
175 std::vector<float> cinds;
176 cinds.resize(Ncolors);
177
178 for (uint i = 0; i < Ncolors; i++) {
179 cinds.at(i) = clocs.at(i) * static_cast<float>(cmapsize - 1);
180 }
181
182 cmap.resize(cmapsize);
183 for (uint c = 0; c < Ncolors - 1; c++) {
184 float cmin = cinds.at(c);
185 float cmax = cinds.at(c + 1);
186
187 for (uint i = 0; i < cmapsize; i++) {
188 auto i_f = static_cast<float>(i);
189
190 if (i_f >= cmin && i_f <= cmax) {
191 cmap.at(i).r = ctable.at(c).r + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).r - ctable.at(c).r);
192 cmap.at(i).g = ctable.at(c).g + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).g - ctable.at(c).g);
193 cmap.at(i).b = ctable.at(c).b + (i_f - cmin) / (cmax - cmin) * (ctable.at(c + 1).b - ctable.at(c).b);
194 }
195 }
196 }
197 }
198
199 [[nodiscard]] helios::RGBcolor query(float x) const {
200 assert(cmapsize>0 && !cmap.empty());
201
202 helios::RGBcolor color;
203
204 uint color_ind;
205 if (minval == maxval) {
206 color_ind = 0;
207 } else {
208 color_ind = std::round((x - minval) / (maxval - minval) * float(cmapsize - 1));
209 }
210
211 if (color_ind > cmapsize - 1) { color_ind = int(cmapsize - 1); }
212 color.r = cmap.at(color_ind).r;
213 color.g = cmap.at(color_ind).g;
214 color.b = cmap.at(color_ind).b;
215
216 return color;
217 }

220 minval = min;
221 maxval = max;
