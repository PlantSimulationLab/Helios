106__device__ float3 d_rotatePoint(const float3 &position, const float &theta, const float &phi) {
107
112 float Ry[3][3], Rz[3][3];
113
114 float st = sinf(theta);
115 float ct = cosf(theta);
116
117 float sp = sinf(phi);
118 float cp = cosf(phi);
119
120 // Setup the rotation matrix, this matrix is based off of the rotation matrix used in glRotatef.
121 Ry[0][0] = ct;
122 Ry[0][1] = 0.f;
123 Ry[0][2] = st;
124 Ry[1][0] = 0.f;
125 Ry[1][1] = 1.f;
126 Ry[1][2] = 0.f;
127 Ry[2][0] = -st;
128 Ry[2][1] = 0.f;
129 Ry[2][2] = ct;
130
131 Rz[0][0] = cp;
132 Rz[0][1] = -sp;
133 Rz[0][2] = 0.f;
134 Rz[1][0] = sp;
135 Rz[1][1] = cp;
136 Rz[1][2] = 0.f;
137 Rz[2][0] = 0.f;
138 Rz[2][1] = 0.f;
139 Rz[2][2] = 1.f;
140
141 // Multiply Ry*Rz
142
143 float rotMat[3][3] = {0.f};
144
145 for( int i=0;i<3;i++){
146 for(int j=0;j<3;j++){
147 for(int k=0;k<3;k++){
148 rotMat[i][j]=rotMat[i][j]+Rz[i][k]*Ry[k][j];
149 }
150 }
151 }
152
153 // Multiply the rotation matrix with the position vector.
154 float3 tmp;
155 tmp.x = rotMat[0][0] * position.x + rotMat[0][1] * position.y + rotMat[0][2] * position.z;
156 tmp.y = rotMat[1][0] * position.x + rotMat[1][1] * position.y + rotMat[1][2] * position.z;
157 tmp.z = rotMat[2][0] * position.x + rotMat[2][1] * position.y + rotMat[2][2] * position.z;
158
159 return tmp;
160
161}

334__device__ float d_calculatePrimitiveArea( const uint primitive_type, const float (&transform_matrix)[16] ){
335
336 if( primitive_type == 0 || primitive_type == 3 ){ //Patch or AlphaMask
337
338 float3 s0 = make_float3(0,0,0);
339 float3 s1 = make_float3(1,1,0);
340 float3 s2 = make_float3(0,1,0);
341 d_transformPoint(transform_matrix,s0);
342 d_transformPoint(transform_matrix,s1);
343 d_transformPoint(transform_matrix,s2);
344 return d_magnitude(s2-s0)*d_magnitude(s2-s1);
345
346 }else if( primitive_type == 1 ){ //Triangle
347
348 float3 v0 = make_float3(0,0,0);
349 d_transformPoint(transform_matrix,v0);
350 float3 v1 = make_float3(0,1,0);
351 d_transformPoint(transform_matrix,v1);
352 float3 v2 = make_float3(1,1,0);
353 d_transformPoint(transform_matrix,v2);
354 float3 A(v1-v0);
355 float3 B(v2-v0);
356 float3 C(v2-v1);
357 float a = d_magnitude(A);
358 float b = d_magnitude(B);
359 float c = d_magnitude(C);
360 float s = 0.5f*( a+b+c );
361 return sqrtf(s*(s-a)*(s-b)*(s-c));
362
363 }else if( primitive_type == 2 ){ //Disk
364
365 float r0 = transform_matrix[0];
366 float r1 = transform_matrix[5];
367 return M_PI*r0*r1;
368
369 }
370
371 return 0;
372
373}

407__device__ void d_sampleDisk( uint& seed, optix::float3& sample ){
408
409 // Map Sample to disk - from Suffern (2007) "Ray tracing fom the ground up" Chap. 6
410
411 optix::float3 sp;
412
413 float Rx = rnd(seed);
414 float Ry = rnd(seed);
415
416 //first map sample point to rectangle [-1,1] [-1,1]
417 sp.x = -1.f + 2.f*Rx;
418 sp.y = -1.f + 2.f*Ry;
419
420 float r,p;
421
422 if( sp.x>-sp.y) {
423 if( sp.x > sp.y ){
424 r = sp.x;
425 p = sp.y/sp.x;
426 }else{
427 r = sp.y;
428 p = 2.f-sp.x/sp.y;
429 }
430 }else{
431 if( sp.x < sp.y ){
432 r = -sp.x;
433 p = 4.f + sp.y/sp.x;
434 }else{
435 r = -sp.y;
436 if( sp.y!=0.f ){ //avoid division by zero at origin
437 p = 6.f - sp.x/sp.y;
438 }else{
439 p = 0.f;
440 }
441 }
442 }
443 p*=0.25f*M_PI;
444
445 //find x,y point on unit disk
446 sp.x = r*cos(p);
447 sp.y = r*sin(p);
448 sp.z = 0.f;
449
450 sample = sp;
451
452}
