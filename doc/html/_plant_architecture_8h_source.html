109 if( distribution!="constant" ) {
110 if (generator == nullptr) {

112 }

115 constval = distribution_parameters.at(0) + unif_distribution(*generator) * (distribution_parameters.at(1) - distribution_parameters.at(0));
116 } else if (distribution == "normal") {
117 std::normal_distribution<float> norm_distribution(distribution_parameters.at(0),distribution_parameters.at(1));
118 constval = norm_distribution(*generator);
119 } else if (distribution == "weibull") {
120 std::weibull_distribution<float> wbull_distribution(distribution_parameters.at(0),distribution_parameters.at(1));
121 constval = wbull_distribution(*generator);
122 }
123 }
124 return constval;
125 }
