67__global__ void bvhTraversalKernel(GPUBVHNode *d_nodes, unsigned int *d_primitive_indices, float3 *d_primitive_aabb_min, float3 *d_primitive_aabb_max, float3 *d_query_aabb_min, float3 *d_query_aabb_max, unsigned int *d_results,
68 unsigned int *d_result_counts, int num_queries, int max_results_per_query) {
69
70 int query_idx = blockIdx.x * blockDim.x + threadIdx.x;
71
72 if (query_idx >= num_queries)
73 return;
74
75 float3 query_min = d_query_aabb_min[query_idx];
76 float3 query_max = d_query_aabb_max[query_idx];
77
78 unsigned int result_count = 0;
79 unsigned int *query_results = &d_results[query_idx * max_results_per_query];
80
81 // Stack-based traversal using shared memory for better performance
82 __shared__ unsigned int node_stack[1024]; // Shared among threads in block
83 int stack_size = 0;
84
85 // Each thread gets its own portion of the shared stack
86 int thread_stack_start = threadIdx.x * 32; // 32 entries per thread
87 unsigned int *thread_stack = &node_stack[thread_stack_start];
88
89 // Start traversal from root node
90 thread_stack[0] = 0;
91 stack_size = 1;
92
93 while (stack_size > 0 && result_count < max_results_per_query) {
94
95 // Pop node from stack
96 stack_size--;
97 unsigned int node_idx = thread_stack[stack_size];
98
99 // Check if node index is valid
100 if (node_idx == 0xFFFFFFFF)
101 continue;
102
103 GPUBVHNode node = d_nodes[node_idx];
104
105 // Test if query AABB intersects node AABB
106 if (!d_aabbIntersect(query_min, query_max, node.aabb_min, node.aabb_max)) {
107 continue;
108 }
109
110 if (node.is_leaf) {
111 // Check each primitive in this leaf individually
112 for (unsigned int i = 0; i < node.primitive_count && result_count < max_results_per_query; i++) {
113 unsigned int primitive_index = node.primitive_start + i;
114 unsigned int primitive_id = d_primitive_indices[primitive_index];
115
116 // Get primitive's AABB from pre-computed arrays (using array position, not UUID)
117 float3 prim_min = d_primitive_aabb_min[primitive_index];
118 float3 prim_max = d_primitive_aabb_max[primitive_index];
119
120 // Only add to results if AABBs actually intersect
121 if (d_aabbIntersect(query_min, query_max, prim_min, prim_max)) {
122 query_results[result_count] = primitive_id;
123 result_count++;
124 }
125 }
126 } else {
127 // Add child nodes to stack
128 if (node.left_child != 0xFFFFFFFF && stack_size < 32) {
129 thread_stack[stack_size] = node.left_child;
130 stack_size++;
131 }
132 if (node.right_child != 0xFFFFFFFF && stack_size < 32) {
133 thread_stack[stack_size] = node.right_child;
134 stack_size++;
135 }
136 }
137 }
138
139 d_result_counts[query_idx] = result_count;
140}
