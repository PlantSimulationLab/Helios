70__global__ void bvhTraversalKernel( GPUBVHNode* d_nodes,
71 unsigned int* d_primitive_indices,
72 float3* d_primitive_aabb_min,
73 float3* d_primitive_aabb_max,
74 float3* d_query_aabb_min,
75 float3* d_query_aabb_max,
76 unsigned int* d_results,
77 unsigned int* d_result_counts,
78 int num_queries,
79 int max_results_per_query ) {
80
81 int query_idx = blockIdx.x * blockDim.x + threadIdx.x;
82
83 if( query_idx >= num_queries ) return;
84
85 float3 query_min = d_query_aabb_min[query_idx];
86 float3 query_max = d_query_aabb_max[query_idx];
87
88 unsigned int result_count = 0;
89 unsigned int* query_results = &d_results[query_idx * max_results_per_query];
90
91 // Stack-based traversal using shared memory for better performance
92 __shared__ unsigned int node_stack[1024]; // Shared among threads in block
93 int stack_size = 0;
94
95 // Each thread gets its own portion of the shared stack
96 int thread_stack_start = threadIdx.x * 32; // 32 entries per thread
97 unsigned int* thread_stack = &node_stack[thread_stack_start];
98
99 // Start traversal from root node
100 thread_stack[0] = 0;
101 stack_size = 1;
102
103 while( stack_size > 0 && result_count < max_results_per_query ) {
104
105 // Pop node from stack
106 stack_size--;
107 unsigned int node_idx = thread_stack[stack_size];
108
109 // Check if node index is valid
110 if( node_idx == 0xFFFFFFFF ) continue;
111
112 GPUBVHNode node = d_nodes[node_idx];
113
114 // Test if query AABB intersects node AABB
115 if( !d_aabbIntersect(query_min, query_max, node.aabb_min, node.aabb_max) ) {
116 continue;
117 }
118
119 if( node.is_leaf ) {
120 // Check each primitive in this leaf individually
121 for( unsigned int i = 0; i < node.primitive_count && result_count < max_results_per_query; i++ ) {
122 unsigned int primitive_index = node.primitive_start + i;
123 unsigned int primitive_id = d_primitive_indices[primitive_index];
124
125 // Get primitive's AABB from pre-computed arrays (using array position, not UUID)
126 float3 prim_min = d_primitive_aabb_min[primitive_index];
127 float3 prim_max = d_primitive_aabb_max[primitive_index];
128
129 // Only add to results if AABBs actually intersect
130 if( d_aabbIntersect(query_min, query_max, prim_min, prim_max) ) {
131 query_results[result_count] = primitive_id;
132 result_count++;
133 }
134 }
135 } else {
136 // Add child nodes to stack
137 if( node.left_child != 0xFFFFFFFF && stack_size < 32 ) {
138 thread_stack[stack_size] = node.left_child;
139 stack_size++;
140 }
141 if( node.right_child != 0xFFFFFFFF && stack_size < 32 ) {
142 thread_stack[stack_size] = node.right_child;
143 stack_size++;
144 }
145 }
146 }
147
148 d_result_counts[query_idx] = result_count;
149}

251void launchBVHTraversal( void* h_nodes, int node_count,
252 unsigned int* h_primitive_indices, int primitive_count,
253 float* h_primitive_aabb_min, float* h_primitive_aabb_max,
254 float* h_query_aabb_min, float* h_query_aabb_max, int num_queries,
255 unsigned int* h_results, unsigned int* h_result_counts,
256 int max_results_per_query ) {
257
258 if( num_queries == 0 ) return;
259
260 // Allocate temporary GPU memory for query data and primitive AABBs
261 float3* d_query_min;
262 float3* d_query_max;
263 float3* d_primitive_min;
264 float3* d_primitive_max;
265 unsigned int* d_results;
266 unsigned int* d_result_counts;
267
268 size_t query_size = num_queries * sizeof(float3);
269 size_t primitive_aabb_size = primitive_count * sizeof(float3);
270 size_t results_size = num_queries * max_results_per_query * sizeof(unsigned int);
271 size_t counts_size = num_queries * sizeof(unsigned int);
272
273 cudaMalloc((void**)&d_query_min, query_size);
274 cudaMalloc((void**)&d_query_max, query_size);
275 cudaMalloc((void**)&d_primitive_min, primitive_aabb_size);
276 cudaMalloc((void**)&d_primitive_max, primitive_aabb_size);
277 cudaMalloc((void**)&d_results, results_size);
278 cudaMalloc((void**)&d_result_counts, counts_size);
279
280 // Convert query data to float3 format
281 std::vector<float3> query_min_vec(num_queries);
282 std::vector<float3> query_max_vec(num_queries);
283 for( int i = 0; i < num_queries; i++ ) {
284 query_min_vec[i] = make_float3(h_query_aabb_min[i*3], h_query_aabb_min[i*3+1], h_query_aabb_min[i*3+2]);
285 query_max_vec[i] = make_float3(h_query_aabb_max[i*3], h_query_aabb_max[i*3+1], h_query_aabb_max[i*3+2]);
286 }
287
288 // Convert primitive AABB data to float3 format
289 std::vector<float3> primitive_min_vec(primitive_count);
290 std::vector<float3> primitive_max_vec(primitive_count);
291 for( int i = 0; i < primitive_count; i++ ) {
292 primitive_min_vec[i] = make_float3(h_primitive_aabb_min[i*3], h_primitive_aabb_min[i*3+1], h_primitive_aabb_min[i*3+2]);
293 primitive_max_vec[i] = make_float3(h_primitive_aabb_max[i*3], h_primitive_aabb_max[i*3+1], h_primitive_aabb_max[i*3+2]);
294 }
295
296 // Copy query and primitive AABB data to GPU
297 cudaMemcpy(d_query_min, query_min_vec.data(), query_size, cudaMemcpyHostToDevice);
298 cudaMemcpy(d_query_max, query_max_vec.data(), query_size, cudaMemcpyHostToDevice);
299 cudaMemcpy(d_primitive_min, primitive_min_vec.data(), primitive_aabb_size, cudaMemcpyHostToDevice);
300 cudaMemcpy(d_primitive_max, primitive_max_vec.data(), primitive_aabb_size, cudaMemcpyHostToDevice);
301
302 // Launch kernel
303 int block_size = 256;
304 int num_blocks = (num_queries + block_size - 1) / block_size;
305
306 bvhTraversalKernel<<<num_blocks, block_size>>>(
307 (GPUBVHNode*)h_nodes, (unsigned int*)h_primitive_indices,
308 d_primitive_min, d_primitive_max,
309 d_query_min, d_query_max, d_results, d_result_counts,
310 num_queries, max_results_per_query
311 );
312
313 cudaDeviceSynchronize();
314
315 // Check for errors
316 cudaError_t err = cudaGetLastError();
317 if( err != cudaSuccess ) {
318 fprintf(stderr, "CUDA kernel launch error: %s\n", cudaGetErrorString(err));
319 }
320
321 // Copy results back
322 cudaMemcpy(h_results, d_results, results_size, cudaMemcpyDeviceToHost);
323 cudaMemcpy(h_result_counts, d_result_counts, counts_size, cudaMemcpyDeviceToHost);
324
325 // Clean up temporary memory
326 cudaFree(d_query_min);
327 cudaFree(d_query_max);
328 cudaFree(d_primitive_min);
329 cudaFree(d_primitive_max);
330 cudaFree(d_results);
331 cudaFree(d_result_counts);
332}
