453 float zo_m = 0.1f*canopy_height_m; // Roughness length for momentum transfer (m)

471 // Update the surface energy balance
472 this->run(UUIDs);

474 // Calculate temperature source term
475 float sensible_source_flux_W_m3;
476 context->calculatePrimitiveDataAreaWeightedSum( UUIDs, "sensible_flux", sensible_source_flux_W_m3 ); //units: Watts
477 sensible_source_flux_W_m3 /= canopy_dimensions.x*canopy_dimensions.y*canopy_height_m; // Convert to W/m^3
478
479 // Calculate moisture source term
480 float moisture_source_flux_W_m3;
481 context->calculatePrimitiveDataAreaWeightedSum( UUIDs, "latent_flux", moisture_source_flux_W_m3 ); //units: Watts
482 moisture_source_flux_W_m3 /= canopy_dimensions.x*canopy_dimensions.y*canopy_height_m; // Convert to W/m^3
483 float moisture_source_flux_mol_s_m3 = moisture_source_flux_W_m3/lambda_mol; // Convert to mol/s/m^3
484
485 float rho_air_mol_m3 = Patm / (R * air_temperature_average);; // Molar density of air (mol/m^3).
486
487 // --- canopy / ABL interface & implicit‐Euler update ---
488
489 // 1) canopy‐scale source fluxes per unit ground area
490 float H_s = sensible_source_flux_W_m3 * canopy_height_m; // W m⁻²
491 float E_s = moisture_source_flux_mol_s_m3 * canopy_height_m; // mol m⁻² s⁻¹
492
493 // 2) neutral friction velocity for Obukhov length
494 float u_star_neutral = von_Karman_constant * wind_speed_reference
495 / std::log((reference_height_m - displacement_height) / zo_m);
496
497 // 3) Obukhov length L [m]
498 float rho_air_kg_m3 = rho_air_mol_m3 * 0.02896f;
499 float cp_air_kg = cp_air_mol / 0.02896f;
500 float L = 1e6f;
501 if (std::fabs(H_s) > 1e-6f) {
502 L = -rho_air_kg_m3 * cp_air_kg * std::pow(u_star_neutral, 3)
503 * air_temperature_reference
504 / (von_Karman_constant * 9.81f * H_s);
505 }
506
507 // 4) stability functions (Businger–Dyer)
508 auto psi_unstable_u = [&](float zeta) {
509 float x = std::pow(1.f - 16.f * zeta, 0.25f);
510 return 2.f * std::log((1.f + x) / 2.f)
511 + std::log((1.f + x * x) / 2.f)
512 - 2.f * std::atan(x)
513 + 0.5f * M_PI;
514 };
515 auto psi_unstable_h = [&](float zeta) {
516 return 2.f * std::log((1.f + std::sqrt(1.f - 16.f * zeta)) / 2.f);
517 };
518 auto psi_stable = [&](float zeta) {
519 return -5.f * zeta;
520 };
521
522 float zeta_r = (reference_height_m - displacement_height) / L;
523 float psi_m_r = (zeta_r < 0.f)
524 ? psi_unstable_u(zeta_r)
525 : psi_stable(zeta_r);
526 float psi_h_r = (zeta_r < 0.f)
527 ? psi_unstable_h(zeta_r)
528 : psi_stable(zeta_r);
529
530 // 5) stability‐corrected friction velocity
531 float u_star = von_Karman_constant * wind_speed_reference
532 / (std::log((reference_height_m - displacement_height) / zo_m) - psi_m_r);
533
534 // 6) Monin–Obukhov scales θ_* and q_*
535 float theta_star = H_s / (rho_air_mol_m3 * cp_air_mol * u_star);
536 float q_star = E_s / (rho_air_mol_m3 * u_star);
537
538 // 7) corrected log‐law lengths
539 float ln_m_corr = std::log((reference_height_m - displacement_height) / zo_m) - psi_m_r;
540 float ln_h_corr = std::log((reference_height_m - displacement_height) / zo_h) - psi_h_r;
541
542 // 8) aerodynamic conductance ga [mol m⁻² s⁻¹]
543 float ga = std::pow(von_Karman_constant, 2) * wind_speed_reference / (ln_m_corr * ln_h_corr) * rho_air_mol_m3;
544
545 // 9) interface values at canopy top (Monin–Obukhov log-law)
546 float air_moisture_reference = esat_Pa(air_temperature_reference) / Patm * air_humidity_reference;
547 float T_int = air_temperature_reference + theta_star / von_Karman_constant * ln_h_corr;
548 float x_int = air_moisture_reference + q_star / von_Karman_constant * ln_h_corr;
549
550 // Cap relative humidity to 100%
551 float x_sat = esat_Pa(T_int)/Patm;
552 if ( x_int > x_sat ) {
553 //std::cerr << "WARNING (EnergyBalanceModel::evaluateAirEnergyBalance): Air moisture exceeds saturation. Capping to saturation value." << std::endl;
554 x_int = 0.99f*x_sat;
555 }
556
557 // 10) entrainment conductance g_e (mol m⁻² s⁻¹)
558 float H_canopy = sensible_source_flux_W_m3 * canopy_height_m;
559 float g_e;
560 if (air_temperature_ABL == 0.f) {
561 // first‐step equilibrium initialization
562 float w_star0 = H_canopy > 0.f ? std::pow(9.81f * H_canopy * canopy_height_m / (rho_air_mol_m3 * cp_air_mol * air_temperature_reference), 1.f / 3.f) : 0.f;
563 const float A_e = 0.02f;
564 float w_e0 = A_e * w_star0;
565 g_e = rho_air_mol_m3 * w_e0;
566 // initialize ABL state
567 air_temperature_ABL = (ga * air_temperature_average
568 + g_e * air_temperature_reference)
569 / (ga + g_e);
570 air_moisture_ABL = (ga * air_moisture_average
571 + g_e * air_moisture_reference)
572 / (ga + g_e);
573 } else {
574 // ongoing entrainment
575 float w_star = 0.f;
576 if (H_canopy > 0.f) {
577 w_star = std::pow(9.81f * H_canopy * canopy_height_m
578 / (rho_air_mol_m3 * cp_air_mol * air_temperature_ABL), 1.f / 3.f);
579 }
580 float u_star_shear = von_Karman_constant * wind_speed_reference
581 / std::log((reference_height_m - displacement_height) / zo_m);
582 float w_e = std::fmax(0.01f * w_star, 0.005f * u_star_shear);
583 g_e = rho_air_mol_m3 * w_e;
584 }
585
586 // if ( g_e>0.05 ) {
587 // std::cerr << "Capping g_e value of " << g_e << std::endl;
588 // g_e = 0.05f;
589 // }
590 g_e = 2.0;
591
592 // 11) canopy→ABL fluxes
593 float sensible_upper_flux_W_m2 = cp_air_mol * ga * (air_temperature_average - T_int);
594
595 // 12) update canopy‐air temperature (implicit Euler)
596 // numerator: T^n + (Δt / (ρ c_p h_c)) * [H_s + c_p g_a T_int]
597 // denominator: 1 + Δt·g_a/(ρ h_c)
598 float numerator_temp = air_temperature_average
599 + dt_actual
600 / (rho_air_mol_m3 * cp_air_mol * canopy_height_m)
601 * (H_s + cp_air_mol * ga * T_int);
602 float denominator_temp = 1.f
603 + dt_actual * ga / (rho_air_mol_m3 * canopy_height_m);
604 air_temperature_average = numerator_temp / denominator_temp;
605
606 // 13) update canopy‐air moisture (implicit Euler)
607
608 // --- canopy implicit‐Euler moisture update ---
609 // update: ρh (x_new - x_old)/dt = E_s - E_top
610 float E_top = ga * (air_moisture_average - x_int);
611 float f = std::pow(1.f - (1.f - 0.622f)*air_moisture_average, 2) / 0.622f;
612 float numer_can_x = air_moisture_average + dt_actual * f / (rho_air_mol_m3 * canopy_height_m) * (E_s + ga * x_int);
613 float denom_can_x = 1.f + dt_actual * f * ga/(rho_air_mol_m3*canopy_height_m);
614 air_moisture_average = numer_can_x/denom_can_x;
615
616 // 14) update ABL‐air temperature (implicit Euler)
617 float denom_abl_T = 1.f + dt_actual * (ga + g_e) / (rho_air_mol_m3 * cp_air_mol * abl_height_m);
618 float num_abl_T = air_temperature_ABL
619 + dt_actual / (rho_air_mol_m3 * cp_air_mol * abl_height_m)
620 * (sensible_upper_flux_W_m2 + cp_air_mol * g_e * (air_temperature_reference - air_temperature_ABL));
621 air_temperature_ABL = num_abl_T / denom_abl_T;
622
623 // 15) update ABL‐air moisture (implicit Euler)
624 // source into ABL storage
625 // ρ_ab·h_ab · (x_ab_new - x_ab_old)/dt = E_top - E_e
626 float E_e = g_e * ( air_moisture_ABL - air_moisture_reference );
627 float numer_abl_x = air_moisture_ABL + dt_actual / (rho_air_mol_m3 * abl_height_m) * ( E_top - E_e );
628 float denom_abl_x = 1.f + dt_actual * (ga + g_e)/ (rho_air_mol_m3 * abl_height_m);
629 air_moisture_ABL = numer_abl_x / denom_abl_x;
630
631 // moisture budget check (mol m⁻² s⁻¹)
632 float canopy_storage = (air_moisture_average - air_moisture_old) * rho_air_mol_m3 * canopy_height_m / dt_actual;
633 float flux_in = E_s; // from latent source
634 float flux_out = E_top; // to ABL
635 std::cerr << std::fixed
636 << "CANOPY MOISTURE BUDGET: in="<<flux_in
637 <<" out="<<flux_out
638 <<" storage="<<canopy_storage
639 <<" residual="<<(flux_in - flux_out - canopy_storage)
640 << std::endl;
641
642 // Cap relative humidity to 100%
643 float esat = esat_Pa(air_temperature_average)/Patm;
644 if ( air_moisture_average > esat ) {
645 std::cerr << "WARNING (EnergyBalanceModel::evaluateAirEnergyBalance): Air moisture exceeds saturation. Capping to saturation value." << std::endl;
646 air_moisture_average = 0.99f*esat;
647 }
648 esat = esat_Pa(air_temperature_ABL)/Patm;
649 if ( air_moisture_ABL > esat ) {
650 air_moisture_ABL = 0.99f*esat;
651 }
652
653 // Check that timestep is not too large based on aerodynamic resistance
654 if ( dt_actual > 0.5f*canopy_height_m*rho_air_mol_m3/ga ) {
655 std::cerr << "WARNING (EnergyBalanceModel::evaluateAirEnergyBalance): Time step is too large. The air energy balance may not converge properly." << std::endl;
656 }
657
658 float heat_from_Htop = dt_actual / (rho_air_mol_m3 * cp_air_mol * abl_height_m) * sensible_upper_flux_W_m2;
659 float heat_from_entrainment = dt_actual * g_e * (air_temperature_reference - air_temperature_ABL) / (rho_air_mol_m3 * abl_height_m);
660
661 std::cerr << "ABL ENERGY UPDATE:\n"
662 << " T_ABL_n = " << air_temperature_ABL << "\n"
663 << " T_ref = " << air_temperature_reference << "\n"
664 << " g_e = " << g_e << " mol/m²/s\n"
665 << " Heat from H_top = " << heat_from_Htop << " K\n"
666 << " Heat from H_e = " << heat_from_entrainment << " K\n"
667 << " Numerator = " << num_abl_T << " K\n"
668 << " Denominator = " << denom_abl_T << "\n"
669 << " T_ABL_new = " << (num_abl_T / denom_abl_T) << "\n";
670
671
672#ifdef HELIOS_DEBUG
673
674 // -- check energy consistency -- //
675 H_s = sensible_source_flux_W_m3 * canopy_height_m; // W m⁻², canopy source
676 float H_top = cp_air_mol * ga * (air_temperature_average - T_int); // W m⁻², canopy→ABL
677 float H_e = cp_air_mol * g_e * (air_temperature_ABL - air_temperature_reference); // W m⁻², ABL→free atm
678
679 E_s = moisture_source_flux_mol_s_m3 * canopy_height_m; // mol m⁻² s⁻¹, canopy source
680 float E_top = ga * (air_moisture_average - x_int); // mol m⁻² s⁻¹, canopy→ABL
681 float E_e = g_e * (air_moisture_ABL - air_moisture_reference); // mol m⁻² s⁻¹, ABL→free atm
682
683 // Print a concise table of key values
684 std::cout << std::fixed << std::setprecision(5)
685 << "step="<<time
686 << " T_c="<<air_temperature_average
687 << " T_int="<<T_int
688 << " T_b="<<air_temperature_ABL
689 << " H_s="<<H_s
690 << " H_top="<<H_top
691 << " H_e="<<H_e
692 << " E_s="<<E_s
693 << " E_top="<<E_top
694 << " E_e="<<E_e
695 << " psi_m="<<psi_m_r
696 << " psi_h="<<psi_h_r
697 << " u*="<<u_star
698 << " L="<<L
699 << std::endl;
700
701 // temperature balance
702
703 const float tol = 1e-5f;
704 float S_can = rho_air_mol_m3 * cp_air_mol * canopy_height_m * (air_temperature_average - air_temperature_old) / dt_actual;
705 if (fabs(H_s - H_top - S_can) > tol)
706 std::cerr << "Canopy budget error: " << (H_s - H_top - S_can) << "\n";
707
708 float S_abl = rho_air_mol_m3 * cp_air_mol * abl_height_m * (air_temperature_ABL - air_temperature_ABL_old) / dt_actual;
709 if (fabs(H_top - H_e - S_abl) > tol)
710 std::cerr << "ABL budget error: " << (H_top - H_e - S_abl) << "\n";
711
712 // moisture balance per unit ground area
713
714 S_can = rho_air_mol_m3 * canopy_height_m * (air_moisture_average - air_moisture_old) / dt_actual;
715
716 const float tol_m = 1e-6f;
717 if (std::fabs(E_s - moisture_upper_flux - S_can) > tol_m) {
718 std::cerr << "CANOPY MOISTURE BUDGET ERROR: " << "E_s - E_up - S_can = " << E_s - moisture_upper_flux - S_can << " mol/m2/s\n";
719 }
720
721 S_abl = rho_air_mol_m3 * abl_height_m * (air_moisture_ABL - air_moisture_ABL_old) / dt_actual;
722
723 if (std::fabs(moisture_upper_flux - E_e - S_abl) > tol_m) {
724 std::cerr << "ABL MOISTURE BUDGET ERROR: " << "E_up - E_e - S_abl = " << moisture_upper_flux - E_e - S_abl << " mol/m2/s\n";
725 }
726
727
728#endif
729
730 // Set primitive data
731 context->setPrimitiveData( UUIDs, "air_temperature", air_temperature_average );
732 float air_humidity = air_moisture_average*Patm/esat_Pa(air_temperature_average);
733 context->setPrimitiveData( UUIDs, "air_humidity", air_humidity );
734
735 // Set global data
736 context->setGlobalData("air_temperature_average", air_temperature_average);
737 context->setGlobalData("air_humidity_average", air_humidity);
738 context->setGlobalData("air_moisture_average", air_moisture_average);
739
740 context->setGlobalData( "air_temperature_ABL", air_temperature_ABL );
741 float air_humidity_ABL = air_moisture_ABL*Patm/esat_Pa(air_temperature_ABL);
742 context->setGlobalData( "air_humidity_ABL", air_humidity_ABL );
743
744 context->setGlobalData( "air_temperature_interface", T_int );
745 float air_humidity_interface = x_int*Patm/esat_Pa(T_int);
746 context->setGlobalData( "air_humidity_interface", air_humidity_interface );
747
748 context->setGlobalData("aerodynamic_resistance", rho_air_mol_m3/ga);
749
750 std::cout << "Computed air temperature: " << air_temperature_average << " " << air_temperature_old << std::endl;
751 std::cout << "Computed air humidity: " << air_humidity << std::endl;
752 std::cout << "Computed air moisture: " << air_moisture_average << " " << air_moisture_old << std::endl;
753
754 if ( time>0 && std::abs(air_temperature_average-air_temperature_old)/air_temperature_old < 0.003f && std::abs(air_moisture_average-air_moisture_old)/air_moisture_old < 0.003f ) {
755 // If the air temperature and humidity have not changed significantly, we can stop iterating
756 std::cout << "Converged" << std::endl;
757 break;
758 }
759 air_temperature_old = air_temperature_average;
760 air_moisture_old = air_moisture_average;
761 air_temperature_ABL_old = air_temperature_ABL;
762 air_moisture_ABL_old = air_moisture_ABL;
763
764 time += dt_actual;
765
766 }

821void EnergyBalanceModel::printDefaultValueReport(const std::vector<uint> &UUIDs) const {
822
823 size_t assumed_default_TL = 0;
824 size_t assumed_default_U = 0;
825 size_t assumed_default_L = 0;
826 size_t assumed_default_p = 0;
827 size_t assumed_default_Ta = 0;
828 size_t assumed_default_rh = 0;
829 size_t assumed_default_gH = 0;
830 size_t assumed_default_gs = 0;
831 size_t assumed_default_Qother = 0;
832 size_t assumed_default_heatcapacity = 0;
833 size_t assumed_default_fs = 0;
834 size_t twosided_0 = 0;
835 size_t twosided_1 = 0;
836 size_t Ne_1 = 0;
837 size_t Ne_2 = 0;
838
839 size_t Nprimitives = UUIDs.size();
840
841 for (uint UUID: UUIDs) {
842
843 //surface temperature (K)
844 if (!context->doesPrimitiveDataExist(UUID, "temperature") ||
845 context->getPrimitiveDataType(UUID, "temperature") != HELIOS_TYPE_FLOAT) {
846 assumed_default_TL++;
847 }
848
849 //air pressure (Pa)
850 if (!context->doesPrimitiveDataExist(UUID, "air_pressure") ||
851 context->getPrimitiveDataType(UUID, "air_pressure") != HELIOS_TYPE_FLOAT) {
852 assumed_default_p++;
853 }
854
855 //air temperature (K)
856 if (!context->doesPrimitiveDataExist(UUID, "air_temperature") ||
857 context->getPrimitiveDataType(UUID, "air_temperature") != HELIOS_TYPE_FLOAT) {
858 assumed_default_Ta++;
859 }
860
861 //air humidity
862 if (!context->doesPrimitiveDataExist(UUID, "air_humidity") ||
863 context->getPrimitiveDataType(UUID, "air_humidity") != HELIOS_TYPE_FLOAT) {
864 assumed_default_rh++;
865 }
866
867 //boundary-layer conductance to heat
868 if (!context->doesPrimitiveDataExist(UUID, "boundarylayer_conductance") || context->getPrimitiveDataType(UUID, "boundarylayer_conductance") != HELIOS_TYPE_FLOAT){
869 assumed_default_gH++;
870 }
871
872 //wind speed
873 if (!context->doesPrimitiveDataExist(UUID, "wind_speed") || context->getPrimitiveDataType(UUID, "wind_speed") != HELIOS_TYPE_FLOAT){
874 assumed_default_U++;
875 }
876
877 //object length
878 if (!context->doesPrimitiveDataExist(UUID, "object_length") || context->getPrimitiveDataType(UUID, "object_length") != HELIOS_TYPE_FLOAT){
879 assumed_default_L++;
880 }
881
882 //moisture conductance
883 if (!context->doesPrimitiveDataExist(UUID, "moisture_conductance") || context->getPrimitiveDataType(UUID, "moisture_conductance") != HELIOS_TYPE_FLOAT){
884 assumed_default_gs++;
885 }
886
887 //Heat capacity
888 if (!context->doesPrimitiveDataExist(UUID, "heat_capacity") || context->getPrimitiveDataType(UUID, "heat_capacity") != HELIOS_TYPE_FLOAT) {
889 assumed_default_heatcapacity++;
890 }
891
892 //"Other" heat fluxes
893 if (!context->doesPrimitiveDataExist(UUID, "other_surface_flux") || context->getPrimitiveDataType(UUID, "other_surface_flux") != HELIOS_TYPE_FLOAT) {
894 assumed_default_Qother++;
895 }
896
897 //two-sided flag
898 if ( context->doesPrimitiveDataExist(UUID, "twosided_flag") && context->getPrimitiveDataType(UUID, "twosided_flag") == HELIOS_TYPE_UINT) {
899 uint twosided;
900 context->getPrimitiveData(UUID, "twosided_flag", twosided);
901 if( twosided==0 ){
902 twosided_0++;
903 }else if( twosided==1 ){
904 twosided_1++;
905 }
906 }else{
907 twosided_1++;
908 }
909
910 //number of evaporating faces
911 if ( context->doesPrimitiveDataExist(UUID, "evaporating_faces") && context->getPrimitiveDataType(UUID, "evaporating_faces") == HELIOS_TYPE_UINT) {
912 uint Ne;
913 context->getPrimitiveData(UUID, "evaporating_faces", Ne);
914 if( Ne==1 ){
915 Ne_1++;
916 }else if( Ne==2 ){
917 Ne_2++;
918 }
919 }else{
920 Ne_1++;
921 }
922
923 //Surface humidity
924 if (!context->doesPrimitiveDataExist(UUID, "surface_humidity") || context->getPrimitiveDataType(UUID, "surface_humidity") != HELIOS_TYPE_FLOAT) {
925 assumed_default_fs++;
926 }
927
928 }
929
930 std::cout << "--- Energy Balance Model Default Value Report ---" << std::endl;
931
932 std::cout << "surface temperature (initial guess): " << assumed_default_TL << " of " << Nprimitives << " used default value of " << temperature_default << " because ""temperature"" primitive data did not exist" << std::endl;
933 std::cout << "air pressure: " << assumed_default_p << " of " << Nprimitives << " used default value of " << pressure_default << " because ""air_pressure"" primitive data did not exist" << std::endl;
934 std::cout << "air temperature: " << assumed_default_Ta << " of " << Nprimitives << " used default value of " << air_temperature_default << " because ""air_temperature"" primitive data did not exist" << std::endl;
935 std::cout << "air humidity: " << assumed_default_rh << " of " << Nprimitives << " used default value of " << air_humidity_default << " because ""air_humidity"" primitive data did not exist" << std::endl;
936 std::cout << "boundary-layer conductance: " << assumed_default_gH << " of " << Nprimitives << " calculated boundary-layer conductance from Polhausen equation because ""boundarylayer_conductance"" primitive data did not exist" << std::endl;
937 if( assumed_default_gH>0 ){
938 std::cout << " - wind speed: " << assumed_default_U << " of " << assumed_default_gH << " using Polhausen equation used default value of " << wind_speed_default << " because ""wind_speed"" primitive data did not exist" << std::endl;
939 std::cout << " - object length: " << assumed_default_L << " of " << assumed_default_gH << " using Polhausen equation used the primitive/object length/area to calculate object length because ""object_length"" primitive data did not exist" << std::endl;
940 }
941 std::cout << "moisture conductance: " << assumed_default_gs << " of " << Nprimitives << " used default value of " << gS_default << " because ""moisture_conductance"" primitive data did not exist" << std::endl;
942 std::cout << "surface humidity: " << assumed_default_fs << " of " << Nprimitives << " used default value of " << surface_humidity_default << " because ""surface_humidity"" primitive data did not exist" << std::endl;
943 std::cout << "two-sided flag: " << twosided_0 << " of " << Nprimitives << " used two-sided flag=0; " << twosided_1 << " of " << Nprimitives << " used two-sided flag=1 (default)" << std::endl;
944 std::cout << "evaporating faces: " << Ne_1 << " of " << Nprimitives << " used Ne = 1 (default); " << Ne_2 << " of " << Nprimitives << " used Ne = 2" << std::endl;
945
946 std::cout << "------------------------------------------------------" << std::endl;
