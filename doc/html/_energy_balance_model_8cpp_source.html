91void EnergyBalanceModel::evaluateAirEnergyBalance(const std::vector<uint> &UUIDs, float dt_sec, float time_advance_sec) {

95 return;

98 float air_temperature_reference = air_temperature_default; // Default air temperature in Kelvin
99 if ( context->doesGlobalDataExist("air_temperature_reference") && context->getGlobalDataType("air_temperature_reference") == helios::HELIOS_TYPE_FLOAT ) {

101 }
102
103 float air_humidity_reference = air_humidity_default; // Default air relative humidity
104 if ( context->doesGlobalDataExist("air_humidity_reference") && context->getGlobalDataType("air_humidity_reference") == helios::HELIOS_TYPE_FLOAT ) {
105 context->getGlobalData("air_humidity_reference", air_humidity_reference);
106 }
107
108 float wind_speed_reference = wind_speed_default; // Default wind speed in m/s
109 if ( context->doesGlobalDataExist("wind_speed_reference") && context->getGlobalDataType("wind_speed_reference") == helios::HELIOS_TYPE_FLOAT ) {
110 context->getGlobalData("wind_speed_reference", wind_speed_reference);
111 }
112
113 // Variables to be set externally via global data
114
115 float Patm;
116 if ( context->doesGlobalDataExist("air_pressure") && context->getGlobalDataType("air_pressure") == helios::HELIOS_TYPE_FLOAT ) {
117 context->getGlobalData("air_pressure", Patm);
118 }else {
119 Patm = pressure_default;
120 }
121
122 float air_temperature_average;
123 if ( context->doesGlobalDataExist("air_temperature_average") && context->getGlobalDataType("air_temperature_average") == helios::HELIOS_TYPE_FLOAT ) {
124 context->getGlobalData("air_temperature_average", air_temperature_average);
125 }else {
126 air_temperature_average = air_temperature_reference;
127 }
128
129 float air_moisture_average;
130 if ( context->doesGlobalDataExist("air_moisture_average") && context->getGlobalDataType("air_moisture_average") == helios::HELIOS_TYPE_FLOAT ) {
131 context->getGlobalData("air_moisture_average", air_moisture_average);
132 std::cerr << "Reading air_moisture_average from global data: " << air_moisture_average << std::endl;
133 }else {
134 air_moisture_average = air_humidity_reference*esat_Pa(air_temperature_reference)/Patm;
135 }
136
137 float air_temperature_ABL = 0; //initialize to 0 as a flag so that if it's not available from global data, we'll initialize it below
138 if ( context->doesGlobalDataExist("air_temperature_ABL") && context->getGlobalDataType("air_temperature_ABL") == helios::HELIOS_TYPE_FLOAT ) {
139 context->getGlobalData("air_temperature_ABL", air_temperature_ABL);
140 }
141
142 float air_moisture_ABL;
143 if ( context->doesGlobalDataExist("air_moisture_ABL") && context->getGlobalDataType("air_moisture_ABL") == helios::HELIOS_TYPE_FLOAT ) {
144 context->getGlobalData("air_moisture_ABL", air_moisture_ABL);
145 std::cerr << "Reading air_moisture_ABL from global data: " << air_moisture_ABL << std::endl;
146 }else {
147 air_moisture_ABL = air_humidity_reference*esat_Pa(air_temperature_reference)/Patm;
148 }
149
150 // Get dimensions of canopy volume
151 if ( canopy_dimensions == make_vec3(0,0,0) ) {
152 vec2 xbounds, ybounds, zbounds;
153 context->getDomainBoundingBox( UUIDs, xbounds, ybounds, zbounds);
154 canopy_dimensions = make_vec3(xbounds.y - xbounds.x, ybounds.y - ybounds.x, zbounds.y - zbounds.x);
155 }
156 assert( canopy_dimensions.x>0 && canopy_dimensions.y>0 && canopy_dimensions.z>0 );
157 if ( canopy_height_m == 0 ) {
158 canopy_height_m = canopy_dimensions.z; // Set the canopy height if not already set
159 }
160 if ( reference_height_m == 0 ) {
161 reference_height_m = canopy_height_m; // Set the reference height if not already set
162 }
163
164 std::cout << "Read in initial values. Ta = " << air_temperature_average << "; e = " << air_moisture_average << "; " << air_moisture_average/esat_Pa(air_temperature_reference)*Patm << std::endl;
165
166 std::cout << "Canopy dimensions: " << canopy_dimensions.x << " x " << canopy_dimensions.y << " x " << canopy_height_m << std::endl;
167
168 float displacement_height = 0.67f*canopy_height_m;
169 float zo_m = 0.1f*canopy_height_m; // Roughness length for momentum transfer (m)
170 float zo_h = 0.01f*canopy_height_m; // Roughness length for heat transfer (m)
171
172 float abl_height_m = 1000.f;
173
174 float time=0;
175 float air_temperature_old = 0;
176 float air_moisture_old = 0;
177 float air_temperature_ABL_old = 0;
178 float air_moisture_ABL_old = 0;
179 while ( time<time_advance_sec ) {
180 float dt_actual = dt_sec;
181 if ( time + dt_actual > time_advance_sec ) {
182 dt_actual = time_advance_sec - time; // Adjust the time step to not exceed the total time
183 }
184
185 std::cout << "Air moisture average: " << air_moisture_average << std::endl;
186
187 // Update the surface energy balance
188 this->run(UUIDs);
189
190 // Calculate temperature source term
191 float sensible_source_flux_W_m3;
192 context->calculatePrimitiveDataAreaWeightedSum( UUIDs, "sensible_flux", sensible_source_flux_W_m3 ); //units: Watts
193 sensible_source_flux_W_m3 /= canopy_dimensions.x*canopy_dimensions.y*canopy_height_m; // Convert to W/m^3
194
195 // Calculate moisture source term
196 float moisture_source_flux_W_m3;
197 context->calculatePrimitiveDataAreaWeightedSum( UUIDs, "latent_flux", moisture_source_flux_W_m3 ); //units: Watts
198 moisture_source_flux_W_m3 /= canopy_dimensions.x*canopy_dimensions.y*canopy_height_m; // Convert to W/m^3
199 float moisture_source_flux_mol_s_m3 = moisture_source_flux_W_m3/lambda_mol; // Convert to mol/s/m^3
200
201 float rho_air_mol_m3 = Patm / (R * air_temperature_average);; // Molar density of air (mol/m^3).
202
203 // --- canopy / ABL interface & implicit‐Euler update ---
204
205 // 1) canopy‐scale source fluxes per unit ground area
206 float H_s = sensible_source_flux_W_m3 * canopy_height_m; // W m⁻²
207 float E_s = moisture_source_flux_mol_s_m3 * canopy_height_m; // mol m⁻² s⁻¹
208
209 // 2) neutral friction velocity for Obukhov length
210 float u_star_neutral = von_Karman_constant * wind_speed_reference
211 / std::log((reference_height_m - displacement_height) / zo_m);
212
213 // 3) Obukhov length L [m]
214 float rho_air_kg_m3 = rho_air_mol_m3 * 0.02896f;
215 float cp_air_kg = cp_air_mol / 0.02896f;
216 float L = 1e6f;
217 if (std::fabs(H_s) > 1e-6f) {
218 L = -rho_air_kg_m3 * cp_air_kg * std::pow(u_star_neutral, 3)
219 * air_temperature_reference
220 / (von_Karman_constant * 9.81f * H_s);
221 }
222
223 // 4) stability functions (Businger–Dyer)
224 auto psi_unstable_u = [&](float zeta) {
225 float x = std::pow(1.f - 16.f * zeta, 0.25f);
226 return 2.f * std::log((1.f + x) / 2.f)
227 + std::log((1.f + x * x) / 2.f)
228 - 2.f * std::atan(x)
229 + 0.5f * M_PI;
230 };
231 auto psi_unstable_h = [&](float zeta) {
232 return 2.f * std::log((1.f + std::sqrt(1.f - 16.f * zeta)) / 2.f);
233 };
234 auto psi_stable = [&](float zeta) {
235 return -5.f * zeta;
236 };
237
238 float zeta_r = (reference_height_m - displacement_height) / L;
239 float psi_m_r = (zeta_r < 0.f)
240 ? psi_unstable_u(zeta_r)
241 : psi_stable(zeta_r);
242 float psi_h_r = (zeta_r < 0.f)
243 ? psi_unstable_h(zeta_r)
244 : psi_stable(zeta_r);
245
246 // 5) stability‐corrected friction velocity
247 float u_star = von_Karman_constant * wind_speed_reference
248 / (std::log((reference_height_m - displacement_height) / zo_m) - psi_m_r);
249
250 // 6) Monin–Obukhov scales θ_* and q_*
251 float theta_star = H_s / (rho_air_mol_m3 * cp_air_mol * u_star);
252 float q_star = E_s / (rho_air_mol_m3 * u_star);
253
254 // 7) corrected log‐law lengths
255 float ln_m_corr = std::log((reference_height_m - displacement_height) / zo_m) - psi_m_r;
256 float ln_h_corr = std::log((reference_height_m - displacement_height) / zo_h) - psi_h_r;
257
258 // 8) aerodynamic conductance ga [mol m⁻² s⁻¹]
259 float ga = std::pow(von_Karman_constant, 2) * wind_speed_reference / (ln_m_corr * ln_h_corr) * rho_air_mol_m3;
260
261 // 9) interface values at canopy top (Monin–Obukhov log-law)
262 float air_moisture_reference = esat_Pa(air_temperature_reference) / Patm * air_humidity_reference;
263 float T_int = air_temperature_reference + theta_star / von_Karman_constant * ln_h_corr;
264 float x_int = air_moisture_reference + q_star / von_Karman_constant * ln_h_corr;
265
266 // Cap relative humidity to 100%
267 float x_sat = esat_Pa(T_int)/Patm;
268 if ( x_int > x_sat ) {
269 //std::cerr << "WARNING (EnergyBalanceModel::evaluateAirEnergyBalance): Air moisture exceeds saturation. Capping to saturation value." << std::endl;
270 x_int = 0.99f*x_sat;
271 }
272
273 // 10) entrainment conductance g_e (mol m⁻² s⁻¹)
274 float H_canopy = sensible_source_flux_W_m3 * canopy_height_m;
275 float g_e;
276 if (air_temperature_ABL == 0.f) {
277 // first‐step equilibrium initialization
278 float w_star0 = H_canopy > 0.f ? std::pow(9.81f * H_canopy * canopy_height_m / (rho_air_mol_m3 * cp_air_mol * air_temperature_reference), 1.f / 3.f) : 0.f;
279 const float A_e = 0.02f;
280 float w_e0 = A_e * w_star0;
281 g_e = rho_air_mol_m3 * w_e0;
282 // initialize ABL state
283 air_temperature_ABL = (ga * air_temperature_average
284 + g_e * air_temperature_reference)
285 / (ga + g_e);
286 air_moisture_ABL = (ga * air_moisture_average
287 + g_e * air_moisture_reference)
288 / (ga + g_e);
289 } else {
290 // ongoing entrainment
291 float w_star = 0.f;
292 if (H_canopy > 0.f) {
293 w_star = std::pow(9.81f * H_canopy * canopy_height_m
294 / (rho_air_mol_m3 * cp_air_mol * air_temperature_ABL), 1.f / 3.f);
295 }
296 float u_star_shear = von_Karman_constant * wind_speed_reference
297 / std::log((reference_height_m - displacement_height) / zo_m);
298 float w_e = std::fmax(0.01f * w_star, 0.005f * u_star_shear);
299 g_e = rho_air_mol_m3 * w_e;
300 }
301
302 // if ( g_e>0.05 ) {
303 // std::cerr << "Capping g_e value of " << g_e << std::endl;
304 // g_e = 0.05f;
305 // }
306 g_e = 2.0;
307
308 // 11) canopy→ABL fluxes
309 float sensible_upper_flux_W_m2 = cp_air_mol * ga * (air_temperature_average - T_int);
310
311 // 12) update canopy‐air temperature (implicit Euler)
312 // numerator: T^n + (Δt / (ρ c_p h_c)) * [H_s + c_p g_a T_int]
313 // denominator: 1 + Δt·g_a/(ρ h_c)
314 float numerator_temp = air_temperature_average
315 + dt_actual
316 / (rho_air_mol_m3 * cp_air_mol * canopy_height_m)
317 * (H_s + cp_air_mol * ga * T_int);
318 float denominator_temp = 1.f
319 + dt_actual * ga / (rho_air_mol_m3 * canopy_height_m);
320 air_temperature_average = numerator_temp / denominator_temp;
321
322 // 13) update canopy‐air moisture (implicit Euler)
323
324 // --- canopy implicit‐Euler moisture update ---
325 // update: ρh (x_new - x_old)/dt = E_s - E_top
326 float E_top = ga * (air_moisture_average - x_int);
327 float f = std::pow(1.f - (1.f - 0.622f)*air_moisture_average, 2) / 0.622f;
328 float numer_can_x = air_moisture_average + dt_actual * f / (rho_air_mol_m3 * canopy_height_m) * (E_s + ga * x_int);
329 float denom_can_x = 1.f + dt_actual * f * ga/(rho_air_mol_m3*canopy_height_m);
330 air_moisture_average = numer_can_x/denom_can_x;
331
332 // 14) update ABL‐air temperature (implicit Euler)
333 float denom_abl_T = 1.f + dt_actual * (ga + g_e) / (rho_air_mol_m3 * cp_air_mol * abl_height_m);
334 float num_abl_T = air_temperature_ABL
335 + dt_actual / (rho_air_mol_m3 * cp_air_mol * abl_height_m)
336 * (sensible_upper_flux_W_m2 + cp_air_mol * g_e * (air_temperature_reference - air_temperature_ABL));
337 air_temperature_ABL = num_abl_T / denom_abl_T;
338
339 // 15) update ABL‐air moisture (implicit Euler)
340 // source into ABL storage
341 // ρ_ab·h_ab · (x_ab_new - x_ab_old)/dt = E_top - E_e
342 float E_e = g_e * ( air_moisture_ABL - air_moisture_reference );
343 float numer_abl_x = air_moisture_ABL + dt_actual / (rho_air_mol_m3 * abl_height_m) * ( E_top - E_e );
344 float denom_abl_x = 1.f + dt_actual * (ga + g_e)/ (rho_air_mol_m3 * abl_height_m);
345 air_moisture_ABL = numer_abl_x / denom_abl_x;
346
347 // moisture budget check (mol m⁻² s⁻¹)
348 float canopy_storage = (air_moisture_average - air_moisture_old) * rho_air_mol_m3 * canopy_height_m / dt_actual;
349 float flux_in = E_s; // from latent source
350 float flux_out = E_top; // to ABL
351 std::cerr << std::fixed
352 << "CANOPY MOISTURE BUDGET: in="<<flux_in
353 <<" out="<<flux_out
354 <<" storage="<<canopy_storage
355 <<" residual="<<(flux_in - flux_out - canopy_storage)
356 << std::endl;
357
358 // Cap relative humidity to 100%
359 float esat = esat_Pa(air_temperature_average)/Patm;
360 if ( air_moisture_average > esat ) {
361 std::cerr << "WARNING (EnergyBalanceModel::evaluateAirEnergyBalance): Air moisture exceeds saturation. Capping to saturation value." << std::endl;
362 air_moisture_average = 0.99f*esat;
363 }
364 esat = esat_Pa(air_temperature_ABL)/Patm;
365 if ( air_moisture_ABL > esat ) {
366 air_moisture_ABL = 0.99f*esat;
367 }
368
369 // Check that timestep is not too large based on aerodynamic resistance
370 if ( dt_actual > 0.5f*canopy_height_m*rho_air_mol_m3/ga ) {
371 std::cerr << "WARNING (EnergyBalanceModel::evaluateAirEnergyBalance): Time step is too large. The air energy balance may not converge properly." << std::endl;
372 }
373
374 float heat_from_Htop = dt_actual / (rho_air_mol_m3 * cp_air_mol * abl_height_m) * sensible_upper_flux_W_m2;
375 float heat_from_entrainment = dt_actual * g_e * (air_temperature_reference - air_temperature_ABL) / (rho_air_mol_m3 * abl_height_m);
376
377 std::cerr << "ABL ENERGY UPDATE:\n"
378 << " T_ABL_n = " << air_temperature_ABL << "\n"
379 << " T_ref = " << air_temperature_reference << "\n"
380 << " g_e = " << g_e << " mol/m²/s\n"
381 << " Heat from H_top = " << heat_from_Htop << " K\n"
382 << " Heat from H_e = " << heat_from_entrainment << " K\n"
383 << " Numerator = " << num_abl_T << " K\n"
384 << " Denominator = " << denom_abl_T << "\n"
385 << " T_ABL_new = " << (num_abl_T / denom_abl_T) << "\n";
386
387
388#ifdef HELIOS_DEBUG
389
390 // -- check energy consistency -- //
391 H_s = sensible_source_flux_W_m3 * canopy_height_m; // W m⁻², canopy source
392 float H_top = cp_air_mol * ga * (air_temperature_average - T_int); // W m⁻², canopy→ABL
393 float H_e = cp_air_mol * g_e * (air_temperature_ABL - air_temperature_reference); // W m⁻², ABL→free atm
394
395 E_s = moisture_source_flux_mol_s_m3 * canopy_height_m; // mol m⁻² s⁻¹, canopy source
396 E_top = ga * (air_moisture_average - x_int); // mol m⁻² s⁻¹, canopy→ABL
397 E_e = g_e * (air_moisture_ABL - air_moisture_reference); // mol m⁻² s⁻¹, ABL→free atm
398
399 // Print a concise table of key values
400 std::cout << std::fixed << std::setprecision(5)
401 << "step="<<time
402 << " T_c="<<air_temperature_average
403 << " T_int="<<T_int
404 << " T_b="<<air_temperature_ABL
405 << " H_s="<<H_s
406 << " H_top="<<H_top
407 << " H_e="<<H_e
408 << " E_s="<<E_s
409 << " E_top="<<E_top
410 << " E_e="<<E_e
411 << " psi_m="<<psi_m_r
412 << " psi_h="<<psi_h_r
413 << " u*="<<u_star
414 << " L="<<L
415 << std::endl;
416
417 // temperature balance
418
419 const float tol = 1e-5f;
420 float S_can = rho_air_mol_m3 * cp_air_mol * canopy_height_m * (air_temperature_average - air_temperature_old) / dt_actual;
421 if (fabs(H_s - H_top - S_can) > tol)
422 std::cerr << "Canopy budget error: " << (H_s - H_top - S_can) << "\n";
423
424 float S_abl = rho_air_mol_m3 * cp_air_mol * abl_height_m * (air_temperature_ABL - air_temperature_ABL_old) / dt_actual;
425 if (fabs(H_top - H_e - S_abl) > tol)
426 std::cerr << "ABL budget error: " << (H_top - H_e - S_abl) << "\n";
427
428 // moisture balance per unit ground area
429
430 S_can = rho_air_mol_m3 * canopy_height_m * (air_moisture_average - air_moisture_old) / dt_actual;
431
432 const float tol_m = 1e-6f;
433 if (std::fabs(E_s - E_top - S_can) > tol_m) {
434 std::cerr << "CANOPY MOISTURE BUDGET ERROR: " << "E_s - E_up - S_can = " << E_s - E_top - S_can << " mol/m2/s\n";
435 }
436
437 S_abl = rho_air_mol_m3 * abl_height_m * (air_moisture_ABL - air_moisture_ABL_old) / dt_actual;
438
439 if (std::fabs(E_top - E_e - S_abl) > tol_m) {
440 std::cerr << "ABL MOISTURE BUDGET ERROR: " << "E_up - E_e - S_abl = " << E_top - E_e - S_abl << " mol/m2/s\n";
441 }
442
443
444#endif
445
446 // Set primitive data
447 context->setPrimitiveData( UUIDs, "air_temperature", air_temperature_average );
448 float air_humidity = air_moisture_average*Patm/esat_Pa(air_temperature_average);
449 context->setPrimitiveData( UUIDs, "air_humidity", air_humidity );
450
451 // Set global data
452 context->setGlobalData("air_temperature_average", air_temperature_average);
453 context->setGlobalData("air_humidity_average", air_humidity);
454 context->setGlobalData("air_moisture_average", air_moisture_average);
455
456 context->setGlobalData( "air_temperature_ABL", air_temperature_ABL );
457 float air_humidity_ABL = air_moisture_ABL*Patm/esat_Pa(air_temperature_ABL);
458 context->setGlobalData( "air_humidity_ABL", air_humidity_ABL );
459
460 context->setGlobalData( "air_temperature_interface", T_int );
461 float air_humidity_interface = x_int*Patm/esat_Pa(T_int);
462 context->setGlobalData( "air_humidity_interface", air_humidity_interface );
463
464 context->setGlobalData("aerodynamic_resistance", rho_air_mol_m3/ga);
465
466 std::cout << "Computed air temperature: " << air_temperature_average << " " << air_temperature_old << std::endl;
467 std::cout << "Computed air humidity: " << air_humidity << std::endl;
468 std::cout << "Computed air moisture: " << air_moisture_average << " " << air_moisture_old << std::endl;
469
470 if ( time>0 && std::abs(air_temperature_average-air_temperature_old)/air_temperature_old < 0.003f && std::abs(air_moisture_average-air_moisture_old)/air_moisture_old < 0.003f ) {
471 // If the air temperature and humidity have not changed significantly, we can stop iterating
472 std::cout << "Converged" << std::endl;
473 break;
474 }

476 air_moisture_old = air_moisture_average;
477 air_temperature_ABL_old = air_temperature_ABL;
478 air_moisture_ABL_old = air_moisture_ABL;

480 time += dt_actual;

537void EnergyBalanceModel::printDefaultValueReport(const std::vector<uint> &UUIDs) const {
538
539 size_t assumed_default_TL = 0;
540 size_t assumed_default_U = 0;
541 size_t assumed_default_L = 0;
542 size_t assumed_default_p = 0;
543 size_t assumed_default_Ta = 0;
544 size_t assumed_default_rh = 0;
545 size_t assumed_default_gH = 0;
546 size_t assumed_default_gs = 0;
547 size_t assumed_default_Qother = 0;
548 size_t assumed_default_heatcapacity = 0;
549 size_t assumed_default_fs = 0;
550 size_t twosided_0 = 0;
551 size_t twosided_1 = 0;
552 size_t Ne_1 = 0;
553 size_t Ne_2 = 0;
554
555 size_t Nprimitives = UUIDs.size();
556
557 for (uint UUID: UUIDs) {
558
559 //surface temperature (K)
560 if (!context->doesPrimitiveDataExist(UUID, "temperature") ||
561 context->getPrimitiveDataType(UUID, "temperature") != HELIOS_TYPE_FLOAT) {
562 assumed_default_TL++;
563 }
564
565 //air pressure (Pa)
566 if (!context->doesPrimitiveDataExist(UUID, "air_pressure") ||
567 context->getPrimitiveDataType(UUID, "air_pressure") != HELIOS_TYPE_FLOAT) {
568 assumed_default_p++;
569 }
570
571 //air temperature (K)
572 if (!context->doesPrimitiveDataExist(UUID, "air_temperature") ||
573 context->getPrimitiveDataType(UUID, "air_temperature") != HELIOS_TYPE_FLOAT) {
574 assumed_default_Ta++;
575 }
576
577 //air humidity
578 if (!context->doesPrimitiveDataExist(UUID, "air_humidity") ||
579 context->getPrimitiveDataType(UUID, "air_humidity") != HELIOS_TYPE_FLOAT) {
580 assumed_default_rh++;
581 }
582
583 //boundary-layer conductance to heat
584 if (!context->doesPrimitiveDataExist(UUID, "boundarylayer_conductance") || context->getPrimitiveDataType(UUID, "boundarylayer_conductance") != HELIOS_TYPE_FLOAT){
585 assumed_default_gH++;
586 }
587
588 //wind speed
589 if (!context->doesPrimitiveDataExist(UUID, "wind_speed") || context->getPrimitiveDataType(UUID, "wind_speed") != HELIOS_TYPE_FLOAT){
590 assumed_default_U++;
591 }
592
593 //object length
594 if (!context->doesPrimitiveDataExist(UUID, "object_length") || context->getPrimitiveDataType(UUID, "object_length") != HELIOS_TYPE_FLOAT){
595 assumed_default_L++;
596 }
597
598 //moisture conductance
599 if (!context->doesPrimitiveDataExist(UUID, "moisture_conductance") || context->getPrimitiveDataType(UUID, "moisture_conductance") != HELIOS_TYPE_FLOAT){
600 assumed_default_gs++;
601 }
602
603 //Heat capacity
604 if (!context->doesPrimitiveDataExist(UUID, "heat_capacity") || context->getPrimitiveDataType(UUID, "heat_capacity") != HELIOS_TYPE_FLOAT) {
605 assumed_default_heatcapacity++;
606 }
607
608 //"Other" heat fluxes
609 if (!context->doesPrimitiveDataExist(UUID, "other_surface_flux") || context->getPrimitiveDataType(UUID, "other_surface_flux") != HELIOS_TYPE_FLOAT) {
610 assumed_default_Qother++;
611 }
612
613 //two-sided flag
614 if ( context->doesPrimitiveDataExist(UUID, "twosided_flag") && context->getPrimitiveDataType(UUID, "twosided_flag") == HELIOS_TYPE_UINT) {
615 uint twosided;
616 context->getPrimitiveData(UUID, "twosided_flag", twosided);
617 if( twosided==0 ){
618 twosided_0++;
619 }else if( twosided==1 ){
620 twosided_1++;
621 }
622 }else{
623 twosided_1++;
624 }
625
626 //number of evaporating faces
627 if ( context->doesPrimitiveDataExist(UUID, "evaporating_faces") && context->getPrimitiveDataType(UUID, "evaporating_faces") == HELIOS_TYPE_UINT) {
628 uint Ne;
629 context->getPrimitiveData(UUID, "evaporating_faces", Ne);
630 if( Ne==1 ){
631 Ne_1++;
632 }else if( Ne==2 ){
633 Ne_2++;
634 }
635 }else{
636 Ne_1++;
637 }
638
639 //Surface humidity
640 if (!context->doesPrimitiveDataExist(UUID, "surface_humidity") || context->getPrimitiveDataType(UUID, "surface_humidity") != HELIOS_TYPE_FLOAT) {
641 assumed_default_fs++;
642 }
643
644 }
645
646 std::cout << "--- Energy Balance Model Default Value Report ---" << std::endl;
647
648 std::cout << "surface temperature (initial guess): " << assumed_default_TL << " of " << Nprimitives << " used default value of " << temperature_default << " because ""temperature"" primitive data did not exist" << std::endl;
649 std::cout << "air pressure: " << assumed_default_p << " of " << Nprimitives << " used default value of " << pressure_default << " because ""air_pressure"" primitive data did not exist" << std::endl;
650 std::cout << "air temperature: " << assumed_default_Ta << " of " << Nprimitives << " used default value of " << air_temperature_default << " because ""air_temperature"" primitive data did not exist" << std::endl;
651 std::cout << "air humidity: " << assumed_default_rh << " of " << Nprimitives << " used default value of " << air_humidity_default << " because ""air_humidity"" primitive data did not exist" << std::endl;
652 std::cout << "boundary-layer conductance: " << assumed_default_gH << " of " << Nprimitives << " calculated boundary-layer conductance from Polhausen equation because ""boundarylayer_conductance"" primitive data did not exist" << std::endl;
653 if( assumed_default_gH>0 ){
654 std::cout << " - wind speed: " << assumed_default_U << " of " << assumed_default_gH << " using Polhausen equation used default value of " << wind_speed_default << " because ""wind_speed"" primitive data did not exist" << std::endl;
655 std::cout << " - object length: " << assumed_default_L << " of " << assumed_default_gH << " using Polhausen equation used the primitive/object length/area to calculate object length because ""object_length"" primitive data did not exist" << std::endl;
656 }
657 std::cout << "moisture conductance: " << assumed_default_gs << " of " << Nprimitives << " used default value of " << gS_default << " because ""moisture_conductance"" primitive data did not exist" << std::endl;
658 std::cout << "surface humidity: " << assumed_default_fs << " of " << Nprimitives << " used default value of " << surface_humidity_default << " because ""surface_humidity"" primitive data did not exist" << std::endl;
659 std::cout << "two-sided flag: " << twosided_0 << " of " << Nprimitives << " used two-sided flag=0; " << twosided_1 << " of " << Nprimitives << " used two-sided flag=1 (default)" << std::endl;
660 std::cout << "evaporating faces: " << Ne_1 << " of " << Nprimitives << " used Ne = 1 (default); " << Ne_2 << " of " << Nprimitives << " used Ne = 2" << std::endl;
661
662 std::cout << "------------------------------------------------------" << std::endl;
