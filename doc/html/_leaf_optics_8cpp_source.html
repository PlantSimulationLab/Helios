91void LeafOptics::prospect(float numberlayers, float chlorophyllcontent, float carotenoidcontent, float anthocyancontent, float brownpigments,
92 float watermass, float drymass, float protein, float carbonconstituents, std::vector<float> &reflectivities_fit, std::vector<float> &transmissivities_fit)
93// Implementation of Prospect-PRO, port of public available matlab code
94{
95 double k;
96 double tau,ralf,r12,talf, t12, t21, r21, denom, Ta, Ra, t, r;
97 // Loop over wavelength, might be a way to be vectorized at least partly
98 for(int i=0; i<LeafOptics::nw; i++) {
99 // k: the mean absorption coefficient of each elementary layer.
100 k = (chlorophyllcontent * absorption_chlorophyll.at(i) + carotenoidcontent * absorption_carotenoid.at(i) + anthocyancontent * absorption_anthocyanin.at(i)
101 + brownpigments * absorption_brown.at(i) + watermass * absorption_water.at(i) + drymass * absorption_drymass.at(i)
102 + protein * absorption_protein.at(i)+carbonconstituents * absorption_carbonconstituents.at(i)) / numberlayers;
103
104 // diffuse transmittance through elementary layer, this integral needs more effort in C++
105 tau = transmittance(k);
106 // surface reflectance at radiated leaf side for near normal incident beam radiation,
107 ralf = R_spec_normal.at(i); // calculate 1-tav in surface function
108 // surface reflectance at radiated leaf side for diffuse radiation
109 r12 = R_spec_diffuse.at(i);
110
111
112 talf = 1 - ralf; //tav90
113 t12 = 1 - r12; //tav
114
115 // transmittance and reflectance for leaf internal diffuse light
116 t21 = t12/(refractiveindex.at(i)*refractiveindex.at(i)); // tav/n^2
117 r21 = 1-t21;
118
119 // top or incident surface side
120
121 denom = 1-r21*r21*tau*tau; // (euqation1 in RPOSPECT)
122 Ta = talf*tau*t21/denom; // transmittance of top surface (euqation2 in RPOSPECT) taua
123 Ra = ralf+r21*tau*Ta; // reflectance of top surface (euqation1 in RPOSPECT) rhoa
124
125 // bottom surface side
126
127 t = t12*tau*t21/denom; // (page 78 paragraph 1 in RPOSPECT) tau90
128 r = r12+r21*tau*t; // (page 78 paragraph 1 in RPOSPECT) rho90
129
130 // reflectance and transmittance of numberlayers layers, Stokes' solution
131 double D, rq,tq,a,b,bNm1,bN2,a2,Rsub,Tsub;
132 D = sqrt((1.+r+t)*(1.+r-t)*(1.-r+t)*(1.-r-t));
133 rq = r*r;
134 tq = t*t;
135 a = (1.+rq-tq+D)/(2*r);
136 b = (1.-rq+tq+D)/(2*t);
137 bNm1 = std::pow(b,(numberlayers - 1));
138 bN2 = bNm1*bNm1;
139 a2 = a*a;
140 denom = a2*bN2-1.;
141 Rsub = a*(bN2-1.)/denom;
142 Tsub = bNm1*(a2-1.)/denom;
143
144 //Case of zero absorption
145
146 if ((r+t) > 1.0) {
147 Tsub = t/(t+(1.-t)*(numberlayers - 1));
148 Rsub = 1-Tsub;
149 }
150
151 // Reflectance and transmittance of the leaf: combine top layer with next numberlayers-1 layers
152
153 denom = 1-Rsub*r;
154 transmissivities_fit.push_back(Ta * Tsub / denom); //(euqation8 in RPOSPEC
155 reflectivities_fit.push_back(Ra + Ta * Rsub * t / denom); //(euqation7 in RPOSPECT)
156 }
157}

193void LeafOptics::surface(float degree, std::vector<float> &reflectivities)
196{
197
198 double rad2degree = 57.2958;
199 // tav is the transmissivity of a dielectric plane surface, averaged over all directions of incidence and over all polarizations.
200 double n2,np,nm,a,k,sinvalue,b1,b2,b,b3,a3,ts,tp1,tp2,tp3,tp4,tp5,tp,tav;
201 for(int i=0; i<LeafOptics::nw; i++) {
202 double n = refractiveindex.at(i); //refractive index
203 n2 = n*n;
204 np = n2 + 1;
205 nm = n2 - 1;
206 a = (n + 1) * (n + 1) / 2;
207 k = -(n2 - 1) * (n2 - 1) / 4;
208 sinvalue = sin(degree / rad2degree);
209 if (degree < 90.0){b1 = sqrt((sinvalue * sinvalue - np / 2) * (sinvalue * sinvalue - np / 2) + k);}
210 else{b1= 0.0;}
211 b2 = sinvalue * sinvalue - np / 2;
212 b = b1 - b2;
213 b3 = b*b*b;
214 a3 = a*a*a;
215 ts = (k*k / (6 * b3) + k/ b - b / 2) - (k*k / (6 * a3) + k/ a - a / 2);
216 tp1 = -2 * n2*(b - a)/ (np*np);
217 tp2 = -2 * n2*np*log(b/ a)/ (nm*nm);
218 tp3 = n2*(1 / b - 1 / a) / 2;
219 tp4 = 16 * n2*n2 * (n2*n2 + 1)*log((2 * np*b - nm*nm)/ (2 * np*a - nm*nm))/ (np*np*np * nm*nm);
220 tp5 = 16 * n2*n2*n2 * (1 / (2 * np*b - nm*nm) - 1 / (2 * np*a - nm*nm))/ (np*np*np);
221 tp = tp1 + tp2 + tp3 + tp4 + tp5;
222 tav = (ts + tp)/ (2 * sinvalue * sinvalue);
223 reflectivities.push_back(1.0 - tav);
224 }
225 return;
