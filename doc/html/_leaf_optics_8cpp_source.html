353void LeafOptics::PROSPECT(float numberlayers, float Chlorophyllcontent, float carotenoidcontent, float anthocyancontent, float brownpigments, float watermass, float drymass, float protein, float carbonconstituents,
354 std::vector<float> &reflectivities_fit, std::vector<float> &transmissivities_fit)
355// Implementation of Prospect-PRO, port of public available matlab code
356{
357 double k;
358 double tau, ralf, r12, talf, t12, t21, r21, denom, Ta, Ra, t, r;
359 // Loop over wavelength, might be a way to be vectorized at least partly
360 for (int i = 0; i < LeafOptics::nw; i++) {
361 // k: the mean absorption coefficient of each elementary layer.
362 k = (Chlorophyllcontent * absorption_chlorophyll.at(i) + carotenoidcontent * absorption_carotenoid.at(i) + anthocyancontent * absorption_anthocyanin.at(i) + brownpigments * absorption_brown.at(i) + watermass * absorption_water.at(i) +
363 drymass * absorption_drymass.at(i) + protein * absorption_protein.at(i) + carbonconstituents * absorption_carbonconstituents.at(i)) /
364 numberlayers;
365
366 // diffuse transmittance through elementary layer, this integral needs more effort in C++
367 tau = transmittance(k);
368 // surface reflectance at radiated leaf side for near normal incident beam radiation,
369 ralf = R_spec_normal.at(i); // calculate 1-tav in surface function
370 // surface reflectance at radiated leaf side for diffuse radiation
371 r12 = R_spec_diffuse.at(i);
372
373
374 talf = 1 - ralf; // tav90
375 t12 = 1 - r12; // tav
376
377 // transmittance and reflectance for leaf internal diffuse light
378 t21 = t12 / (refractiveindex.at(i) * refractiveindex.at(i)); // tav/n^2
379 r21 = 1 - t21;
380
381 // top or incident surface side
382
383 denom = 1 - r21 * r21 * tau * tau; // (euqation1 in RPOSPECT)
384 Ta = talf * tau * t21 / denom; // transmittance of top surface (euqation2 in RPOSPECT) taua
385 Ra = ralf + r21 * tau * Ta; // reflectance of top surface (euqation1 in RPOSPECT) rhoa
386
387 // bottom surface side
388
389 t = t12 * tau * t21 / denom; // (page 78 paragraph 1 in RPOSPECT) tau90
390 r = r12 + r21 * tau * t; // (page 78 paragraph 1 in RPOSPECT) rho90
391
392 // reflectance and transmittance of numberlayers layers, Stokes' solution
393 double D, rq, tq, a, b, bNm1, bN2, a2, Rsub, Tsub;
394 D = sqrt((1. + r + t) * (1. + r - t) * (1. - r + t) * (1. - r - t));
395 rq = r * r;
396 tq = t * t;
397 a = (1. + rq - tq + D) / (2 * r);
398 b = (1. - rq + tq + D) / (2 * t);
399 bNm1 = std::pow(b, (numberlayers - 1));
400 bN2 = bNm1 * bNm1;
401 a2 = a * a;
402 denom = a2 * bN2 - 1.;
403 Rsub = a * (bN2 - 1.) / denom;
404 Tsub = bNm1 * (a2 - 1.) / denom;
405
406 // Case of zero absorption
407
408 if ((r + t) > 1.0) {
409 Tsub = t / (t + (1. - t) * (numberlayers - 1));
410 Rsub = 1 - Tsub;
411 }
412
413 // Reflectance and transmittance of the leaf: combine top layer with next numberlayers-1 layers
414
415 denom = 1 - Rsub * r;
416 transmissivities_fit.push_back(Ta * Tsub / denom); //(euqation8 in RPOSPEC
417 reflectivities_fit.push_back(Ra + Ta * Rsub * t / denom); //(euqation7 in RPOSPECT)
418 }
419}

451void LeafOptics::surface(float degree, std::vector<float> &reflectivities)
454{
455
456 double rad2degree = 57.2958;
457 // tav is the transmissivity of a dielectric plane surface, averaged over all directions of incidence and over all polarizations.
458 double n2, np, nm, a, k, sinvalue, b1, b2, b, b3, a3, ts, tp1, tp2, tp3, tp4, tp5, tp, tav;
459 for (int i = 0; i < LeafOptics::nw; i++) {
460 double n = refractiveindex.at(i); // refractive index
461 n2 = n * n;
462 np = n2 + 1;
463 nm = n2 - 1;
464 a = (n + 1) * (n + 1) / 2;
465 k = -(n2 - 1) * (n2 - 1) / 4;
466 sinvalue = sin(degree / rad2degree);
467 if (degree < 90.0) {
468 b1 = sqrt((sinvalue * sinvalue - np / 2) * (sinvalue * sinvalue - np / 2) + k);
469 } else {
470 b1 = 0.0;
471 }
472 b2 = sinvalue * sinvalue - np / 2;
473 b = b1 - b2;
474 b3 = b * b * b;
475 a3 = a * a * a;
476 ts = (k * k / (6 * b3) + k / b - b / 2) - (k * k / (6 * a3) + k / a - a / 2);
477 tp1 = -2 * n2 * (b - a) / (np * np);
478 tp2 = -2 * n2 * np * log(b / a) / (nm * nm);
479 tp3 = n2 * (1 / b - 1 / a) / 2;
480 tp4 = 16 * n2 * n2 * (n2 * n2 + 1) * log((2 * np * b - nm * nm) / (2 * np * a - nm * nm)) / (np * np * np * nm * nm);
481 tp5 = 16 * n2 * n2 * n2 * (1 / (2 * np * b - nm * nm) - 1 / (2 * np * a - nm * nm)) / (np * np * np);
482 tp = tp1 + tp2 + tp3 + tp4 + tp5;
483 tav = (ts + tp) / (2 * sinvalue * sinvalue);
484 reflectivities.push_back(1.0 - tav);
485 }
486 return;
487}

680LeafOpticsProperties LeafOptics::computePropertiesFromNitrogen(float N_area_gN_m2, const LeafOpticsProperties_Nauto &params) {
681 LeafOpticsProperties props;
682
683 // Convert nitrogen concentration to chlorophyll content
684 // N (g/m2) -> N (ug/cm2): multiply by 100
685 // Then apply photosynthetic fraction and empirical coefficient
686 float N_area_ug_cm2 = N_area_gN_m2 * 100.0f;
687 props.chlorophyllcontent = N_area_ug_cm2 * params.f_photosynthetic * params.N_to_Cab_coefficient;
688
689 // Clamp chlorophyll to physically reasonable range [5, 80] ug/cm2
690 props.chlorophyllcontent = std::max(5.0f, std::min(80.0f, props.chlorophyllcontent));
691
692 // Calculate carotenoids from chlorophyll ratio
693 props.carotenoidcontent = props.chlorophyllcontent * params.Car_to_Cab_ratio;
694
695 // Copy fixed parameters from Nauto struct
696 props.numberlayers = params.numberlayers;
697 props.anthocyancontent = params.anthocyancontent;
698 props.brownpigments = params.brownpigments;
699 props.watermass = params.watermass;
700 props.drymass = params.drymass;
701 props.protein = params.protein;
702 props.carbonconstituents = params.carbonconstituents;
703
704 return props;
705}

734void LeafOptics::createAdaptiveBins(const std::vector<float> &nitrogen_values) {
735 nitrogen_bins.clear();
736
737 if (nitrogen_values.empty()) {
738 return;
739 }
740
741 // Sort nitrogen values for quantile calculation
742 std::vector<float> sorted_N = nitrogen_values;
743 std::sort(sorted_N.begin(), sorted_N.end());
744
745 uint count = sorted_N.size();
746 uint target_bins = nitrogen_params.num_bins;
747
748 // Adjust number of bins if fewer unique values than requested bins
749 if (count < target_bins) {
750 target_bins = count;
751 }
752
753 // Create quantile-based bin centers
754 std::vector<float> bin_centers;
755 for (uint i = 0; i < target_bins; i++) {
756 // Calculate index for quantile center
757 uint idx = (i * count / target_bins) + (count / (2 * target_bins));
758 if (idx >= count) {
759 idx = count - 1;
760 }
761 float center = sorted_N[idx];
762
763 // Only add if not a duplicate (within tolerance)
764 bool is_duplicate = false;
765 for (float existing: bin_centers) {
766 if (std::abs(center - existing) < 0.001f) {
767 is_duplicate = true;
768 break;
769 }
770 }
771 if (!is_duplicate) {
772 bin_centers.push_back(center);
773 }
774 }
775
776 // Generate PROSPECT spectrum for each bin
777 for (uint i = 0; i < bin_centers.size(); i++) {
778 SpectrumBin bin;
779 bin.N_center = bin_centers[i];
780 bin.spectrum_label = "Nauto_" + std::to_string(i);
781
782 // Compute PROSPECT properties and generate spectrum
783 LeafOpticsProperties props = computePropertiesFromNitrogen(bin.N_center, nitrogen_params);
784 run(props, bin.spectrum_label);
785
786 nitrogen_bins.push_back(bin);
787 }
788
789 if (message_flag) {
790 std::cout << "LeafOptics: Created " << nitrogen_bins.size() << " nitrogen-based spectrum bins." << std::endl;
791 }
792}

794uint LeafOptics::findNearestBin(float N_value) {
795 if (nitrogen_bins.empty()) {
796 helios_runtime_error("ERROR (LeafOptics::findNearestBin): No nitrogen bins have been created.");
797 }
798
799 uint best_bin = 0;
800 float best_distance = std::abs(N_value - nitrogen_bins[0].N_center);
801
802 for (uint i = 1; i < nitrogen_bins.size(); i++) {
803 float distance = std::abs(N_value - nitrogen_bins[i].N_center);
804 if (distance < best_distance) {
805 best_distance = distance;
806 best_bin = i;
807 }
808 }
809
810 return best_bin;

813bool LeafOptics::shouldReassign(float current_N, uint current_bin) {
814 if (current_bin >= nitrogen_bins.size()) {
815 return false;
816 }
817
818 float bin_center = nitrogen_bins[current_bin].N_center;
819 float absolute_change = std::abs(current_N - bin_center);
820
821 // Check absolute threshold first
822 if (absolute_change < nitrogen_params.min_reassignment_change) {
823 return false;
824 }
825
826 // Check relative threshold
827 float relative_change = (bin_center > 0.0f) ? (absolute_change / bin_center) : 0.0f;
828 return relative_change > nitrogen_params.reassignment_threshold;
829}

831bool LeafOptics::isSignificantImprovement(float current_N, uint old_bin, uint new_bin) {
832 if (old_bin >= nitrogen_bins.size() || new_bin >= nitrogen_bins.size()) {
833 return false;
834 }
835
836 float old_distance = std::abs(current_N - nitrogen_bins[old_bin].N_center);
837 float new_distance = std::abs(current_N - nitrogen_bins[new_bin].N_center);
838
839 // Require improvement of at least half the minimum change threshold (hysteresis)
840 return (old_distance - new_distance) > nitrogen_params.min_reassignment_change * 0.5f;
841}

882void LeafOptics::run(const std::vector<uint> &UUIDs, const LeafOpticsProperties_Nauto &params) {
883 if (UUIDs.empty()) {
884 if (message_flag) {
885 std::cout << "LeafOptics: Empty UUID list provided to nitrogen mode, nothing to do." << std::endl;
886 }
887 return;
888 }
889
890 // Group primitives by parent object
891 std::map<uint, std::vector<uint>> object_groups = groupPrimitivesByObject(UUIDs);
892
893 if (object_groups.empty()) {
894 if (message_flag) {
895 std::cerr << "WARNING (LeafOptics::run): No valid objects found for nitrogen-based leaf optics." << std::endl;
896 }
897 return;
898 }
899
900 // Collect nitrogen values from all objects
901 std::vector<float> nitrogen_values;
902 std::map<uint, float> object_nitrogen;
903
904 for (const auto &pair: object_groups) {
905 uint objID = pair.first;
906
907 // Check if nitrogen data exists on this object
908 if (!context->doesObjectDataExist(objID, "leaf_nitrogen_gN_m2")) {
909 helios_runtime_error("ERROR (LeafOptics::run): Object " + std::to_string(objID) + " does not have 'leaf_nitrogen_gN_m2' data. Enable PlantArchitecture nitrogen model first.");
910 }
911
912 float N_area;
913 context->getObjectData(objID, "leaf_nitrogen_gN_m2", N_area);
914 nitrogen_values.push_back(N_area);
915 object_nitrogen[objID] = N_area;
916 }
917
918 if (!nitrogen_mode_active) {
919 // First call: Initialize nitrogen mode
920 nitrogen_params = params;
921
922 // Create adaptive bins based on nitrogen distribution
923 createAdaptiveBins(nitrogen_values);
924
925 // Assign each object to nearest bin
926 for (const auto &pair: object_groups) {
927 uint objID = pair.first;
928 const std::vector<uint> &obj_UUIDs = pair.second;
929 float N_area = object_nitrogen[objID];
930
931 uint best_bin = findNearestBin(N_area);
932 assignSpectrumToPrimitives(obj_UUIDs, best_bin);
933
934 // Track assignment
935 ObjectAssignment assignment;
936 assignment.bin_index = best_bin;
937 assignment.N_at_assignment = N_area;
938 assignment.primitive_UUIDs = obj_UUIDs;
939 object_assignments[objID] = assignment;
940
941 // Track primitive to object mapping
942 for (uint UUID: obj_UUIDs) {
943 primitive_to_object[UUID] = objID;
944 }
945 }
946
947 nitrogen_mode_active = true;
948
949 if (message_flag) {
950 std::cout << "LeafOptics: Nitrogen mode initialized with " << object_assignments.size() << " leaf objects assigned to " << nitrogen_bins.size() << " spectrum bins." << std::endl;
951 }
952
953 } else {
954 // Subsequent call: Update assignments
955
956 // Build set of current object IDs
957 std::set<uint> current_objects;
958 for (const auto &pair: object_groups) {
959 current_objects.insert(pair.first);
960 }
961
962 // Build set of previously tracked objects
963 std::set<uint> tracked_objects;
964 for (const auto &pair: object_assignments) {
965 tracked_objects.insert(pair.first);
966 }
967
968 // Find removed objects (tracked but not in current)
969 std::vector<uint> removed_objects;
970 for (uint objID: tracked_objects) {
971 if (current_objects.find(objID) == current_objects.end()) {
972 removed_objects.push_back(objID);
973 }
974 }
975
976 // Remove tracking for removed objects
977 for (uint objID: removed_objects) {
978 const ObjectAssignment &assignment = object_assignments[objID];
979 for (uint UUID: assignment.primitive_UUIDs) {
980 primitive_to_object.erase(UUID);
981 }
982 object_assignments.erase(objID);
983 }
984
985 // Process current objects
986 for (const auto &pair: object_groups) {
987 uint objID = pair.first;
988 const std::vector<uint> &obj_UUIDs = pair.second;
989 float N_area = object_nitrogen[objID];
990
991 if (tracked_objects.find(objID) == tracked_objects.end()) {
992 // New object: assign to nearest existing bin
993 uint best_bin = findNearestBin(N_area);
994 assignSpectrumToPrimitives(obj_UUIDs, best_bin);
995
996 ObjectAssignment assignment;
997 assignment.bin_index = best_bin;
998 assignment.N_at_assignment = N_area;
999 assignment.primitive_UUIDs = obj_UUIDs;
1000 object_assignments[objID] = assignment;
1001
1002 for (uint UUID: obj_UUIDs) {
1003 primitive_to_object[UUID] = objID;
1004 }
1005
1006 } else {
1007 // Existing object: check for reassignment
1008 ObjectAssignment &assignment = object_assignments[objID];
1009
1010 // Update primitive list (in case it changed)
1011 assignment.primitive_UUIDs = obj_UUIDs;
1012 for (uint UUID: obj_UUIDs) {
1013 primitive_to_object[UUID] = objID;
1014 }
1015
1016 if (shouldReassign(N_area, assignment.bin_index)) {
1017 uint best_bin = findNearestBin(N_area);
1018
1019 if (best_bin != assignment.bin_index && isSignificantImprovement(N_area, assignment.bin_index, best_bin)) {
1020 assignSpectrumToPrimitives(obj_UUIDs, best_bin);
1021 assignment.bin_index = best_bin;
1022 assignment.N_at_assignment = N_area;
1023 }
1024 }
1025 }
1026 }
1027 }
