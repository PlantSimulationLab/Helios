178void LeafOptics::PROSPECT(float numberlayers, float Chlorophyllcontent, float carotenoidcontent, float anthocyancontent, float brownpigments,
179 float watermass, float drymass, float protein, float carbonconstituents, std::vector<float> &reflectivities_fit, std::vector<float> &transmissivities_fit)
180// Implementation of Prospect-PRO, port of public available matlab code
181{
182 double k;
183 double tau,ralf,r12,talf, t12, t21, r21, denom, Ta, Ra, t, r;
184 // Loop over wavelength, might be a way to be vectorized at least partly
185 for(int i=0; i<LeafOptics::nw; i++) {
186 // k: the mean absorption coefficient of each elementary layer.
187 k = (Chlorophyllcontent * absorption_chlorophyll.at(i) + carotenoidcontent * absorption_carotenoid.at(i) + anthocyancontent * absorption_anthocyanin.at(i)
188 + brownpigments * absorption_brown.at(i) + watermass * absorption_water.at(i) + drymass * absorption_drymass.at(i)
189 + protein * absorption_protein.at(i) +carbonconstituents * absorption_carbonconstituents.at(i)) / numberlayers;
190
191 // diffuse transmittance through elementary layer, this integral needs more effort in C++
192 tau = transmittance(k);
193 // surface reflectance at radiated leaf side for near normal incident beam radiation,
194 ralf = R_spec_normal.at(i); // calculate 1-tav in surface function
195 // surface reflectance at radiated leaf side for diffuse radiation
196 r12 = R_spec_diffuse.at(i);
197
198
199 talf = 1 - ralf; //tav90
200 t12 = 1 - r12; //tav
201
202 // transmittance and reflectance for leaf internal diffuse light
203 t21 = t12/(refractiveindex.at(i)*refractiveindex.at(i)); // tav/n^2
204 r21 = 1-t21;
205
206 // top or incident surface side
207
208 denom = 1-r21*r21*tau*tau; // (euqation1 in RPOSPECT)
209 Ta = talf*tau*t21/denom; // transmittance of top surface (euqation2 in RPOSPECT) taua
210 Ra = ralf+r21*tau*Ta; // reflectance of top surface (euqation1 in RPOSPECT) rhoa
211
212 // bottom surface side
213
214 t = t12*tau*t21/denom; // (page 78 paragraph 1 in RPOSPECT) tau90
215 r = r12+r21*tau*t; // (page 78 paragraph 1 in RPOSPECT) rho90
216
217 // reflectance and transmittance of numberlayers layers, Stokes' solution
218 double D, rq,tq,a,b,bNm1,bN2,a2,Rsub,Tsub;
219 D = sqrt((1.+r+t)*(1.+r-t)*(1.-r+t)*(1.-r-t));
220 rq = r*r;
221 tq = t*t;
222 a = (1.+rq-tq+D)/(2*r);
223 b = (1.-rq+tq+D)/(2*t);
224 bNm1 = std::pow(b,(numberlayers - 1));
225 bN2 = bNm1*bNm1;
226 a2 = a*a;
227 denom = a2*bN2-1.;
228 Rsub = a*(bN2-1.)/denom;
229 Tsub = bNm1*(a2-1.)/denom;
230
231 //Case of zero absorption
232
233 if ((r+t) > 1.0) {
234 Tsub = t/(t+(1.-t)*(numberlayers - 1));
235 Rsub = 1-Tsub;
236 }
237
238 // Reflectance and transmittance of the leaf: combine top layer with next numberlayers-1 layers
239
240 denom = 1-Rsub*r;
241 transmissivities_fit.push_back(Ta * Tsub / denom); //(euqation8 in RPOSPEC
242 reflectivities_fit.push_back(Ra + Ta * Rsub * t / denom); //(euqation7 in RPOSPECT)
243 }
244}

279void LeafOptics::surface(float degree, std::vector<float> &reflectivities)
282{
283
284 double rad2degree = 57.2958;
285 // tav is the transmissivity of a dielectric plane surface, averaged over all directions of incidence and over all polarizations.
286 double n2,np,nm,a,k,sinvalue,b1,b2,b,b3,a3,ts,tp1,tp2,tp3,tp4,tp5,tp,tav;
287 for(int i=0; i<LeafOptics::nw; i++) {
288 double n = refractiveindex.at(i); //refractive index
289 n2 = n*n;
290 np = n2 + 1;
291 nm = n2 - 1;
292 a = (n + 1) * (n + 1) / 2;
293 k = -(n2 - 1) * (n2 - 1) / 4;
294 sinvalue = sin(degree / rad2degree);
295 if (degree < 90.0){b1 = sqrt((sinvalue * sinvalue - np / 2) * (sinvalue * sinvalue - np / 2) + k);}
296 else{b1= 0.0;}
297 b2 = sinvalue * sinvalue - np / 2;
298 b = b1 - b2;
299 b3 = b*b*b;
300 a3 = a*a*a;
301 ts = (k*k / (6 * b3) + k/ b - b / 2) - (k*k / (6 * a3) + k/ a - a / 2);
302 tp1 = -2 * n2*(b - a)/ (np*np);
303 tp2 = -2 * n2*np*log(b/ a)/ (nm*nm);
304 tp3 = n2*(1 / b - 1 / a) / 2;
305 tp4 = 16 * n2*n2 * (n2*n2 + 1)*log((2 * np*b - nm*nm)/ (2 * np*a - nm*nm))/ (np*np*np * nm*nm);
306 tp5 = 16 * n2*n2*n2 * (1 / (2 * np*b - nm*nm) - 1 / (2 * np*a - nm*nm))/ (np*np*np);
307 tp = tp1 + tp2 + tp3 + tp4 + tp5;
308 tav = (ts + tp)/ (2 * sinvalue * sinvalue);
309 reflectivities.push_back(1.0 - tav);
310 }
311 return;
