20uint buildGenericLeafPrototype(helios::Context *context_ptr, uint subdivisions, const std::string &leaf_texture, float leaf_aspect_ratio, float midrib_fold_fraction, float longitudinal_curvature, float lateral_curvature, float petiole_roll, float wave_period,
21 float wave_amplitude, bool build_petiolule) {
22
23 // -- main leaf generation code -- //
24
25 std::vector<uint> UUIDs;
26
27 uint Nx = subdivisions; //number of leaf subdivisions in the x-direction (longitudinal)
28 uint Ny = ceil(leaf_aspect_ratio*float(Nx)); //number of leaf subdivisions in the y-direction (lateral)
29
30 if ( Ny % 2 != 0){ //Ny must be even
31 Ny = Ny + 1;
32 }
33
34 float dx = 1.f/float(Nx); //length of leaf subdivision in the x-direction
35 float dy = leaf_aspect_ratio/float(Ny); //length of leaf subdivision in the y-direction
36
37 for (int i = 0; i < Nx; i++) {
38 for (int j = 0; j < Ny; j++) {
39
40 float x = float(i) * dx; //x-coordinate of leaf subdivision
41 float y = float(j) * dy - 0.5f * leaf_aspect_ratio; //y-coordinate of leaf subdivision
42
43 //midrib leaf folding
44 float y_fold = cosf(0.5f*midrib_fold_fraction*M_PI)*y;
45 float y_fold_jplus = cosf(0.5f*midrib_fold_fraction*M_PI)*(y+dy);
46 float z_fold = sinf(0.5f*midrib_fold_fraction*M_PI)*fabs(y);
47 float z_fold_jplus = sinf(0.5f*midrib_fold_fraction*M_PI)*fabs(y+dy);
48
49 //x-curvature
50 float z_xcurve = longitudinal_curvature * powf(x, 4);
51 float z_xcurve_iplus = longitudinal_curvature * powf(x + dx, 4);
52
53 //y-curvature
54 float z_ycurve = lateral_curvature * powf(y / leaf_aspect_ratio, 4);
55 float z_ycurve_jplus = lateral_curvature * powf((y + dy) / leaf_aspect_ratio, 4);
56
57 //petiole roll
58 float z_petiole_0 = 0, z_petiole_1 = 0, z_petiole_2 = 0, z_petiole_3 = 0;
59 if( petiole_roll != 0.0f ) {
60 z_petiole_0 = fmin(0.1f, petiole_roll * powf(7.f * y / leaf_aspect_ratio, 4) * exp(-70.f * (x))) - 0.01 * petiole_roll / fabs(petiole_roll);
61 z_petiole_1 = fmin(0.1f, petiole_roll * powf(7.f * y / leaf_aspect_ratio, 4) * exp(-70.f * (x + dx))) - 0.01 * petiole_roll / fabs(petiole_roll);
62 z_petiole_2 = fmin(0.1f, petiole_roll * powf(7.f * (y + dy) / leaf_aspect_ratio, 4) * exp(-70.f * (x + dx))) - 0.01 * petiole_roll / fabs(petiole_roll);
63 z_petiole_3 = fmin(0.1f, petiole_roll * powf(7.f * (y + dy) / leaf_aspect_ratio, 4) * exp(-70.f * (x))) - 0.01 * petiole_roll / fabs(petiole_roll);
64 }
65
66 //vertical displacement for leaf wave at each of the four subdivision vertices
67 float z_wave_0 = 0, z_wave_1 = 0, z_wave_2 = 0, z_wave_3 = 0;
68 if( wave_period > 0.0f && wave_amplitude > 0.0f ) {
69 z_wave_0 = (2.f * fabs(y) * wave_amplitude * sinf((x + 0.5f * float(j >= 0.5 * Ny)) * M_PI / wave_period));
70 z_wave_1 = (2.f * fabs(y) * wave_amplitude * sinf((x + 0.5f * float(j >= 0.5 * Ny) + dx) * M_PI / wave_period));
71 z_wave_2 = (2.f * fabs(y + dy) * wave_amplitude * sinf((x + 0.5f * float(j >= 0.5 * Ny) + dx) * M_PI / wave_period));
72 z_wave_3 = (2.f * fabs(y + dy) * wave_amplitude * sinf((x + 0.5f * float(j >= 0.5 * Ny)) * M_PI / wave_period));
73 }
74
75 //define the four vertices of the leaf subdivision
76 vec3 v0(x, y_fold, z_fold+z_xcurve+z_ycurve+z_wave_0+z_petiole_0);
77
78 vec3 v1(x + dx, y_fold, z_fold+z_xcurve_iplus+z_ycurve+z_wave_1+z_petiole_1);
79
80 vec3 v2(x + dx, y_fold_jplus, z_fold_jplus+z_xcurve_iplus+z_ycurve_jplus+z_wave_2+z_petiole_2);
81
82 vec3 v3(x, y_fold_jplus, z_fold_jplus+z_xcurve+z_ycurve_jplus+z_wave_3+z_petiole_3);
83
84 vec2 uv0(x, (y+0.5f*leaf_aspect_ratio) / leaf_aspect_ratio);
85 vec2 uv1(x + dx, (y+0.5f*leaf_aspect_ratio) / leaf_aspect_ratio);
86 vec2 uv2(x + dx, (y + dy + 0.5f*leaf_aspect_ratio) / leaf_aspect_ratio);
87 vec2 uv3(x, (y + dy + 0.5f*leaf_aspect_ratio) / leaf_aspect_ratio);
88
89 UUIDs.push_back(context_ptr->addTriangle(v0, v1, v2, leaf_texture.c_str(), uv0, uv1, uv2));
90 UUIDs.push_back(context_ptr->addTriangle(v0, v2, v3, leaf_texture.c_str(), uv0, uv2, uv3));
91
92 }
93 }
94
95 if( build_petiolule ){
96 std::vector<uint> UUIDs_petiolule = context_ptr->loadOBJ("plugins/plantarchitecture/assets/obj/PetiolulePrototype.obj", make_vec3(0,0,0), 0, nullrotation, RGB::black, "ZUP", true);
97 context_ptr->translatePrimitive(UUIDs, make_vec3(0.07,0,0.005) );
98 UUIDs.insert(UUIDs.end(), UUIDs_petiolule.begin(), UUIDs_petiolule.end());
99 }
100
101 return context_ptr->addPolymeshObject( UUIDs );
102}

540uint SorghumPaniclePrototype( helios::Context* context_ptr, uint subdivisions, float time_since_fruit_set ){
541
542 if( subdivisions<=1 ){
543 subdivisions = 3;
544 }
545
546 float panicle_height = 1;
547 float panicle_width = 0.08;
548 float width_seed = 0.08;
549 float height_seed = 0.25;
550 float seed_tilt = 50;
551 subdivisions = 6;
552
553 std::string seed_texture_file = "plugins/plantarchitecture/assets/textures/SorghumSeed.jpeg";
554 RGBcolor stem_color(0.45,0.55,0.42);
555
556 std::vector<uint> UUIDs;
557
558 panicle_height -= 0.8*height_seed;
559
560 std::vector<vec3> nodes_panicle;
561 std::vector<float> radius_panicle;
562
563 for (int n = 0; n < subdivisions ; n++) {
564 float x = 0;
565 float y = 0;
566 float z;
567 if( n==0 ){
568 z = 0.5f*height_seed/float(subdivisions-1);
569 }else if( n==subdivisions-1 ){
570 z = (subdivisions-1.5f)*height_seed/float(subdivisions-1);
571 }else{
572 z = n*height_seed/float(subdivisions-1);
573 }
574
575 float angle = n * M_PI /float(subdivisions-1);
576 float dr = 0.5f * width_seed * sin(angle);
577
578 nodes_panicle.push_back(make_vec3(x, y, z));
579 radius_panicle.push_back(dr);
580
581 }
582
583 std::vector<uint> UUIDs_seed_ptype = context_ptr->addTube( subdivisions, nodes_panicle, radius_panicle,seed_texture_file.c_str());
584
585 int Ntheta = ceil( 6.f*panicle_height/height_seed );
586 int Nphi = ceil( 2.f*M_PI*panicle_width/width_seed);
587
588 for(int j=0; j < Nphi; j++ ) {
589 for (int i = 0; i < Ntheta; i++) {
590
591 if( i==0 && j==0 ){
592 continue;
593 }
594
595 std::vector<uint> UUIDs_copy = context_ptr->copyPrimitive(UUIDs_seed_ptype);
596 context_ptr->scalePrimitive( UUIDs_copy, make_vec3(1,1,1)*context_ptr->randu(0.9f,1.1f));
597
598 float phi = 2.f * M_PI * float(j + 0.5f*float(i%2)) / float(Nphi);
599 float theta = acos(1 - 2 * float(i + float(j)/float(Nphi)) / float(Ntheta));
600 float x = sin(theta) * cos(phi);
601 float y = sin(theta) * sin(phi);
602 float z = 0.5f + 0.5f*cos(theta);
603
604 x *= 0.5f * panicle_width;
605 y *= 0.5f * panicle_width;
606 z *= panicle_height;
607
608 float tilt = - deg2rad(seed_tilt) * sqrtf(1.f - z / panicle_height);
609
610 context_ptr->rotatePrimitive(UUIDs_copy, tilt, "x");
611 context_ptr->rotatePrimitive(UUIDs_copy, phi - 0.5f * M_PI, "z");
612
613 context_ptr->translatePrimitive(UUIDs_copy, make_vec3(x, y, z));
614 UUIDs.insert(UUIDs.end(), UUIDs_copy.begin(), UUIDs_copy.end());
615
616 }
617 }
618
619 context_ptr->deletePrimitive(UUIDs_seed_ptype);
620
621 std::vector<uint> UUIDs_sphere = context_ptr->addSphere( 10, make_vec3(0,0,0.5*panicle_height), 0.5f, seed_texture_file.c_str() );
622 context_ptr->scalePrimitiveAboutPoint( UUIDs_sphere, make_vec3(1.9*panicle_width, 1.9*panicle_width, 0.8*panicle_height), make_vec3(0,0,0.5*panicle_height));
623 UUIDs.insert(UUIDs.end(), UUIDs_sphere.begin(), UUIDs_sphere.end());
624
625 context_ptr->rotatePrimitive(UUIDs, 0.5f*M_PI, "y");
626 context_ptr->translatePrimitive( UUIDs, make_vec3(-0.2,0,0));
627
628 uint objID = context_ptr->addPolymeshObject( UUIDs );
629 return objID;
630
631}
