127uint SorghumLeafPrototype( helios::Context* context_ptr, uint subdivisions, int flag ){
128
129 std::vector<uint> UUIDs;
130
131 float leaf_aspect = 0.1; //ratio of leaf width to leaf length
132
133 //parameters for leaf wave/wrinkles
134 float wave_period = 0.1f; //period factor of leaf waves
135 float wave_amplitude = 0.015f; // amplitude of leaf waves
136
137 //parameters for leaf curvature
138 float leaf_amplitude = 0.25f; //amplitude of leaf curvature
139
140 int Nx = 50; //number of leaf subdivisions in the x-direction (longitudinal)
141 int Ny = ceil(leaf_aspect*float(Nx)); //number of leaf subdivisions in the y-direction (lateral)
142
143 if ( Ny % 2 != 0){ //Ny must be even
144 Ny = Ny + 1;
145 }
146
147 float dx = 1.f/float(Nx); //length of leaf subdivision in the x-direction
148 float dy = leaf_aspect/float(Ny); //length of leaf subdivision in the y-direction
149
150 for (int i = 0; i < Nx; i++) {
151 for (int j = 0; j < Ny; j++) {
152
153 float y_frac, y_frac_plus;
154 if (float(j) / float(Ny) >= 0.5) {
155 y_frac = (1 - float(j) / float(Ny)) * 2;
156 y_frac_plus = y_frac - 2.f / float(Ny);
157 } else {
158 y_frac = float(j) / float(Ny) * 2;
159 y_frac_plus = y_frac + 2.f / float(Ny);
160 }
161 y_frac = 1.f - y_frac;
162 y_frac_plus = 1.f - y_frac_plus;
163
164 float x = float(i) * dx; //x-coordinate of leaf subdivision
165 float y = float(j) * dy; //y-coordinate of leaf subdivision
166 float z = 0;
167
168 //vertical displacement for leaf wave at each of the four subdivision vertices
169 float z_i = x * M_PI / wave_period;
170 float z_iplus = (x + dx) * M_PI / wave_period;
171
172 float leaf_wave_zoffset_0 = (wave_amplitude * sinf(z_i) + 0.01f) * y_frac;
173 float leaf_wave_zoffset_1 = (wave_amplitude * sinf(z_iplus) + 0.01f) * y_frac;
174 float leaf_wave_zoffset_2 = (wave_amplitude * sinf(z_iplus) + 0.01f) * y_frac_plus;
175 float leaf_wave_zoffset_3 = (wave_amplitude * sinf(z_i) + 0.01f) * y_frac_plus;
176
177 //vertical displacement due to leaf curvature at each of the four subdivision vertices
178 float x_i = x * M_PI;
179 float x_iplus = (x + dx) * M_PI;
180
181 float leaf_curvature_zoffset_0 = leaf_amplitude * sin(x_i);
182 float leaf_curvature_zoffset_1 = leaf_amplitude * sin(x_iplus);
183 float leaf_curvature_zoffset_2 = leaf_amplitude * sin(x_iplus);
184 float leaf_curvature_zoffset_3 = leaf_amplitude * sin(x_i);
185
186 //define the four vertices of the leaf subdivision
187 z = leaf_curvature_zoffset_0 + leaf_wave_zoffset_0;
188 vec3 v0(x, y - 0.5f * leaf_aspect, z);
189
190 z = leaf_curvature_zoffset_1 + leaf_wave_zoffset_1;
191 vec3 v1(x + dx, y - 0.5f * leaf_aspect, z);
192
193 z = leaf_curvature_zoffset_2 + leaf_wave_zoffset_2;
194 vec3 v2(x + dx, y + dy - 0.5f * leaf_aspect, z);
195
196 z = leaf_curvature_zoffset_3 + leaf_wave_zoffset_3;
197 vec3 v3(x, y + dy - 0.5f * leaf_aspect, z);
198
199 vec2 uv0(x, y / leaf_aspect);
200 vec2 uv1(x + dx, y / leaf_aspect);
201 vec2 uv2(x + dx, (y + dy) / leaf_aspect);
202 vec2 uv3(x, (y + dy) / leaf_aspect);
203
204 UUIDs.push_back(context_ptr->addTriangle(v0, v1, v2, "plugins/plantarchitecture/assets/textures/SorghumLeaf.png", uv0, uv1, uv2));
205 UUIDs.push_back(context_ptr->addTriangle(v0, v2, v3, "plugins/plantarchitecture/assets/textures/SorghumLeaf.png", uv0, uv2, uv3));
206
207 }
208 }
209
210 return context_ptr->addPolymeshObject( UUIDs );
211}
