418float random_float(float min, float max) {
419 return min + static_cast<float>(rand()) / (static_cast<float>(RAND_MAX/(max-min)));

422// Function to check if two spheres overlap
423bool spheres_overlap(const helios::vec3& center1, float radius1, const helios::vec3& center2, float radius2) {
424 float distance = std::sqrt(
425 std::pow(center1.x - center2.x, 2) +
426 std::pow(center1.y - center2.y, 2) +
427 std::pow(center1.z - center2.z, 2)
428 );
429 return distance < (radius1 + radius2);
430}

432uint GrapevineFruitPrototype( helios::Context* context_ptr, uint subdivisions ){
433
434 int num_grapes = 60;
435 float height = 6.0f; // Height of the cluster
436 float base_radius = 2.f; // Base radius of the cluster
437 float taper_factor = 0.6f; // Taper factor (higher means more taper)
438 float grape_radius = 0.25f; // Fixed radius for each grape
439
440 std::vector<std::pair<helios::vec3, float>> grapes;
441 float z_step = height / num_grapes;
442
443 // Place the first grape at the bottom center
444 helios::vec3 first_center(0.0f, 0.0f, 0.0f);
445 grapes.push_back({first_center, grape_radius});
446
447 // Attempt to place each subsequent grape close to an existing grape
448 int max_attempts = 100; // Number of retries to find a tight fit
449
450 for (int i = 1; i < num_grapes; ++i) {
451 float z = i * z_step;
452 // Tapered radius based on height (denser at the top, sparser at the bottom)
453 float taper_radius = base_radius * (1.0f - taper_factor * (z / height));
454
455 bool placed = false;
456 int attempts = 0;
457 while (!placed && attempts < max_attempts) {
458 // Randomly select an existing grape as the reference point
459 int reference_idx = rand() % grapes.size();
460 const helios::vec3& reference_center = grapes[reference_idx].first;
461
462 // Pick a random offset direction from the reference grape
463 float angle = random_float(0, 2 * M_PI);
464 float distance = random_float(1.2 * grape_radius, 1.3 * grape_radius); // Keep grapes close but not overlapping
465
466 // Compute the new potential center for the grape
467 helios::vec3 new_center(
468 reference_center.x + distance * cos(angle),
469 reference_center.y + distance * sin(angle),
470 random_float(z - 0.5f * z_step, z + 0.5f * z_step)
471 );
472
473 // Check that the new center is within the allowable radius (for tapering)
474 float new_center_distance = std::sqrt(new_center.x * new_center.x + new_center.y * new_center.y);
475 if (new_center_distance > taper_radius) {
476 attempts++;
477 continue; // Skip if the new position exceeds the tapered radius
478 }
479
480 // Check for collisions with existing grapes
481 bool collision = false;
482 for (const auto& grape : grapes) {
483 if (spheres_overlap(new_center, grape_radius, grape.first, grape.second)) {
484 collision = true;
485 break;
486 }
487 }
488
489 // If no collision, place the grape
490 if (!collision) {
491 grapes.push_back({new_center, grape_radius});
492 placed = true;
493 }
494
495 attempts++;
496 }
497
498 }
499
500 std::vector<uint> UUIDs;

722uint SorghumPaniclePrototype( helios::Context* context_ptr, uint subdivisions ){
723
724 if( subdivisions<=1 ){
725 subdivisions = 3;
726 }
727
728 float panicle_height = 1;
729 float panicle_width = 0.08;
730 float width_seed = 0.08;
731 float height_seed = 0.25;
732 float seed_tilt = 50;
733 subdivisions = 6;
734
735 std::string seed_texture_file = "plugins/plantarchitecture/assets/textures/SorghumSeed.jpeg";
736 RGBcolor stem_color(0.45,0.55,0.42);
737
738 std::vector<uint> UUIDs;
739
740 panicle_height -= 0.8*height_seed;
741
742 std::vector<vec3> nodes_panicle;
743 std::vector<float> radius_panicle;
744
745 for (int n = 0; n < subdivisions ; n++) {
746 float x = 0;
747 float y = 0;
748 float z;
749 if( n==0 ){
750 z = 0.5f*height_seed/float(subdivisions-1);
751 }else if( n==subdivisions-1 ){
752 z = (subdivisions-1.5f)*height_seed/float(subdivisions-1);
753 }else{
754 z = n*height_seed/float(subdivisions-1);
755 }
756
757 float angle = float(n) * M_PI /float(subdivisions-1);
758 float dr = std::fmax(0.f, 0.5f * width_seed * sin(angle));
759
760 nodes_panicle.push_back(make_vec3(x, y, z));
761 radius_panicle.push_back(dr);
762
763 }
764
765 std::vector<uint> UUIDs_seed_ptype = context_ptr->addTube( subdivisions, nodes_panicle, radius_panicle,seed_texture_file.c_str());
766
767 int Ntheta = ceil( 6.f*panicle_height/height_seed );
768 int Nphi = ceil( 2.f*M_PI*panicle_width/width_seed);
769
770 for(int j=0; j < Nphi; j++ ) {
771 for (int i = 0; i < Ntheta; i++) {
772
773 if( i==0 && j==0 ){
774 continue;
775 }
776
777 std::vector<uint> UUIDs_copy = context_ptr->copyPrimitive(UUIDs_seed_ptype);
778 context_ptr->scalePrimitive( UUIDs_copy, make_vec3(1,1,1)*context_ptr->randu(0.9f,1.1f));
779
780 float phi = 2.f * M_PI * float(j + 0.5f*float(i%2)) / float(Nphi);
781 float theta = acos(1 - 2 * float(i + float(j)/float(Nphi)) / float(Ntheta));
782 float x = sin(theta) * cos(phi);
783 float y = sin(theta) * sin(phi);
784 float z = 0.5f + 0.5f*cos(theta);
785
786 x *= 0.5f * panicle_width;
787 y *= 0.5f * panicle_width;
788 z *= panicle_height;
789
790 float tilt = - deg2rad(seed_tilt) * sqrtf(1.f - z / panicle_height);
791
792 context_ptr->rotatePrimitive(UUIDs_copy, tilt, "x");
793 context_ptr->rotatePrimitive(UUIDs_copy, phi - 0.5f * M_PI, "z");
794
795 context_ptr->translatePrimitive(UUIDs_copy, make_vec3(x, y, z));
796 UUIDs.insert(UUIDs.end(), UUIDs_copy.begin(), UUIDs_copy.end());
797
798 }
799 }
800
801 context_ptr->deletePrimitive(UUIDs_seed_ptype);
802
803 std::vector<uint> UUIDs_sphere = context_ptr->addSphere( 10, make_vec3(0,0,0.5*panicle_height), 0.5f, seed_texture_file.c_str() );
804 context_ptr->scalePrimitiveAboutPoint( UUIDs_sphere, make_vec3(1.9*panicle_width, 1.9*panicle_width, 0.8*panicle_height), make_vec3(0,0,0.5*panicle_height));
805 UUIDs.insert(UUIDs.end(), UUIDs_sphere.begin(), UUIDs_sphere.end());
806
807 context_ptr->rotatePrimitive(UUIDs, 0.5f*M_PI, "y");
808 context_ptr->translatePrimitive( UUIDs, make_vec3(-0.2,0,0));
809
810 uint objID = context_ptr->addPolymeshObject( UUIDs );
811 return objID;
812
813}
