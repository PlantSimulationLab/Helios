164uint SorghumLeafPrototype( helios::Context* context_ptr, uint subdivisions, int flag ){
165
166 std::vector<uint> UUIDs;
167
168 float leaf_aspect = 0.1; //ratio of leaf width to leaf length
169
170 //parameters for leaf wave/wrinkles
171 float wave_period = 0.1f; //period factor of leaf waves
172 float wave_amplitude = 0.015f; // amplitude of leaf waves
173
174 //parameters for leaf curvature
175 float leaf_amplitude = 0.25f; //amplitude of leaf curvature
176
177 int Nx = 50; //number of leaf subdivisions in the x-direction (longitudinal)
178 int Ny = ceil(leaf_aspect*float(Nx)); //number of leaf subdivisions in the y-direction (lateral)
179
180 if ( Ny % 2 != 0){ //Ny must be even
181 Ny = Ny + 1;
182 }
183
184 float dx = 1.f/float(Nx); //length of leaf subdivision in the x-direction
185 float dy = leaf_aspect/float(Ny); //length of leaf subdivision in the y-direction
186
187 for (int i = 0; i < Nx; i++) {
188 for (int j = 0; j < Ny; j++) {
189
190 float y_frac, y_frac_plus;
191 if (float(j) / float(Ny) >= 0.5) {
192 y_frac = (1 - float(j) / float(Ny)) * 2;
193 y_frac_plus = y_frac - 2.f / float(Ny);
194 } else {
195 y_frac = float(j) / float(Ny) * 2;
196 y_frac_plus = y_frac + 2.f / float(Ny);
197 }
198 y_frac = 1.f - y_frac;
199 y_frac_plus = 1.f - y_frac_plus;
200
201 float x = float(i) * dx; //x-coordinate of leaf subdivision
202 float y = float(j) * dy; //y-coordinate of leaf subdivision
203 float z = 0;
204
205 //vertical displacement for leaf wave at each of the four subdivision vertices
206 float z_i = x * M_PI / wave_period;
207 float z_iplus = (x + dx) * M_PI / wave_period;
208
209 float leaf_wave_zoffset_0 = (wave_amplitude * sinf(z_i) + 0.01f) * y_frac;
210 float leaf_wave_zoffset_1 = (wave_amplitude * sinf(z_iplus) + 0.01f) * y_frac;
211 float leaf_wave_zoffset_2 = (wave_amplitude * sinf(z_iplus) + 0.01f) * y_frac_plus;
212 float leaf_wave_zoffset_3 = (wave_amplitude * sinf(z_i) + 0.01f) * y_frac_plus;
213
214 //vertical displacement due to leaf curvature at each of the four subdivision vertices
215 float x_i = x * M_PI;
216 float x_iplus = (x + dx) * M_PI;
217
218 float leaf_curvature_zoffset_0 = leaf_amplitude * sin(x_i);
219 float leaf_curvature_zoffset_1 = leaf_amplitude * sin(x_iplus);
220 float leaf_curvature_zoffset_2 = leaf_amplitude * sin(x_iplus);
221 float leaf_curvature_zoffset_3 = leaf_amplitude * sin(x_i);
222
223 //define the four vertices of the leaf subdivision
224 z = leaf_curvature_zoffset_0 + leaf_wave_zoffset_0;
225 vec3 v0(x, y - 0.5f * leaf_aspect, z);
226
227 z = leaf_curvature_zoffset_1 + leaf_wave_zoffset_1;
228 vec3 v1(x + dx, y - 0.5f * leaf_aspect, z);
229
230 z = leaf_curvature_zoffset_2 + leaf_wave_zoffset_2;
231 vec3 v2(x + dx, y + dy - 0.5f * leaf_aspect, z);
232
233 z = leaf_curvature_zoffset_3 + leaf_wave_zoffset_3;
234 vec3 v3(x, y + dy - 0.5f * leaf_aspect, z);
235
236 vec2 uv0(x, y / leaf_aspect);
237 vec2 uv1(x + dx, y / leaf_aspect);
238 vec2 uv2(x + dx, (y + dy) / leaf_aspect);
239 vec2 uv3(x, (y + dy) / leaf_aspect);
240
241 UUIDs.push_back(context_ptr->addTriangle(v0, v1, v2, "plugins/plantarchitecture/assets/textures/SorghumLeaf.png", uv0, uv1, uv2));
242 UUIDs.push_back(context_ptr->addTriangle(v0, v2, v3, "plugins/plantarchitecture/assets/textures/SorghumLeaf.png", uv0, uv2, uv3));
243
244 }
245 }
246
247 return context_ptr->addPolymeshObject( UUIDs );
248}
