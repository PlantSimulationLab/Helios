1081void Visualizer::updatePerspectiveTransformation(bool shadow) {
1082 float dist = glm::distance(glm_vec3(camera_lookat_center), glm_vec3(camera_eye_location));
1083 float nearPlane = std::max(0.1f, 0.05f * dist); // avoid 0
1084 if (shadow) {
1085 float farPlane = std::max(5.f * camera_eye_location.z, 2.0f * dist);
1086 cameraProjectionMatrix = glm::perspective(glm::radians(camera_FOV), float(Wframebuffer) / float(Hframebuffer), nearPlane, farPlane);
1087 } else {
1088 cameraProjectionMatrix = glm::infinitePerspective(glm::radians(camera_FOV), float(Wframebuffer) / float(Hframebuffer), nearPlane);
1089 }
1090 cameraViewMatrix = glm::lookAt(glm_vec3(camera_eye_location), glm_vec3(camera_lookat_center), glm::vec3(0, 0, 1));
1091
1092 perspectiveTransformationMatrix = cameraProjectionMatrix * cameraViewMatrix;
1093}

1526std::vector<size_t> Visualizer::addSphereByCenter(float radius, const vec3 &center, uint Ndivisions, const RGBAcolor &color, CoordinateSystem coordinate_system) {
1527 float dtheta = PI_F / scast<float>(Ndivisions);
1528 float dphi = 2.f * PI_F / scast<float>(Ndivisions);
1529
1530 std::vector<size_t> UUIDs;
1531 UUIDs.reserve(2 * Ndivisions + 2 * (Ndivisions - 2) * (Ndivisions - 1));
1532
1533 //bottom cap
1534 for (int j = 0; j < Ndivisions; j++) {
1535 float phi = scast<float>(j) * dphi;
1536 float phi_plus = scast<float>(j + 1) * dphi;
1537
1538 vec3 v0 = center + sphere2cart(make_SphericalCoord(radius, -0.5f * PI_F, 0));
1539 vec3 v1 = center + sphere2cart(make_SphericalCoord(radius, -0.5f * PI_F + dtheta, phi));
1540 vec3 v2 = center + sphere2cart(make_SphericalCoord(radius, -0.5f * PI_F + dtheta, phi_plus));
1541
1542 UUIDs.push_back(addTriangle(v0, v1, v2, color, coordinate_system));
1543 }
1544
1545 //top cap
1546 for (int j = 0; j < Ndivisions; j++) {
1547 float phi = scast<float>(j) * dphi;
1548 float phi_plus = scast<float>(j + 1) * dphi;

2647glm::mat4 Visualizer::computeShadowDepthMVP() const {
2648 glm::vec3 lightDir = -glm::normalize(glm_vec3(light_direction));
2649
2650 const float margin = 0.01;
2651
2652 // Get the eight corners of the camera frustum in world space (NDC cube corners → clip → view → world)
2653
2654 // NDC cube
2655 static const std::array<glm::vec4, 8> ndcCorners =
2656 {
2657 glm::vec4(-1, -1, -1, 1), glm::vec4(+1, -1, -1, 1),
2658 glm::vec4(+1, +1, -1, 1), glm::vec4(-1, +1, -1, 1),
2659 glm::vec4(-1, -1, +1, 1), glm::vec4(+1, -1, +1, 1),
2660 glm::vec4(+1, +1, +1, 1), glm::vec4(-1, +1, +1, 1)
2661 };
2662
2663 glm::mat4 invCam = glm::inverse(this->perspectiveTransformationMatrix);
2664
2665 std::array<glm::vec3, 8> frustumWs;
2666 for (std::size_t i = 0; i < 8; i++) {
2667 glm::vec4 ws = invCam * ndcCorners[i];
2668 frustumWs[i] = glm::vec3(ws) / ws.w;
2669 }
2670
2671 // Build a light-view matrix (orthographic, directional light) We choose an arbitrary but stable "up" vector.
2672 glm::vec3 lightUp(0.0f, 1.0f, 0.0f);
2673 if (glm::abs(glm::dot(lightUp, lightDir)) > 0.9f) // almost collinear
2674 lightUp = glm::vec3(1, 0, 0);
2675
2676 // Position the "camera" that generates the shadow map so that every
2677 // frustum corner is in front of it. We place it on the negative light
2678 // direction, centered on the frustum’s centroid.
2679 glm::vec3 centroid(0);
2680 for (auto &c: frustumWs) centroid += c;
2681 centroid /= 8.0f;
2682
2683 glm::vec3 lightPos = centroid - lightDir * 100.0f; // 100 is arbitrary,
2684 // we will tighten z
2685 glm::mat4 lightView = glm::lookAt(lightPos, centroid, lightUp);
2686
2687 // Transform frustum corners to light space and find min/max extents
2688 glm::vec3 minL(std::numeric_limits<float>::infinity());
2689 glm::vec3 maxL(-std::numeric_limits<float>::infinity());
2690
2691 for (auto &c: frustumWs) {
2692 glm::vec3 p = glm::vec3(lightView * glm::vec4(c, 1));
2693 minL = glm::min(minL, p);
2694 maxL = glm::max(maxL, p);
2695 }
2696
2697 // Build orthographic projection that exactly fits the camera frustum and enlarge slightly to avoid clipping due to kernel offsets.
2698 glm::vec3 extent = maxL - minL;
2699 minL -= extent * margin;
2700 maxL += extent * margin;
2701
2702 float zNear = -maxL.z; // light space points toward -z
2703 float zFar = -minL.z;
2704
2705 glm::mat4 lightProj = glm::ortho(minL.x, maxL.x, minL.y, maxL.y, zNear, zFar);
2706
2707 // Transform into [0,1] texture space (bias matrix)
2708 const glm::mat4 bias(
2709 0.5f, 0.0f, 0.0f, 0.0f,
2710 0.0f, 0.5f, 0.0f, 0.0f,
2711 0.0f, 0.0f, 0.5f, 0.0f,
2712 0.5f, 0.5f, 0.5f, 1.0f
2713 );
2714
2715 return bias * lightProj * lightView;
2716}

3566 for (int y = 0; y < new_height; ++y) {
3567 float srcY = y * y_ratio;
3568 int y0 = std::min(scast<int>(std::floor(srcY)), old_height - 1);
3569 int y1 = std::min(y0 + 1, old_height - 1);
3570 float dy = srcY - y0;
3571
3572 for (int x = 0; x < new_width; ++x) {
3573 float srcX = x * x_ratio;
3574 int x0 = std::min(scast<int>(std::floor(srcX)), old_width - 1);
3575 int x1 = std::min(x0 + 1, old_width - 1);
3576 float dx = srcX - x0;
3577
3578 // for each channel, fetch 4 neighbors and lerp
3579 for (int c = 0; c < num_channels; ++c) {
3580 float p00 = texture_data[(y0 * old_width + x0) * num_channels + c];
3581 float p10 = texture_data[(y0 * old_width + x1) * num_channels + c];
3582 float p01 = texture_data[(y1 * old_width + x0) * num_channels + c];
3583 float p11 = texture_data[(y1 * old_width + x1) * num_channels + c];
3584
3585 float top = p00 * (1.f - dx) + p10 * dx;
3586 float bottom = p01 * (1.f - dx) + p11 * dx;
3587 float value = top * (1.f - dy) + bottom * dy;
3588
3589 new_data[(y * new_width + x) * num_channels + c] =
3590 scast<unsigned char>(clamp(std::round(value + 0.5f), 0.f, 255.f));
3591 }
3592 }
3593 }
3594
3595 texture_data = std::move(new_data);
3596 texture_resolution = new_image_resolution;
3597}
