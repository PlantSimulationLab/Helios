834 ybounds.y = verts.y;
835 }

837 zbounds.x = verts.z;

1124 for(auto vertex : vertices){
1125 if(vertex.x < 0.f || vertex.x > 1.f ){
1126 if( message_flag ){

1129 }else if(vertex.y < 0.f || vertex.y > 1.f ){
1130 if( message_flag ){

1132 }
1133 }else if(vertex.z < -1.f || vertex.z > 1.f ){
1134 if( message_flag ){
1135 std::cout << "WARNING: Rectangle `z' position ( " << vertex.z << " ) is outside of drawable area." << std::endl;
1136 }
1137 }
1138 }
1139
1140 //NOTE for vertex positions: OpenGL window coordinates range from -1 to 1, but our rectangle coordinates are from 0 to 1 ---- need to convert
1141 for( uint i=0; i<vertices.size(); i++ ){
1142 v.at(i).x = 2.f*v.at(i).x - 1.f;
1143 v.at(i).y = 2.f*v.at(i).y - 1.f;
1144 }
