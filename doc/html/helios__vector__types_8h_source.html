1272
1273 int month = -1;
1274
1275 int skips_leap[] = {0, 31, 60, 91, 121, 152, 182, 213, 244, 274, 305, 335};
1276 int skips_nonleap[] = {0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334};
1277 const int* skips = nullptr;
1278
1279 if( JulianDay<1 || JulianDay>366 ){
1280 throw(std::runtime_error("ERROR (Julian2Calendar): Julian day of " + std::to_string(JulianDay) + " is out of range (should be >0 and <367)."));
1281 }
1282
1283 if( (year-2000)%4 == 0 ){ //leap year
1284 skips=skips_leap;
1285 }else{ //non-leap year
1286 skips=skips_nonleap;
1287 }
1288
1289 for( int i=1; i<=12; i++ ){
1290 if( i==12 ){
1291 month = 12;
1292 }else if( JulianDay>skips[i-1] && JulianDay<=skips[i] ){
1293 month = i;
1294 break;
1295 }
1296 }
1297 assert( month>0 && month<13 );
