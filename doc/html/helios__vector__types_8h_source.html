1700
1701 int month = -1;
1702
1703 int skips_leap[] = {0, 31, 60, 91, 121, 152, 182, 213, 244, 274, 305, 335};
1704 int skips_nonleap[] = {0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334};
1705 const int *skips = nullptr;
1706
1707 if (JulianDay < 1 || JulianDay > 366) {
1708 throw(std::runtime_error("ERROR (Julian2Calendar): Julian day of " + std::to_string(JulianDay) + " is out of range (should be >0 and <367)."));
1709 }
1710
1711 if ((year - 2000) % 4 == 0) { // leap year
1712 skips = skips_leap;
1713 } else { // non-leap year
1714 skips = skips_nonleap;
1715 }
1716
1717 for (int i = 1; i <= 12; i++) {
1718 if (i == 12) {
1719 month = 12;
1720 } else if (JulianDay > skips[i - 1] && JulianDay <= skips[i]) {
1721 month = i;
1722 break;
1723 }
1724 }
1725 assert(month > 0 && month < 13);
1726
1727 int day = JulianDay - skips[month - 1];
1728
1729 assert(day > 0 && day < 32);
1730
1731 return {day, month, year};
1732 }
