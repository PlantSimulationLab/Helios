119__device__ float3 LIDAR_CUDA::d_rotatePoint(const float3 &position, const float &theta, const float &phi) {
120
121 float Ry[3][3], Rz[3][3];
122
123 float st = sinf(theta);
124 float ct = cosf(theta);
125
126 float sp = sinf(phi);
127 float cp = cosf(phi);
128
129 // Setup the rotation matrix, this matrix is based off of the rotation matrix used in glRotatef.
130 Ry[0][0] = ct;
131 Ry[0][1] = 0.f;
132 Ry[0][2] = st;
133 Ry[1][0] = 0.f;
134 Ry[1][1] = 1.f;
135 Ry[1][2] = 0.f;
136 Ry[2][0] = -st;
137 Ry[2][1] = 0.f;
138 Ry[2][2] = ct;
139
140 Rz[0][0] = cp;
141 Rz[0][1] = -sp;
142 Rz[0][2] = 0.f;
143 Rz[1][0] = sp;
144 Rz[1][1] = cp;
145 Rz[1][2] = 0.f;
146 Rz[2][0] = 0.f;
147 Rz[2][1] = 0.f;
148 Rz[2][2] = 1.f;
149
150 // Multiply Ry*Rz
151
152 float rotMat[3][3] = {0.f};
153
154 for( int i=0;i<3;i++){
155 for(int j=0;j<3;j++){
156 for(int k=0;k<3;k++){
157 rotMat[i][j]=rotMat[i][j]+Rz[i][k]*Ry[k][j];
158 }
159 }
160 }
161
162 // Multiply the rotation matrix with the position vector.
163 float3 tmp;
164 tmp.x = rotMat[0][0] * position.x + rotMat[0][1] * position.y + rotMat[0][2] * position.z;
165 tmp.y = rotMat[1][0] * position.x + rotMat[1][1] * position.y + rotMat[1][2] * position.z;
166 tmp.z = rotMat[2][0] * position.x + rotMat[2][1] * position.y + rotMat[2][2] * position.z;
167
168 return tmp;
169

228__device__ float2 LIDAR_CUDA::d_sampleDisk( float radius, uint seed ){
229
230 // use Box-Muller tranform approach
231 // the input radius is the place where the intensity decreases to 1/e^2 (~0.1353) of the intensity at the beam center
232
233 // sample from uniform random distribution [0,1] for radial distance and angle
234 float U1 = rnd(seed);
235 float U2 = rnd(seed);
236
237 // calculate the standard deviation of a normal distribution where intensity decreases to 1/e^2 at the input radius
238 float sd = sqrt(0.25*radius*radius);
239
240 // formula from https://www.alanzucconi.com/2015/09/16/how-to-sample-from-a-gaussian-distribution/
241 float R = sqrt(-2.f*logf(U1))*sd;
242 float angle = 2.f*M_PI*U2;
243
244 float2 sp;
245 //find x,y point on unit disk
246 sp.x = R*cosf(angle);
247 sp.y = R*sinf(angle);
248 return sp;

643 hit_table2D.at(column).push_back( hit_table_clean.at(r) );
644 //for small scans (like the rectangle test case, this needs to change to < 0 or some smaller angle (that is larger than noise))
645 // if( dtheta_clean.at(r) < 0 ){
646 // for normal scans, this threshold allows for 10 degrees drops in theta within a given sweep as noise. This can be adjusted as appropriate.
647 if( dtheta_clean.at(r) < -0.1745329f ){
648 column++;
649 hit_table2D.resize(column+1);
650 }
651
652 }
653
654 // calculate average dt and dtheta for subsequent points
655
656 //calculate average dt
657 float dt_avg = 0;
658 int dt_sum = 0;
659
660 //calculate the average dtheta to use for extrapolation
661 float dtheta_avg = 0;
662 int dtheta_sum = 0;
663
664 for( int j=0; j<hit_table2D.size(); j++ ){
665 for( int i=0; i<hit_table2D.at(j).size(); i++ ){
666 int r = int(hit_table2D.at(j).at(i).at(0));
667 if( dt_clean.at(r)>=dt_clean_min && dt_clean.at(r)<1.5*dt_clean_min ){
668 dt_avg += dt_clean.at(r);
669 dt_sum ++;
670
671 //calculate the average dtheta to use for extrapolation
672 dtheta_avg += dtheta_clean.at(r);
673 dtheta_sum ++;
674
675 }
676 }
677 }
678
679 dt_avg = dt_avg/float(dt_sum);
680 //Calculate the average dtheta to use for extrapolation
681 dtheta_avg = dtheta_avg/float(dtheta_sum);

684// std::cout << "dtheta_avg = " << dtheta_avg << std::endl;
685// std::cout << "hit_table2D.size() = "<< hit_table2D.size() << std::endl;

687 //identify gaps and fill
688 for( int j=0; j<hit_table2D.size(); j++ ){

691 {
692 for( int i=0; i<hit_table2D.at(j).size()-1; i++ ){

694 double dt = hit_table2D.at(j).at(i+1).at(1)-hit_table2D.at(j).at(i).at(1);
695
696 if( dt>1.5f*dt_clean_min ){ //missing hit(s)

698 //calculate number of missing hits
699 int Ngap = round(dt/dt_avg)-1;
700
701 //fill missing points
702 for( int k=1; k<=Ngap; k++ ){
703
704 float timestep = hit_table2D.at(j).at(i).at(1) + dt_avg*float(k);
705
706 //interpolate theta and phi
707 float theta = hit_table2D.at(j).at(i).at(2) + (hit_table2D.at(j).at(i+1).at(2)-hit_table2D.at(j).at(i).at(2))*float(k)/float(Ngap+1);
708 float phi = hit_table2D.at(j).at(i).at(3) + (hit_table2D.at(j).at(i+1).at(3)-hit_table2D.at(j).at(i).at(3))*float(k)/float(Ngap+1);
709 if( phi>2.f*M_PI ){
710 phi = phi-2.f*M_PI;
711 }
712
713 //calculate the (x,y,z) position of the filled hit point
714 helios::SphericalCoord spherical(gap_distance,0.5*M_PI-theta,phi);
715 helios::vec3 xyz = origin+helios::sphere2cart(spherical);
716
717 xyz_filled.push_back(xyz);
718
719 //ERK
720 std::map<std::string, double> data;
721 data.insert(std::pair<std::string, double>("timestamp", timestep));
722 data.insert(std::pair<std::string, double>("target_index", min_tindex));
723 data.insert(std::pair<std::string, double>("nRaysHit", 500));
724 if(add_flags)
725 {

727 data.insert(std::pair<std::string, double>("gapfillMisses_code", 1.0));
728 }
729 addHitPoint(scanID, xyz, spherical, data);
730
731 }
732 }
733 }
734 }
735
736 }
737 uint npointsfilled = xyz_filled.size();

741
742 if(gapfill_grid_only == true)
743 {

751 grid_vertices.push_back(helios::make_vec3(boxmin.x, boxmin.y, boxmax.z));
752 grid_vertices.push_back(helios::make_vec3(boxmax.x, boxmax.y, boxmin.z));
753 grid_vertices.push_back(helios::make_vec3(boxmin.x, boxmax.y, boxmin.z));
754 grid_vertices.push_back(helios::make_vec3(boxmin.x, boxmax.y, boxmax.z));
755 grid_vertices.push_back(helios::make_vec3(boxmax.x, boxmin.y, boxmin.z));
756 grid_vertices.push_back(helios::make_vec3(boxmax.x, boxmin.y, boxmax.z));

758 float max_theta = 0;
759 float min_theta = M_PI;
760 float max_phi = 0;
761 float min_phi = 2*M_PI;
762 for(uint gg=0;gg<grid_vertices.size();gg++)

766 if(sc.azimuth < min_phi)
767 {

769 }
770
771 if(sc.azimuth > max_phi)

774 }

779 }
780
781 if(sc.zenith > max_theta)

783 max_theta = sc.zenith;

796 max_theta = theta_range.y;

804 // extrapolate missing points
805 for( int j=0; j<hit_table2D.size(); j++ ){
806
807 if(hit_table2D.at(j).size() > 0)
808 {
809
810 //upward edge points
811 if( hit_table2D.at(j).front().at(2)>theta_range.x ){
812
813 float dtheta = dtheta_avg;
814 float theta = hit_table2D.at(j).at(0).at(2) - dtheta;
815 //just use the last value of phi in the sweep
816 float phi = hit_table2D.at(j).at(0).at(3);
817 float timestep = hit_table2D.at(j).at(0).at(1) - dt_avg;
818 if( dtheta==0 ){
819 continue;
820 }
821
822 while( theta>theta_range.x ){
823
824 helios::SphericalCoord spherical(gap_distance,0.5*M_PI-theta,phi);
825 helios::vec3 xyz = origin+helios::sphere2cart(spherical);
826
827 xyz_filled.push_back(xyz);
828
829 std::map<std::string, double> data;
830 data.insert(std::pair<std::string, double>("timestamp", timestep));
831 data.insert(std::pair<std::string, double>("target_index", min_tindex));
832 data.insert(std::pair<std::string, double>("nRaysHit", 500));
833 if(add_flags)
834 {
835 // gapfillMisses_code = 3: upward edge points
836 data.insert(std::pair<std::string, double>("gapfillMisses_code", 3.0));
837
838 }
839
840 addHitPoint(scanID, xyz, spherical, data);
841
842
843 theta = theta - dtheta;
844 timestep = timestep - dt_avg;
845
846
847 }
848 }
849
850 //downward edge points
851 if( hit_table2D.at(j).back().at(2)<theta_range.y ){
852
853 int sz = hit_table2D.at(j).size();
854 // same concept as above for downward edge points
855 float dtheta = dtheta_avg;
856 float theta = hit_table2D.at(j).at(sz-1).at(2) + dtheta;
857 float phi = hit_table2D.at(j).at(sz-1).at(3);
858 float timestep = hit_table2D.at(j).at(sz-1).at(1) + dt_avg;
859 while( theta<theta_range.y ){
860
861 helios::SphericalCoord spherical(gap_distance,0.5*M_PI-theta,phi);
862 helios::vec3 xyz = origin+helios::sphere2cart(spherical);
863
864 xyz_filled.push_back(xyz);
865
866 std::map<std::string, double> data;
867 data.insert(std::pair<std::string, double>("timestamp", timestep));
868 data.insert(std::pair<std::string, double>("target_index", min_tindex));
869 data.insert(std::pair<std::string, double>("nRaysHit", 500));
870 if(add_flags)
871 {
872 // gapfillMisses_code = 2: downward edge points
873 data.insert(std::pair<std::string, double>("gapfillMisses_code", 2.0));
874 }
875
876 addHitPoint(scanID, xyz, spherical, data);
877
878 theta = theta + dtheta;
879 timestep = timestep + dt_avg;
880
881 }
882 }
883 }
884
885 }
886
887 uint npointsextrapolated = xyz_filled.size() - npointsfilled;
888
889 if( printmessages ) {
890 std::cout << "filled " << xyz_filled.size() << " points." << std::endl;
891 }
892 return xyz_filled;
