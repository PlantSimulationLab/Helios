218uint PlantArchitecture::buildAlmondTree(const helios::vec3 &base_position, float age) {
219
220 if( shoot_types.empty() ){
221 //automatically initialize almond tree shoots
222 initializeAlmondTreeShoots();
223 }
224
225 uint plantID = addPlantInstance(base_position, age);
226
227 uint uID_trunk = addBaseStemShoot(plantID, 3, make_AxisRotation(context_ptr->randu(0.f, 0.025f * M_PI), context_ptr->randu(0.f, 2.f * M_PI), 0.f * M_PI), 0.01, 0.1, 1.f, 1.f, "trunk");
228
229 plant_instances.at(plantID).shoot_tree.at(uID_trunk)->meristem_is_alive = false;
230
231 auto phytomers = plant_instances.at(plantID).shoot_tree.at(uID_trunk)->phytomers;
232 for( const auto & phytomer : phytomers ){
233 phytomer->removeLeaf();
234 phytomer->setVegetativeBudState(BUD_DEAD);
235 phytomer->setFloralBudState(BUD_DEAD);
236 }
237
238 uint Nscaffolds = context_ptr->randu(4,5);
239
240 for( int i=0; i<Nscaffolds; i++ ) {
241 uint uID_shoot = appendShoot(plantID, uID_trunk, context_ptr->randu(6, 8), make_AxisRotation(context_ptr->randu(deg2rad(35), deg2rad(45)), (float(i) + context_ptr->randu(-0.1f, 0.1f)) / float(Nscaffolds) * 2 * M_PI, 0), 0.003, 0.02, 1.f, 1.f,
242 "scaffold");
243
244 plant_instances.at(plantID).shoot_tree.at(uID_shoot)->breakDormancy();
245
246 uint blind_nodes = context_ptr->randu(2,3);
247 for( int b=0; b<blind_nodes; b++){
248 if( b<plant_instances.at(plantID).shoot_tree.at(uID_shoot)->phytomers.size() ) {
249 plant_instances.at(plantID).shoot_tree.at(uID_shoot)->phytomers.at(b)->removeLeaf();
250 plant_instances.at(plantID).shoot_tree.at(uID_shoot)->phytomers.at(b)->setFloralBudState(BUD_DEAD);
251 plant_instances.at(plantID).shoot_tree.at(uID_shoot)->phytomers.at(b)->setVegetativeBudState(BUD_DEAD);
252 }
253 }
254
255 }
256
257 setPlantPhenologicalThresholds(plantID, 1, 1, 0, 3, 7, 12);
258
259 return plantID;
