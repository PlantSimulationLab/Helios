182uint PlantArchitecture::buildAlmondTree(const helios::vec3 &base_position, float age) {
183
184 if( shoot_types.empty() ){
185 //automatically initialize almond tree shoots
186 initializeAlmondTreeShoots();
187 }
188
189 uint plantID = addPlantInstance(base_position, age);
190
191 uint uID_trunk = addBaseStemShoot(plantID, 3, 0.1, make_AxisRotation(context_ptr->randu(0.f, 0.05f * M_PI), context_ptr->randu(0.f, 2.f * M_PI), 0.f * M_PI), "trunk");
192
193 plant_instances.at(plantID).shoot_tree.at(uID_trunk)->meristem_is_alive = false;
194
195 auto phytomers = plant_instances.at(plantID).shoot_tree.at(uID_trunk)->phytomers;
196 for( const auto & phytomer : phytomers ){
197 phytomer->removeLeaf();
198 phytomer->flower_bud_state = BUD_DEAD;
199 phytomer->vegetative_bud_state = BUD_DEAD;
200 }
201
202 uint Nscaffolds = context_ptr->randu(4,5);
203
204 for( int i=0; i<Nscaffolds; i++ ) {
205 uint uID_shoot = appendShoot(plantID, uID_trunk, context_ptr->randu(6, 8), 0.02, make_AxisRotation(context_ptr->randu(deg2rad(35), deg2rad(45)), (float(i) + context_ptr->randu(-0.1f, 0.1f)) / float(Nscaffolds) * 2 * M_PI, 0), "scaffold");
206
207 plant_instances.at(plantID).shoot_tree.at(uID_shoot)->breakDormancy();
208
209 uint blind_nodes = context_ptr->randu(2,3);
210 for( int b=0; b<blind_nodes; b++){
211 if( b<plant_instances.at(plantID).shoot_tree.at(uID_shoot)->phytomers.size() ) {
212 plant_instances.at(plantID).shoot_tree.at(uID_shoot)->phytomers.at(b)->removeLeaf();
213 plant_instances.at(plantID).shoot_tree.at(uID_shoot)->phytomers.at(b)->flower_bud_state = BUD_DEAD;
214 plant_instances.at(plantID).shoot_tree.at(uID_shoot)->phytomers.at(b)->vegetative_bud_state = BUD_DEAD;
215 }
216 }
217
218 }
219
220 setPlantPhenologicalThresholds(plantID, 1, 1, 3, 7, 12);
221
222 return plantID;
223
