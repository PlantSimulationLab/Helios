100 struct RayPacket {
101 // Ray data (Structure-of-Arrays layout)
102 std::vector<helios::vec3> origins;
103 std::vector<helios::vec3> directions;
104 std::vector<float> max_distances;
105 std::vector<std::vector<uint>> target_UUIDs;
106
107 // Results (output)
108 std::vector<HitResult> results;
109
110 size_t ray_count = 0;
111
112 RayPacket() = default;
113
117 void reserve(size_t capacity) {
118 origins.reserve(capacity);
119 directions.reserve(capacity);
120 max_distances.reserve(capacity);
121 target_UUIDs.reserve(capacity);
122 results.reserve(capacity);
123 }
124
128 void addRay(const RayQuery& query) {
129 origins.push_back(query.origin);
130 directions.push_back(query.direction);
131 max_distances.push_back(query.max_distance);
132 target_UUIDs.push_back(query.target_UUIDs);
133 results.emplace_back(); // Initialize empty result
134 ray_count++;
135 }
136
140 void clear() {
141 origins.clear();
142 directions.clear();
143 max_distances.clear();
144 target_UUIDs.clear();
145 results.clear();
146 ray_count = 0;
147 }
148
152 std::vector<RayQuery> toRayQueries() const {
153 std::vector<RayQuery> queries;
154 queries.reserve(ray_count);
155 for (size_t i = 0; i < ray_count; ++i) {
156 queries.emplace_back(origins[i], directions[i], max_distances[i], target_UUIDs[i]);
157 }
158 return queries;
159 }
160
164 size_t getMemoryUsage() const {
165 size_t base_memory = (origins.size() + directions.size()) * sizeof(helios::vec3) +
166 max_distances.size() * sizeof(float) +
167 results.size() * sizeof(HitResult);
168
169 // Add memory for target UUIDs
170 for (const auto& targets : target_UUIDs) {
171 base_memory += targets.size() * sizeof(uint);
172 }
173
174 return base_memory;
175 }
176 };

182 struct RayStream {
183 std::vector<RayPacket> packets;
184 size_t current_packet = 0;
185 size_t total_rays = 0;
186
190 void addRays(const std::vector<RayQuery>& queries) {
191 for (const auto& query : queries) {
192 // Create new packet if current one is full
193 if (packets.empty() || packets.back().ray_count >= RAY_BATCH_SIZE) {
194 packets.emplace_back();
195 packets.back().reserve(RAY_BATCH_SIZE);
196 }
197
198 packets.back().addRay(query);
199 total_rays++;
200 }
201 }
202
206 std::vector<HitResult> getAllResults() const {
207 std::vector<HitResult> all_results;
208 all_results.reserve(total_rays);
209
210 for (const auto& packet : packets) {
211 all_results.insert(all_results.end(), packet.results.begin(), packet.results.end());
212 }
213
214 return all_results;
215 }
216
220 void clear() {
221 packets.clear();
222 current_packet = 0;
223 total_rays = 0;
224 }
225
229 size_t getMemoryUsage() const {
230 size_t total_memory = 0;
231 for (const auto& packet : packets) {
232 total_memory += packet.getMemoryUsage();
233 }
234 return total_memory;
235 }
