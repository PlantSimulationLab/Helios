66 float dx = 1.f/float(Nx);
67 float dy = 0.5f/float(Ny);
68
69 std::vector<uint> UUIDs;
70
71 float a0 = 0.6;
72 float e0 = 0.4f;
73
74 for( int i=0; i<Nx; i++ ){
75 for( int j=0; j<Ny; j++ ){
76
77 float x = i*dx;
78 float y = j*dy;
79
80 float mag, z;
81
82 mag = sqrt( x*x + 2*y*y );
83 z = a0*mag/(e0+mag);
84 vec3 v0( x, y, z );
85
86 mag = sqrt( (x+dx)*(x+dx) + 2*y*y );
87 z = a0*mag/(e0+mag);
88 vec3 v1( x+dx, y, z );
89
90 mag = sqrt( (x+dx)*(x+dx) + 2*(y+dy)*(y+dy) );
91 z = a0*mag/(e0+mag);
92 vec3 v2( x+dx, y+dy, z );
93
94 mag = sqrt( x*x + 2*(y+dy)*(y+dy) );
95 z = a0*mag/(e0+mag);
96 vec3 v3( x, y+dy, z );
97
98 vec2 uv0( 1.f-x, 0.5+j*dy);
99 vec2 uv1( 1.f-x-dx, 0.5+j*dy );

101 vec2 uv3( 1.f-x, 0.5+(j+1)*dy );
102
103 UUIDs.push_back( context->addTriangle( v0, v1, v2, params.leaf_texture_file.c_str(), uv0, uv1, uv2 ) );
104 UUIDs.push_back( context->addTriangle( v0, v2, v3, params.leaf_texture_file.c_str(), uv0, uv2, uv3 ) );
105

108 v2.y = -v2.y;
109 v3.y = -v3.y;

111 uv0 = make_vec2( 1.f-x, 0.5-j*dy);
112 uv1 = make_vec2( 1.f-x-dx, 0.5-j*dy );
113 uv2 = make_vec2( 1.f-x-dx, 0.5-(j+1)*dy );
114 uv3 = make_vec2( 1.f-x, 0.5-(j+1)*dy );

116 UUIDs.push_back( context->addTriangle( v0, v2, v1, params.leaf_texture_file.c_str(), uv0, uv2, uv1 ) );
