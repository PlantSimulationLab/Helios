22inline bool validateDoctestArguments(int argc, char** argv) {
23 // List of valid doctest arguments (short and long forms)
24 static const std::vector<std::string> valid_prefixes = {
25 "-?", "--help", "-h",
26 "-v", "--version",
27 "-c", "--count",
28 "-ltc", "--list-test-cases",
29 "-lts", "--list-test-suites",
30 "-lr", "--list-reporters",
31 "-tc", "--test-case",
32 "-tce", "--test-case-exclude",
33 "-sf", "--source-file",
34 "-sfe", "--source-file-exclude",
35 "-ts", "--test-suite",
36 "-tse", "--test-suite-exclude",
37 "-sc", "--subcase",
38 "-sce", "--subcase-exclude",
39 "-r", "--reporters",
40 "-o", "--out",
41 "-ob", "--order-by",
42 "-rs", "--rand-seed",
43 "-f", "--first",
44 "-l", "--last",
45 "-aa", "--abort-after",
46 "-scfl", "--subcase-filter-levels",
47 // dt- prefixed versions
48 "-dt-?", "--dt-help", "-dt-h",
49 "-dt-v", "--dt-version",
50 "-dt-c", "--dt-count",
51 "-dt-ltc", "--dt-list-test-cases",
52 "-dt-lts", "--dt-list-test-suites",
53 "-dt-lr", "--dt-list-reporters",
54 "-dt-tc", "--dt-test-case",
55 "-dt-tce", "--dt-test-case-exclude",
56 "-dt-sf", "--dt-source-file",
57 "-dt-sfe", "--dt-source-file-exclude",
58 "-dt-ts", "--dt-test-suite",
59 "-dt-tse", "--dt-test-suite-exclude",
60 "-dt-sc", "--dt-subcase",
61 "-dt-sce", "--dt-subcase-exclude",
62 "-dt-r", "--dt-reporters",
63 "-dt-o", "--dt-out",
64 "-dt-ob", "--dt-order-by",
65 "-dt-rs", "--dt-rand-seed",
66 "-dt-f", "--dt-first",
67 "-dt-l", "--dt-last",
68 "-dt-aa", "--dt-abort-after",
69 "-dt-scfl", "--dt-subcase-filter-levels"
70 };

72 auto isValidDoctestArgument = [&](const std::string& arg) -> bool {
73 // Check for exact matches or prefix matches with =
74 for (const auto& prefix : valid_prefixes) {
75 if (arg == prefix || (arg.length() > prefix.length() &&
76 arg.substr(0, prefix.length()) == prefix &&
77 arg[prefix.length()] == '=')) {
78 return true;
79 }
80 }
81 return false;
82 };
83
84 // Validate command line arguments
85 std::vector<std::string> invalid_args;
86 for (int i = 1; i < argc; ++i) { // Skip program name (argv[0])
87 std::string arg(argv[i]);
88 if (!isValidDoctestArgument(arg)) {
89 invalid_args.push_back(arg);
90 }
91 }
92
93 // Warn about invalid arguments
94 if (!invalid_args.empty()) {
95 std::cerr << "WARNING: Invalid or unrecognized test arguments detected:" << std::endl;
96 for (const auto& invalid_arg : invalid_args) {
97 std::cerr << " " << invalid_arg << std::endl;
98 }
99 std::cerr << std::endl;
100 std::cerr << "Common valid patterns:" << std::endl;
101 std::cerr << " -tc=\"pattern\" : Run tests matching pattern" << std::endl;
102 std::cerr << " --test-case=\"pattern\" : Run tests matching pattern" << std::endl;
103 std::cerr << " -c : Count matching tests" << std::endl;
104 std::cerr << " --help : Show full help" << std::endl;
105 std::cerr << std::endl;
106
107 // Try to suggest corrections for common mistakes
108 for (const auto& invalid_arg : invalid_args) {
109 if (!invalid_arg.empty() && invalid_arg[0] != '-') {
110 std::cerr << "Suggestion: Did you mean -tc=\"" << invalid_arg << "\" ?" << std::endl;
111 }
112 }
113 std::cerr << std::endl;
114 std::cerr << "Proceeding with test execution..." << std::endl;
115 std::cerr << std::endl;
116
117 return false;
118 }
119
120 return true;
121}

133inline bool checkTestFiltersMatchTests(int argc, char** argv) {
134 // Create a separate context to count matching tests
135 doctest::Context count_context;
136 count_context.applyCommandLine(argc, argv);
137
138 // Add the count option to see how many tests match
139 count_context.setOption("count", true);
140 count_context.setOption("no-run", true);
141
142 // Capture the output by redirecting cout temporarily
143 std::ostringstream count_output;
144 std::streambuf* orig_cout = std::cout.rdbuf();
145 std::cout.rdbuf(count_output.rdbuf());
146
147 // Run in count mode
148 int count_result = count_context.run();
149
150 // Restore cout
151 std::cout.rdbuf(orig_cout);
152
153 // Parse the output to get the count
154 std::string output = count_output.str();
155
156 // Look for patterns indicating 0 tests found
157 if (output.find("unskipped test cases passing the current filters: 0") != std::string::npos ||
158 output.find("number of tests: 0") != std::string::npos) {
159 // Check if any test filters were actually specified
160 bool has_test_filters = false;
161 for (int i = 1; i < argc; ++i) {
162 std::string arg(argv[i]);
163 if (arg.find("-tc=") == 0 || arg.find("--test-case=") == 0 ||
164 arg.find("-ts=") == 0 || arg.find("--test-suite=") == 0 ||
165 arg.find("-sc=") == 0 || arg.find("--subcase=") == 0 ||
166 arg.find("-dt-tc=") == 0 || arg.find("--dt-test-case=") == 0 ||
167 arg.find("-dt-ts=") == 0 || arg.find("--dt-test-suite=") == 0 ||
168 arg.find("-dt-sc=") == 0 || arg.find("--dt-subcase=") == 0) {
169 has_test_filters = true;
170 break;
171 }
172 }
173
174 if (has_test_filters) {
175 std::cerr << "WARNING: No tests match the specified filters!" << std::endl;
176 std::cerr << "Your filter criteria resulted in 0 matching tests." << std::endl;
177 std::cerr << "Use -ltc or --list-test-cases to see available tests." << std::endl;
178 std::cerr << std::endl;
179 return false;
180 }
181 }
182
183 return true;
184}

197template<typename ContextType = doctest::Context>
198inline int runDoctestWithValidation(int argc, char** argv) {
199 // Validate arguments first
200 validateDoctestArguments(argc, argv);
201
202 // Check if test filters match any tests
203 checkTestFiltersMatchTests(argc, argv);
204
205 // Run the tests with command line arguments
206 ContextType context;
207 context.applyCommandLine(argc, argv);
208 int res = context.run();
209
210 if (context.shouldExit()) {
211 return res;
212 }
213
214 return res;
215}
