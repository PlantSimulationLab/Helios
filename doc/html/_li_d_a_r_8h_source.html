140
141 //calculate area
142 helios::vec3 s0 = vertex1-vertex0;
143 helios::vec3 s1 = vertex2-vertex0;
144 helios::vec3 s2 = vertex2-vertex1;
145
146 float a = s0.magnitude();
147 float b = s1.magnitude();
148 float c = s2.magnitude();
