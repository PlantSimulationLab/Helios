120 vertex2 = helios::make_vec3(0,0,0);
121 ID0 = 0;

126 color = helios::RGB::green;

129 Triangulation( int __scanID, helios::vec3 __vertex0, helios::vec3 __vertex1, helios::vec3 __vertex2, int __ID0, int __ID1, int __ID2, helios::RGBcolor __color, int __gridcell ){
130 scanID = __scanID;
131 vertex0 = __vertex0;
132 vertex1 = __vertex1;
133 vertex2 = __vertex2;
134 ID0 = __ID0;
135 ID1 = __ID1;
136 ID2 = __ID2;
137 gridcell = __gridcell;
138 color = __color;
139
140 //calculate area
141 helios::vec3 s0 = vertex1-vertex0;
142 helios::vec3 s1 = vertex2-vertex0;
143 helios::vec3 s2 = vertex2-vertex1;
144
145 float a = s0.magnitude();
146 float b = s1.magnitude();
147 float c = s2.magnitude();
148 float s = 0.5f*(a+b+c);
149
150 area = sqrt( s*(s-a)*(s-b)*(s-c) );
