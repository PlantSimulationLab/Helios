51 if (i >= 0 && i < Ntheta && j >= 0 && j < Nphi) {
52 return data.at(j).at(i);

60 data.at(j).at(i) = value;

62 std::cerr << "ERROR (hit_map.set): set index out of range. Attempting to set index map at (" << i << "," << j << "), but size of scan is " << Ntheta << " x " << Nphi << "." << std::endl;

64 }
65 }

67 Ntheta = nx;
68 Nphi = ny;
69 data.resize(Nphi);
70 for (int j = 0; j < Nphi; j++) {

108struct Triangulation {
109 helios::vec3 vertex0, vertex1, vertex2;
110 int ID0, ID1, ID2;
111 int scanID;
112 int gridcell;
113 helios::RGBcolor color;
114 float area;
115 Triangulation(void) {
116 vertex0 = helios::make_vec3(0, 0, 0);
117 vertex1 = helios::make_vec3(0, 0, 0);
118 vertex2 = helios::make_vec3(0, 0, 0);
119 ID0 = 0;
120 ID1 = 0;
121 ID2 = 0;
122 scanID = -1;
123 gridcell = -2;
124 color = helios::RGB::green;
125 area = 0;
126 }
127 Triangulation(int __scanID, helios::vec3 __vertex0, helios::vec3 __vertex1, helios::vec3 __vertex2, int __ID0, int __ID1, int __ID2, helios::RGBcolor __color, int __gridcell) {
128 scanID = __scanID;
129 vertex0 = __vertex0;
130 vertex1 = __vertex1;
131 vertex2 = __vertex2;
132 ID0 = __ID0;
133 ID1 = __ID1;
134 ID2 = __ID2;
135 gridcell = __gridcell;
136 color = __color;
137
138 // calculate area
139 helios::vec3 s0 = vertex1 - vertex0;
140 helios::vec3 s1 = vertex2 - vertex0;
141 helios::vec3 s2 = vertex2 - vertex1;
142
143 float a = s0.magnitude();
144 float b = s1.magnitude();
145 float c = s2.magnitude();
146 float s = 0.5f * (a + b + c);
147
148 area = sqrt(s * (s - a) * (s - b) * (s - c));
149 }
150};
