32__device__ float evaluateEnergyBalance( float T, float R, float Qother, float eps, float Ta, float ea, float pressure, float gH, float gS, uint Nsides, char Ntranspire, float heatcapacity, float surfacehumidity, float dt, float Tprev ){
33
34 //Outgoing emission flux
35 float Rout = float(Nsides)*eps*5.67e-8F*T*T*T*T;
36
37 //Sensible heat flux
38 float cp = 29.25f; //Molar specific heat of air. Units: J/mol
39 float QH = cp*gH*(T-Ta); // (see Campbell and Norman Eq. 6.8)
40
41 //Latent heat flux
42 float es = 611.f*exp(17.502f*(T-273.f)/((T-273.f)+240.97f)); // This is Clausius-Clapeyron equation (See Campbell and Norman pp. 41 Eq. 3.8). Note that temperature must be in Kelvin, and result is in Pascals
43 float gM = float(Ntranspire)*1.08f*(gH/float(Nsides))*gS/(1.08f*(gH/float(Nsides))+gS);
44 if( gH==0 && gS==0 ){//if somehow both go to zero, can get NaN
45 gM = 0;
46 }
47 float lambda = 44000.f; //Latent heat of vaporization for water. Units: J/mol
48 float QL = gM*lambda*(es-ea*surfacehumidity)/pressure;
49
50 //Storage heat flux
51 float storage = 0.f;
52 if (dt>0){
53 storage=heatcapacity*(T-Tprev)/dt;
54 }
55
56 //Residual
57 return R-Rout-QH-QL-Qother-storage;
58
59}
