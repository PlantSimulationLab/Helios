59  float st = sinf(theta);
60  float ct = cosf(theta);
61 
62  float sp = sinf(phi);
63  float cp = cosf(phi);
64

66  Ry[0][0] = ct;
67  Ry[0][1] = 0.f;
68  Ry[0][2] = st;
69  Ry[1][0] = 0.f;
70  Ry[1][1] = 1.f;
71  Ry[1][2] = 0.f;
72  Ry[2][0] = -st;
73  Ry[2][1] = 0.f;
74  Ry[2][2] = ct;
75 
76  Rz[0][0] = cp;
77  Rz[0][1] = -sp;
78  Rz[0][2] = 0.f;
79  Rz[1][0] = sp;
80  Rz[1][1] = cp;
81  Rz[1][2] = 0.f;
82  Rz[2][0] = 0.f;
83  Rz[2][1] = 0.f;
84  Rz[2][2] = 1.f;
85 
86  // Multiply Ry*Rz
87 
88  float rotMat[3][3] = {0.f};
89 
90  for( int i=0;i<3;i++){
91  for(int j=0;j<3;j++){
92  for(int k=0;k<3;k++){
93  rotMat[i][j]=rotMat[i][j]+Rz[i][k]*Ry[k][j];
94  }
95  }
96  }
97 
98  // Multiply the rotation matrix with the position vector.
99  float3 tmp;
100  tmp.x = rotMat[0][0] * position.x + rotMat[0][1] * position.y + rotMat[0][2] * position.z;
101  tmp.y = rotMat[1][0] * position.x + rotMat[1][1] * position.y + rotMat[1][2] * position.z;
102  tmp.z = rotMat[2][0] * position.x + rotMat[2][1] * position.y + rotMat[2][2] * position.z;
