44__device__ float3 d_rotatePoint_vi(const float3 &position, const float &theta, const float &phi) {
45
46 float Ry[3][3], Rz[3][3];
47
48 float st = sinf(theta);
49 float ct = cosf(theta);
50
51 float sp = sinf(phi);
52 float cp = cosf(phi);
53
54 // Setup the rotation matrix, this matrix is based off of the rotation matrix used in glRotatef.
55 Ry[0][0] = ct;
56 Ry[0][1] = 0.f;
57 Ry[0][2] = st;
58 Ry[1][0] = 0.f;
59 Ry[1][1] = 1.f;
60 Ry[1][2] = 0.f;
61 Ry[2][0] = -st;
62 Ry[2][1] = 0.f;
63 Ry[2][2] = ct;
64
65 Rz[0][0] = cp;
66 Rz[0][1] = -sp;
67 Rz[0][2] = 0.f;
68 Rz[1][0] = sp;
69 Rz[1][1] = cp;
70 Rz[1][2] = 0.f;
71 Rz[2][0] = 0.f;
72 Rz[2][1] = 0.f;
73 Rz[2][2] = 1.f;
74
75 // Multiply Ry*Rz
76
77 float rotMat[3][3] = {0.f};
78
79 for( int i=0;i<3;i++){
80 for(int j=0;j<3;j++){
81 for(int k=0;k<3;k++){
82 rotMat[i][j]=rotMat[i][j]+Rz[i][k]*Ry[k][j];
83 }
84 }
85 }
86
87 // Multiply the rotation matrix with the position vector.
88 float3 tmp;
89 tmp.x = rotMat[0][0] * position.x + rotMat[0][1] * position.y + rotMat[0][2] * position.z;
90 tmp.y = rotMat[1][0] * position.x + rotMat[1][1] * position.y + rotMat[1][2] * position.z;
91 tmp.z = rotMat[2][0] * position.x + rotMat[2][1] * position.y + rotMat[2][2] * position.z;
92
93 return tmp;
94
95}

140 float y0 = center.y - 0.5f*size.y; float y1 = center.y + 0.5f*size.y;
