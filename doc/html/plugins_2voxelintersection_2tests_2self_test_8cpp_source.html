85DOCTEST_TEST_CASE("VoxelIntersection - Areas after slicing (non-textured)") {
86 Context context;
87
88 // Create patch using center and size instead of vertices
89 vec3 patch_center = make_vec3(0, 0, 0);
90 vec2 patch_size = make_vec2(2, 2);

95
96 vec3 grid_center = make_vec3(0, 0, 0);

99 VoxelIntersection voxelintersection(&context);
