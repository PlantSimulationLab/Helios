56float runNetPhotosynthesisKernal( float sourceflux, float Vcmax, float Jmax, float Ci, float Kc, float Ko, float Oi, float Gamma_star,float Rd, float alpha){
57
58 float Kco = Kc*(1+Oi/Ko);
59 float Wc = Vcmax*Ci/(Ci+Kco);
60
61 float J = sourceflux*alpha/sqrt(1+pow(sourceflux*alpha/Jmax,2));
62 float Wj = J/4*Ci/(Ci+2*Gamma_star);
63 float W;
64 if (Wc<Wj){
65 W = Wc;
66 }
67 else{
68 W = Wj;
69 }
70
71 float A = W*(1-Gamma_star/Ci)-Rd;
72 return A;

84
85 float avc = photosynthesisParameters.avc;
86 float bvc = photosynthesisParameters.bvc;
87 float Vcmax25 = avc*chlorophyllcontent+bvc;
88
89 float ajv = photosynthesisParameters.ajv;

155void Annotation::getDepthImage(const std::string &filename, const helios::vec3 &camera_position, const helios::vec3 &camera_lookat) {
156
157 // Set depth for all primitives
158 float maxdepth = 0;
159 float mindepth = -1;
160 for( uint UUIDr : pixel_UUIDs ){
161 uint UUID = UUIDr-1;
162 if (context->doesPrimitiveExist(UUID)) {
163 std::vector<vec3> vertices = context->getPrimitiveVertices(UUID);//Centre
164 vec3 vertex = vertices.at(0) - camera_position;
165 vec3 cameraplane_direct = camera_lookat - camera_position;
166 float dotproductvc = vertex * cameraplane_direct;
167 float cam_norm = cameraplane_direct.magnitude();
168 float depth = dotproductvc / cam_norm;
169 if (depth > maxdepth) {
170 maxdepth = depth;
171 }
172 if (mindepth == -1) {
173 mindepth = depth;
174 } else if (depth < mindepth) {
175 mindepth = depth;
176 }
177 context->setPrimitiveData(UUID, "depth", depth);
178 }
179 }
180
181 for (uint UUIDr : pixel_UUIDs ){
182 uint UUID = UUIDr-1;
183 if (context->doesPrimitiveExist(UUID)){
184 float depth;
185 context->getPrimitiveData(UUID,"depth",depth);
186 float depth_norm = (depth-mindepth)/(maxdepth-mindepth);
187 context->setPrimitiveData(UUID,"depth_norm",depth_norm);
188 }
189 }
190
191 // Output depth image in ".txt" format
192 Annotation::getBasicLabel(filename, "depth_norm", HELIOS_TYPE_FLOAT,1);
