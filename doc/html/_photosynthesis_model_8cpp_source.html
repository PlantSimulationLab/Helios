154void PhotosynthesisModel::setModelCoefficients(const std::string &material_label, const FarquharModelCoefficients &coeffs) {
155 // Get temperature response parameters
156 PhotosyntheticTemperatureResponseParameters vcmax_resp = const_cast<FarquharModelCoefficients &>(coeffs).getVcmaxTempResponse();
157 PhotosyntheticTemperatureResponseParameters jmax_resp = const_cast<FarquharModelCoefficients &>(coeffs).getJmaxTempResponse();
158 PhotosyntheticTemperatureResponseParameters rd_resp = const_cast<FarquharModelCoefficients &>(coeffs).getRdTempResponse();
159 PhotosyntheticTemperatureResponseParameters alpha_resp = const_cast<FarquharModelCoefficients &>(coeffs).getQuantumEfficiencyTempResponse();
160
161 // Serialize temperature response parameters for Vcmax
162 context->setMaterialData(material_label, LABEL_FQ_Vcmax, vcmax_resp.value_at_25C);
163 context->setMaterialData(material_label, "photo_fq_Vcmax_dHa", vcmax_resp.dHa);
164 context->setMaterialData(material_label, "photo_fq_Vcmax_dHd", vcmax_resp.dHd);
165 context->setMaterialData(material_label, "photo_fq_Vcmax_Topt", vcmax_resp.Topt);
166
167 // Serialize temperature response parameters for Jmax
168 context->setMaterialData(material_label, LABEL_FQ_Jmax, jmax_resp.value_at_25C);
169 context->setMaterialData(material_label, "photo_fq_Jmax_dHa", jmax_resp.dHa);
170 context->setMaterialData(material_label, "photo_fq_Jmax_dHd", jmax_resp.dHd);
171 context->setMaterialData(material_label, "photo_fq_Jmax_Topt", jmax_resp.Topt);
172
173 // Serialize temperature response parameters for Rd
174 context->setMaterialData(material_label, LABEL_FQ_Rd, rd_resp.value_at_25C);
175 context->setMaterialData(material_label, "photo_fq_Rd_dHa", rd_resp.dHa);
176 context->setMaterialData(material_label, "photo_fq_Rd_dHd", rd_resp.dHd);
177 context->setMaterialData(material_label, "photo_fq_Rd_Topt", rd_resp.Topt);
178
179 // Serialize temperature response parameters for alpha
180 context->setMaterialData(material_label, LABEL_FQ_alpha, alpha_resp.value_at_25C);
181 context->setMaterialData(material_label, "photo_fq_alpha_dHa", alpha_resp.dHa);
182
183 // Serialize other parameters
184 context->setMaterialData(material_label, LABEL_FQ_O, coeffs.O);

366FarquharModelCoefficients PhotosynthesisModel::getCoefficientsForPrimitive_Farquhar(uint UUID) const {
367 uint materialID = context->getPrimitiveMaterialID(UUID);
368
369 // Check cache first
370 if (material_coefficient_cache_farquhar.find(materialID) != material_coefficient_cache_farquhar.end()) {
371 return material_coefficient_cache_farquhar.at(materialID);
372 }
373
374 // Try to load from material data
375 FarquharModelCoefficients coeffs;
376 bool found = true;
377
378 try {
379 const Material &mat = context->getMaterial(materialID);
380 if (mat.doesMaterialDataExist(LABEL_FQ_Vcmax) && mat.doesMaterialDataExist(LABEL_FQ_Jmax) && mat.doesMaterialDataExist(LABEL_FQ_Rd) && mat.doesMaterialDataExist(LABEL_FQ_alpha)) {
381
382 // Reconstruct temperature response parameters for Vcmax
383 float vcmax_25C, vcmax_dHa, vcmax_dHd, vcmax_Topt;
384 mat.getMaterialData(LABEL_FQ_Vcmax, vcmax_25C);
385 mat.getMaterialData("photo_fq_Vcmax_dHa", vcmax_dHa);
386 mat.getMaterialData("photo_fq_Vcmax_dHd", vcmax_dHd);
387 mat.getMaterialData("photo_fq_Vcmax_Topt", vcmax_Topt);
388
389 // Reconstruct for Jmax
390 float jmax_25C, jmax_dHa, jmax_dHd, jmax_Topt;
391 mat.getMaterialData(LABEL_FQ_Jmax, jmax_25C);
392 mat.getMaterialData("photo_fq_Jmax_dHa", jmax_dHa);
393 mat.getMaterialData("photo_fq_Jmax_dHd", jmax_dHd);
394 mat.getMaterialData("photo_fq_Jmax_Topt", jmax_Topt);
395
396 // Reconstruct for Rd
397 float rd_25C, rd_dHa, rd_dHd, rd_Topt;
398 mat.getMaterialData(LABEL_FQ_Rd, rd_25C);
399 mat.getMaterialData("photo_fq_Rd_dHa", rd_dHa);
400 mat.getMaterialData("photo_fq_Rd_dHd", rd_dHd);
401 mat.getMaterialData("photo_fq_Rd_Topt", rd_Topt);
402
403 // Reconstruct for alpha
404 float alpha_25C, alpha_dHa;
405 mat.getMaterialData(LABEL_FQ_alpha, alpha_25C);
406 mat.getMaterialData("photo_fq_alpha_dHa", alpha_dHa);
407
408 // Use setter methods to properly populate the coefficient struct
409 if (vcmax_Topt < 1000.f) { // Has optimum
410 coeffs.setVcmax(vcmax_25C, vcmax_dHa, vcmax_Topt - 273.15f, vcmax_dHd);
411 } else if (vcmax_dHa > 0.f) { // Monotonic increase
412 coeffs.setVcmax(vcmax_25C, vcmax_dHa);
413 } else { // Constant
414 coeffs.setVcmax(vcmax_25C);
415 }
416
417 if (jmax_Topt < 1000.f) {
418 coeffs.setJmax(jmax_25C, jmax_dHa, jmax_Topt - 273.15f, jmax_dHd);
419 } else if (jmax_dHa > 0.f) {
420 coeffs.setJmax(jmax_25C, jmax_dHa);
421 } else {
422 coeffs.setJmax(jmax_25C);
423 }
424
425 if (rd_Topt < 1000.f) {
426 coeffs.setRd(rd_25C, rd_dHa, rd_Topt - 273.15f, rd_dHd);
427 } else if (rd_dHa > 0.f) {
428 coeffs.setRd(rd_25C, rd_dHa);
429 } else {
430 coeffs.setRd(rd_25C);
431 }
432
433 if (alpha_dHa > 0.f) {
434 coeffs.setQuantumEfficiency_alpha(alpha_25C, alpha_dHa);
435 } else {
436 coeffs.setQuantumEfficiency_alpha(alpha_25C);
437 }
438
439 // Set O and TPU_flag
440 mat.getMaterialData(LABEL_FQ_O, coeffs.O);
441 mat.getMaterialData(LABEL_FQ_TPU_flag, coeffs.TPU_flag);
442
443 material_coefficient_cache_farquhar[materialID] = coeffs;
444 return coeffs;
445 } else {
446 found = false;
447 }
448 } catch (const std::exception &) {
449 found = false;
450 }
451
452 // Fallback to legacy UUID map
453 if (!found && farquharmodel_coefficients.find(UUID) != farquharmodel_coefficients.end()) {
454 return farquharmodel_coefficients.at(UUID);

489void PhotosynthesisModel::run(const std::vector<uint> &lUUIDs) {
490
491 WarningAggregator warnings;
492 warnings.setEnabled(message_flag); // Respect existing message flag
493
494 for (uint UUID: lUUIDs) {
495
496 float i_PAR;
497 if (context->doesPrimitiveDataExist(UUID, "radiation_flux_PAR") && context->getPrimitiveDataType("radiation_flux_PAR") == HELIOS_TYPE_FLOAT) {
498 context->getPrimitiveData(UUID, "radiation_flux_PAR", i_PAR);
499 i_PAR = i_PAR * 4.57f; // umol/m^2-s (ref https://www.controlledenvironments.org/wp-content/uploads/sites/6/2017/06/Ch01.pdf)
500 if (i_PAR < 0) {
501 i_PAR = 0;
502 warnings.addWarning("negative_par_clipped", "PAR flux value provided was negative. Clipping to zero.");
503 }
504 } else {
505 i_PAR = i_PAR_default;
506 }
507
508 float TL;
509 if (context->doesPrimitiveDataExist(UUID, "temperature") && context->getPrimitiveDataType("temperature") == HELIOS_TYPE_FLOAT) {
510 context->getPrimitiveData(UUID, "temperature", TL);
511 if (TL < 200) {
512 warnings.addWarning("low_temperature", "Primitive temperature value was very low (" + std::to_string(TL) + "K). Using default. Are you using absolute temperature units?");
513 TL = TL_default;
514 }
515 } else {
516 TL = TL_default;
517 }
518
519 float CO2;
520 if (context->doesPrimitiveDataExist(UUID, "air_CO2") && context->getPrimitiveDataType("air_CO2") == HELIOS_TYPE_FLOAT) {
521 context->getPrimitiveData(UUID, "air_CO2", CO2);
522 if (CO2 < 0) {
523 CO2 = 0;
524 warnings.addWarning("negative_co2_clipped", "CO2 concentration value provided was negative. Clipping to zero.");
525 }
526 } else {
527 CO2 = CO2_default;
528 }
529
530 float gM;
531 if (context->doesPrimitiveDataExist(UUID, "moisture_conductance") && context->getPrimitiveDataType("moisture_conductance") == HELIOS_TYPE_FLOAT) {
532 context->getPrimitiveData(UUID, "moisture_conductance", gM);
533 if (gM < 0) {
534 gM = 0;
535 warnings.addWarning("negative_moisture_conductance_clipped", "Moisture conductance value provided was negative. Clipping to zero.");
536 }
537 } else {
538 gM = gM_default;
539 }
540
541 // Number of sides - check material first, then primitive data
542 uint twosided_flag = context->getPrimitiveTwosidedFlag(UUID, 1);
543 uint Nsides = (twosided_flag == 0) ? 1 : 2;
544
545 float stomatal_sidedness = 0.f; // default all stomata on one side (hypostomatous)
546 if (Nsides == 2 && context->doesPrimitiveDataExist(UUID, "stomatal_sidedness") && context->getPrimitiveDataType("stomatal_sidedness") == HELIOS_TYPE_FLOAT) {
547 context->getPrimitiveData(UUID, "stomatal_sidedness", stomatal_sidedness);
548 }
549
550 float gH;
551 if (context->doesPrimitiveDataExist(UUID, "boundarylayer_conductance") && context->getPrimitiveDataType("boundarylayer_conductance") == HELIOS_TYPE_FLOAT) {
552 context->getPrimitiveData(UUID, "boundarylayer_conductance", gH);
553 } else if (context->doesPrimitiveDataExist(UUID, "boundarylayer_conductance_out") && context->getPrimitiveDataType("boundarylayer_conductance_out") == HELIOS_TYPE_FLOAT) {
554 context->getPrimitiveData(UUID, "boundarylayer_conductance_out", gH);
555 } else {
556 gH = gH_default;
557 }
558 if (gH < 0) {
559 gH = 0;
560 warnings.addWarning("negative_boundarylayer_conductance_clipped", "Boundary-layer conductance value provided was negative. Clipping to zero.");
561 }
562
563 // combine stomatal (gM) and boundary-layer (gH) conductances
564 if (gH == 0 && gM == 0) { // if somehow both go to zero, can get NaN
565 gM = 0;
566 } else {
567 gM = 1.08f * gH * gM * (stomatal_sidedness / (1.08f * gH + gM * stomatal_sidedness) + (1.f - stomatal_sidedness) / (1.08f * gH + gM * (1.f - stomatal_sidedness)));
568 }
569
570 float A, Ci, Gamma, J_over_Jmax = 0.f;
571 int limitation_state, TPU_flag = 0;
572
573 if (model == "farquhar") { // Farquhar-von Caemmerer-Berry Model
574
575 // Check for manual Ci override first
576 if (manual_Ci.find(UUID) != manual_Ci.end()) {
577 // Use manual Ci, bypassing iterative calculation
578 Ci = manual_Ci.at(UUID);
579
580 FarquharModelCoefficients coeffs = getCoefficientsForPrimitive_Farquhar(UUID);
581
582 // Set up variables vector for evaluateCi_Farquhar
583 std::vector<float> variables{CO2, i_PAR, TL, gM, 0.f, 0.f, 0.f, float(TPU_flag), 0.f};
584
585 // Call evaluateCi_Farquhar directly to compute A for the manual Ci
586 // This bypasses the fzero iteration in evaluateFarquharModel
587 evaluateCi_Farquhar(Ci, variables, &coeffs); // Return value (residual) is unused
588
589 // Extract results from variables vector
590 A = variables[4];
591 limitation_state = (int) variables[5];
592 Gamma = variables[6];
593 J_over_Jmax = variables[8];
594
595 // Store in previous_Ci for consistency
596 previous_Ci[UUID] = Ci;
597
598 } else {
599 // Normal iterative calculation
600
601 // Initialize Ci with previous timestep value for temporal continuity, or reasonable default
602 if (previous_Ci.find(UUID) != previous_Ci.end()) {
603 Ci = previous_Ci.at(UUID); // Use previous timestep's Ci
604 } else {
605 Ci = CO2 * 0.7f; // Default initial guess (typical Ci/Ca ratio)
606 }
607
608 FarquharModelCoefficients coeffs = getCoefficientsForPrimitive_Farquhar(UUID);
609 A = evaluateFarquharModel(coeffs, i_PAR, TL, CO2, gM, Ci, Gamma, limitation_state, TPU_flag, J_over_Jmax, warnings);
610
611 // Store computed Ci for next timestep (temporal continuity)
612 previous_Ci[UUID] = Ci;
613 }
614
615 } else { // Empirical Model
616
617 EmpiricalModelCoefficients coeffs = getCoefficientsForPrimitive_Empirical(UUID);
618 A = evaluateEmpiricalModel(coeffs, i_PAR, TL, CO2, gM);
619 }
620
621 if (A == 0) {
622 warnings.addWarning("convergence_failure", "Solution did not converge for primitive " + std::to_string(UUID) + ".");
623 }
624
625 context->setPrimitiveData(UUID, "net_photosynthesis", A);
626
627 for (const auto &data: output_prim_data) {
628 if (data == "Ci" && model == "farquhar") {
629 context->setPrimitiveData(UUID, "Ci", Ci);
630 } else if (data == "limitation_state" && model == "farquhar") {
631 context->setPrimitiveData(UUID, "limitation_state", limitation_state);
632 } else if (data == "Gamma_CO2" && model == "farquhar") {
633 context->setPrimitiveData(UUID, "Gamma_CO2", Gamma);
634 } else if (data == "electron_transport_ratio" && model == "farquhar") {
635 context->setPrimitiveData(UUID, "electron_transport_ratio", J_over_Jmax);
636 }
637 }
638 }
639
640 warnings.report(std::cerr);
641}

643float PhotosynthesisModel::evaluateCi_Empirical(const EmpiricalModelCoefficients &params, float Ci, float CO2, float fL, float Rd, float gM) const {
644
645
646 //--- CO2 Response Function --- //
647
648 float fC = params.kC * Ci / params.Ci_ref;
649
650
651 //--- Assimilation Rate --- //
652
653 float A = params.Asat * fL * fC - Rd;
654
655 //--- Calculate error and update --- //
656
657 float resid = 0.75f * gM * (CO2 - Ci) - A;
658
659
660 return resid;
661}

663float PhotosynthesisModel::evaluateEmpiricalModel(const EmpiricalModelCoefficients &params, float i_PAR, float TL, float CO2, float gM) {
664
665 // initial guess for intercellular CO2
666 float Ci = CO2;
667
668 //--- Light Response Function --- //
669
670 float fL = i_PAR / (params.theta + i_PAR);
671
672 assert(fL >= 0 && fL <= 1);
673
674 //--- Respiration Rate --- //
675
676 float Rd = params.R * sqrt(TL - 273.f) * exp(-params.ER / TL);
677
678 float Ci_old = Ci;
679 float Ci_old_old = 0.95f * Ci;
680
681 float resid_old = evaluateCi_Empirical(params, Ci_old, CO2, fL, Rd, gM);
682 float resid_old_old = evaluateCi_Empirical(params, Ci_old_old, CO2, fL, Rd, gM);
683
684 float err = 10000, err_max = 0.01;
685 int iter = 0, max_iter = 100;
686 float resid;
687 while (err > err_max && iter < max_iter) {
688
689 if (resid_old == resid_old_old) { // this condition will cause NaN
690 break;
691 }
692
693 Ci = fabs((Ci_old_old * resid_old - Ci_old * resid_old_old) / (resid_old - resid_old_old));
694
695 resid = evaluateCi_Empirical(params, Ci, CO2, fL, Rd, gM);
696
697 resid_old_old = resid_old;
698 resid_old = resid;
699
700 err = fabs(resid);
701
702 Ci_old_old = Ci_old;
703 Ci_old = Ci;
704
705 iter++;
706 }
707
708 float A;
709 if (err > err_max) {
710 A = 0;
711 } else {
712 float fC = params.kC * Ci / params.Ci_ref;
713 A = params.Asat * fL * fC - Rd;
714 }
715
716 return A;
717}

719float PhotosynthesisModel::evaluateCi_Farquhar(float Ci, std::vector<float> &variables, const void *parameters) {
720
721 const FarquharModelCoefficients modelcoeffs = *reinterpret_cast<const FarquharModelCoefficients *>(parameters);
722
723 float Vcmax, Jmax, TPU, Rd, alpha, theta;
724
725 float CO2 = variables[0];
726 float Q = variables[1];
727 float TL = variables[2];
728 float gM = variables[3];
729 int TPUflag = int(variables[4]);
730
731 float R = 0.0083144598; // molar gas constant (kJ/K/mol)
732 float O = 213.5; // ambient oxygen concentration (mmol/mol)
733 float c_Gamma = 19.02;
734 float dH_Gamma = 37.83;
735 float c_Kc = 38.05;
736 float dH_Kc = 79.43;
737 float c_Ko = 20.30;
738 float dH_Ko = 36.38;
739
740 float invDiffRT = (1.f / 298.15f - 1.f / TL) / R;
741
742 if (modelcoeffs.Vcmax > 0) {
743 Vcmax = modelcoeffs.Vcmax * expf(modelcoeffs.dH_Vcmax * (invDiffRT));
744 } else {
745 Vcmax = respondToTemperature(&modelcoeffs.VcmaxTempResponse, TL);
746 }
747
748 if (modelcoeffs.Jmax > 0) {
749 Jmax = modelcoeffs.Jmax * expf(modelcoeffs.dH_Jmax * (invDiffRT));
750 } else {
751 Jmax = respondToTemperature(&modelcoeffs.JmaxTempResponse, TL);
752 }
753
754 if (modelcoeffs.Rd > 0) {
755 Rd = modelcoeffs.Rd * expf(modelcoeffs.dH_Rd * (invDiffRT));
756 } else {
757 Rd = respondToTemperature(&modelcoeffs.RdTempResponse, TL);
758 }
759
760 if (modelcoeffs.alpha > 0) {
761 alpha = modelcoeffs.alpha;
762 } else {
763 alpha = respondToTemperature(&modelcoeffs.alphaTempResponse, TL);
764 }
765
766
767 TPU = respondToTemperature(&modelcoeffs.TPUTempResponse, TL);
768 theta = respondToTemperature(&modelcoeffs.thetaTempResponse, TL);
769
770
771 float Gamma_star = exp(c_Gamma - dH_Gamma / (R * TL));
772 float Kc = exp(c_Kc - dH_Kc / (R * TL));
773 float Ko = exp(c_Ko - dH_Ko / (R * TL));
774 float Kco = Kc * (1.f + O / Ko);
775
776 // thetaJ^2 + -(alphaQ+Jmax)J + alphaQJmax = 0
777 double a = std::max(theta, 0.0001f);
778 double ia = 1.000f / a;
779 double b = -(alpha * Q + Jmax);
780 double c = alpha * Q * Jmax;
781 double J = (-b - sqrt(pow(b, 2.000f) - 4.f * a * c)) * 0.5f * ia;
782 // J = Jmax * alpha * Q / (alpha * Q + Jmax);
783
784 // Store J/Jmax ratio for fluorescence calculations
785 float J_over_Jmax = (Jmax > 0.f) ? static_cast<float>(J / Jmax) : 0.f;
786 variables[8] = J_over_Jmax;
787
788 float Wc = Vcmax * Ci / (Ci + Kco);
789 float Wj = J * Ci / (4.f * Ci + 8.f * Gamma_star);
790 float Wp = 3.f * TPU * Ci / (Ci - Gamma_star);
791
792 float smooth_factor = 0.99f;
793 float s = helios::clamp(0.5f + 0.5f * (Wc - Wj) / smooth_factor, 0.0f, 1.0f);
794 float smooth_min = Wc * (1.f - s) + Wj * s - smooth_factor * s * (1.f - s);
795
796 if (TPUflag == 1) {
797 smooth_factor = 0.99f;
798 s = helios::clamp(0.5f + 0.5f * (smooth_min - Wp) / smooth_factor, 0.0f, 1.0f);
799 smooth_min = smooth_min * (1.f - s) + Wp * s - smooth_factor * s * (1.f - s);
800 }
801 float A = (1.f - Gamma_star / Ci) * smooth_min - Rd;
802
803
804 float limitation_state;
805 if (Wj > Wc) { // Rubisco limited
806 limitation_state = 0;
807 } else { // Electron transport limited
808 limitation_state = 1;
809 }
810
811 //--- Calculate error and update --- //
812
813 float resid = 0.75f * gM * (CO2 - Ci) - A;
814
815 variables[4] = A;
816 variables[5] = limitation_state;
817
818 float Gamma = (Gamma_star + Kco * Rd / Vcmax) / (1.f - Rd / Vcmax); // Equation 39 of Farquhar et al. (1980)
819 variables[6] = Gamma;
820
821 return resid;
822}

825float PhotosynthesisModel::respondToTemperature(const PhotosyntheticTemperatureResponseParameters *params, float T_in_Kelvin) {
826 float T = T_in_Kelvin;
827 float R = 0.0083144598f;
828 float v25 = params->value_at_25C;
829 float dHa = params->dHa;
830 float dHd = params->dHd;
831 float Topt = params->Topt;
832 if (dHa == 0) {
833 return v25;
834 } else {
835 float logterm = logf(dHd / dHa - 1.f);
836 float t1 = 1.f + expf(dHd / R * (1.f / Topt - 1.f / 298.15f) - logterm);
837 float t2 = 1.f + expf(dHd / R * (1.f / Topt - 1.f / T) - logterm);
838 return v25 * exp(dHa / R * (1.f / 298.15f - 1.f / T)) * t1 / t2;
839 }

843float PhotosynthesisModel::evaluateFarquharModel(const FarquharModelCoefficients &params, float i_PAR, float TL, float CO2, float gM, float &Ci, float &Gamma, int &limitation_state, int &TPU_flag, float &J_over_Jmax,
844 helios::WarningAggregator &warnings) {
845
846 float A = 0;
847 std::vector<float> variables{CO2, i_PAR, TL, gM, A, float(limitation_state), Gamma, float(TPU_flag), 0.f};
848
849 // Temporal continuity approach: use previous Ci as initial guess, with fallbacks
850 // This leverages the fact that Ci changes slowly between timesteps
851 std::vector<float> initial_guesses;
852
853 // First try previous Ci if available (temporal continuity)
854 if (Ci > 0 && std::isfinite(Ci)) {
855 initial_guesses.push_back(Ci);
856 }
857
858 // Add fallback guesses based on research (physiologically reasonable values)
859 initial_guesses.insert(initial_guesses.end(), {CO2 * 0.7f, 100.0f, CO2 * 0.3f, CO2 * 0.9f, 50.0f});
860
861 bool overall_converged = false;
862 for (float guess: initial_guesses) {
863 // Reset variables for each attempt
864 std::vector<float> vars_attempt = variables;
865 bool attempt_converged = false;
866 float Ci_attempt = fzero(evaluateCi_Farquhar, vars_attempt, &params, guess, attempt_converged, 0.001f, 200);
867
868 // Check if solution converged and is physiologically reasonable
869 if (attempt_converged && Ci_attempt > 0 && std::isfinite(Ci_attempt)) {
870 Ci = Ci_attempt;
871 variables = vars_attempt;
872 overall_converged = true;
873 break;
874 }
875 }
876
877 // If all initial guesses failed, add warning to aggregator
878 if (!overall_converged) {
879 warnings.addWarning("photosynthesis_ci_convergence_failure", "Photosynthesis model failed to converge for Ci calculation after trying multiple initial guesses.");
880 // Use best available estimate from last attempt
881 bool final_converged = false;
882 Ci = fzero(evaluateCi_Farquhar, variables, &params, 100.0f, final_converged, 0.01f, 500);
883 }
884
885 A = variables[4];
886 limitation_state = (int) variables[5];
887 Gamma = variables[6];
888 J_over_Jmax = variables[8];
889
890 return A;
891}
