319float PhotosynthesisModel::evaluateEmpiricalModel( const EmpiricalModelCoefficients &params, float i_PAR, float TL, float CO2, float gM ){
320
321//initial guess for intercellular CO2
322 float Ci = CO2;
323
324//--- Light Response Function --- //
325
326 float fL = i_PAR/(params.theta+i_PAR);
327
328 assert( fL>=0 && fL<=1 );
329
330//--- Respiration Rate --- //
331
332 float Rd = params.R*sqrt(TL-273.f)*exp(-params.ER/TL);
333
334 float Ci_old = Ci;
335 float Ci_old_old = 0.95f*Ci;
336
337 float resid_old = evaluateCi_Empirical( params, Ci_old, CO2, fL, Rd, gM );
338 float resid_old_old = evaluateCi_Empirical( params, Ci_old_old, CO2, fL, Rd, gM );
339
340 float err = 10000, err_max = 0.01;
341 int iter = 0, max_iter = 100;
342 float resid;
343 while( err>err_max && iter<max_iter ){
344
345 if( resid_old==resid_old_old ){//this condition will cause NaN
346 break;
347 }
348
349 Ci = fabs((Ci_old_old*resid_old-Ci_old*resid_old_old)/(resid_old-resid_old_old));
350
351 resid = evaluateCi_Empirical( params, Ci, CO2, fL, Rd, gM );
352
353 resid_old_old = resid_old;
354 resid_old = resid;
355
356 err = fabs(resid);
357
358 Ci_old_old = Ci_old;
359 Ci_old = Ci;
360
361 iter++;
362
363 }
364
365 float A;
366 if( err>err_max ){
367 A = 0;
368 }else{
369 float fC = params.kC*Ci/params.Ci_ref;
370 A = params.Asat*fL*fC-Rd;
371 }
372
373 return A;

377float PhotosynthesisModel::evaluateCi_Farquhar( float Ci, std::vector<float> &variables, const void* parameters ){
378
379 const FarquharModelCoefficients modelcoeffs = *reinterpret_cast<const FarquharModelCoefficients*>(parameters);
380
381 float CO2 = variables[0];
382 float i_PAR = variables[1];
383 float TL = variables[2];
384 float gM = variables[3];
385
386//molar gas constant (kJ/K/mol)
387 float R = 0.0083144598;
388
389 float Rd = modelcoeffs.Rd*exp(modelcoeffs.c_Rd-modelcoeffs.dH_Rd/(R*TL));
390 float Vcmax = modelcoeffs.Vcmax*exp(modelcoeffs.c_Vcmax-modelcoeffs.dH_Vcmax/(R*TL));
391 float Jmax = modelcoeffs.Jmax*exp(modelcoeffs.c_Jmax-modelcoeffs.dH_Jmax/(R*TL));
392
393 float Gamma_star = exp(modelcoeffs.c_Gamma-modelcoeffs.dH_Gamma/(R*TL));
394 float Kc = exp(modelcoeffs.c_Kc-modelcoeffs.dH_Kc/(R*TL));
395 float Ko = exp(modelcoeffs.c_Ko-modelcoeffs.dH_Ko/(R*TL));
396
397 float Kco = Kc*(1.f+modelcoeffs.O/Ko);
398
399 float Wc = Vcmax*Ci/(Ci+Kco);
400
401 float J = Jmax*i_PAR*modelcoeffs.alpha/(i_PAR*modelcoeffs.alpha+Jmax);
402 float Wj = J*Ci/(4.f*Ci+8.f*Gamma_star);
403
404 float A = (1-Gamma_star/Ci)*fmin(Wc,Wj)-Rd;
405
406 float limitation_state;
407 if( Wj>Wc ){ //Rubisco limited
408 limitation_state = 0;
409 }else{ //Electron transport limited
410 limitation_state = 1;
411 }
412
413//--- Calculate error and update --- //
414
415 float resid = 0.75f*gM*(CO2-Ci) - A;
416
417 variables[4] = A;
418 variables[5] = limitation_state;
419
420 float Gamma = (Gamma_star+Kco*Rd/Vcmax)/(1-Rd/Vcmax); //Equation 39 of Farquhar et al. (1980)
421 variables[6] = Gamma;
422
423 return resid;
424
425}
