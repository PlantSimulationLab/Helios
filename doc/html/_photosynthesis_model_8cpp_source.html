223

225
226 float i_PAR;
227 if (context->doesPrimitiveDataExist(UUID, "radiation_flux_PAR") && context->getPrimitiveDataType("radiation_flux_PAR") == HELIOS_TYPE_FLOAT) {
228 context->getPrimitiveData(UUID, "radiation_flux_PAR", i_PAR);

231 i_PAR = 0;
232 if (message_flag) {

237 i_PAR = i_PAR_default;
238 }
239

244 if (message_flag) {

264 }
265
266 float gM;
267 if (context->doesPrimitiveDataExist(UUID, "moisture_conductance") && context->getPrimitiveDataType("moisture_conductance") == HELIOS_TYPE_FLOAT) {

269 if (gM < 0) {

271 if (message_flag) {
272 std::cout << "WARNING (PhotosynthesisModel::run): Moisture conductance value provided was negative. Clipping to zero." << std::endl;
273 }
274 }
275 } else {
276 gM = gM_default;
277 }
278
279 // Number of sides
280 uint Nsides = 2; // default is 2
281 if (context->doesPrimitiveDataExist(UUID, "twosided_flag") && context->getPrimitiveDataType("twosided_flag") == HELIOS_TYPE_UINT) {
282 uint flag;
283 context->getPrimitiveData(UUID, "twosided_flag", flag);

288
289 float stomatal_sidedness = 0.f; // default all stomata on one side (hypostomatous)
290 if (Nsides == 2 && context->doesPrimitiveDataExist(UUID, "stomatal_sidedness") && context->getPrimitiveDataType("stomatal_sidedness") == HELIOS_TYPE_FLOAT) {
291 context->getPrimitiveData(UUID, "stomatal_sidedness", stomatal_sidedness);
292 }
293
294 float gH;

296 context->getPrimitiveData(UUID, "boundarylayer_conductance", gH);
297 } else if (context->doesPrimitiveDataExist(UUID, "boundarylayer_conductance_out") && context->getPrimitiveDataType("boundarylayer_conductance_out") == HELIOS_TYPE_FLOAT) {

339 } else { // Empirical Model

355 }
356

358
359 for (const auto &data: output_prim_data) {

362 } else if (data == "limitation_state" && model == "farquhar") {

371float PhotosynthesisModel::evaluateCi_Empirical(const EmpiricalModelCoefficients &params, float Ci, float CO2, float fL, float Rd, float gM) const {
372
373
374 //--- CO2 Response Function --- //
375
376 float fC = params.kC * Ci / params.Ci_ref;
377
378
379 //--- Assimilation Rate --- //
380
381 float A = params.Asat * fL * fC - Rd;
382
383 //--- Calculate error and update --- //
384
385 float resid = 0.75f * gM * (CO2 - Ci) - A;
386
387
388 return resid;
389}

391float PhotosynthesisModel::evaluateEmpiricalModel(const EmpiricalModelCoefficients &params, float i_PAR, float TL, float CO2, float gM) {
392
393 // initial guess for intercellular CO2
394 float Ci = CO2;
395
396 //--- Light Response Function --- //
397
398 float fL = i_PAR / (params.theta + i_PAR);
399
400 assert(fL >= 0 && fL <= 1);
401
402 //--- Respiration Rate --- //
403
404 float Rd = params.R * sqrt(TL - 273.f) * exp(-params.ER / TL);
405
406 float Ci_old = Ci;
407 float Ci_old_old = 0.95f * Ci;
408
409 float resid_old = evaluateCi_Empirical(params, Ci_old, CO2, fL, Rd, gM);
410 float resid_old_old = evaluateCi_Empirical(params, Ci_old_old, CO2, fL, Rd, gM);
411
412 float err = 10000, err_max = 0.01;
413 int iter = 0, max_iter = 100;
414 float resid;
415 while (err > err_max && iter < max_iter) {
416
417 if (resid_old == resid_old_old) { // this condition will cause NaN
418 break;
419 }
420
421 Ci = fabs((Ci_old_old * resid_old - Ci_old * resid_old_old) / (resid_old - resid_old_old));
422
423 resid = evaluateCi_Empirical(params, Ci, CO2, fL, Rd, gM);
424
425 resid_old_old = resid_old;
426 resid_old = resid;
427
428 err = fabs(resid);
429
430 Ci_old_old = Ci_old;
431 Ci_old = Ci;
432
433 iter++;
434 }
435
436 float A;
437 if (err > err_max) {
438 A = 0;
439 } else {
440 float fC = params.kC * Ci / params.Ci_ref;
441 A = params.Asat * fL * fC - Rd;
442 }
443
444 return A;
445}

447float PhotosynthesisModel::evaluateCi_Farquhar(float Ci, std::vector<float> &variables, const void *parameters) {
448
449 const FarquharModelCoefficients modelcoeffs = *reinterpret_cast<const FarquharModelCoefficients *>(parameters);
450
451 float Vcmax, Jmax, TPU, Rd, alpha, theta;
452
453 float CO2 = variables[0];
454 float Q = variables[1];
455 float TL = variables[2];
456 float gM = variables[3];
457 int TPUflag = int(variables[4]);
458
459 float R = 0.0083144598; // molar gas constant (kJ/K/mol)
460 float O = 213.5; // ambient oxygen concentration (mmol/mol)
461 float c_Gamma = 19.02;
462 float dH_Gamma = 37.83;
463 float c_Kc = 38.05;
464 float dH_Kc = 79.43;
465 float c_Ko = 20.30;
466 float dH_Ko = 36.38;
467
468 float invDiffRT = (1.f / 298.15f - 1.f / TL) / R;
469
470 if (modelcoeffs.Vcmax > 0) {
471 Vcmax = modelcoeffs.Vcmax * expf(modelcoeffs.dH_Vcmax * (invDiffRT));
472 } else {
473 Vcmax = respondToTemperature(&modelcoeffs.VcmaxTempResponse, TL);
474 }
475
476 if (modelcoeffs.Jmax > 0) {
477 Jmax = modelcoeffs.Jmax * expf(modelcoeffs.dH_Jmax * (invDiffRT));
478 } else {
479 Jmax = respondToTemperature(&modelcoeffs.JmaxTempResponse, TL);
480 }
481
482 if (modelcoeffs.Rd > 0) {
483 Rd = modelcoeffs.Rd * expf(modelcoeffs.dH_Rd * (invDiffRT));
484 } else {
485 Rd = respondToTemperature(&modelcoeffs.RdTempResponse, TL);
486 }
487
488 if (modelcoeffs.alpha > 0) {
489 alpha = modelcoeffs.alpha;
490 } else {
491 alpha = respondToTemperature(&modelcoeffs.alphaTempResponse, TL);
492 }
493
494
495 TPU = respondToTemperature(&modelcoeffs.TPUTempResponse, TL);
496 theta = respondToTemperature(&modelcoeffs.thetaTempResponse, TL);
497
498
499 float Gamma_star = exp(c_Gamma - dH_Gamma / (R * TL));
500 float Kc = exp(c_Kc - dH_Kc / (R * TL));
501 float Ko = exp(c_Ko - dH_Ko / (R * TL));
502 float Kco = Kc * (1.f + O / Ko);
503
504 // thetaJ^2 + -(alphaQ+Jmax)J + alphaQJmax = 0
505 double a = std::max(theta, 0.0001f);
506 double ia = 1.000f / a;
507 double b = -(alpha * Q + Jmax);
508 double c = alpha * Q * Jmax;
509 double J = (-b - sqrt(pow(b, 2.000f) - 4.f * a * c)) * 0.5f * ia;
510 // J = Jmax * alpha * Q / (alpha * Q + Jmax);
511
512 float Wc = Vcmax * Ci / (Ci + Kco);
513 float Wj = J * Ci / (4.f * Ci + 8.f * Gamma_star);
514 float Wp = 3.f * TPU * Ci / (Ci - Gamma_star);
515
516 float smooth_factor = 0.99f;
517 float s = helios::clamp(0.5f + 0.5f * (Wc - Wj) / smooth_factor, 0.0f, 1.0f);
518 float smooth_min = Wc * (1.f - s) + Wj * s - smooth_factor * s * (1.f - s);
519
520 if (TPUflag == 1) {
521 smooth_factor = 0.99f;
522 s = helios::clamp(0.5f + 0.5f * (smooth_min - Wp) / smooth_factor, 0.0f, 1.0f);
523 smooth_min = smooth_min * (1.f - s) + Wp * s - smooth_factor * s * (1.f - s);
524 }
525 float A = (1.f - Gamma_star / Ci) * smooth_min - Rd;
526
527
528 float limitation_state;
529 if (Wj > Wc) { // Rubisco limited
530 limitation_state = 0;
531 } else { // Electron transport limited
532 limitation_state = 1;
533 }
534
535 //--- Calculate error and update --- //
536
537 float resid = 0.75f * gM * (CO2 - Ci) - A;
538
539 variables[4] = A;
540 variables[5] = limitation_state;
541
542 float Gamma = (Gamma_star + Kco * Rd / Vcmax) / (1.f - Rd / Vcmax); // Equation 39 of Farquhar et al. (1980)
543 variables[6] = Gamma;
544
545 return resid;
546}

549float PhotosynthesisModel::respondToTemperature(const PhotosyntheticTemperatureResponseParameters *params, float T_in_Kelvin) {
550 float T = T_in_Kelvin;
551 float R = 0.0083144598f;
552 float v25 = params->value_at_25C;
553 float dHa = params->dHa;
554 float dHd = params->dHd;
555 float Topt = params->Topt;
556 if (dHa == 0) {
557 return v25;
558 } else {
559 float logterm = logf(dHd / dHa - 1.f);
560 float t1 = 1.f + expf(dHd / R * (1.f / Topt - 1.f / 298.15f) - logterm);
561 float t2 = 1.f + expf(dHd / R * (1.f / Topt - 1.f / T) - logterm);
562 return v25 * exp(dHa / R * (1.f / 298.15f - 1.f / T)) * t1 / t2;
563 }
564}

567float PhotosynthesisModel::evaluateFarquharModel(const FarquharModelCoefficients &params, float i_PAR, float TL, float CO2, float gM, float &Ci, float &Gamma, int &limitation_state, int &TPU_flag) {
568
569 float A = 0;
570 std::vector<float> variables{CO2, i_PAR, TL, gM, A, float(limitation_state), Gamma, float(TPU_flag)};
571
572 // Temporal continuity approach: use previous Ci as initial guess, with fallbacks
573 // This leverages the fact that Ci changes slowly between timesteps
574 std::vector<float> initial_guesses;
575
576 // First try previous Ci if available (temporal continuity)
577 if (Ci > 0 && std::isfinite(Ci)) {
578 initial_guesses.push_back(Ci);
579 }
580
581 // Add fallback guesses based on research (physiologically reasonable values)
582 initial_guesses.insert(initial_guesses.end(), {CO2 * 0.7f, 100.0f, CO2 * 0.3f, CO2 * 0.9f, 50.0f});
583
584 bool overall_converged = false;
585 for (float guess : initial_guesses) {
586 // Reset variables for each attempt
587 std::vector<float> vars_attempt = variables;
588 bool attempt_converged = false;
589 float Ci_attempt = fzero(evaluateCi_Farquhar, vars_attempt, &farquharmodelcoeffs, guess, attempt_converged, 0.001f, 200);
590
591 // Check if solution converged and is physiologically reasonable
592 if (attempt_converged && Ci_attempt > 0 && std::isfinite(Ci_attempt)) {
593 Ci = Ci_attempt;
594 variables = vars_attempt;
595 overall_converged = true;
596 break;
597 }
598 }
599
600 // If all initial guesses failed, issue a warning message that will be captured in tests
601 if (!overall_converged) {
602 std::cerr << "WARNING: Photosynthesis model failed to converge for Ci calculation after trying multiple initial guesses.\n";
603 // Use best available estimate from last attempt
604 bool final_converged = false;
605 Ci = fzero(evaluateCi_Farquhar, variables, &farquharmodelcoeffs, 100.0f, final_converged, 0.01f, 500);
606 }
607
608 A = variables[4];
609 limitation_state = (int) variables[5];
610 Gamma = variables[6];
611
612 return A;
613}
