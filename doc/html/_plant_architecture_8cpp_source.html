22float PlantArchitecture::interpolateTube(const std::vector<float> &P, const float frac) {
23 assert(frac >= 0 && frac <= 1);
24 assert(!P.empty());
25
26 float dl = 1.f / float(P.size());
27
28 float f = 0;
29 for (int i = 0; i < P.size() - 1; i++) {
30 float fplus = f + dl;
31
32 if (fplus >= 1.f) {
33 fplus = 1.f + 1e-3;
34 }
35
36 if (frac >= f && (frac <= fplus || fabs(frac - fplus) < 0.0001)) {
37 float V = P.at(i) + (frac - f) / (fplus - f) * (P.at(i + 1) - P.at(i));
38
39 return V;
40 }
41
42 f = fplus;
43 }
44
45 return P.front();
46}

48vec3 PlantArchitecture::interpolateTube(const std::vector<vec3> &P, const float frac) {
49 assert(frac >= 0 && frac <= 1);
50 assert(!P.empty());
51
52 float dl = 0.f;
53 for (int i = 0; i < P.size() - 1; i++) {
54 dl += (P.at(i + 1) - P.at(i)).magnitude();
55 }
56
57 float f = 0;
58 for (int i = 0; i < P.size() - 1; i++) {
59 float dseg = (P.at(i + 1) - P.at(i)).magnitude();

290helios::vec3 Phytomer::getAxisVector(const float stem_fraction, const std::vector<helios::vec3> &axis_vertices) {
291 assert(stem_fraction >= 0 && stem_fraction <= 1);
292
293 float df = 0.1f;
294 float frac_plus, frac_minus;
295 if (stem_fraction + df <= 1) {
296 frac_minus = stem_fraction;
297 frac_plus = stem_fraction + df;
298 } else {
299 frac_minus = stem_fraction - df;
300 frac_plus = stem_fraction;
301 }
302
303 const vec3 node_minus = PlantArchitecture::interpolateTube(axis_vertices, frac_minus);
304 const vec3 node_plus = PlantArchitecture::interpolateTube(axis_vertices, frac_plus);
305
306 vec3 norm = node_plus - node_minus;
307 norm.normalize();
308
309 return norm;
310}

396void Phytomer::setFloralBudState(BudState state, FloralBud &fbud) {
397 // If state is already at the desired state, do nothing
398 if (fbud.state == state) {
399 return;
400 } else if (state == BUD_DORMANT || state == BUD_ACTIVE) {
401 fbud.state = state;
402 return;
403 }
404
405 // Calculate carbon cost
406 if (plantarchitecture_ptr->carbon_model_enabled) {
407 if (state == BUD_FLOWER_CLOSED || (fbud.state == BUD_ACTIVE && state == BUD_FLOWER_OPEN)) { // state went from active to closed flower or open flower
408 float flower_cost = calculateFlowerConstructionCosts(fbud);
409 plantarchitecture_ptr->plant_instances.at(this->plantID).shoot_tree.at(this->parent_shoot_ID)->carbohydrate_pool_molC -= flower_cost;
410 } else if (state == BUD_FRUITING) { // adding a fruit
411 float fruit_cost = calculateFruitConstructionCosts(fbud);
412 fbud.previous_fruit_scale_factor = fbud.current_fruit_scale_factor;
413 if (plantarchitecture_ptr->plant_instances.at(this->plantID).shoot_tree.at(this->parent_shoot_ID)->carbohydrate_pool_molC > fruit_cost) {
414 plantarchitecture_ptr->plant_instances.at(this->plantID).shoot_tree.at(this->parent_shoot_ID)->carbohydrate_pool_molC -= fruit_cost;
415 } else {
416 setFloralBudState(BUD_DEAD, fbud);
417 }
418 }
419 }
420
421 // Delete geometry from previous reproductive state (if present)
422 context_ptr->deleteObject(fbud.inflorescence_objIDs);
423 fbud.inflorescence_objIDs.resize(0);
424 fbud.inflorescence_bases.resize(0);
425
426 if (plantarchitecture_ptr->build_context_geometry_peduncle) {
427 context_ptr->deleteObject(fbud.peduncle_objIDs);
428 fbud.peduncle_objIDs.resize(0);
429 }
430
431 fbud.state = state;
432
433 if (state != BUD_DEAD) { // add new reproductive organs
434
435 updateInflorescence(fbud);
436 fbud.time_counter = 0;
437 if (fbud.state == BUD_FRUITING) {
438 setInflorescenceScaleFraction(fbud, 0.25);
439 }
440 }

922 // initialize leaf variables
923 leaf_bases.resize(phytomer_parameters.petiole.petioles_per_internode);
924 leaf_objIDs.resize(phytomer_parameters.petiole.petioles_per_internode);

931 leaf_rotation.at(petiole).resize(leaves_per_petiole);

976 }

979

983 // }

1001 if ((fabs(parent_shoot->gravitropic_curvature) > 0 || parent_shoot_parameters.tortuosity.val() > 0) && shoot_index.x > 0) {

1016 float yaw_angle = deg2rad((parent_shoot->yaw_perturbation));

1029 } else { // if not the first phytomer on shoot, don't insert the first node because it's already defined on the previous phytomer

1031 parent_shoot_ptr->shoot_internode_radii.emplace_back(phytomer_internode_radii.begin() + 1, phytomer_internode_radii.end());
1032 }

1035 if (build_context_geometry_internode) {

1038 float length = 0; // shoot length prior to this phytomer
1039 for (auto &phytomer: parent_shoot_ptr->phytomers) {
1040 length += phytomer->internode_length_max;
1041 }
1042 std::vector<float> uv_y(phytomer_internode_vertices.size());
1043 float dy = internode_length_max / float(uv_y.size() - 1);
1044 for (int j = 0; j < uv_y.size(); j++) {
1045 uv_y.at(j) = (length + j * dy) / texture_repeat_length - std::floor((length + j * dy) / texture_repeat_length);
1046 }
1047
1048 if (!context_ptr->doesObjectExist(parent_shoot->internode_tube_objID)) { // first internode on shoot
1049 if (!phytomer_parameters.internode.image_texture.empty()) {
1050 parent_shoot->internode_tube_objID = context_ptr->addTubeObject(Ndiv_internode_radius, phytomer_internode_vertices, phytomer_internode_radii, phytomer_parameters.internode.image_texture.c_str(), uv_y);
1051 } else {
1052 parent_shoot->internode_tube_objID = context_ptr->addTubeObject(Ndiv_internode_radius, phytomer_internode_vertices, phytomer_internode_radii, internode_colors);
1053 }
1054 context_ptr->setPrimitiveData(context_ptr->getObjectPrimitiveUUIDs(parent_shoot->internode_tube_objID), "object_label", "shoot");
1055 } else { // appending internode to shoot
1056 for (int inode_segment = 1; inode_segment <= Ndiv_internode_length; inode_segment++) {
1057 if (!phytomer_parameters.internode.image_texture.empty()) {
1058 context_ptr->appendTubeSegment(parent_shoot->internode_tube_objID, phytomer_internode_vertices.at(inode_segment), phytomer_internode_radii.at(inode_segment), phytomer_parameters.internode.image_texture.c_str(),
1059 {uv_y.at(inode_segment - 1), uv_y.at(inode_segment)});
1060 } else {
1061 context_ptr->appendTubeSegment(parent_shoot->internode_tube_objID, phytomer_internode_vertices.at(inode_segment), phytomer_internode_radii.at(inode_segment), internode_colors.at(inode_segment));
1062 }
1063 }
1064 context_ptr->setPrimitiveData(context_ptr->getObjectPrimitiveUUIDs(parent_shoot->internode_tube_objID), "object_label", "shoot");
1065 }
1066 }
1067
1068 //--- create petiole ---//
1069
1070 for (int petiole = 0; petiole < phytomer_parameters.petiole.petioles_per_internode; petiole++) { // looping over petioles
1071
1072 vec3 petiole_axis = internode_axis;
1073
1074 // petiole pitch rotation
1075 petiole_pitch.at(petiole) = deg2rad(phytomer_parameters.petiole.pitch.val());
1076 phytomer_parameters.petiole.pitch.resample();
1077 if (fabs(petiole_pitch.at(petiole)) < deg2rad(5.f)) {
1078 petiole_pitch.at(petiole) = deg2rad(5.f);
1079 }
1080 petiole_axis = rotatePointAboutLine(petiole_axis, nullorigin, petiole_rotation_axis, std::abs(petiole_pitch.at(petiole)));
1081
1082 // petiole yaw rotation
1083 if (phytomer_index != 0 && internode_phyllotactic_angle != 0) { // not first phytomer along shoot
1084 petiole_axis = rotatePointAboutLine(petiole_axis, nullorigin, internode_axis, internode_phyllotactic_angle);
1085 petiole_rotation_axis = rotatePointAboutLine(petiole_rotation_axis, nullorigin, internode_axis, internode_phyllotactic_angle);
1086 }
1087
1088 // petiole curvature
1089 petiole_curvature.at(petiole) = phytomer_parameters.petiole.curvature.val();
1090 phytomer_parameters.petiole.curvature.resample();
1091
1092 vec3 petiole_rotation_axis_actual = petiole_rotation_axis;
1093 vec3 petiole_axis_actual = petiole_axis;
1094
1095 if (petiole > 0) {
1096 float budrot = float(petiole) * 2.f * PI_F / float(phytomer_parameters.petiole.petioles_per_internode);
1097 petiole_axis_actual = rotatePointAboutLine(petiole_axis_actual, nullorigin, internode_axis, budrot);
1098 petiole_rotation_axis_actual = rotatePointAboutLine(petiole_rotation_axis_actual, nullorigin, internode_axis, budrot);
1099 }
1100
1101 petiole_vertices.at(petiole).at(0) = phytomer_internode_vertices.back();
1102
1103 for (int j = 1; j <= Ndiv_petiole_length; j++) {
1104 if (fabs(petiole_curvature.at(petiole)) > 0) {
1105 petiole_axis_actual = rotatePointAboutLine(petiole_axis_actual, nullorigin, petiole_rotation_axis_actual, -deg2rad(petiole_curvature.at(petiole) * dr_petiole_max.at(petiole)));
1106 }
1107
1108 petiole_vertices.at(petiole).at(j) = petiole_vertices.at(petiole).at(j - 1) + dr_petiole.at(petiole) * petiole_axis_actual;
1109
1110 petiole_radii.at(petiole).at(j) = leaf_scale_factor_fraction * phytomer_parameters.petiole.radius.val() * (1.f - phytomer_parameters.petiole.taper.val() / float(Ndiv_petiole_length) * float(j));
1111 petiole_colors.at(j) = phytomer_parameters.petiole.color;
1112
1113 assert( !std::isnan(petiole_vertices.at(petiole).at(j).x) && std::isfinite(petiole_vertices.at(petiole).at(j).x) );
1114 assert( !std::isnan(petiole_radii.at(petiole).at(j)) && std::isfinite(petiole_radii.at(petiole).at(j)) );
1115 }
1116
1117 if (build_context_geometry_petiole && petiole_radii.at(petiole).front() > 0.f) {
1118 petiole_objIDs.at(petiole) = makeTubeFromCones(Ndiv_petiole_radius, petiole_vertices.at(petiole), petiole_radii.at(petiole), petiole_colors, context_ptr);
1119 context_ptr->setPrimitiveData(context_ptr->getObjectPrimitiveUUIDs(petiole_objIDs.at(petiole)), "object_label", "petiole");
1120 }
1121
1122 //--- create buds ---//
1123
1124 std::vector<VegetativeBud> vegetative_buds_new;
1125 vegetative_buds_new.resize(phytomer_parameters.internode.max_vegetative_buds_per_petiole.val());
1126 phytomer_parameters.internode.max_vegetative_buds_per_petiole.resample();
1127
1128 axillary_vegetative_buds.push_back(vegetative_buds_new);
1129
1130 std::vector<FloralBud> floral_buds_new;
1131 floral_buds_new.resize(phytomer_parameters.internode.max_floral_buds_per_petiole.val());
1132 phytomer_parameters.internode.max_floral_buds_per_petiole.resample();
1133
1134 uint index = 0;
1135 for (auto &fbud: floral_buds_new) {
1136 fbud.bud_index = index;
1137 fbud.parent_index = petiole;
1138 float pitch_adjustment = fbud.bud_index * 0.1f * PI_F / float(axillary_vegetative_buds.size());
1139 float yaw_adjustment = -0.25f * PI_F + fbud.bud_index * 0.5f * PI_F / float(axillary_vegetative_buds.size());
1140 fbud.base_rotation = make_AxisRotation(pitch_adjustment, yaw_adjustment, 0);
1141 fbud.base_position = phytomer_internode_vertices.back();
1142 fbud.bending_axis = shoot_bending_axis;
1143 index++;
1144 }
1145
1146 floral_buds.push_back(floral_buds_new);
1147
1148 //--- create leaves ---//
1149
1150 if (phytomer_parameters.leaf.prototype.prototype_function == nullptr) {
1151 helios_runtime_error("ERROR (PlantArchitecture::Phytomer): Leaf prototype function was not defined for shoot type " + parent_shoot->shoot_type_label + ".");
1152 }
1153
1154 vec3 petiole_tip_axis = getPetioleAxisVector(1.f, petiole);
1155
1156 // Create unique leaf prototypes for each shoot type so we can simply copy them for each leaf
1157 assert(phytomer_parameters.leaf.prototype.unique_prototype_identifier != 0);
1158 if (phytomer_parameters.leaf.prototype.unique_prototypes > 0 &&
1159 plantarchitecture_ptr->unique_leaf_prototype_objIDs.find(phytomer_parameters.leaf.prototype.unique_prototype_identifier) == plantarchitecture_ptr->unique_leaf_prototype_objIDs.end()) {
1160 plantarchitecture_ptr->unique_leaf_prototype_objIDs[phytomer_parameters.leaf.prototype.unique_prototype_identifier].resize(phytomer_parameters.leaf.prototype.unique_prototypes);
1161 for (int prototype = 0; prototype < phytomer_parameters.leaf.prototype.unique_prototypes; prototype++) {
1162 for (int leaf = 0; leaf < leaves_per_petiole; leaf++) {
1163 float ind_from_tip = float(leaf) - float(leaves_per_petiole - 1) / 2.f;
1164 uint objID_leaf = phytomer_parameters.leaf.prototype.prototype_function(context_ptr, &phytomer_parameters.leaf.prototype, ind_from_tip);
1165 if (phytomer_parameters.leaf.prototype.prototype_function == GenericLeafPrototype) {
1166 context_ptr->setPrimitiveData(context_ptr->getObjectPrimitiveUUIDs(objID_leaf), "object_label", "leaf");
1167 }
1168 plantarchitecture_ptr->unique_leaf_prototype_objIDs.at(phytomer_parameters.leaf.prototype.unique_prototype_identifier).at(prototype).push_back(objID_leaf);
1169 std::vector<uint> petiolule_UUIDs = context_ptr->filterPrimitivesByData(context_ptr->getObjectPrimitiveUUIDs(objID_leaf), "object_label", "petiolule");
1170 context_ptr->setPrimitiveColor(petiolule_UUIDs, phytomer_parameters.petiole.color);
1171 context_ptr->hideObject(objID_leaf);
1172 }
1173 }
1174 }
1175
1176 for (int leaf = 0; leaf < leaves_per_petiole; leaf++) {
1177 float ind_from_tip = float(leaf) - float(leaves_per_petiole - 1) / 2.f;
1178
1179 uint objID_leaf;
1180 if (phytomer_parameters.leaf.prototype.unique_prototypes > 0) { // copy the existing prototype
1181 int prototype = context_ptr->randu(0, phytomer_parameters.leaf.prototype.unique_prototypes - 1);
1182 assert(plantarchitecture_ptr->unique_leaf_prototype_objIDs.find(phytomer_parameters.leaf.prototype.unique_prototype_identifier) != plantarchitecture_ptr->unique_leaf_prototype_objIDs.end());

1185 objID_leaf = context_ptr->copyObject(plantarchitecture_ptr->unique_leaf_prototype_objIDs.at(phytomer_parameters.leaf.prototype.unique_prototype_identifier).at(prototype).at(leaf));
1186 } else { // load a new prototype
1187 objID_leaf = phytomer_parameters.leaf.prototype.prototype_function(context_ptr, &phytomer_parameters.leaf.prototype, ind_from_tip);
1188 }
1189
1190 // -- leaf scaling -- //
1191
1192 if (leaves_per_petiole > 0 && phytomer_parameters.leaf.leaflet_scale.val() != 1.f && ind_from_tip != 0) {
1193 leaf_size_max.at(petiole).at(leaf) = powf(phytomer_parameters.leaf.leaflet_scale.val(), fabs(ind_from_tip)) * phytomer_parameters.leaf.prototype_scale.val();
1194 } else {
1195 leaf_size_max.at(petiole).at(leaf) = phytomer_parameters.leaf.prototype_scale.val();
1196 }
1197 vec3 leaf_scale = leaf_scale_factor_fraction * leaf_size_max.at(petiole).at(leaf) * make_vec3(1, 1, 1);
1198
1199 context_ptr->scaleObject(objID_leaf, leaf_scale);
1200
1201 float compound_rotation = 0;
1202 if (leaves_per_petiole > 1) {
1203 if (phytomer_parameters.leaf.leaflet_offset.val() == 0) {
1204 float dphi = PI_F / (floor(0.5 * float(leaves_per_petiole - 1)) + 1);
1205 compound_rotation = -float(PI_F) + dphi * (leaf + 0.5f);
1206 } else {
1207 if (leaf == float(leaves_per_petiole - 1) / 2.f) { // tip leaf
1208 compound_rotation = 0;
1209 } else if (leaf < float(leaves_per_petiole - 1) / 2.f) {
1210 compound_rotation = -0.5 * PI_F;
1211 } else {
1212 compound_rotation = 0.5 * PI_F;
1213 }
1214 }
1215 }
1216
1217 // -- leaf rotations -- //
1218
1219 // leaf roll rotation
1220 float roll_rot = 0;
1221 if (leaves_per_petiole == 1) {
1222 int sign = (shoot_index.x % 2 == 0) ? 1 : -1;
1223 roll_rot = (acos_safe(internode_axis.z) - deg2rad(phytomer_parameters.leaf.roll.val())) * sign;
1224 } else if (ind_from_tip != 0) {
1225 roll_rot = (asin_safe(petiole_tip_axis.z) + deg2rad(phytomer_parameters.leaf.roll.val())) * compound_rotation / std::fabs(compound_rotation);
1226 }
1227 leaf_rotation.at(petiole).at(leaf).roll = deg2rad(phytomer_parameters.leaf.roll.val());
1228 phytomer_parameters.leaf.roll.resample();
1229 context_ptr->rotateObject(objID_leaf, roll_rot, "x");
1230
1231 // leaf pitch rotation
1232 leaf_rotation.at(petiole).at(leaf).pitch = deg2rad(phytomer_parameters.leaf.pitch.val());
1233 float pitch_rot = leaf_rotation.at(petiole).at(leaf).pitch;
1234 phytomer_parameters.leaf.pitch.resample();
1235 if (ind_from_tip == 0) {
1236 pitch_rot += asin_safe(petiole_tip_axis.z);
1237 }
1238 context_ptr->rotateObject(objID_leaf, -pitch_rot, "y");
1239
1240 // leaf yaw rotation
1241 if (ind_from_tip != 0) {
1242 float sign = -compound_rotation / fabs(compound_rotation);
1243 leaf_rotation.at(petiole).at(leaf).yaw = sign * deg2rad(phytomer_parameters.leaf.yaw.val());
1244 float yaw_rot = leaf_rotation.at(petiole).at(leaf).yaw;
1245 phytomer_parameters.leaf.yaw.resample();
1246 context_ptr->rotateObject(objID_leaf, yaw_rot, "z");
1247 } else {
1248 leaf_rotation.at(petiole).at(leaf).yaw = 0;
1249 }
1250
1251 // rotate leaf to azimuth of petiole
1252 context_ptr->rotateObject(objID_leaf, -std::atan2(petiole_tip_axis.y, petiole_tip_axis.x) + compound_rotation, "z");
1253
1254
1255 // -- leaf translation -- //
1256
1257 vec3 leaf_base = petiole_vertices.at(petiole).back();
1258 if (leaves_per_petiole > 1 && phytomer_parameters.leaf.leaflet_offset.val() > 0) {
1259 if (ind_from_tip != 0) {
1260 float offset = (fabs(ind_from_tip) - 0.5f) * phytomer_parameters.leaf.leaflet_offset.val() * phytomer_parameters.petiole.length.val();
1261 leaf_base = PlantArchitecture::interpolateTube(petiole_vertices.at(petiole), 1.f - offset / phytomer_parameters.petiole.length.val());
1262 }
1263 }
1264
1265 context_ptr->translateObject(objID_leaf, leaf_base);
1266
1267 leaf_objIDs.at(petiole).push_back(objID_leaf);
1268 leaf_bases.at(petiole).push_back(leaf_base);
1269 }
1270 phytomer_parameters.leaf.prototype_scale.resample();
1271
1272 if (petiole_axis_actual == make_vec3(0, 0, 1)) {
1273 inflorescence_bending_axis = make_vec3(1, 0, 0);
1274 } else {
1275 inflorescence_bending_axis = cross(make_vec3(0, 0, 1), petiole_axis_actual);
1276 }
1277 }
1278}

1280float Phytomer::calculatePhytomerVolume(uint node_number) const {
1281 // Get the radii of this phytomer from the parent shoot
1282 const auto &segment = parent_shoot_ptr->shoot_internode_radii.at(node_number);
1283
1284 // Find the average radius
1285 float avg_radius = 0.0f;
1286 for (float radius: segment) {
1287 avg_radius += radius;
1288 }
1289 avg_radius /= scast<float>(segment.size());
1290
1291 // Get the length of the phytomer
1292 float length = getInternodeLength();
1293
1294 // Calculate the volume of the cylinder
1295 float volume = PI_F * avg_radius * avg_radius * length;
1296
1297 return volume;
1298}

1422 objID_fruit = context_ptr->copyObject(plantarchitecture_ptr->unique_open_flower_prototype_objIDs.at(phytomer_parameters.inflorescence.flower_prototype_function).at(prototype));
1423 } else { // load new prototype
1424 objID_fruit = phytomer_parameters.inflorescence.flower_prototype_function(context_ptr, 1, flower_is_open);
1425 }
1426 }
1427 fruit_scale = phytomer_parameters.inflorescence.flower_prototype_scale.val() * make_vec3(1, 1, 1);
1428 phytomer_parameters.inflorescence.flower_prototype_scale.resample();
1429 }
1430
1431 float ind_from_tip = fabs(fruit - float(phytomer_parameters.inflorescence.flowers_per_peduncle.val() - 1) / float(phytomer_parameters.petiole.petioles_per_internode));
1432
1433 context_ptr->scaleObject(objID_fruit, fruit_scale);
1434
1435 // if we have more than one flower/fruit, we need to adjust the base position of the fruit
1436 vec3 fruit_base = peduncle_vertices.back();
1437 float frac = 1;
1438 if (phytomer_parameters.inflorescence.flowers_per_peduncle.val() > 1 && phytomer_parameters.inflorescence.flower_offset.val() > 0) {
1439 if (ind_from_tip != 0) {
1440 float offset = offset = (ind_from_tip - 0.5f) * phytomer_parameters.inflorescence.flower_offset.val() * phytomer_parameters.peduncle.length.val();
1441 if (phytomer_parameters.peduncle.length.val() > 0) {
1442 frac = 1.f - offset / phytomer_parameters.peduncle.length.val();
1443 }

1445 }
1446 }
1447
1448 // if we have more than one flower/fruit, we need to adjust the rotation about the peduncle
1449 float compound_rotation = 0;
1450 if (phytomer_parameters.inflorescence.flowers_per_peduncle.val() > 1) {
1451 if (phytomer_parameters.inflorescence.flower_offset.val() == 0) { // flowers/fruit are all at the tip, so just equally distribute them about the azimuth
1452 float dphi = PI_F / (floor(0.5 * float(phytomer_parameters.inflorescence.flowers_per_peduncle.val() - 1)) + 1);
1453 compound_rotation = -float(PI_F) + dphi * (fruit + 0.5f);
1454 } else {
1455 compound_rotation = deg2rad(phytomer_parameters.internode.phyllotactic_angle.val()) * float(ind_from_tip) + 2.f * PI_F / float(phytomer_parameters.petiole.petioles_per_internode) * float(fruit);
1456 phytomer_parameters.internode.phyllotactic_angle.resample();
1457 }
1458 }
1459
1460 peduncle_axis = getAxisVector(frac, peduncle_vertices);
1461
1462 vec3 fruit_axis = peduncle_axis;
1463
1464 // roll rotation
1465 if (phytomer_parameters.inflorescence.roll.val() != 0.f) {
1466 context_ptr->rotateObject(objID_fruit, deg2rad(phytomer_parameters.inflorescence.roll.val()), "x");
1467 phytomer_parameters.inflorescence.roll.resample();
1468 }
1469
1470 // pitch rotation
1471 float pitch_inflorescence = -asin_safe(peduncle_axis.z) + deg2rad(phytomer_parameters.inflorescence.pitch.val());
1472 phytomer_parameters.inflorescence.pitch.resample();
1473 if (fbud.state == BUD_FRUITING) { // gravity effect for fruit
1474 pitch_inflorescence = pitch_inflorescence + phytomer_parameters.inflorescence.fruit_gravity_factor_fraction.val() * (0.5f * PI_F - pitch_inflorescence);
1475 }
1476 context_ptr->rotateObject(objID_fruit, pitch_inflorescence, "y");
1477 fruit_axis = rotatePointAboutLine(fruit_axis, nullorigin, make_vec3(1, 0, 0), pitch_inflorescence);
1478
1479 // rotate flower/fruit to azimuth of peduncle
1480 context_ptr->rotateObject(objID_fruit, -std::atan2(peduncle_axis.y, peduncle_axis.x), "z");
1481 fruit_axis = rotatePointAboutLine(fruit_axis, nullorigin, make_vec3(0, 0, 1), -std::atan2(peduncle_axis.y, peduncle_axis.x));
1482
1483 context_ptr->translateObject(objID_fruit, fruit_base);
1484
1485 // rotate flower/fruit about peduncle (roll)
1486 if (phytomer_parameters.inflorescence.fruit_gravity_factor_fraction.val() != 0 && fbud.state == BUD_FRUITING) {
1487 context_ptr->rotateObject(objID_fruit, deg2rad(phytomer_parameters.peduncle.roll.val()) + compound_rotation, fruit_base, make_vec3(0, 0, 1));
1488 } else {
1489 context_ptr->rotateObject(objID_fruit, deg2rad(phytomer_parameters.peduncle.roll.val()) + compound_rotation, fruit_base, peduncle_axis);
1490 fruit_axis = rotatePointAboutLine(fruit_axis, nullorigin, peduncle_axis, deg2rad(phytomer_parameters.peduncle.roll.val()) + compound_rotation);
1491 }
1492 phytomer_parameters.inflorescence.fruit_gravity_factor_fraction.resample();
1493
1494 fbud.inflorescence_bases.push_back(fruit_base);
1495
1496 fbud.inflorescence_objIDs.push_back(objID_fruit);
1497 }
1498 phytomer_parameters.inflorescence.flowers_per_peduncle.resample();
1499 phytomer_parameters.peduncle.roll.resample();
1500

1502 context_ptr->setObjectData(fbud.peduncle_objIDs, "rank", rank);
1503 context_ptr->setObjectData(fbud.inflorescence_objIDs, "rank", rank);

1508 context_ptr->setObjectData(objID, "peduncleID", (int) objID);
1509 }
1510 }

1514 } else if (fbud.state == BUD_FLOWER_OPEN && plantarchitecture_ptr->output_object_data.at("openflowerID")) {

1516 context_ptr->setObjectData(objID, "openflowerID", (int) objID);

1557void Phytomer::rotateLeaf(uint petiole_index, uint leaf_index, const AxisRotation &rotation) {

1561 helios_runtime_error("ERROR (PlantArchitecture::Phytomer): Invalid leaf index.");
1562 }
1563
1564 vec3 petiole_axis = getPetioleAxisVector(1.f, petiole_index);
1565 // note: this is not exactly correct because it should get the axis at the leaf position and not the tip
1566
1567 vec3 internode_axis = getInternodeAxisVector(1.f);
1568
1569 vec3 pitch_axis = -1 * cross(internode_axis, petiole_axis);
1570
1571 int leaves_per_petiole = leaf_rotation.at(petiole_index).size();

1573 float roll;
1574 float compound_rotation = 0;
1575 if (leaves_per_petiole > 1 && leaf_index == float(leaves_per_petiole - 1) / 2.f) { // tip leaflet of compound leaf
1576 roll = 0;
1577 yaw = 0;
1578 compound_rotation = 0;
1579 } else if (leaves_per_petiole > 1 && leaf_index < float(leaves_per_petiole - 1) / 2.f) { // lateral leaflet of compound leaf

1581 roll = -rotation.roll;
1582 compound_rotation = -0.5 * PI_F;
1583 } else { // not a compound leaf

1598 pitch_axis = rotatePointAboutLine(pitch_axis, nullorigin, {0, 0, 1}, -compound_rotation);
1599 context_ptr->rotateObject(leaf_objIDs.at(petiole_index).at(leaf_index), rotation.pitch, leaf_bases.at(petiole_index).at(leaf_index), pitch_axis);
1600 leaf_rotation.at(petiole_index).at(leaf_index).pitch += rotation.pitch;
1601 }

1605 context_ptr->rotateObject(leaf_objIDs.at(petiole_index).at(leaf_index), yaw, leaf_bases.at(petiole_index).at(leaf_index), {0, 0, 1});

1610void Phytomer::setInternodeLengthScaleFraction(const float internode_scale_factor_fraction, const bool update_context_geometry) {
1611 assert(internode_scale_factor_fraction >= 0 && internode_scale_factor_fraction <= 1);
1612
1613 if (internode_scale_factor_fraction == current_internode_scale_factor) {
1614 return;
1615 }
1616
1617 float delta_scale = internode_scale_factor_fraction / current_internode_scale_factor;
1618
1619 current_internode_scale_factor = internode_scale_factor_fraction;
1620
1621 int p = shoot_index.x;
1622 int s_start = (p == 0) ? 1 : 0; // skip the first node at the base of the shoot
1623
1624 for (int s = s_start; s < parent_shoot_ptr->shoot_internode_vertices.at(p).size(); s++) { // looping over all segments within this phytomer internode
1625
1626 int p_minus = p;
1627 int s_minus = s - 1;
1628 if (s_minus < 0) {
1629 p_minus--;
1630 s_minus = static_cast<int>(parent_shoot_ptr->shoot_internode_vertices.at(p_minus).size() - 1);
1631 }
1632
1633 vec3 central_axis = (parent_shoot_ptr->shoot_internode_vertices.at(p).at(s) - parent_shoot_ptr->shoot_internode_vertices.at(p_minus).at(s_minus));
1634 float current_length = central_axis.magnitude();
1635 central_axis = central_axis / current_length;
1636 vec3 dL = central_axis * current_length * (delta_scale - 1);
1637
1638 // apply shift to all downstream nodes
1639 for (int p_downstream = p; p_downstream < parent_shoot_ptr->shoot_internode_vertices.size(); p_downstream++) {
1640 int sd_start = (p_downstream == p) ? s : 0;
1641 for (int s_downstream = sd_start; s_downstream < parent_shoot_ptr->shoot_internode_vertices.at(p_downstream).size(); s_downstream++) {
1642 parent_shoot_ptr->shoot_internode_vertices.at(p_downstream).at(s_downstream) += dL;
1643 }
1644 }
1645 }
1646
1647 parent_shoot_ptr->updateShootNodes(update_context_geometry);

1668void Phytomer::setLeafScaleFraction(uint petiole_index, float leaf_scale_factor_fraction) {
1669
1670 assert(leaf_scale_factor_fraction >= 0 && leaf_scale_factor_fraction <= 1);
1671
1672 if (current_leaf_scale_factor.size() <= petiole_index) {
1673 helios_runtime_error("ERROR (PlantArchitecture::Phytomer): Invalid petiole index for leaf scale factor.");
1674 }
1675
1676 // If the leaf is already at leaf_scale_factor_fraction, or there are no petioles/leaves, nothing to do.
1677 if (leaf_scale_factor_fraction == current_leaf_scale_factor.at(petiole_index) || (leaf_objIDs.at(petiole_index).empty() && petiole_objIDs.at(petiole_index).empty())) {
1678 return;
1679 }

1682

1684
1685 current_leaf_scale_factor.at(petiole_index) = leaf_scale_factor_fraction;
1686
1687 assert(leaf_objIDs.size() == leaf_bases.size());
1688
1689 // scale the petiole
1690
1691 if (!petiole_objIDs.at(petiole_index).empty()) {
1692 int node = 0;
1693 vec3 last_base = petiole_vertices.at(petiole_index).front(); // looping over petioles
1694 for (uint objID: petiole_objIDs.at(petiole_index)) { // looping over cones/segments within petiole
1695 context_ptr->getConeObjectPointer(objID)->scaleLength(delta_scale);
1696 context_ptr->getConeObjectPointer(objID)->scaleGirth(delta_scale);
1697 petiole_radii.at(petiole_index).at(node) *= delta_scale;
1698 if (node > 0) {

1700 context_ptr->translateObject(objID, last_base - new_base);
1701 } else {
1702 petiole_vertices.at(petiole_index).at(0) = context_ptr->getConeObjectNode(objID, 0);
1703 }
1704 last_base = context_ptr->getConeObjectNode(objID, 1);
1705 petiole_vertices.at(petiole_index).at(node + 1) = last_base;
1706 node++;
1707 }
1708 }
1709
1710 // scale and translate leaves
1711 assert(leaf_objIDs.at(petiole_index).size() == leaf_bases.at(petiole_index).size());

1714
1715 context_ptr->translateObject(leaf_objIDs.at(petiole_index).at(leaf), -1 * leaf_bases.at(petiole_index).at(leaf));
1716 context_ptr->scaleObject(leaf_objIDs.at(petiole_index).at(leaf), delta_scale * make_vec3(1, 1, 1));
1717 if (ind_from_tip == 0) {
1718 context_ptr->translateObject(leaf_objIDs.at(petiole_index).at(leaf), petiole_vertices.at(petiole_index).back());
1719 leaf_bases.at(petiole_index).at(leaf) = petiole_vertices.at(petiole_index).back();
1720 } else {
1721 float offset = (fabs(ind_from_tip) - 0.5f) * phytomer_parameters.leaf.leaflet_offset.val() * phytomer_parameters.petiole.length.val();
1722 vec3 leaf_base = PlantArchitecture::interpolateTube(petiole_vertices.at(petiole_index), 1.f - offset / phytomer_parameters.petiole.length.val());
1723 context_ptr->translateObject(leaf_objIDs.at(petiole_index).at(leaf), leaf_base);
1724 leaf_bases.at(petiole_index).at(leaf) = leaf_base;
1725 }
1726 }

2866// Rodrigues formula: rotate v about unit‐axis k by angle α
2867static vec3 rodrigues(const vec3 &v, const vec3 &k, float a) {
2868 float c = std::cos(a);
2869 float s = std::sin(a);
2870 // dot = k·v
2871 float kv = k * v;
2872 return v * c + cross(k, v) * s + k * (kv * (1.f - c));
