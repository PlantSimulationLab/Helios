21helios::vec3 interpolateTube( const std::vector<vec3> &P, float frac ){
22
23 assert( frac>=0 && frac<=1 );
24 assert( !P.empty() );
25
26 float dl=0.f;
27 for( int i=0; i<P.size()-1; i++ ){
28 dl+=(P.at(i+1)-P.at(i)).magnitude();
29 }
30
31 float f = 0;
32 for( int i=0; i<P.size()-1; i++ ){
33
34 float dseg = (P.at(i+1)-P.at(i)).magnitude();
35
36 float fplus = f+dseg/dl;
37
38 if( fplus>=1.f ){
39 fplus = 1.f+1e-3;
40 }
41
42 if( frac>=f && (frac<=fplus || fabs(frac-fplus)<0.0001 ) ){
43
44 vec3 V = P.at(i) + (frac-f)/(fplus-f)*(P.at(i+1)-P.at(i));
45
46 return V;
47 }
48
49 f=fplus;
50 }
51
52 return P.front();
53
54}

115helios::vec3 Phytomer::getAxisVector( float stem_fraction, const std::vector<helios::vec3> &axis_vertices ) const{
116
117 if( stem_fraction<0 || stem_fraction>1 ){
118 std::cerr << "ERROR: Stem fraction must be between 0 and 1." << std::endl;
119 }
120
121 float df = 0.1f;
122 float frac_plus, frac_minus;
123 if( stem_fraction+df<=1 ) {
124 frac_minus = stem_fraction;
125 frac_plus = stem_fraction + df;
126 }else{
127 frac_minus = stem_fraction-df;
128 frac_plus = stem_fraction;
129 }
130
131 vec3 node_minus = interpolateTube( axis_vertices, frac_minus );
132 vec3 node_plus = interpolateTube( axis_vertices, frac_plus );
133
134 vec3 norm = node_plus-node_minus;
135 norm.normalize();
136
137 return norm;

231

277 petiole_axis = rotatePointAboutLine(petiole_axis, petiole_vertices.at(j-1), petiole_rotation_axis, deg2rad(phytomer_parameters.petiole.curvature.val()*dr_petiole) );

291

298 uint objID_leaf = phytomer_parameters.leaf.prototype_function(context_ptr,1,0);

303

305 if( phytomer_parameters.leaf.leaves_per_petiole>0 && phytomer_parameters.leaf.leaflet_scale!=1.f && ind_from_tip!=0 ){

307 }

309 context_ptr->scaleObject( objID_leaf, leaf_scale );
310
311 float compound_rotation = 0;
312 if( phytomer_parameters.leaf.leaves_per_petiole>1 ) {
313 if (phytomer_parameters.leaf.leaflet_offset == 0) {
314 float dphi = M_PI / (floor(0.5 * float(phytomer_parameters.leaf.leaves_per_petiole - 1)) + 1);
315 compound_rotation = -float(M_PI) + dphi * (leaf + 0.5f);
316 } else {
317 if( leaf == float(phytomer_parameters.leaf.leaves_per_petiole-1)/2.f ){ //tip leaf
318 compound_rotation = 0;
319 }else if( leaf < float(phytomer_parameters.leaf.leaves_per_petiole-1)/2.f ) {

332 //pitch rotation
333 float pitch_rot = phytomer_parameters.leaf.rotation.pitch;
334 if( ind_from_tip==0 ){
335 pitch_rot += asin_safe(petiole_tip_axis.z);
336 }
337 context_ptr->rotateObject(objID_leaf, -pitch_rot , "y" );
338
339 //yaw rotation
340 if( ind_from_tip!=0 ){
341 float yaw_rot = -phytomer_parameters.leaf.rotation.yaw*compound_rotation/fabs(compound_rotation);
342 context_ptr->rotateObject( objID_leaf, yaw_rot, "z" );

344
345 //roll rotation
346 if( ind_from_tip!= 0){
347 float roll_rot = (asin_safe(petiole_tip_axis.z)+phytomer_parameters.leaf.rotation.roll)*compound_rotation/std::fabs(compound_rotation);
348 context_ptr->rotateObject(objID_leaf, roll_rot, "x" );
349 }
350
351 //rotate to azimuth of petiole
352 context_ptr->rotateObject( objID_leaf, -std::atan2(petiole_tip_axis.y, petiole_tip_axis.x)+compound_rotation, "z" );
353
354
355 // -- translation -- //
356
357 vec3 leaf_base = petiole_vertices.back();
358 if( phytomer_parameters.leaf.leaves_per_petiole>1 && phytomer_parameters.leaf.leaflet_offset>0 ){
359 if( ind_from_tip != 0 ) {
360 float offset = (ind_from_tip - 0.5f) * phytomer_parameters.leaf.leaflet_offset * phytomer_parameters.petiole.length;
361 leaf_base = interpolateTube(petiole_vertices, 1.f - offset / phytomer_parameters.petiole.length);
362 }
363 }
364
365 context_ptr->translateObject( objID_leaf, leaf_base );
366
367 UUIDs = context_ptr->getObjectPrimitiveUUIDs( objID_leaf );
368 leaf_UUIDs.insert( leaf_UUIDs.end(), UUIDs.begin(), UUIDs.end() );
369
370 }
371
372 //--- create inflorescence ---//
373
374 if( phytomer_parameters.inflorescence.reproductive_state!=0 ) {
375 vec3 inflorescence_bending_axis;
376 if( petiole_axis==make_vec3(0,0,1) ) {
377 inflorescence_bending_axis = make_vec3(1, 0, 0);
378 }else{
379 inflorescence_bending_axis = cross(make_vec3(0, 0, 1), petiole_axis);
380 }
381 addInfluorescence(internode_vertices.back(), make_AxisRotation(0, 0, 0), inflorescence_bending_axis);
382 }
383
384 }
