22float interpolateTube( const std::vector<float> &P, float frac ){
23
24 assert( frac>=0 && frac<=1 );
25 assert( !P.empty() );
26
27 float dl=1.f/float(P.size());
28
29 float f = 0;
30 for( int i=0; i<P.size()-1; i++ ){
31
32 float fplus = f+dl;
33
34 if( fplus>=1.f ){
35 fplus = 1.f+1e-3;
36 }
37
38 if( frac>=f && (frac<=fplus || fabs(frac-fplus)<0.0001 ) ){
39
40 float V = P.at(i) + (frac-f)/(fplus-f)*(P.at(i+1)-P.at(i));
41
42 return V;
43 }
44
45 f=fplus;
46 }
47
48 return P.front();
49

225helios::vec3 Phytomer::getAxisVector( float stem_fraction, const std::vector<helios::vec3> &axis_vertices ) const{
226
227 assert( stem_fraction>=0 && stem_fraction<=1 );
228
229 float df = 0.1f;
230 float frac_plus, frac_minus;
231 if( stem_fraction+df<=1 ) {
232 frac_minus = stem_fraction;
233 frac_plus = stem_fraction + df;
234 }else{
235 frac_minus = stem_fraction-df;
236 frac_plus = stem_fraction;
237 }
238
239 vec3 node_minus = interpolateTube( axis_vertices, frac_minus );
240 vec3 node_plus = interpolateTube( axis_vertices, frac_plus );
241
242 vec3 norm = node_plus-node_minus;
243 norm.normalize();
244

590 std::vector<float> dr_petiole_max(phytomer_parameters.petiole.petioles_per_internode);
591 for( int p=0; p<phytomer_parameters.petiole.petioles_per_internode; p++ ) {
592
593 petiole_vertices.at(p).resize(Ndiv_petiole_length + 1);
594 petiole_radii.at(p).resize(Ndiv_petiole_length + 1);
595
596 petiole_length.at(p) = leaf_scale_factor_fraction * phytomer_parameters.petiole.length.val();
597 dr_petiole.at(p) = petiole_length.at(p) / float(phytomer_parameters.petiole.length_segments);
598 dr_petiole_max.at(p) = phytomer_parameters.petiole.length.val() / float(phytomer_parameters.petiole.length_segments);
599 phytomer_parameters.petiole.length.resample();
600
601 petiole_radii.at(p).at(0) = leaf_scale_factor_fraction * phytomer_parameters.petiole.radius.val();
602
603 }
604 if( build_context_geometry_petiole ) {
605 petiole_objIDs.resize(phytomer_parameters.petiole.petioles_per_internode);
606 }
607
608 //initialize leaf variables
609 leaf_bases.resize(phytomer_parameters.petiole.petioles_per_internode);
610 leaf_objIDs.resize(phytomer_parameters.petiole.petioles_per_internode);
611 leaf_size_max.resize(phytomer_parameters.petiole.petioles_per_internode);
612 leaf_rotation.resize(phytomer_parameters.petiole.petioles_per_internode);
613 for( uint petiole=0; petiole<phytomer_parameters.petiole.petioles_per_internode; petiole++ ) {
614 leaf_rotation.at(petiole).resize(phytomer_parameters.leaf.leaves_per_petiole);
615 }
616
617 internode_colors.resize(Ndiv_internode_length + 1 );
618 internode_colors.at(0) = phytomer_parameters.internode.color;
619 petiole_colors.resize(Ndiv_petiole_length + 1 );
620 petiole_colors.at(0) = phytomer_parameters.petiole.color;
621
622 vec3 internode_axis = parent_internode_axis;
623
624 vec3 petiole_rotation_axis = cross(parent_internode_axis, parent_petiole_axis );
625 if(petiole_rotation_axis == make_vec3(0, 0, 0) ){
626 petiole_rotation_axis = make_vec3(1, 0, 0);
627 }
628
629 if( phytomer_index==0 ){ //if this is the first phytomer along a shoot, apply the origin rotation about the parent axis)
630
631 //pitch rotation for phytomer base
632 if( internode_pitch!=0.f ) {
633 internode_axis = rotatePointAboutLine(internode_axis, nullorigin, petiole_rotation_axis, 0.5f*internode_pitch );
634 }
635
636 //roll rotation for shoot base rotation
637 float roll_nudge = 0.f;
638 if( shoot_base_rotation.roll/180.f == floor(shoot_base_rotation.roll/180.f) ) {
639 roll_nudge = 0.2;
640 }
641 if( shoot_base_rotation.roll!=0.f || roll_nudge!=0.f ){
642 petiole_rotation_axis = rotatePointAboutLine(petiole_rotation_axis, nullorigin, parent_internode_axis,shoot_base_rotation.roll + roll_nudge ); //small additional rotation is to make sure the petiole is not exactly vertical
643 internode_axis = rotatePointAboutLine(internode_axis, nullorigin, parent_internode_axis,shoot_base_rotation.roll + roll_nudge );
644 }
645
646 vec3 base_pitch_axis = -1*cross(parent_internode_axis, parent_petiole_axis );
647
648 //pitch rotation for shoot base rotation
649 if( shoot_base_rotation.pitch!=0.f ) {
650 petiole_rotation_axis = rotatePointAboutLine(petiole_rotation_axis, nullorigin, base_pitch_axis, -shoot_base_rotation.pitch);
651 internode_axis = rotatePointAboutLine(internode_axis, nullorigin, base_pitch_axis, -shoot_base_rotation.pitch);
652 }
653
654 //yaw rotation for shoot base rotation
655 if( shoot_base_rotation.yaw!=0 ){
656 petiole_rotation_axis = rotatePointAboutLine(petiole_rotation_axis, nullorigin, parent_internode_axis, shoot_base_rotation.yaw);
657 internode_axis = rotatePointAboutLine(internode_axis, nullorigin, parent_internode_axis, shoot_base_rotation.yaw );
658 }
659
660 }else {
661
662 //pitch rotation for phytomer base
663 if ( internode_pitch != 0) {
664 internode_axis = rotatePointAboutLine(internode_axis, nullorigin, petiole_rotation_axis,-1.25f*internode_pitch );
665 }
666
667 }
668
669 vec3 shoot_bending_axis = cross( internode_axis, make_vec3(0,0,1) );
670
671 internode_axis.normalize();
672 if( internode_axis==make_vec3(0,0,1) ){
673 shoot_bending_axis = make_vec3(0,1,0);
674 }
675
676 // create internode
677 for(int i=1; i <= Ndiv_internode_length; i++ ){
678
679 //apply curvature
680 if( fabs(parent_shoot_parameters.gravitropic_curvature.val()) > 0 ) {
681 float dt = 1.f / float(Ndiv_internode_length);
682 parent_shoot->curvature_perturbation += - 0.5f*parent_shoot->curvature_perturbation*dt + 5*parent_shoot_parameters.tortuosity.val()*context_ptr->randn()*sqrt(dt);
683 float curvature_angle = deg2rad((parent_shoot_parameters.gravitropic_curvature.val()+parent_shoot->curvature_perturbation) * dr_internode_max);
684 internode_axis = rotatePointAboutLine(internode_axis, nullorigin, shoot_bending_axis, curvature_angle);
685 }
686
687 internode_vertices.at(i) = internode_vertices.at(i - 1) + dr_internode * internode_axis;
688
689 internode_radii.at(i) = internode_radius;
690 internode_colors.at(i) = phytomer_parameters.internode.color;
691
692 }
693
694 if( build_context_geometry_internode ) {
695 internode_objIDs = makeTubeFromCones(Ndiv_internode_radius, internode_vertices, internode_radii, internode_colors, context_ptr);
696 }
697
698 //--- create petiole ---//
699
700 vec3 petiole_axis = internode_axis;
701
702 //petiole pitch rotation
703 petiole_pitch = deg2rad(phytomer_parameters.petiole.pitch.val());
704 phytomer_parameters.petiole.pitch.resample();
705 if( fabs(petiole_pitch)<deg2rad(5.f) ) {
706 petiole_pitch = deg2rad(5.f);
707 }
708 petiole_axis = rotatePointAboutLine(petiole_axis, nullorigin, petiole_rotation_axis, std::abs(petiole_pitch) );
709
710 //petiole yaw rotation
711 if( phytomer_index!=0 && internode_phyllotactic_angle!=0 ){ //not first phytomer along shoot
712 petiole_axis = rotatePointAboutLine(petiole_axis, nullorigin, internode_axis, internode_phyllotactic_angle );
713 petiole_rotation_axis = rotatePointAboutLine(petiole_rotation_axis, nullorigin, internode_axis, internode_phyllotactic_angle );
714 }
715
716 for(int petiole=0; petiole < phytomer_parameters.petiole.petioles_per_internode; petiole++ ) { //looping over petioles
717
718 if( petiole > 0 ) {
719 float budrot = float(petiole) * 2.f * M_PI / float(phytomer_parameters.petiole.petioles_per_internode);
720 petiole_axis = rotatePointAboutLine(petiole_axis, nullorigin, internode_axis, budrot );
721 petiole_rotation_axis = rotatePointAboutLine(petiole_rotation_axis, nullorigin, internode_axis, budrot );
722 }
723
724 petiole_vertices.at(petiole).at(0) = internode_vertices.back();
725
726 for (int j = 1; j <= Ndiv_petiole_length; j++) {
727
728 if( fabs(phytomer_parameters.petiole.curvature.val())>0 ) {
729 petiole_axis = rotatePointAboutLine(petiole_axis, nullorigin, petiole_rotation_axis, -deg2rad(phytomer_parameters.petiole.curvature.val() * dr_petiole_max.at(petiole)));
730 phytomer_parameters.petiole.curvature.resample();
731 }
732
733 petiole_vertices.at(petiole).at(j) = petiole_vertices.at(petiole).at(j - 1) + dr_petiole.at(petiole) * petiole_axis;
734
735 petiole_radii.at(petiole).at(j) = leaf_scale_factor_fraction * phytomer_parameters.petiole.radius.val() * (1.f - phytomer_parameters.petiole.taper.val() / float(Ndiv_petiole_length) * float(j) );
736 petiole_colors.at(j) = phytomer_parameters.petiole.color;
737
738 }
739
740 if( build_context_geometry_petiole && petiole_radii.at(petiole).front() > 0.f ) {
741 petiole_objIDs.at(petiole) = makeTubeFromCones(Ndiv_petiole_radius, petiole_vertices.at(petiole), petiole_radii.at(petiole), petiole_colors, context_ptr);
742 }
743
744 //--- create buds ---//
745
746 std::vector<VegetativeBud> vegetative_buds_new;
747 vegetative_buds_new.resize( phytomer_parameters.internode.max_vegetative_buds_per_petiole.val() );
748 phytomer_parameters.internode.max_vegetative_buds_per_petiole.resample();
749
750 vegetative_buds.push_back(vegetative_buds_new);
751
752 std::vector<FloralBud> floral_buds_new;
753 floral_buds_new.resize( phytomer_parameters.internode.max_floral_buds_per_petiole.val() );
754 phytomer_parameters.internode.max_floral_buds_per_petiole.resample();
755
756 floral_buds.push_back(floral_buds_new);
757 resize_vector(inflorescence_objIDs, 0, phytomer_parameters.internode.max_floral_buds_per_petiole.val(), phytomer_parameters.petiole.petioles_per_internode);
758 if( build_context_geometry_peduncle ) {
759 resize_vector(peduncle_objIDs, 0, phytomer_parameters.internode.max_floral_buds_per_petiole.val(), phytomer_parameters.petiole.petioles_per_internode);
760 }
761 resize_vector( inflorescence_bases, 0, phytomer_parameters.internode.max_floral_buds_per_petiole.val(), phytomer_parameters.petiole.petioles_per_internode );
762
763 //--- create leaves ---//
764
765 if( phytomer_parameters.leaf.prototype_function == nullptr ){
766 helios_runtime_error("ERROR (PlantArchitecture::Phytomer): Leaf prototype function was not defined for shoot type " + parent_shoot->shoot_type_label + ".");
767 }
768
769 vec3 petiole_tip_axis = getPetioleAxisVector(1.f, petiole);
770
771 vec3 leaf_rotation_axis = cross(internode_axis, petiole_tip_axis );
772 leaf_size_max.at(petiole) = phytomer_parameters.leaf.prototype_scale.val();
773
774 for(int leaf=0; leaf < phytomer_parameters.leaf.leaves_per_petiole; leaf++ ){
775
776 float ind_from_tip = float(leaf)-float(phytomer_parameters.leaf.leaves_per_petiole-1)/2.f;
777
778 uint objID_leaf = phytomer_parameters.leaf.prototype_function(context_ptr, phytomer_parameters.leaf.subdivisions, ind_from_tip, shoot_index.x, shoot_index.y );
779
780 // -- scaling -- //
781
782 vec3 leaf_scale = leaf_scale_factor_fraction * leaf_size_max.at(petiole) * make_vec3(1,1,1);
783 if( phytomer_parameters.leaf.leaves_per_petiole>0 && phytomer_parameters.leaf.leaflet_scale.val()!=1.f && ind_from_tip!=0 ){
784 leaf_scale = powf(phytomer_parameters.leaf.leaflet_scale.val(),fabs(ind_from_tip))*leaf_scale;
785 }
786
787 context_ptr->scaleObject( objID_leaf, leaf_scale );
788
789 float compound_rotation = 0;
790 if( phytomer_parameters.leaf.leaves_per_petiole>1 ) {
791 if (phytomer_parameters.leaf.leaflet_offset.val() == 0) {
792 float dphi = M_PI / (floor(0.5 * float(phytomer_parameters.leaf.leaves_per_petiole - 1)) + 1);
793 compound_rotation = -float(M_PI) + dphi * (leaf + 0.5f);
794 } else {
795 if( leaf == float(phytomer_parameters.leaf.leaves_per_petiole-1)/2.f ){ //tip leaf
796 compound_rotation = 0;
797 }else if( leaf < float(phytomer_parameters.leaf.leaves_per_petiole-1)/2.f ) {
798 compound_rotation = -0.5*M_PI;
799 }else{
800 compound_rotation = 0.5*M_PI;
801 }
802 }
803 }
804
805 // -- rotations -- //
806
807 //roll rotation
808 float roll_rot = 0;
809 if( phytomer_parameters.leaf.leaves_per_petiole==1 ){
810 int sign = (shoot_index.x%2==0) ? 1 : -1;
811 roll_rot = (acos_safe(internode_axis.z)-deg2rad(phytomer_parameters.leaf.roll.val()))*sign;
812 } else if( ind_from_tip!= 0){
813 roll_rot = (asin_safe(petiole_tip_axis.z)+deg2rad(phytomer_parameters.leaf.roll.val()))*compound_rotation/std::fabs(compound_rotation);
814 }
815 leaf_rotation.at(petiole).at(leaf).roll = roll_rot;
816 phytomer_parameters.leaf.roll.resample();
817 context_ptr->rotateObject(objID_leaf, roll_rot, "x" );
818
819 //pitch rotation
820 leaf_rotation.at(petiole).at(leaf).pitch = deg2rad(phytomer_parameters.leaf.pitch.val());
821 float pitch_rot = leaf_rotation.at(petiole).at(leaf).pitch;
822 phytomer_parameters.leaf.pitch.resample();
823 if( ind_from_tip==0 ){
824 pitch_rot += asin_safe(petiole_tip_axis.z);
825 }
826 context_ptr->rotateObject(objID_leaf, -pitch_rot , "y" );
827
828 //yaw rotation
829 if( ind_from_tip!=0 ){
830 leaf_rotation.at(petiole).at(leaf).yaw = -deg2rad(phytomer_parameters.leaf.yaw.val()/fabs(compound_rotation));
831 float yaw_rot = leaf_rotation.at(petiole).at(leaf).yaw;
832 phytomer_parameters.leaf.yaw.resample();
833 context_ptr->rotateObject( objID_leaf, yaw_rot, "z" );
834 }else{
835 leaf_rotation.at(petiole).at(leaf).yaw = 0;
836 }
837
838 //rotate to azimuth of petiole
839 context_ptr->rotateObject( objID_leaf, -std::atan2(petiole_tip_axis.y, petiole_tip_axis.x)+compound_rotation, "z" );
840
841
842 // -- translation -- //
843
844 vec3 leaf_base = petiole_vertices.at(petiole).back();
845 if( phytomer_parameters.leaf.leaves_per_petiole>1 && phytomer_parameters.leaf.leaflet_offset.val()>0 ){
846 if( ind_from_tip != 0 ) {
847 float offset = (fabs(ind_from_tip) - 0.5f) * phytomer_parameters.leaf.leaflet_offset.val() * phytomer_parameters.petiole.length.val();
848 leaf_base = interpolateTube(petiole_vertices.at(petiole), 1.f - offset / phytomer_parameters.petiole.length.val() );
849 }
850 }
851
852 context_ptr->translateObject( objID_leaf, leaf_base );
853
854 leaf_objIDs.at(petiole).push_back(objID_leaf );
855 leaf_bases.at(petiole).push_back(leaf_base );
856
857 }
858 phytomer_parameters.leaf.prototype_scale.resample();
859
860 if( petiole_axis==make_vec3(0,0,1) ) {
861 inflorescence_bending_axis = make_vec3(1, 0, 0);
862 }else{
863 inflorescence_bending_axis = cross(make_vec3(0, 0, 1), petiole_axis);
864 }
865
866 }
867
868}

870void Phytomer::addInflorescence(const helios::vec3 &base_position, const AxisRotation &base_rotation, const helios::vec3 &a_inflorescence_bending_axis, FloralBud &fbud) {
871
872 uint Ndiv_rachis_length = std::max(uint(1), phytomer_parameters.peduncle.length_segments);
873 uint Ndiv_rachis_radius = std::max(uint(3), phytomer_parameters.peduncle.radial_subdivisions);
874 if( phytomer_parameters.peduncle.length_segments==0 || phytomer_parameters.peduncle.radial_subdivisions<3 ){
875 build_context_geometry_peduncle = false;
876 }
877
878 float dr_peduncle = phytomer_parameters.peduncle.length.val() / float(Ndiv_rachis_length);
879 phytomer_parameters.peduncle.length.resample();
880
881 std::vector<vec3> peduncle_vertices(phytomer_parameters.peduncle.length_segments + 1);
882 peduncle_vertices.at(0) = base_position;
883 std::vector<float> peduncle_radii(phytomer_parameters.peduncle.length_segments + 1);
884 peduncle_radii.at(0) = phytomer_parameters.peduncle.radius.val();
885 std::vector<RGBcolor> peduncle_colors(phytomer_parameters.peduncle.length_segments + 1);
886 peduncle_colors.at(0) = phytomer_parameters.internode.color;
887
888 vec3 peduncle_axis = getAxisVector(1.f, internode_vertices );
889
890 //peduncle pitch rotation
891 if( phytomer_parameters.peduncle.pitch.val()!=0.f || base_rotation.pitch!=0.f ) {
892 peduncle_axis = rotatePointAboutLine(peduncle_axis, nullorigin, inflorescence_bending_axis, deg2rad(phytomer_parameters.peduncle.pitch.val())+base_rotation.pitch );
893 phytomer_parameters.peduncle.pitch.resample();
894 }
895
896 //rotate peduncle to azimuth of petiole and apply peduncle base yaw rotation
897 vec3 internode_axis = getAxisVector(1.f, internode_vertices );
898 vec3 parent_petiole_base_axis = getPetioleAxisVector(0.f, fbud.parent_petiole_index);
899 float parent_petiole_azimuth = -std::atan2(parent_petiole_base_axis.y, parent_petiole_base_axis.x);
900 float current_peduncle_azimuth = -std::atan2(peduncle_axis.y, peduncle_axis.x);
901 peduncle_axis = rotatePointAboutLine( peduncle_axis, nullorigin, internode_axis, (current_peduncle_azimuth-parent_petiole_azimuth) + base_rotation.yaw );
902
903 float theta_base = fabs(cart2sphere(peduncle_axis).zenith);
904
905 for(int i=1; i<=phytomer_parameters.peduncle.length_segments; i++ ){
906
907 if( phytomer_parameters.peduncle.curvature.val()!=0.f ) {
908 float theta_curvature = -deg2rad(phytomer_parameters.peduncle.curvature.val() * dr_peduncle);
909 phytomer_parameters.peduncle.curvature.resample();
910 if (fabs(theta_curvature) * float(i) < M_PI - theta_base) {
911 peduncle_axis = rotatePointAboutLine(peduncle_axis, nullorigin, inflorescence_bending_axis, theta_curvature);
912 } else {
913 peduncle_axis = make_vec3(0, 0, -1);
914 }
915 }
916
917 peduncle_vertices.at(i) = peduncle_vertices.at(i - 1) + dr_peduncle * peduncle_axis;
918
919 peduncle_radii.at(i) = phytomer_parameters.peduncle.radius.val();
920 peduncle_colors.at(i) = phytomer_parameters.internode.color;
921
922 }
923 phytomer_parameters.peduncle.radius.resample();
924
925 if( build_context_geometry_peduncle) {
926 peduncle_objIDs.at(fbud.parent_petiole_index).at(fbud.bud_index).push_back(context_ptr->addTubeObject(Ndiv_rachis_radius, peduncle_vertices, peduncle_radii, peduncle_colors));
927 }
928
929 for(int fruit=0; fruit < phytomer_parameters.inflorescence.flowers_per_rachis.val(); fruit++ ){
930
931 uint objID_fruit;
932 helios::vec3 fruit_scale;
933
934 if(fbud.state == BUD_FRUITING ){
935 objID_fruit = phytomer_parameters.inflorescence.fruit_prototype_function(context_ptr,1,fbud.time_counter);
936 fruit_scale = phytomer_parameters.inflorescence.fruit_prototype_scale.val()*make_vec3(1,1,1);
937 phytomer_parameters.inflorescence.fruit_prototype_scale.resample();
938 }else{
939 bool flower_is_open;
940 if(fbud.state == BUD_FLOWER_CLOSED ) {
941 flower_is_open = false;
942 }else{
943 flower_is_open = true;
944 }
945 objID_fruit = phytomer_parameters.inflorescence.flower_prototype_function(context_ptr,1,flower_is_open);
946 fruit_scale = phytomer_parameters.inflorescence.flower_prototype_scale.val()*make_vec3(1,1,1);
947 phytomer_parameters.inflorescence.flower_prototype_scale.resample();
948 }
949
950 float ind_from_tip = fabs(fruit- float(phytomer_parameters.inflorescence.flowers_per_rachis.val() - 1) / 2.f);
951
952 context_ptr->scaleObject( objID_fruit, fruit_scale );
953
954 //if we have more than one flower/fruit, we need to adjust the base position of the fruit
955 vec3 fruit_base = peduncle_vertices.back();
956 float frac = 1;
957 if(phytomer_parameters.inflorescence.flowers_per_rachis.val() > 1 && phytomer_parameters.inflorescence.flower_offset.val() > 0 ){
958 if( ind_from_tip != 0 ) {
959 float offset = 0;
960 if(phytomer_parameters.inflorescence.flower_arrangement_pattern == "opposite" ){
961 offset = (ind_from_tip - 0.5f) * phytomer_parameters.inflorescence.flower_offset.val() * phytomer_parameters.peduncle.length.val();
962 }else if(phytomer_parameters.inflorescence.flower_arrangement_pattern == "alternate" ){
963 offset = (ind_from_tip - 0.5f + 0.5f*float(fruit> float(phytomer_parameters.inflorescence.flowers_per_rachis.val() - 1) / 2.f) ) * phytomer_parameters.inflorescence.flower_offset.val() * phytomer_parameters.peduncle.length.val();
964 }else{
965 helios_runtime_error("ERROR (PlantArchitecture::addInflorescence): Invalid fruit arrangement pattern.");
966 }
967 if( phytomer_parameters.peduncle.length.val()>0 ){
968 frac = 1.f - offset / phytomer_parameters.peduncle.length.val();
969 }
970 fruit_base = interpolateTube(peduncle_vertices, frac);
971 }
972 }
973
974 //if we have more than one flower/fruit, we need to adjust the rotation about the peduncle
975 float compound_rotation = 0;
976 if(phytomer_parameters.inflorescence.flowers_per_rachis.val() > 1 ) {
977 if (phytomer_parameters.inflorescence.flower_offset.val() == 0) { //flowers/fruit are all at the tip, so just equally distribute them about the azimuth
978 float dphi = M_PI / (floor(0.5 * float(phytomer_parameters.inflorescence.flowers_per_rachis.val() - 1)) + 1);
979 compound_rotation = -float(M_PI) + dphi * (fruit + 0.5f);
980 } else {
981 if( fruit < float(phytomer_parameters.inflorescence.flowers_per_rachis.val() - 1) / 2.f ) {
982 compound_rotation = 0;
983 }else {
984 compound_rotation = M_PI;
985 }
986 }
987 }
988
989 peduncle_axis = getAxisVector(frac, peduncle_vertices );
990
991 vec3 fruit_axis = peduncle_axis;
992
993 //roll rotation
994 if( phytomer_parameters.inflorescence.roll.val()!=0.f ) {
995 context_ptr->rotateObject(objID_fruit, deg2rad(phytomer_parameters.inflorescence.roll.val()), "x" );
996 phytomer_parameters.inflorescence.roll.resample();
997 }
998
999 //pitch rotation
1000 float pitch_inflorescence = -asin_safe(peduncle_axis.z) + deg2rad(phytomer_parameters.inflorescence.pitch.val());
1001 phytomer_parameters.inflorescence.pitch.resample();
1002 if(fbud.state == BUD_FRUITING ) { //gravity effect for fruit
1003 pitch_inflorescence = pitch_inflorescence + phytomer_parameters.inflorescence.fruit_gravity_factor_fraction.val() * (0.5f * M_PI - pitch_inflorescence);
1004 }
1005 context_ptr->rotateObject(objID_fruit, pitch_inflorescence, "y");
1006 fruit_axis = rotatePointAboutLine(fruit_axis, nullorigin, make_vec3(1, 0, 0), pitch_inflorescence);
1007
1008 //rotate flower/fruit to azimuth of peduncle
1009 context_ptr->rotateObject(objID_fruit, -std::atan2(peduncle_axis.y, peduncle_axis.x), "z" );
1010 fruit_axis = rotatePointAboutLine( fruit_axis, nullorigin, make_vec3(0,0,1), -std::atan2(peduncle_axis.y, peduncle_axis.x) );
1011
1012 context_ptr->translateObject( objID_fruit, fruit_base );
1013
1014 //rotate flower/fruit about peduncle (roll)
1015 if( phytomer_parameters.inflorescence.fruit_gravity_factor_fraction.val()!=0 && fbud.state == BUD_FRUITING ) {
