84 }
85 ray_direction = ray_direction / ray_direction.magnitude();

88 const_cast<CollisionDetection*>(this)->ensureBVHCurrent();

100 return castRayBVHTraversal(query);
101 }

103 // FALLBACK: Brute-force primitive testing (only when no BVH available or target UUIDs specified)
104 // This is the slow path that was causing the performance regression
105 std::vector<uint> search_primitives;
106 if (target_UUIDs.empty()) {
107 // Use cached primitives if available (thread-safe), otherwise use context (thread-unsafe)
108 if (!primitive_cache.empty()) {
109 // Use cached primitive IDs for thread-safe operation
110 search_primitives.reserve(primitive_cache.size());
111 for (const auto& cached_pair : primitive_cache) {
112 search_primitives.push_back(cached_pair.first);
113 }
114 } else {
115 // Fall back to context call (thread-unsafe)
116 search_primitives = context->getAllUUIDs();
117 }
118 } else {
119 search_primitives = target_UUIDs;
120 }
121
122 // Find nearest intersection
123 float nearest_distance = std::numeric_limits<float>::max();
124 if (max_distance > 0) {
125 nearest_distance = max_distance;
126 }
127
128 uint hit_primitive = 0;
129 bool found_intersection = false;
130
131 // Test each primitive for intersection
132 for (uint candidate_uuid : search_primitives) {
133 float intersection_distance;
134 bool hit = false;
135
136 // Use thread-safe cached intersection if available
137 if (!primitive_cache.empty()) {
138 HitResult primitive_result = intersectPrimitiveThreadSafe(origin, ray_direction, candidate_uuid, max_distance);
139 if (primitive_result.hit) {
140 hit = true;
141 intersection_distance = primitive_result.distance;
142 }
143 } else {
144 // Fall back to thread-unsafe context call
145 if (!context->doesPrimitiveExist(candidate_uuid)) {
146 continue;
147 }
148 hit = rayPrimitiveIntersection(origin, ray_direction, candidate_uuid, intersection_distance);
149 }
150
151 if (hit) {
152 // Check distance constraints
153 if (intersection_distance > 1e-6f && // Avoid self-intersection
154 intersection_distance < nearest_distance) { // Find nearest
155
156 nearest_distance = intersection_distance;
157 hit_primitive = candidate_uuid;
158 found_intersection = true;
159 }
160 }
161 }
162
163 // Fill result
164 if (found_intersection) {
165 result.hit = true;
166 result.distance = nearest_distance;
167 result.primitive_UUID = hit_primitive;
168 result.intersection_point = origin + ray_direction * nearest_distance;
169
170 // Calculate surface normal
171 try {
172 PrimitiveType type = context->getPrimitiveType(hit_primitive);
173 std::vector<vec3> vertices = context->getPrimitiveVertices(hit_primitive);
174
175 if (type == PRIMITIVE_TYPE_TRIANGLE && vertices.size() >= 3) {
176 // Calculate triangle normal
177 vec3 v0 = vertices[0];
178 vec3 v1 = vertices[1];
179 vec3 v2 = vertices[2];
180 vec3 edge1 = v1 - v0;
181 vec3 edge2 = v2 - v0;
182 result.normal = cross(edge1, edge2);
183 result.normal = result.normal / result.normal.magnitude();
184 } else if (type == PRIMITIVE_TYPE_PATCH && vertices.size() >= 4) {
185 // Calculate patch normal (assuming quad)
186 vec3 v0 = vertices[0];
187 vec3 v1 = vertices[1];
188 vec3 v2 = vertices[2];
189 vec3 edge1 = v1 - v0;
190 vec3 edge2 = v2 - v0;
191 result.normal = cross(edge1, edge2);
192 result.normal = result.normal / result.normal.magnitude();
193 } else {
194 // Default normal (pointing back along ray)
195 result.normal = make_vec3(-ray_direction.x, -ray_direction.y, -ray_direction.z);
196 }
197 } catch (const std::exception& e) {
198 // If we can't get surface normal, use default
199 result.normal = make_vec3(-ray_direction.x, -ray_direction.y, -ray_direction.z);
200 }
201 }
202
203 return result;

643 if (++traversal_steps > MAX_TRAVERSAL_STEPS) {
644 if (printmessages) {
645 std::cout << "WARNING: BVH traversal exceeded maximum steps, terminating ray" << std::endl;
646 }
647 break;
648 }
649
650 size_t node_idx = node_stack.top();
651 node_stack.pop();
652 stats.bvh_nodes_visited++;
653
654 // Bounds check for node index
655 if (node_idx >= bvh_nodes_soa.node_count) {
656 if (printmessages) {
657 std::cout << "WARNING: Invalid node index " << node_idx << " >= " << bvh_nodes_soa.node_count << std::endl;
658 }
659 continue;
660 }
661
662 // Prefetch data for better cache performance
663 if (node_idx + 1 < bvh_nodes_soa.node_count) {
664#ifdef __GNUC__
665 __builtin_prefetch(&bvh_nodes_soa.aabb_mins[node_idx + 1], 0, 1);
666 __builtin_prefetch(&bvh_nodes_soa.aabb_maxs[node_idx + 1], 0, 1);
667#elif defined(_MSC_VER)
668 _mm_prefetch(reinterpret_cast<const char*>(&bvh_nodes_soa.aabb_mins[node_idx + 1]), _MM_HINT_T1);
669 _mm_prefetch(reinterpret_cast<const char*>(&bvh_nodes_soa.aabb_maxs[node_idx + 1]), _MM_HINT_T1);
670#endif
671 }
672
673 // AABB intersection test using SoA layout
674 if (!aabbIntersectSoA(query.origin, query.direction, closest_distance, node_idx)) {
675 continue;
676 }
677
678 // Check if leaf node
679 if (bvh_nodes_soa.is_leaf_flags[node_idx]) {
680 // Process primitives in this leaf
681 uint32_t primitive_start = bvh_nodes_soa.primitive_starts[node_idx];
682 uint32_t primitive_count = bvh_nodes_soa.primitive_counts[node_idx];
683
684 for (uint32_t i = 0; i < primitive_count; ++i) {
685 uint primitive_id = primitive_indices[primitive_start + i];
686
687 // Skip if not in target list (if specified)
688 if (!query.target_UUIDs.empty()) {
689 bool found = false;
690 for (uint target : query.target_UUIDs) {
691 if (primitive_id == target) {
692 found = true;
693 break;
694 }
695 }
696 if (!found) continue;
697 }
698
699 // Use high-performance cached primitive intersection
700 HitResult primitive_result = intersectPrimitiveThreadSafe(query.origin, query.direction, primitive_id, closest_distance);
701 if (primitive_result.hit && primitive_result.distance < closest_distance) {
702 result = primitive_result;
703 closest_distance = primitive_result.distance;
704 }
705 }
706 } else {
707 // Internal node - add children to stack
708 uint32_t left_child = bvh_nodes_soa.left_children[node_idx];
709 uint32_t right_child = bvh_nodes_soa.right_children[node_idx];
710
711 if (left_child != 0xFFFFFFFF && left_child < bvh_nodes_soa.node_count) {
712 node_stack.push(left_child);
713 }
714 if (right_child != 0xFFFFFFFF && right_child < bvh_nodes_soa.node_count) {
715 node_stack.push(right_child);
716 }
717 }
718 }
719
720 return result;
721}

724bool CollisionDetection::aabbIntersectSoA(const helios::vec3& ray_origin, const helios::vec3& ray_direction, float max_distance, size_t node_index) const {
725 // Direct access to SoA arrays for optimal memory usage
726 const vec3& aabb_min = bvh_nodes_soa.aabb_mins[node_index];
727 const vec3& aabb_max = bvh_nodes_soa.aabb_maxs[node_index];
728
729#ifdef __SSE4_1__
730 // SIMD-optimized ray-AABB intersection for better performance
731 __m128 ray_orig = _mm_set_ps(0.0f, ray_origin.z, ray_origin.y, ray_origin.x);
732 __m128 ray_dir = _mm_set_ps(0.0f, ray_direction.z, ray_direction.y, ray_direction.x);
733 __m128 aabb_min_vec = _mm_set_ps(0.0f, aabb_min.z, aabb_min.y, aabb_min.x);
734 __m128 aabb_max_vec = _mm_set_ps(0.0f, aabb_max.z, aabb_max.y, aabb_max.x);
735
736 // Compute inverse ray direction
737 __m128 inv_dir = _mm_div_ps(_mm_set1_ps(1.0f), ray_dir);
738
739 // Compute t1 and t2 for all axes
740 __m128 t1 = _mm_mul_ps(_mm_sub_ps(aabb_min_vec, ray_orig), inv_dir);
741 __m128 t2 = _mm_mul_ps(_mm_sub_ps(aabb_max_vec, ray_orig), inv_dir);
742
743 // Get min and max for each axis
744 __m128 tmin = _mm_min_ps(t1, t2);
745 __m128 tmax = _mm_max_ps(t1, t2);
746
747 // Extract components
748 float tmin_vals[4], tmax_vals[4];
749 _mm_store_ps(tmin_vals, tmin);
750 _mm_store_ps(tmax_vals, tmax);
751
752 float t_near = std::max({tmin_vals[0], tmin_vals[1], tmin_vals[2], 0.0f});
753 float t_far = std::min({tmax_vals[0], tmax_vals[1], tmax_vals[2], max_distance});
754
755 return t_near <= t_far;
756#else
757 // Fallback scalar implementation
758 float inv_dir_x = 1.0f / ray_direction.x;
759 float inv_dir_y = 1.0f / ray_direction.y;
760 float inv_dir_z = 1.0f / ray_direction.z;
761
762 float t1_x = (aabb_min.x - ray_origin.x) * inv_dir_x;
763 float t2_x = (aabb_max.x - ray_origin.x) * inv_dir_x;
764 float t1_y = (aabb_min.y - ray_origin.y) * inv_dir_y;
765 float t2_y = (aabb_max.y - ray_origin.y) * inv_dir_y;
766 float t1_z = (aabb_min.z - ray_origin.z) * inv_dir_z;
767 float t2_z = (aabb_max.z - ray_origin.z) * inv_dir_z;
768
769 float tmin_x = std::min(t1_x, t2_x);
770 float tmax_x = std::max(t1_x, t2_x);
771 float tmin_y = std::min(t1_y, t2_y);
772 float tmax_y = std::max(t1_y, t2_y);
773 float tmin_z = std::min(t1_z, t2_z);
774 float tmax_z = std::max(t1_z, t2_z);
775
776 float t_near = std::max({tmin_x, tmin_y, tmin_z, 0.0f});
777 float t_far = std::min({tmax_x, tmax_y, tmax_z, max_distance});
778
779 return t_near <= t_far;
780#endif
781}

784// Basic BVH traversal using standard node structure
785CollisionDetection::HitResult CollisionDetection::castRayBVHTraversal(const RayQuery &query) {
786 HitResult result;
787
788 if (bvh_nodes.empty()) {
789 return result; // No BVH built
790 }
791
792 // Stack-based traversal using standard BVH nodes
793 std::stack<size_t> node_stack;
794 node_stack.push(0); // Start from root
795
796 float closest_distance = (query.max_distance > 0) ? query.max_distance : std::numeric_limits<float>::max();
797
798 while (!node_stack.empty()) {
799 size_t node_idx = node_stack.top();
800 node_stack.pop();
801
802 if (node_idx >= bvh_nodes.size()) {
803 if (printmessages) {
804 std::cout << "ERROR: Invalid BVH node index " << node_idx << " >= " << bvh_nodes.size() << " nodes" << std::endl;
805 }
806 result.hit = false;
807 return result;
808 }
809
810 const BVHNode& node = bvh_nodes[node_idx];
811
812 // AABB intersection test - this provides the early miss detection that was missing
813 if (!rayAABBIntersect(query.origin, query.direction, node.aabb_min, node.aabb_max)) {
814 continue; // Ray misses this node's bounding box - skip entire subtree
815 }
816
817 if (node.is_leaf) {
818 // Process primitives in this leaf
819 for (uint32_t i = 0; i < node.primitive_count; ++i) {
820 if (node.primitive_start + i >= primitive_indices.size()) {
821 if (printmessages) {
822 std::cout << "ERROR: Invalid BVH primitive index " << (node.primitive_start + i)
823 << " >= " << primitive_indices.size() << " primitives" << std::endl;
824 }
825 result.hit = false;
826 return result;
827 }
828
829 uint primitive_id = primitive_indices[node.primitive_start + i];
830
831 // Skip if not in target list (if specified)
832 if (!query.target_UUIDs.empty()) {
833 bool found = false;
834 for (uint target : query.target_UUIDs) {
835 if (primitive_id == target) {
836 found = true;
837 break;
838 }
839 }
840 if (!found) continue;
841 }
842
843 // Perform primitive intersection test
844 HitResult primitive_hit = intersectPrimitive(query, primitive_id);
845 if (primitive_hit.hit && primitive_hit.distance < closest_distance) {
846 result = primitive_hit;
847 closest_distance = primitive_hit.distance;
848 }
849 }
850 } else {
851 // Internal node - add children to stack for traversal
852 if (node.left_child != 0xFFFFFFFF && node.left_child < bvh_nodes.size()) {
853 node_stack.push(node.left_child);
854 }
855 if (node.right_child != 0xFFFFFFFF && node.right_child < bvh_nodes.size()) {
856 node_stack.push(node.right_child);
857 }
858 }
859 }
860
861 return result;
862}

864// Helper method to test ray-AABB intersection
865bool CollisionDetection::rayAABBIntersect(const vec3& ray_origin, const vec3& ray_direction,
866 const vec3& aabb_min, const vec3& aabb_max) const {
867 // Robust ray-AABB intersection using slab method with proper axis-aligned ray handling
868 const float EPSILON = 1e-8f;
869
870 float tmin = 0.0f; // Ray starts at origin
871 float tmax = std::numeric_limits<float>::max(); // No maximum distance limit
872
873 // Handle X slab
874 if (std::abs(ray_direction.x) > EPSILON) {
875 float inv_dir_x = 1.0f / ray_direction.x;
876 float t1 = (aabb_min.x - ray_origin.x) * inv_dir_x;
877 float t2 = (aabb_max.x - ray_origin.x) * inv_dir_x;
878
879 float slab_tmin = std::min(t1, t2);
880 float slab_tmax = std::max(t1, t2);
881
882 tmin = std::max(tmin, slab_tmin);
883 tmax = std::min(tmax, slab_tmax);
884
885 if (tmin > tmax) return false; // Early exit if no intersection
886 } else {
887 // Ray is parallel to X slab - check if ray origin is within X bounds
888 if (ray_origin.x < aabb_min.x || ray_origin.x > aabb_max.x) {
889 return false;
890 }
891 }
892
893 // Handle Y slab
894 if (std::abs(ray_direction.y) > EPSILON) {
895 float inv_dir_y = 1.0f / ray_direction.y;
896 float t1 = (aabb_min.y - ray_origin.y) * inv_dir_y;
897 float t2 = (aabb_max.y - ray_origin.y) * inv_dir_y;
898
899 float slab_tmin = std::min(t1, t2);
900 float slab_tmax = std::max(t1, t2);
901
902 tmin = std::max(tmin, slab_tmin);
903 tmax = std::min(tmax, slab_tmax);
904
905 if (tmin > tmax) return false; // Early exit if no intersection
906 } else {
907 // Ray is parallel to Y slab - check if ray origin is within Y bounds
908 if (ray_origin.y < aabb_min.y || ray_origin.y > aabb_max.y) {
909 return false;
910 }
911 }
912
913 // Handle Z slab
914 if (std::abs(ray_direction.z) > EPSILON) {
915 float inv_dir_z = 1.0f / ray_direction.z;
916 float t1 = (aabb_min.z - ray_origin.z) * inv_dir_z;
917 float t2 = (aabb_max.z - ray_origin.z) * inv_dir_z;
918
919 float slab_tmin = std::min(t1, t2);
920 float slab_tmax = std::max(t1, t2);
921
922 tmin = std::max(tmin, slab_tmin);
923 tmax = std::min(tmax, slab_tmax);
924
925 if (tmin > tmax) return false; // Early exit if no intersection
926 } else {
927 // Ray is parallel to Z slab - check if ray origin is within Z bounds
928 if (ray_origin.z < aabb_min.z || ray_origin.z > aabb_max.z) {
929 return false;
930 }
931 }
932
933 return tmin <= tmax;
934}

1091 }
1092 }
1093 }
1094 } else if (cached.type == PRIMITIVE_TYPE_VOXEL && cached.vertices.size() == 8) {
1095 // Voxel (AABB) intersection using slab method
1096 // Calculate AABB from 8 vertices
1097 vec3 aabb_min = cached.vertices[0];
1098 vec3 aabb_max = cached.vertices[0];
1099
1100 for (int i = 1; i < 8; i++) {
1101 aabb_min.x = std::min(aabb_min.x, cached.vertices[i].x);
1102 aabb_min.y = std::min(aabb_min.y, cached.vertices[i].y);
1103 aabb_min.z = std::min(aabb_min.z, cached.vertices[i].z);
1104 aabb_max.x = std::max(aabb_max.x, cached.vertices[i].x);
1105 aabb_max.y = std::max(aabb_max.y, cached.vertices[i].y);
1106 aabb_max.z = std::max(aabb_max.z, cached.vertices[i].z);
1107 }
1108
1109 // Ray-AABB intersection using slab method
1110 float t_near = -std::numeric_limits<float>::max();
1111 float t_far = std::numeric_limits<float>::max();
1112
1113 // Check intersection with each slab (X, Y, Z)
1114 for (int axis = 0; axis < 3; axis++) {
1115 float ray_dir_component = (axis == 0) ? direction.x : (axis == 1) ? direction.y : direction.z;
1116 float ray_orig_component = (axis == 0) ? origin.x : (axis == 1) ? origin.y : origin.z;
1117 float aabb_min_component = (axis == 0) ? aabb_min.x : (axis == 1) ? aabb_min.y : aabb_min.z;
1118 float aabb_max_component = (axis == 0) ? aabb_max.x : (axis == 1) ? aabb_max.y : aabb_max.z;
1119

1121 // Ray is parallel to slab
1122 if (ray_orig_component < aabb_min_component || ray_orig_component > aabb_max_component) {
1123 return result; // Ray is outside slab and parallel - no intersection
1124 }
1125 } else {
1126 // Calculate intersection distances for this slab
1127 float t1 = (aabb_min_component - ray_orig_component) / ray_dir_component;
1128 float t2 = (aabb_max_component - ray_orig_component) / ray_dir_component;
1129
1130 // Ensure t1 <= t2
1131 if (t1 > t2) {
1132 std::swap(t1, t2);
1133 }
1134
1135 // Update near and far intersection distances
1136 t_near = std::max(t_near, t1);
1137 t_far = std::min(t_far, t2);
1138
1139 // Early exit if no intersection possible
1140 if (t_near > t_far) {
1141 return result;
1142 }
1143 }
1144 }
1145
1146 // Check if intersection is in front of ray origin and within max distance
1147 if (t_far >= 0.0f) {
1148 // Use t_near if it's positive (ray starts outside box), otherwise t_far (ray starts inside box)
1149 float intersection_distance = (t_near >= 1e-6f) ? t_near : t_far;
1150 if (intersection_distance >= 1e-6f && (max_distance <= 0 || intersection_distance < max_distance)) {
1151 result.hit = true;
1152 result.distance = intersection_distance;
1153 result.primitive_UUID = primitive_id;
1154 result.intersection_point = origin + direction * intersection_distance;
1155
1156 // Calculate normal based on which face was hit
1157 // Determine which face of the voxel was hit by examining intersection point
1158 vec3 hit_point = result.intersection_point;
1159 vec3 box_center = (aabb_min + aabb_max) * 0.5f;
1160 vec3 box_extent = (aabb_max - aabb_min) * 0.5f;
1161
1162 // Find which face the hit point is closest to
1163 vec3 local_hit = hit_point - box_center;
1164 vec3 abs_local_hit = make_vec3(std::abs(local_hit.x), std::abs(local_hit.y), std::abs(local_hit.z));
1165
1166 // Determine which axis has the largest relative coordinate (closest to face)
1167 float rel_x = abs_local_hit.x / box_extent.x;
1168 float rel_y = abs_local_hit.y / box_extent.y;
1169 float rel_z = abs_local_hit.z / box_extent.z;
1170
1171 if (rel_x >= rel_y && rel_x >= rel_z) {
1172 // Hit X face
1173 result.normal = make_vec3((local_hit.x > 0) ? 1.0f : -1.0f, 0.0f, 0.0f);
1174 } else if (rel_y >= rel_z) {
1175 // Hit Y face
1176 result.normal = make_vec3(0.0f, (local_hit.y > 0) ? 1.0f : -1.0f, 0.0f);

1220bool CollisionDetection::triangleIntersect(const vec3 &origin, const vec3 &direction,
1221 const vec3 &v0, const vec3 &v1, const vec3 &v2, float &distance) {
1222 // Möller-Trumbore triangle intersection algorithm (optimized - no vec3 temporaries)
1223 const float EPSILON = 1e-8f;
1224
1225 // Compute triangle edges directly as components (avoid vec3 constructors)
1226 float edge1_x = v1.x - v0.x, edge1_y = v1.y - v0.y, edge1_z = v1.z - v0.z;
1227 float edge2_x = v2.x - v0.x, edge2_y = v2.y - v0.y, edge2_z = v2.z - v0.z;
1228
1229 // Cross product: h = direction × edge2 (computed directly)
1230 float h_x = direction.y * edge2_z - direction.z * edge2_y;
1231 float h_y = direction.z * edge2_x - direction.x * edge2_z;
1232 float h_z = direction.x * edge2_y - direction.y * edge2_x;
1233
1234 // Dot product: a = edge1 · h
1235 float a = edge1_x * h_x + edge1_y * h_y + edge1_z * h_z;
1236
1237 if (a > -EPSILON && a < EPSILON) {
1238 return false; // Ray is parallel to triangle
1239 }
1240
1241 float f = 1.0f / a;
1242
1243 // Vector s = origin - v0 (computed as components)
1244 float s_x = origin.x - v0.x, s_y = origin.y - v0.y, s_z = origin.z - v0.z;
1245
1246 // u = f * (s · h)
1247 float u = f * (s_x * h_x + s_y * h_y + s_z * h_z);
1248
1249 if (u < -EPSILON || u > 1.0f + EPSILON) {
1250 return false;
1251 }
1252
1253 // Cross product: q = s × edge1 (computed directly)
1254 float q_x = s_y * edge1_z - s_z * edge1_y;
1255 float q_y = s_z * edge1_x - s_x * edge1_z;
1256 float q_z = s_x * edge1_y - s_y * edge1_x;
1257
1258 // v = f * (direction · q)
1259 float v = f * (direction.x * q_x + direction.y * q_y + direction.z * q_z);
1260
1261 if (v < -EPSILON || u + v > 1.0f + EPSILON) {
1262 return false;
1263 }
1264
1265 // t = f * (edge2 · q) - computed directly as dot product
1266 float t = f * (edge2_x * q_x + edge2_y * q_y + edge2_z * q_z);
1267
1268 if (t > EPSILON) {
1269 distance = t;
1270 return true;
1271 }
1272
1273 return false; // Line intersection but not ray intersection
1274}

1276bool CollisionDetection::patchIntersect(const vec3 &origin, const vec3 &direction,
1277 const vec3 &v0, const vec3 &v1, const vec3 &v2, const vec3 &v3, float &distance) {
1278 // Patch (quadrilateral) intersection using radiation model algorithm
1279 const float EPSILON = 1e-8f;
1280
1281 // Calculate patch vectors and normal (same as radiation model)
1282 vec3 anchor = v0;
1283 vec3 normal = cross(v1 - v0, v2 - v0);
1284 normal.normalize();
1285
1286 vec3 a = v1 - v0; // First edge vector
1287 vec3 b = v3 - v0; // Second edge vector
1288
1289 // Ray-plane intersection
1290 float denom = direction * normal;
1291 if (std::abs(denom) > EPSILON) { // Not parallel to plane
1292 float t = (anchor - origin) * normal / denom;
1293
1294 if (t > EPSILON && t < 1e8f) { // Valid intersection distance
1295 // Find intersection point
1296 vec3 p = origin + direction * t;
1297 vec3 d = p - anchor;
1298
1299 // Project onto patch coordinate system
1300 float ddota = d * a;
1301 float ddotb = d * b;
1302
1303 // Check if point is within patch bounds (with epsilon tolerance for edge cases)
1304 if (ddota >= -EPSILON && ddota <= (a * a) + EPSILON &&
1305 ddotb >= -EPSILON && ddotb <= (b * b) + EPSILON) {
1306 distance = t;
1307 return true;
1308 }
1309 }
1310 }
1311
1312 return false;
1313}

1769 // Per-ray data structures - using aligned arrays for SIMD efficiency
1770 alignas(32) uint32_t node_stacks[8][MAX_STACK_SIZE]; // 8 stacks for up to 8 rays
1771 alignas(32) uint32_t stack_tops[8] = {0}; // Current stack positions
1772 alignas(32) float closest_distances[8]; // Closest hit distances per ray

1774
1775 // Initialize per-ray state
1776 for (int i = 0; i < count; ++i) {
1777 node_stacks[i][0] = 0; // Start with root node
1778 stack_tops[i] = 1;
1779 closest_distances[i] = std::numeric_limits<float>::max();

1781 results[i] = HitResult(); // Initialize as miss
1782 }
1783
1784 // Main traversal loop - continue while any ray is active
1785 while (true) {
1786 bool any_active = false;
1787 for (int i = 0; i < count; ++i) {
1788 if (ray_active[i] && stack_tops[i] > 0) {
1789 any_active = true;
1790 break;
1791 }
1792 }
1793 if (!any_active) break;
1794
1795 // Collect next nodes to test for active rays
1796 alignas(32) vec3 test_aabb_mins[8];
1797 alignas(32) vec3 test_aabb_maxs[8];
1798 alignas(32) uint32_t test_node_indices[8];
