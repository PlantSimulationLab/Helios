393size_t LiDARcloud::loadASCIIFile( uint scanID, ScanMetadata &scandata ){
394
395 ifstream datafile(scandata.data_file); //open the file
396
397 if(!datafile.is_open()){ //check that file exists
398 throw( std::runtime_error("ERROR (LiDARcloud::loadASCIIFile): ASCII data file '" + scandata.data_file + "' does not exist.") );
399 }
400
401 vec3 temp_xyz;
402 SphericalCoord temp_direction;
403 RGBcolor temp_rgb;
404 float temp_row, temp_column;
405 double temp_data;
406 std::map<std::string, double> data;
407
408 vector<unsigned int> row, column;
409 std::size_t hit_count = 0;
410 while ( datafile.good() ){ //loop through file to read scan data
411
412 temp_xyz = make_vec3(-9999,-9999,-9999);
413 temp_rgb = make_RGBcolor(1,0,0); //default color: red
414 temp_row = -1;
415 temp_column = -1;
416 temp_direction = make_SphericalCoord(-9999,-9999);
417
418 for( uint i=0; i<scandata.columnFormat.size(); i++ ){
419 if( scandata.columnFormat.at(i) == "row" ){
420 datafile >> temp_row;
421 }else if( scandata.columnFormat.at(i) == "column" ){
422 datafile >> temp_column;
423 }else if( scandata.columnFormat.at(i) == "zenith" ){
424 datafile >> temp_direction.zenith;
425 }else if( scandata.columnFormat.at(i) == "azimuth" ){
426 datafile >> temp_direction.azimuth;
427 }else if( scandata.columnFormat.at(i) == "zenith_rad" ){
428 datafile >> temp_direction.zenith;
429 temp_direction.zenith = deg2rad(temp_direction.zenith);
430 }else if( scandata.columnFormat.at(i) == "azimuth_rad" ){
431 datafile >> temp_direction.azimuth;
432 temp_direction.azimuth = deg2rad(temp_direction.azimuth);
433 }else if( scandata.columnFormat.at(i) == "x" ){
434 datafile >> temp_xyz.x;
435 }else if( scandata.columnFormat.at(i) == "y" ){
436 datafile >> temp_xyz.y;
437 }else if( scandata.columnFormat.at(i) == "z" ){
438 datafile >> temp_xyz.z;
439 }else if( scandata.columnFormat.at(i) == "r" ){
440 datafile >> temp_rgb.r;
441 }else if( scandata.columnFormat.at(i) == "g" ){
442 datafile >> temp_rgb.g;
443 }else if( scandata.columnFormat.at(i) == "b" ){
444 datafile >> temp_rgb.b;
445 }else if( scandata.columnFormat.at(i) == "r255" ){
446 datafile >> temp_rgb.r;
447 temp_rgb.r/=255.f;
448 }else if( scandata.columnFormat.at(i) == "g255" ){
449 datafile >> temp_rgb.g;
450 temp_rgb.g/=255.f;
451 }else if( scandata.columnFormat.at(i) == "b255" ){
452 datafile >> temp_rgb.b;
453 temp_rgb.b/=255.f;
454 }else{ //assume that rest is data
455 datafile >> temp_data;
456 data[ scandata.columnFormat.at(i) ] = temp_data;
457 }
458 }
459
460 if( !datafile.good() ){//if the whole line was not read successfully, stop
461 if( hit_count==0 ){
462 std::cerr << "WARNING: Something is likely wrong with the data file " << scandata.data_file << ". Check that the format is consistent with that specified in the XML metadata file." << std::endl;
463 }
464 break;
465 }
466
467 // -- Checks to make sure everything was specified correctly -- //
468
469 //hit point
470 if( temp_xyz.x==-9999 ){
471 throw( std::runtime_error("ERROR (LiDARcloud::loadASCIIFile): x-coordinate not specified for hit point #" + std::to_string(hit_count) + " of scan #" + std::to_string(scanID) ) );
472 }else if( temp_xyz.y==-9999 ){
473 throw( std::runtime_error("ERROR (LiDARcloud::loadASCIIFile): y-coordinate not specified for hit point #" + std::to_string(hit_count) + " of scan #" + std::to_string(scanID) ) );
474 }else if( temp_xyz.z==-9999 ){
475 throw( std::runtime_error("ERROR (LiDARcloud::loadASCIIFile): z-coordinate not specified for hit point #" + std::to_string(hit_count) + " of scan #" + std::to_string(scanID) ) );
476 }
477
478 //direction
479 if( temp_direction.elevation==-9999 || temp_direction.azimuth==-9999 ){
480 temp_direction = cart2sphere( temp_xyz - scandata.origin );
481 }
482
483 //add hit point to the scan
484 addHitPoint( scanID, temp_xyz, temp_direction, temp_rgb, data );
485
486 hit_count++;
487
488 }
489
490 datafile.close();
491
492 return hit_count;
493

608void LiDARcloud::exportTriangleInclinationDistribution( const char* filename, uint Nbins ){
609
610 std::vector<std::vector<float>> inclinations( getGridCellCount() );
611 for( int i=0; i<getGridCellCount(); i++ ){
612 inclinations.at(i).resize( Nbins );
613 }
614
615 float db = 0.5f*M_PI/float(Nbins); //bin width
616
617 for( std::size_t t=0; t<triangles.size(); t++ ){
618
619 Triangulation tri = triangles.at(t);
620
621 int cell = tri.gridcell;
622
623 if( cell<0 ){
624 continue;
625 }
626
627 vec3 v0 = tri.vertex0;
628 vec3 v1 = tri.vertex1;
629 vec3 v2 = tri.vertex2;
630
631 vec3 normal = cross( v1-v0, v2-v0 );
632 normal.normalize();
633
634 float angle = acos_safe(fabs(normal.z));
635
636 float area = tri.area;
637
638 uint bin = floor(angle/db);
639 if( bin>=Nbins ){
640 bin = Nbins-1;
641 }
642
643 inclinations.at(cell).at(bin) += area;
644
645 }
646
647 ofstream file;
648
649 file.open(filename);
650
651 if( !file.is_open() ){
652 throw( std::runtime_error("ERROR (LiDARcloud::exportTriangleInclinationDistribution): Could not open file '" + std::string(filename) + "' for writing.") );
653 }
654
655 for( int cell=0; cell<getGridCellCount(); cell++ ){
656 for( int bin=0; bin<Nbins; bin++ ) {
657 file << inclinations.at(cell).at(bin) << " ";
658 }
659 file << std::endl;
660 }

662 file.close();

754void LiDARcloud::exportPointCloud( const char* filename, uint scanID ){
755
756 if( scanID> getScanCount()){
757 std::cerr << "ERROR (LiDARcloud::exportPointCloud): Cannot export scan " << scanID << " because this scan does not exist." << std::endl;
758 throw 1;
759 }
760
761 ofstream file;
762
763 file.open(filename);
764
765 if( !file.is_open() ) {
766 throw (std::runtime_error("ERROR (LiDARcloud::exportPointCloud): Could not open file '" + std::string(filename) + "' for writing."));
767 }
768
769 std::vector<std::string> hit_data;
770 for( int r=0; r< getHitCount(); r++ ){
771 std::map<std::string,double> data = hits.at(r).data;
772 for( std::map<std::string,double>::iterator iter=data.begin(); iter!=data.end(); ++iter ){
773 std::vector<std::string>::iterator it = find(hit_data.begin(),hit_data.end(),iter->first);
774 if( it==hit_data.end() ){
775 hit_data.push_back(iter->first);
776 }
777 }
778 }
779
780 std::vector<std::string> ASCII_format = getScanColumnFormat(scanID);
781
782 if( ASCII_format.size()==0 ){
783 ASCII_format.push_back("x");
784 ASCII_format.push_back("y");
785 ASCII_format.push_back("z");
786 }
787
788 for( int r=0; r< getHitCount(); r++ ){
789
790 if( getHitScanID(r) != scanID ){
791 continue;
792 }
793
794 vec3 xyz = getHitXYZ(r);
795 RGBcolor color = getHitColor(r);
796
797 for( int c=0; c<ASCII_format.size(); c++ ){
798
799 if( ASCII_format.at(c).compare("x")==0 ){
800 file << xyz.x;
801 }else if( ASCII_format.at(c).compare("y")==0 ){
802 file << xyz.y;
803 }else if( ASCII_format.at(c).compare("z")==0 ){
804 file << xyz.z;
805 }else if( ASCII_format.at(c).compare("r")==0 ){
806 file << color.r;
807 }else if( ASCII_format.at(c).compare("g")==0 ){
808 file << color.g;
809 }else if( ASCII_format.at(c).compare("b")==0 ){
810 file << color.b;
811 }else if( ASCII_format.at(c).compare("r255")==0 ){
812 file << round(color.r*255);
813 }else if( ASCII_format.at(c).compare("g255")==0 ){
814 file << round(color.g*255);
815 }else if( ASCII_format.at(c).compare("b255")==0 ){
816 file << round(color.b*255);
817 }else if( ASCII_format.at(c).compare("zenith")==0 ){
818 file << getHitRaydir(r).zenith;
819 }else if( ASCII_format.at(c).compare("azimuth")==0 ){
820 file << getHitRaydir(r).azimuth;
821 }else if( hits.at(r).data.find(ASCII_format.at(c))!=hits.at(r).data.end() ){ //hit scalar data
822 file << getHitData(r,ASCII_format.at(c).c_str());
823 }else{
824 file << -9999;
825 }
826
827 if( c<ASCII_format.size()-1 ){
828 file << " ";
829 }
830
831 }
832
833 file << std::endl;
834
835 }

841void LiDARcloud::exportPointCloudPTX( const char* filename, uint scanID ){
842
843 if( scanID> getScanCount()){
844 std::cerr << "ERROR (LiDARcloud::exportPointCloudPTX): Cannot export scan " << scanID << " because this scan does not exist." << std::endl;
845 throw 1;
846 }
847
848 ofstream file;
849
850 file.open(filename);
851
852 if( !file.is_open() ) {
853 throw (std::runtime_error("ERROR (LiDARcloud::exportPointCloudPTX): Could not open file '" + std::string(filename) + "' for writing."));
854 }
855
856 std::vector<std::string> ASCII_format = getScanColumnFormat(scanID);
857
858 uint Nx = getScanSizeTheta(scanID);
859 uint Ny = getScanSizePhi(scanID);
860
861 file << Nx << std::endl;
862 file << Ny << std::endl;
863 file << "0 0 0" << std::endl;
864 file << "1 0 0" << std::endl;
865 file << "0 1 0" << std::endl;
866 file << "0 0 1" << std::endl;
867 file << "1 0 0 0" << std::endl;
868 file << "0 1 0 0" << std::endl;
869 file << "0 0 1 0" << std::endl;
870 file << "0 0 0 1" << std::endl;
871
872 std::vector<std::vector<vec4> > xyzi(Ny);
873 for( int j=0; j<Ny; j++ ){
874 xyzi.at(j).resize(Nx);
875 for( int i=0; i<Nx; i++ ){
876 xyzi.at(j).at(i) = make_vec4(0,0,0,1);
877 }

886 }

889
890 int2 row_column = scans.at(scanID).direction2rc(raydir);

893
894 vec3 xyz = getHitXYZ(r);
895
896 if( (xyz-origin).magnitude()>=1e4 ){
897 continue;
898 }

900 float intensity = 1.f;
901 if( hits.at(r).data.find("intensity")!=hits.at(r).data.end() ) {
902 intensity = getHitData(r, "intensity");
903 }
904
905 xyzi.at(row_column.y).at(row_column.x) = make_vec4( xyz.x, xyz.y, xyz.z, intensity );
906
907 }
908
909 for( int j=0; j<Ny; j++ ){
910 for( int i=0; i<Nx; i++ ){
911 file << xyzi.at(j).at(i).x << " " << xyzi.at(j).at(i).y << " " << xyzi.at(j).at(i).z << " " << xyzi.at(j).at(i).w << std::endl;
912 }
913 }

915 file.close();
