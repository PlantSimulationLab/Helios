21ParametersToOptimize readParametersFromFile(const std::string& filename) {
22 ParametersToOptimize params;
23 std::ifstream infile(filename);
24 if (!infile.is_open()) {
25 throw std::runtime_error("Failed to open parameter file: " + filename);
26 }
27
28 std::string line;
29 while (std::getline(infile, line)) {
30 if (line.empty()) continue;
31
32 std::istringstream ss(line);
33 std::string name;
34 float value, min, max;
35
36 if (std::getline(ss, name, ',') &&
37 ss >> value &&
38 ss.ignore(1) && ss >> min &&
39 ss.ignore(1) && ss >> max) {
40 params[name] = ParameterToOptimize{value, min, max};
41 }
42 }
43
44 return params;
45}

48static ParameterOptimization::Result evaluatePopulation(
49 const std::vector<ParametersToOptimize>& pop,
50 const std::function<float(const ParametersToOptimize&)>& simulation){
51
52 ParameterOptimization::Result best;
53 best.fitness = std::numeric_limits<float>::max();
54 for(const auto& ind : pop){
55 float fit = simulation(ind);
56 if(fit < best.fitness){
57 best.parameters = ind;
58 best.fitness = fit;
59 }
60 }
61 return best;
62}

64ParameterOptimization::Result ParameterOptimization::run(
65 std::function<float(const ParametersToOptimize&)> simulation,
66 const ParametersToOptimize& parameters,
67 const OptimizationSettings& settings)
68{
69 std::mt19937 rng( std::random_device{}() );
70 std::uniform_real_distribution<float> uni01(0.f,1.f);
71
72 std::ofstream outProgress;
73 if (!settings.write_progress_to_file.empty()) {
74 outProgress=std::ofstream(settings.write_progress_to_file);
75 outProgress.clear();
76 }
77
78
79 size_t popsize = std::max<size_t>(2, settings.population_size);
80 std::vector<ParametersToOptimize> population(popsize);
81 for (auto& ind : population) {
82 ind = parameters;
83 for (auto& [name, param] : ind) {
84 param.value = param.min + (param.max - param.min) * uni01(rng);
85 }
86 }
87
88 if (!settings.read_input_from_file.empty()) {
89 try {
90 ParametersToOptimize best_from_file = readParametersFromFile(settings.read_input_from_file);
91 population[0] = best_from_file; // seed with known best
92 } catch (const std::exception& e) {
93 std::cerr << "Warning: " << e.what() << "\n";
94 }
95 }
96
97 Result best = evaluatePopulation(population, simulation);
98
99 for (size_t g = 0; g < settings.generations; g++) {
100 std::vector<ParametersToOptimize> new_pop(popsize);
101
102 // Elitism, top 5%
103 size_t elitism_count = std::max<size_t>(1, popsize * settings.elitism_rate);
104 std::partial_sort(population.begin(), population.begin() + elitism_count, population.end(),
105 [&](const auto& a, const auto& b) {
106 return simulation(a) < simulation(b);
107 });
108 for (size_t e = 0; e < elitism_count; ++e) {
109 new_pop[e] = population[e];
110 }
111
112
113 std::uniform_int_distribution<size_t> dist(0, popsize - 1);
114
115 for (size_t i = 1; i < popsize; i++) {
116 //Selection, Tournament
117 auto tournamentSelect = [&](int k = 3) -> const ParametersToOptimize& {
118 const ParametersToOptimize* bestSelection = &population[dist(rng)];
119 for (int j = 1; j < k; ++j) {
120 const ParametersToOptimize* challenger = &population[dist(rng)];
121 if (simulation(*challenger) < simulation(*bestSelection)) {
122 bestSelection = challenger;
123 }
124 }
125 return *bestSelection;
126 };
127
128 const auto& parent1 = tournamentSelect();
129 const auto& parent2 = tournamentSelect();
130
131 ParametersToOptimize child = parameters;
132
133 for (auto& [name, param] : child) {
134 const auto& p1 = parent1.at(name);
135 const auto& p2 = parent2.at(name);
136
137 float val = p1.value;
138
139 //Crossover, Blended alpha
140 if (uni01(rng) < settings.crossover_rate) {
141 float alpha = 0.5f;
142 float minval = std::min(p1.value, p2.value);
143 float maxval = std::max(p1.value, p2.value);
144 float range = maxval - minval;
145 val = minval - alpha * range + uni01(rng) * (1.f + 2.f * alpha) * range;
146 }
147
148 //Mutation
149 if (uni01(rng) < settings.mutation_rate) {
150 std::normal_distribution<float> gauss(0.f, 0.1f * (param.max - param.min));
151 val += gauss(rng);
152 }
153
154
155 param.value = std::min(std::max(val, param.min), param.max);
156 }
157 new_pop[i] = child;
158 }
159
160 population = std::move(new_pop);
161 Result generation_best = evaluatePopulation(population, simulation);
162 if(generation_best.fitness <= best.fitness){
163 best = generation_best;
164 fitness_over_time.push_back(best.fitness);
165 if (settings.print_progress) {
166 std::cout << "Generation " << g
167 << " - Best Fitness: " << best.fitness << "\n";
168 for (const auto& [name, param] : best.parameters) {
169 std::cout << " " << name << ": " << param.value << "\n";
170 }
171 }
172 if (!settings.write_progress_to_file.empty()) {
173 if (outProgress.is_open()) {
174 outProgress << "generation,fitness,parameter,value,min,max\n";
175 for (const auto& [name, param] : best.parameters) {
176 outProgress << g << "," << best.fitness << "," << name << "," << param.value << "," << param.min << "," << param.max << "\n";
177 }
178 if (message_flag) {
179 std::cout << "Best parameters progress written to: " << settings.write_progress_to_file << "\n";
180 }
181 } else {
182 std::cerr << "Failed to write to file: " << settings.write_progress_to_file << "\n";
183 }
184 }
185 }
186 }
187
188 if(settings.print_progress) {
189 printf("Generation Fitness\n");
190 for(int i=0;i<fitness_over_time.size();i++) {
191 printf("%d %f\n",i,fitness_over_time.at(i));
192 }
193 }
194
195 if (!settings.write_result_to_file.empty()) {
196 std::ofstream out(settings.write_result_to_file);
197 if (out.is_open()) {
198 out << "parameter,value,min,max\n";
199 for (const auto& [name, param] : best.parameters) {
200 out << name << "," << param.value << "," << param.min << "," << param.max << "\n";
201 }
202 out.close();
203 if (message_flag) {
204 std::cout << "Best parameters written to: " << settings.write_result_to_file << "\n";
205 }
206 } else {
207 std::cerr << "Failed to write to file: " << settings.write_result_to_file << "\n";
208 }
209 }
210
211 if (!settings.write_progress_to_file.empty()) {
212 outProgress.close();
213 }
214
215
216 return best;
217}
