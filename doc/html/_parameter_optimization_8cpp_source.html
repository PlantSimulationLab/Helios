67 const int graph_height = 8;

74 float max_fitness = *std::max_element(fitness_history.begin(), fitness_history.end());

186 // Cholesky decomposition (in-place, lower triangular)
187 bool choleskyDecompose(std::vector<std::vector<float>> &mat) {
188 size_t n = mat.size();
189 for (size_t i = 0; i < n; ++i) {
190 for (size_t j = 0; j <= i; ++j) {
191 float sum = 0.0f;
192 for (size_t k = 0; k < j; ++k) {
193 sum += mat[i][k] * mat[j][k];
194 }
195 if (i == j) {
196 float val = mat[i][i] - sum;
197 if (val <= 0.0f)
198 return false; // Not positive definite
199 mat[i][j] = std::sqrt(val);
200 } else {
201 mat[i][j] = (mat[i][j] - sum) / mat[j][j];
202 }
203 }
204 }
205 return true;
206 }

208 // Solve L*y = b where L is lower triangular (forward substitution)
209 std::vector<float> solveTriangular(const std::vector<std::vector<float>> &L, const std::vector<float> &b, bool lower = true) {
210 size_t n = b.size();
211 std::vector<float> y(n);
212 if (lower) {
213 for (size_t i = 0; i < n; ++i) {
214 float sum = 0.0f;
215 for (size_t j = 0; j < i; ++j) {
216 sum += L[i][j] * y[j];
217 }
218 y[i] = (b[i] - sum) / L[i][i];
219 }
220 } else {
221 for (int i = n - 1; i >= 0; --i) {
222 float sum = 0.0f;
223 for (size_t j = i + 1; j < n; ++j) {
224 sum += L[i][j] * y[j];
225 }
226 y[i] = (b[i] - sum) / L[i][i];
227 }
228 }
229 return y;

232 // Jacobi eigendecomposition for symmetric matrices.
233 // On return, eigenvalues contains the eigenvalues and eigenvectors has
234 // eigenvectors as columns (eigenvectors[row][col] where col is the eigenvector index).
235 void jacobiEigen(std::vector<std::vector<float>> A, std::vector<float> &eigenvalues, std::vector<std::vector<float>> &eigenvectors) {
236 size_t n = A.size();
237 eigenvectors.assign(n, std::vector<float>(n, 0.0f));
238 for (size_t i = 0; i < n; ++i)
239 eigenvectors[i][i] = 1.0f;
240
241 const int max_sweeps = 100;
242 for (int sweep = 0; sweep < max_sweeps; ++sweep) {
243 // Check convergence: sum of squares of off-diagonal elements
244 float off_diag = 0.0f;
245 for (size_t i = 0; i < n; ++i)
246 for (size_t j = i + 1; j < n; ++j)
247 off_diag += A[i][j] * A[i][j];
248 if (off_diag < 1e-12f)
249 break;
250
251 for (size_t p = 0; p < n; ++p) {
252 for (size_t q = p + 1; q < n; ++q) {
253 if (std::abs(A[p][q]) < 1e-12f)
254 continue;
255
256 float t, c, s;
257 if (std::abs(A[p][p] - A[q][q]) < 1e-12f) {
258 t = 1.0f;
259 } else {
260 float tau = (A[q][q] - A[p][p]) / (2.0f * A[p][q]);
261 t = (tau >= 0 ? 1.0f : -1.0f) / (std::abs(tau) + std::sqrt(1.0f + tau * tau));
262 }
263 c = 1.0f / std::sqrt(1.0f + t * t);
264 s = t * c;
265
266 // Rotate A
267 float A_pp = A[p][p], A_qq = A[q][q], A_pq = A[p][q];
268 A[p][p] = c * c * A_pp - 2 * s * c * A_pq + s * s * A_qq;
269 A[q][q] = s * s * A_pp + 2 * s * c * A_pq + c * c * A_qq;
270 A[p][q] = 0.0f;
271 A[q][p] = 0.0f;
272
273 for (size_t i = 0; i < n; ++i) {
274 if (i == p || i == q)
275 continue;
276 float A_ip = A[i][p], A_iq = A[i][q];
277 A[i][p] = c * A_ip - s * A_iq;
278 A[p][i] = A[i][p];
279 A[i][q] = s * A_ip + c * A_iq;
280 A[q][i] = A[i][q];
281 }
282
283 // Accumulate eigenvectors
284 for (size_t i = 0; i < n; ++i) {
285 float V_ip = eigenvectors[i][p], V_iq = eigenvectors[i][q];
286 eigenvectors[i][p] = c * V_ip - s * V_iq;
287 eigenvectors[i][q] = s * V_ip + c * V_iq;
288 }
289 }
290 }
291 }
292 eigenvalues.resize(n);
293 for (size_t i = 0; i < n; ++i)
294 eigenvalues[i] = A[i][i];
295 }

297 // PCA state computed from a population, used for BLXPCA crossover.
298 struct PCAState {
299 std::vector<float> mean;
300 std::vector<std::vector<float>> eigenvectors; // columns are eigenvectors
301 std::vector<float> eigenvalues;
302 std::vector<std::string> param_names; // ordered names of numeric params in PCA
303 bool valid = false;
304
305 // Forward: y = diag(1/sqrt(lambda)) * P^T * (x - mean)
306 std::vector<float> transform(const ParametersToOptimize &params) const {
307 size_t n = param_names.size();
308 std::vector<float> centered(n);
309 for (size_t i = 0; i < n; ++i)
310 centered[i] = params.at(param_names[i]).value - mean[i];
311
312 std::vector<float> result(n, 0.0f);
313 for (size_t i = 0; i < n; ++i) {
314 float dot = 0.0f;
315 for (size_t j = 0; j < n; ++j)
316 dot += eigenvectors[j][i] * centered[j]; // P^T row i = column i of P
317 float scale = (eigenvalues[i] > 1e-10f) ? 1.0f / std::sqrt(eigenvalues[i]) : 0.0f;
318 result[i] = dot * scale;
319 }
320 return result;
321 }
322
323 // Inverse: x = P * diag(sqrt(lambda)) * y + mean
324 std::vector<float> inverseTransform(const std::vector<float> &y) const {
325 size_t n = param_names.size();
326 std::vector<float> scaled(n);
327 for (size_t i = 0; i < n; ++i)
328 scaled[i] = y[i] * std::sqrt(std::max(eigenvalues[i], 1e-10f));
329
330 std::vector<float> result(n, 0.0f);
331 for (size_t i = 0; i < n; ++i)
332 for (size_t j = 0; j < n; ++j)
333 result[i] += eigenvectors[i][j] * scaled[j];
334
335 for (size_t i = 0; i < n; ++i)
336 result[i] += mean[i];
337 return result;
338 }
339 };

341 // Compute PCA from the numeric (FLOAT/INTEGER) parameters of a population.
342 PCAState computePCA(const std::vector<ParametersToOptimize> &population, const ParametersToOptimize &template_params) {
343 PCAState pca;
344
345 // Collect sorted names of numeric params
346 for (const auto &[name, param]: template_params) {
347 if (param.type != ParameterType::CATEGORICAL)
348 pca.param_names.push_back(name);
349 }
350 std::sort(pca.param_names.begin(), pca.param_names.end());
351 size_t n = pca.param_names.size();
352 size_t m = population.size();
353
354 if (n == 0 || m < 2)
355 return pca;
356
357 // Extract values (m individuals × n dimensions)
358 std::vector<std::vector<float>> data(m, std::vector<float>(n));
359 for (size_t j = 0; j < m; ++j)
360 for (size_t i = 0; i < n; ++i)
361 data[j][i] = population[j].at(pca.param_names[i]).value;
362
363 // Mean
364 pca.mean.assign(n, 0.0f);
365 for (size_t j = 0; j < m; ++j)
366 for (size_t i = 0; i < n; ++i)
367 pca.mean[i] += data[j][i];
368 for (size_t i = 0; i < n; ++i)
369 pca.mean[i] /= m;
370
371 // Center
372 for (size_t j = 0; j < m; ++j)
373 for (size_t i = 0; i < n; ++i)
374 data[j][i] -= pca.mean[i];
375
376 // Covariance: S = X^T X / (m-1)
377 std::vector<std::vector<float>> cov(n, std::vector<float>(n, 0.0f));
378 for (size_t j = 0; j < m; ++j)
379 for (size_t i1 = 0; i1 < n; ++i1)
380 for (size_t i2 = i1; i2 < n; ++i2)
381 cov[i1][i2] += data[j][i1] * data[j][i2];
382 for (size_t i1 = 0; i1 < n; ++i1)
383 for (size_t i2 = i1; i2 < n; ++i2) {
384 cov[i1][i2] /= (m - 1);
385 cov[i2][i1] = cov[i1][i2];
386 }
387
388 // Eigendecompose
389 jacobiEigen(cov, pca.eigenvalues, pca.eigenvectors);
390 pca.valid = true;
391 return pca;
392 }

395 class GaussianProcess {

397 GaussianProcess(size_t n_dims, float length_scale_base = 1.0f, float signal_variance = 1.0f, float noise_variance = 1e-4f) : n_dims_(n_dims), signal_variance_(signal_variance), noise_variance_(noise_variance) {

416 // Compute negative log marginal likelihood (for hyperparameter optimization)
417 // NLML = 0.5 * y^T * K^{-1} * y + 0.5 * log|K| + (n/2) * log(2π)
418 float computeNLML(const std::vector<std::vector<float>> &X, const std::vector<float> &y) const {
419 size_t n = X.size();
420 if (n == 0)
421 return 1e10f;
422
423 // Build kernel matrix K
424 std::vector<std::vector<float>> K(n, std::vector<float>(n));
425 for (size_t i = 0; i < n; ++i) {
426 for (size_t j = 0; j <= i; ++j) {
427 float k = kernel(X[i], X[j]);
428 K[i][j] = k;
429 K[j][i] = k;
430 }
431 K[i][i] += noise_variance_; // Add noise to diagonal
432 }
433
434 // Cholesky decomposition K = L * L^T
435 std::vector<std::vector<float>> L = K;
436 if (!choleskyDecompose(L)) {
437 // Not positive definite - return high penalty
438 return 1e10f;
439 }
440
441 // Compute data fit term: 0.5 * y^T * K^{-1} * y = 0.5 * ||L^{-1} * y||²
442 std::vector<float> Ly_inv = solveTriangular(L, y, true);
443 float data_fit = 0.0f;
444 for (float val: Ly_inv) {
445 data_fit += val * val;
446 }
447 data_fit *= 0.5f;
448
449 // Compute complexity penalty: 0.5 * log|K| = sum(log(diag(L)))
450 float log_det = 0.0f;
451 for (size_t i = 0; i < n; ++i) {
452 log_det += std::log(std::max(L[i][i], 1e-10f));
453 }
454
455 // Compute normalization constant: (n/2) * log(2π)
456 float normalization = 0.5f * n * std::log(2.0f * M_PI);
457
458 return data_fit + log_det + normalization;
459 }

461 void fit(const std::vector<std::vector<float>> &X, const std::vector<float> &y) {
462 X_ = X;
463 y_ = y;
464
465 size_t n = X.size();
466 K_ = std::vector<std::vector<float>>(n, std::vector<float>(n));
467
468 // Compute kernel matrix
469 for (size_t i = 0; i < n; ++i) {
470 for (size_t j = 0; j <= i; ++j) {
471 float k = kernel(X[i], X[j]);
472 K_[i][j] = k;
473 K_[j][i] = k;
474 }
475 K_[i][i] += noise_variance_; // Add noise to diagonal
476 }
477
478 // Cholesky decomposition
479 L_ = K_;
480 if (!choleskyDecompose(L_)) {
481 // If decomposition fails, add more noise to diagonal
482 for (size_t i = 0; i < n; ++i) {
483 K_[i][i] += 1e-3f;
484 }
485 L_ = K_;
486 choleskyDecompose(L_);
487 }
488
489 // Compute alpha = K^-1 * y = L^-T * L^-1 * y
490 std::vector<float> Ly_inv = solveTriangular(L_, y_, true);
491
492 // Transpose L for upper triangular solve
493 std::vector<std::vector<float>> LT(n, std::vector<float>(n, 0.0f));
494 for (size_t i = 0; i < n; ++i) {
495 for (size_t j = 0; j <= i; ++j) {
496 LT[j][i] = L_[i][j];
497 }
498 }
499 alpha_ = solveTriangular(LT, Ly_inv, false);
500 }

502 std::pair<float, float> predict(const std::vector<float> &x) const {
503 size_t n = X_.size();
504 std::vector<float> k_star(n);
505
506 for (size_t i = 0; i < n; ++i) {
507 k_star[i] = kernel(x, X_[i]);
508 }
509
510 // Mean prediction: k_star^T * alpha
511 float mean = 0.0f;
512 for (size_t i = 0; i < n; ++i) {
513 mean += k_star[i] * alpha_[i];
514 }
515
516 // Variance: k(x,x) - k_star^T * K^-1 * k_star
517 float k_xx = kernel(x, x);
518 std::vector<float> v = solveTriangular(L_, k_star, true);
519 float variance = k_xx;
520 for (size_t i = 0; i < n; ++i) {
521 variance -= v[i] * v[i];
522 }
523 variance = std::max(0.0f, variance); // Numerical stability
524
525 return {mean, std::sqrt(variance)};
526 }

528 private:
529 float kernel(const std::vector<float> &x1, const std::vector<float> &x2) const {
530 float dist_sq = 0.0f;
531 for (size_t i = 0; i < x1.size(); ++i) {
532 float diff = x1[i] - x2[i];
533 dist_sq += diff * diff;
534 }
535 return signal_variance_ * std::exp(-dist_sq / (2.0f * length_scale_ * length_scale_));
536 }

601 // Latin Hypercube Sampling for initial points
602 std::vector<std::vector<float>> latinHypercubeSampling(size_t n_samples, size_t n_dims, std::mt19937 &rng) {
603 std::vector<std::vector<float>> samples(n_samples, std::vector<float>(n_dims));
604 std::uniform_real_distribution<float> uni01(0.0f, 1.0f);
605
606 for (size_t d = 0; d < n_dims; ++d) {
607 std::vector<size_t> indices(n_samples);
608 for (size_t i = 0; i < n_samples; ++i)
609 indices[i] = i;
610 std::shuffle(indices.begin(), indices.end(), rng);
611
612 for (size_t i = 0; i < n_samples; ++i) {
613 float segment = 1.0f / n_samples;
614 samples[i][d] = (indices[i] + uni01(rng)) * segment;
615 }
616 }
617 return samples;
618 }

620 // Optimize GP hyperparameters using random search
621 // Returns optimal [length_scale, signal_variance, noise_variance]
622 std::vector<float> optimizeGPHyperparameters(const std::vector<std::vector<float>> &X, const std::vector<float> &y, size_t n_dims, std::mt19937 &rng, size_t n_trials = 50, const std::vector<float> &warm_start = {}) {
623
624 // Search ranges (log-scale for better coverage)
625 std::uniform_real_distribution<float> log_length_scale(std::log(0.01f), std::log(3.0f));
626 std::uniform_real_distribution<float> log_signal_var(std::log(0.1f), std::log(10.0f));
627 std::uniform_real_distribution<float> log_noise_var(std::log(1e-6f), std::log(0.1f));
628
629 float best_nlml = std::numeric_limits<float>::max();
630 std::vector<float> best_params = {
631 1.0f * std::sqrt(static_cast<float>(n_dims)), // Default length_scale
632 1.0f, // Default signal_variance
633 1e-4f // Default noise_variance
634 };
635
636 // Try default first
637 GaussianProcess gp_test(n_dims);
638 float default_nlml = gp_test.computeNLML(X, y);
639 best_nlml = default_nlml;
640
641 // Try warm-start hyperparameters from previous iteration (if provided)
642 if (warm_start.size() == 3) {
643 GaussianProcess gp_warm(n_dims);
644 gp_warm.setHyperparameters(warm_start[0], warm_start[1], warm_start[2]);
645 float warm_nlml = gp_warm.computeNLML(X, y);
646 if (warm_nlml < best_nlml) {
647 best_nlml = warm_nlml;
648 best_params = warm_start;
649 }
650 }
651
652 // Random search
653 for (size_t trial = 0; trial < n_trials; ++trial) {
654 float length_scale = std::exp(log_length_scale(rng)) * std::sqrt(static_cast<float>(n_dims));
655 float signal_var = std::exp(log_signal_var(rng));
656 float noise_var = std::exp(log_noise_var(rng));
657
658 GaussianProcess gp_trial(n_dims);
659 gp_trial.setHyperparameters(length_scale, signal_var, noise_var);
660 float nlml = gp_trial.computeNLML(X, y);
661
662 if (nlml < best_nlml) {
663 best_nlml = nlml;
664 best_params = {length_scale, signal_var, noise_var};
665 }
666 }
667
668 return best_params;
669 }
670} // namespace

687static ParameterOptimization::Result runBayesianOptimization(std::function<float(const ParametersToOptimize &)> simulation, const ParametersToOptimize &parameters, const BayesianOptimization &bo, bool print_progress,
688 const std::string &write_progress_to_file, std::ofstream &outProgress, bool message_flag) {
689
690 std::mt19937 rng(std::random_device{}());
691 helios::Timer timer;
692 timer.tic();
693
694 validateParameters(parameters);
695
696 // Get parameter names in consistent order
697 std::vector<std::string> param_names;
698 for (const auto &[name, _]: parameters) {
699 param_names.push_back(name);
700 }
701 std::sort(param_names.begin(), param_names.end());
702 size_t n_dims = param_names.size();
703
704 // Determine initial samples
705 size_t n_initial = bo.initial_samples > 0 ? bo.initial_samples : std::max<size_t>(2 * n_dims, 10);
706 n_initial = std::min(n_initial, bo.max_evaluations);
707
708 // Print starting message
709 if (print_progress) {
710 std::cout << "\n========================================\n";
711 std::cout << "Beginning Bayesian Optimization\n";
712 std::cout << "========================================\n";
713 std::cout << "Parameters to optimize: " << n_dims << "\n";
714 std::cout << "Max iterations: " << bo.max_evaluations << "\n";
715 std::cout << "Initial samples: " << n_initial << "\n";
716 std::cout << "UCB kappa (exploration): " << bo.ucb_kappa << "\n";
717 std::cout << "Acquisition samples: " << bo.acquisition_samples << "\n";
718 std::cout << "========================================\n\n";
719 }
720
721 // Generate initial samples using Latin Hypercube Sampling
722 auto initial_points = latinHypercubeSampling(n_initial, n_dims, rng);
723
724 // Evaluate initial points
725 std::vector<std::vector<float>> X;
726 std::vector<float> y;
727 std::vector<float> fitness_history; // Track all best fitness values for graphing
728 ParameterOptimization::Result best;
729 best.fitness = std::numeric_limits<float>::max();
730
731 for (size_t i = 0; i < n_initial; ++i) {
732 ParametersToOptimize params = vectorToParams(initial_points[i], parameters, param_names);
733 float fitness = simulation(params);
734
735 X.push_back(initial_points[i]);
736 y.push_back(fitness);
737
738 if (fitness < best.fitness) {
739 best.fitness = fitness;
740 best.parameters = params;
741 }
742
743 fitness_history.push_back(best.fitness);
744
745 int display_freq = computeDisplayFrequency(i + 1, bo.max_evaluations);
746
747 // Display graph at appropriate frequency or on first/last
748 if (print_progress && (i == 0 || (i + 1) % display_freq == 0 || i + 1 == n_initial)) {
749 double elapsed_time = timer.toc("mute");
750 displayOptimizationGraph(fitness_history, i + 1, bo.max_evaluations, elapsed_time, i == 0);
751 }
752 }
753
754 // Bayesian Optimization loop
755 std::vector<float> prev_hyperparams; // Warm-start for hyperparameter optimization
756 for (size_t iter = n_initial; iter < bo.max_evaluations; ++iter) {
757 // Limit GP training data if needed
758 if (X.size() > bo.max_gp_samples) {
759 // Keep best samples
760 std::vector<size_t> indices(X.size());
761 for (size_t i = 0; i < X.size(); ++i)
762 indices[i] = i;
763 std::partial_sort(indices.begin(), indices.begin() + bo.max_gp_samples, indices.end(), [&](size_t a, size_t b) { return y[a] < y[b]; });
764
765 std::vector<std::vector<float>> X_new(bo.max_gp_samples);
766 std::vector<float> y_new(bo.max_gp_samples);
767 for (size_t i = 0; i < bo.max_gp_samples; ++i) {
768 X_new[i] = X[indices[i]];
769 y_new[i] = y[indices[i]];
770 }
771 X = X_new;
772 y = y_new;
773 }
774
775 // Recompute normalization statistics from current data each iteration
776 float y_mean = 0.0f, y_std = 0.0f;
777 for (float val: y)
778 y_mean += val;
779 y_mean /= y.size();
780 for (float val: y)
781 y_std += (val - y_mean) * (val - y_mean);
782 y_std = std::sqrt(y_std / y.size());
783 if (y_std < 1e-6f)
784 y_std = 1.0f;
785
786 std::vector<float> y_normalized(y.size());
787 for (size_t i = 0; i < y.size(); ++i) {
788 y_normalized[i] = (y[i] - y_mean) / y_std;
789 }
790
791 // Optimize GP hyperparameters — full search periodically, reuse previous otherwise
792 if (prev_hyperparams.empty() || (iter - n_initial) % 20 == 0) {
793 prev_hyperparams = optimizeGPHyperparameters(X, y_normalized, n_dims, rng, 50, prev_hyperparams);
794 }
795
796 // Fit Gaussian Process with optimized hyperparameters
797 GaussianProcess gp(n_dims);
798 gp.setHyperparameters(prev_hyperparams[0], prev_hyperparams[1], prev_hyperparams[2]);
799 gp.fit(X, y_normalized);
800
801 // Optimize acquisition function (UCB)
802 std::uniform_real_distribution<float> uni01(0.0f, 1.0f);
803 std::vector<float> best_candidate;
804 float best_acquisition = -std::numeric_limits<float>::max();
805
806 for (size_t i = 0; i < bo.acquisition_samples; ++i) {
807 std::vector<float> candidate(n_dims);
808 for (size_t d = 0; d < n_dims; ++d) {
809 candidate[d] = uni01(rng);
810 }
811
812 auto [mean, std] = gp.predict(candidate);
813 // Convert back to original scale (approximately)
814 mean = mean * y_std + y_mean;
815 std = std * y_std;
816
817 // UCB for minimization: lower mean is better, higher uncertainty is better
818 // acq = -mean + kappa * std (higher is better)
819 float acq = -mean + bo.ucb_kappa * std;
820
821 if (acq > best_acquisition) {
822 best_acquisition = acq;
823 best_candidate = candidate;
824 }
825 }
826
827 // Evaluate best candidate
828 ParametersToOptimize next_params = vectorToParams(best_candidate, parameters, param_names);
829 float fitness = simulation(next_params);
830
831 X.push_back(best_candidate);
832 y.push_back(fitness);
833
834 if (fitness < best.fitness) {
835 best.fitness = fitness;
836 best.parameters = next_params;
837 }
838
839 fitness_history.push_back(best.fitness);
840
841 int display_freq = computeDisplayFrequency(iter + 1, bo.max_evaluations);
842
843 // Display graph at appropriate frequency or on last iteration
844 if (print_progress && ((iter + 1) % display_freq == 0 || iter + 1 == bo.max_evaluations)) {
845 double elapsed_time = timer.toc("mute");
846 displayOptimizationGraph(fitness_history, iter + 1, bo.max_evaluations, elapsed_time, false);
847 }
848
849 // Write to progress file every iteration (not just on improvement)
850 if (!write_progress_to_file.empty() && outProgress.is_open()) {
851 for (const auto &[name, param]: best.parameters) {
852 outProgress << iter + 1 << "," << best.fitness << "," << name << "," << param.value << "," << param.min << "," << param.max << "\n";
853 }
854 }
855 }
856
857 // Print final summary
858 if (print_progress) {
859 double total_time = timer.toc("mute");
860 std::cout << "\n========================================\n";
861 std::cout << "Bayesian Optimization Complete\n";
862 std::cout << "========================================\n";
863 std::cout << "Total time: " << std::fixed << std::setprecision(2) << total_time << " s\n";
864 std::cout << "Best fitness: " << std::setprecision(6) << best.fitness << "\n";
865 std::cout << "\nBest parameters:\n";
866 for (const auto &[name, param]: best.parameters) {
867 std::cout << " " << name << ": " << param.value << "\n";
868 }
869 std::cout << "========================================\n";
870 }
871
872 return best;
873}

875// Genetic Algorithm implementation
876static ParameterOptimization::Result runGeneticAlgorithm(std::function<float(const ParametersToOptimize &)> simulation, const ParametersToOptimize &parameters, const GeneticAlgorithm &ga, bool print_progress,
877 const std::string &write_progress_to_file, const std::string &write_result_to_file, const std::string &read_input_from_file, std::ofstream &outProgress, bool message_flag) {
878
879 // Genetic Algorithm
880 helios::Timer timer;
881 timer.tic();
882
883 std::mt19937 rng(std::random_device{}());
884 std::uniform_real_distribution<float> uni01(0.f, 1.f);
885
886 validateParameters(parameters);
887
888 // Fitness cache to avoid redundant evaluations
889 std::unordered_map<ParametersToOptimize, float, ParametersHash, ParametersEqual> fitness_cache;
890 size_t cache_hits = 0;
891 size_t total_queries = 0;
892
893 // Cached simulation wrapper
894 auto cached_simulation = [&](const ParametersToOptimize &params) -> float {
895 total_queries++;
896 auto it = fitness_cache.find(params);
897 if (it != fitness_cache.end()) {
898 cache_hits++;
899 return it->second;
900 }
901 float fitness = simulation(params);
902 fitness_cache[params] = fitness;
903 return fitness;
904 };
905
906 size_t num_generations = ga.generations;
907 size_t popsize = std::max<size_t>(2, ga.population_size);
908
909 // Extract mutation rate from the mutation variant
910 float mutation_rate = std::visit([](const auto &m) { return m.rate; }, ga.mutation);
911
912 // Print starting message
913 if (print_progress) {
914 std::cout << "\n========================================\n";
915 std::cout << "Beginning Genetic Algorithm Optimization\n";
916 std::cout << "========================================\n";
917 std::cout << "Parameters to optimize: " << parameters.size() << "\n";
918 std::cout << "Generations: " << num_generations << "\n";
919 std::cout << "Population size: " << popsize << "\n";
920 std::cout << "Mutation rate: " << mutation_rate << "\n";
921 std::cout << "Crossover rate: " << ga.crossover_rate << "\n";
922 std::cout << "Elitism rate: " << ga.elitism_rate << "\n";
923 std::cout << "========================================\n\n";
924 }
925 std::vector<ParametersToOptimize> population(popsize);
926 for (auto &ind: population) {
927 ind = parameters;
928 for (auto &[name, param]: ind) {
929 if (param.type == ParameterType::CATEGORICAL) {
930 std::uniform_int_distribution<size_t> cat_dist(0, param.categories.size() - 1);
931 param.value = param.categories[cat_dist(rng)];
932 } else {
933 param.value = param.min + (param.max - param.min) * uni01(rng);
934 param.value = roundIfInteger(param.value, param.type);
935 }
936 }
937 }
938
939 if (!read_input_from_file.empty()) {
940 ParametersToOptimize best_from_file = readParametersFromFile(read_input_from_file);
941 population[0] = best_from_file; // seed with known best
942 }
943
944 ParameterOptimization::Result best = evaluatePopulation(population, cached_simulation);
945
946 std::vector<double> generation_times;
947 helios::Timer gen_timer;
948
949 PCAState pca; // Persists across generations for BLXPCA and HybridMutation
950
951 // Determine PCA update needs
952 bool crossover_needs_pca = std::holds_alternative<BLXPCACrossover>(ga.crossover);
953 bool mutation_needs_pca = std::holds_alternative<HybridMutation>(ga.mutation);
954 bool needs_pca = crossover_needs_pca || mutation_needs_pca;
955 size_t pca_interval = SIZE_MAX;
956 if (crossover_needs_pca)
957 pca_interval = std::min(pca_interval, std::get<BLXPCACrossover>(ga.crossover).pca_update_interval);
958 if (mutation_needs_pca)
959 pca_interval = std::min(pca_interval, std::get<HybridMutation>(ga.mutation).pca_update_interval);
960
961 for (size_t g = 0; g < num_generations; g++) {
962 gen_timer.tic();
963 std::vector<ParametersToOptimize> new_pop(popsize);
964
965 // Pre-compute fitness for entire population (pure lookup, no side effects in comparators)
966 std::vector<float> pop_fitness(popsize);
967 for (size_t i = 0; i < popsize; ++i) {
968 pop_fitness[i] = cached_simulation(population[i]);
969 }
970
971 // Elitism: keep top individuals
972 size_t elitism_count = std::max<size_t>(1, popsize * ga.elitism_rate);
973 std::vector<size_t> elite_indices(popsize);
974 for (size_t i = 0; i < popsize; ++i)
975 elite_indices[i] = i;
976 std::partial_sort(elite_indices.begin(), elite_indices.begin() + elitism_count, elite_indices.end(), [&](size_t a, size_t b) { return pop_fitness[a] < pop_fitness[b]; });
977 for (size_t e = 0; e < elitism_count; ++e) {
978 new_pop[e] = population[elite_indices[e]];
979 }
980
981 std::uniform_int_distribution<size_t> dist(0, popsize - 1);
982
983 // Compute PCA periodically if needed
984 if (needs_pca && g % pca_interval == 0) {
985 pca = computePCA(population, parameters);
986 }
987
988 for (size_t i = elitism_count; i < popsize; i++) {
989 // Selection, Tournament (uses pre-computed fitness)
990 auto tournamentSelect = [&](int k = 3) -> const ParametersToOptimize & {
991 size_t bestIdx = dist(rng);
992 for (int j = 1; j < k; ++j) {
993 size_t challengerIdx = dist(rng);
994 if (pop_fitness[challengerIdx] < pop_fitness[bestIdx]) {
995 bestIdx = challengerIdx;
996 }
997 }
998 return population[bestIdx];
999 };
1000
1001 const auto &parent1 = tournamentSelect();
1002 const auto &parent2 = tournamentSelect();
1003
1004 ParametersToOptimize child = parameters;
1005
1006 // Categorical parameters: always per-gene crossover (unaffected by crossover type)
1007 for (auto &[name, param]: child) {
1008 if (param.type == ParameterType::CATEGORICAL) {
1009 const auto &p1 = parent1.at(name);
1010 param.value = p1.value;
1011 if (uni01(rng) < ga.crossover_rate) {
1012 param.value = (uni01(rng) < 0.5f) ? p1.value : parent2.at(name).value;
1013 }
1014 if (uni01(rng) < mutation_rate) {
1015 std::uniform_int_distribution<size_t> cat_dist(0, param.categories.size() - 1);
1016 param.value = param.categories[cat_dist(rng)];
1017 }
1018 }
1019 }
1020
1021 // Phase 1: Crossover for numeric parameters
1022 std::visit(
1023 [&](const auto &crossover_op) {
1024 using T = std::decay_t<decltype(crossover_op)>;
1025 if constexpr (std::is_same_v<T, BLXPCACrossover>) {
1026 // BLXPCA: transform parents to PCA space, blend, transform back
1027 if (pca.valid) {
1028 auto z1 = pca.transform(parent1);
1029 auto z2 = pca.transform(parent2);
1030 size_t n_pca = pca.param_names.size();
1031
1032 std::vector<float> z_child(n_pca);
1033 bool do_crossover = uni01(rng) < ga.crossover_rate;
1034 for (size_t d = 0; d < n_pca; ++d) {
1035 if (do_crossover) {
1036 float alpha = crossover_op.alpha;
1037 float minval = std::min(z1[d], z2[d]);
1038 float maxval = std::max(z1[d], z2[d]);
1039 float range = maxval - minval;
1040 z_child[d] = minval - alpha * range + uni01(rng) * (1.f + 2.f * alpha) * range;
1041 } else {
1042 z_child[d] = z1[d];
1043 }
1044 }
1045
1046 // Transform back to original space
1047 auto vals = pca.inverseTransform(z_child);
1048 for (size_t d = 0; d < n_pca; ++d) {
1049 child.at(pca.param_names[d]).value = vals[d];
1050 }
1051 } else {
1052 // Fallback to BLX-alpha if PCA not valid
1053 if (g == 0 && i == elitism_count && print_progress) {
1054 std::cout << "WARNING (ParameterOptimization): PCA computation failed; BLXPCA crossover falling back to standard BLX-alpha.\n";
1055 }
1056 for (auto &[name, param]: child) {
1057 if (param.type == ParameterType::CATEGORICAL)
1058 continue;
1059 const auto &p1 = parent1.at(name);
1060 const auto &p2 = parent2.at(name);
1061
1062 float val = p1.value;
1063 if (uni01(rng) < ga.crossover_rate) {
1064 float alpha = crossover_op.alpha;
1065 float minval = std::min(p1.value, p2.value);
1066 float maxval = std::max(p1.value, p2.value);
1067 float range = maxval - minval;
1068 val = minval - alpha * range + uni01(rng) * (1.f + 2.f * alpha) * range;
1069 }
1070 param.value = val;
1071 }
1072 }
1073 } else {
1074 // BLXAlphaCrossover: per-gene crossover for numeric params
1075 for (auto &[name, param]: child) {
1076 if (param.type == ParameterType::CATEGORICAL)
1077 continue;
1078 const auto &p1 = parent1.at(name);
1079 const auto &p2 = parent2.at(name);
1080
1081 float val = p1.value;
1082 if (uni01(rng) < ga.crossover_rate) {
1083 float alpha = crossover_op.alpha;
1084 float minval = std::min(p1.value, p2.value);
1085 float maxval = std::max(p1.value, p2.value);
1086 float range = maxval - minval;
1087 val = minval - alpha * range + uni01(rng) * (1.f + 2.f * alpha) * range;
1088 }
1089 param.value = val;
1090 }
1091 }
1092 },
1093 ga.crossover);
1094
1095 // Phase 2: Mutation for numeric parameters
1096 std::visit(
1097 [&](const auto &mutation_op) {
1098 using T = std::decay_t<decltype(mutation_op)>;
1099 if constexpr (std::is_same_v<T, PerGeneMutation>) {
1100 // Per-gene Gaussian mutation
1101 for (auto &[name, param]: child) {
1102 if (param.type == ParameterType::CATEGORICAL)
1103 continue;
1104 if (uni01(rng) < mutation_op.rate) {
1105 std::normal_distribution<float> gauss(0.f, 0.1f * (param.max - param.min));
1106 param.value += gauss(rng);
1107 }
1108 }
1109 } else if constexpr (std::is_same_v<T, IsotropicMutation>) {
1110 // Isotropic: single gate, mutate all genes together
1111 bool do_mutate = uni01(rng) < mutation_op.rate;
1112 if (do_mutate) {
1113 for (auto &[name, param]: child) {
1114 if (param.type == ParameterType::CATEGORICAL)
1115 continue;
1116 std::normal_distribution<float> gauss(0.f, 0.1f * (param.max - param.min));
1117 param.value += gauss(rng);
1118 }
1119 }
1120 } else {
1121 // HybridMutation
1122 bool do_mutate = uni01(rng) < mutation_op.rate;
1123 if (do_mutate && pca.valid) {
1124 size_t n_pca = pca.param_names.size();
1125 // Roll sub-strategy
1126 float strategy_roll = uni01(rng);
1127 if (strategy_roll < mutation_op.pca_gaussian_prob) {
1128 // PCA-Gaussian
1129 auto z_child = pca.transform(child);
1130 for (size_t d = 0; d < n_pca; ++d) {
1131 std::normal_distribution<float> gauss(0.f, mutation_op.sigma_pca);
1132 z_child[d] += gauss(rng);
1133 }
1134 auto vals = pca.inverseTransform(z_child);
1135 for (size_t d = 0; d < n_pca; ++d) {
1136 child.at(pca.param_names[d]).value = vals[d];
1137 }
1138 } else if (strategy_roll < mutation_op.pca_gaussian_prob + mutation_op.pca_cauchy_prob) {
1139 // PCA-Cauchy (heavy-tailed)
1140 auto z_child = pca.transform(child);
1141 std::cauchy_distribution<float> cauchy(0.f, mutation_op.gamma_cauchy);
1142 for (size_t d = 0; d < n_pca; ++d) {
1143 z_child[d] += cauchy(rng);
1144 }
1145 auto vals = pca.inverseTransform(z_child);
1146 for (size_t d = 0; d < n_pca; ++d) {
1147 child.at(pca.param_names[d]).value = vals[d];
1148 }
1149 } else {
1150 // Random Direction
1151 std::vector<float> direction(n_pca);
1152 float norm = 0.0f;
1153 for (size_t d = 0; d < n_pca; ++d) {
1154 std::normal_distribution<float> gauss(0.f, 1.f);
1155 direction[d] = gauss(rng);
1156 norm += direction[d] * direction[d];
1157 }
1158 norm = std::sqrt(norm);
1159 for (size_t d = 0; d < n_pca; ++d)
1160 direction[d] /= norm;
1161
1162 // Scale by half-normal step size × avg parameter range
1163 float avg_range = 0.0f;
1164 for (const auto &pname: pca.param_names) {
1165 const auto &p = child.at(pname);
1166 avg_range += (p.max - p.min);
1167 }
1168 avg_range /= pca.param_names.size();
1169
1170 std::normal_distribution<float> half_normal(0.f, 1.f);
1171 float step_size = std::abs(half_normal(rng)) * mutation_op.sigma_random * avg_range;
1172
1173 // Apply
1174 for (size_t d = 0; d < n_pca; ++d) {
1175 child.at(pca.param_names[d]).value += step_size * direction[d];
1176 }
1177 }
1178 } else if (do_mutate && !pca.valid) {
1179 // Fallback to per-gene mutation when PCA not available
1180 if (g == 0 && i == elitism_count && print_progress) {
1181 std::cout << "WARNING (ParameterOptimization): PCA computation failed; HybridMutation falling back to per-gene mutation.\n";
1182 }
1183 for (auto &[name, param]: child) {
1184 if (param.type == ParameterType::CATEGORICAL)
1185 continue;
1186 std::normal_distribution<float> gauss(0.f, 0.1f * (param.max - param.min));
1187 param.value += gauss(rng);
1188 }
1189 }
1190 }
1191 },
1192 ga.mutation);
1193
1194 // Phase 3: Clamp and round all numeric params
1195 for (auto &[name, param]: child) {
1196 if (param.type != ParameterType::CATEGORICAL) {
1197 param.value = std::min(std::max(param.value, param.min), param.max);
1198 param.value = roundIfInteger(param.value, param.type);
1199 }
1200 }
1201
1202 new_pop[i] = child;
1203 }
1204
1205 population = std::move(new_pop);
1206 ParameterOptimization::Result generation_best = evaluatePopulation(population, cached_simulation);
1207
1208 // Record generation time
1209 double gen_time = gen_timer.toc("mute");
1210 generation_times.push_back(gen_time);
1211
1212 if (generation_best.fitness <= best.fitness) {
1213 best = generation_best;
1214 if (print_progress) {
1215 double elapsed_time = timer.toc("mute");
1216
1217 // Calculate average time per generation and estimate time remaining
1218 double avg_gen_time = 0.0;
1219 for (double t: generation_times) {
1220 avg_gen_time += t;
1221 }
1222 avg_gen_time /= generation_times.size();
1223
1224 size_t remaining_gens = num_generations - g - 1;
1225 double estimated_remaining = avg_gen_time * remaining_gens;
1226

1310// CMA-ES (Covariance Matrix Adaptation Evolution Strategy) implementation
1311// Based on Hansen & Ostermeier 2001, Hansen 2016 tutorial
1312static ParameterOptimization::Result runCMAES(std::function<float(const ParametersToOptimize &)> simulation, const ParametersToOptimize &parameters, const CMAES &cmaes, bool print_progress, const std::string &write_progress_to_file,
1313 const std::string &write_result_to_file, std::ofstream &outProgress, bool message_flag) {
1314
1315 std::mt19937 rng(std::random_device{}());
1316 helios::Timer timer;
1317 timer.tic();
1318
1319 validateParameters(parameters);
1320
1321 // Get parameter names in consistent order
1322 std::vector<std::string> param_names;
1323 for (const auto &[name, _]: parameters) {
1324 param_names.push_back(name);
1325 }
1326 std::sort(param_names.begin(), param_names.end());
1327 size_t n = param_names.size();
1328
1329 // CMA-ES Strategy Parameters (following Hansen)
1330 size_t lambda = cmaes.lambda > 0 ? cmaes.lambda : (4 + static_cast<size_t>(floor(3.0 * log(static_cast<double>(n)))));
1331 size_t mu = lambda / 2; // Number of parents
1332
1333 // Recombination weights
1334 std::vector<float> weights(mu);
1335 float sum_weights = 0.0f;
1336 for (size_t i = 0; i < mu; ++i) {
1337 weights[i] = log(mu + 0.5f) - log(i + 1.0f);
1338 sum_weights += weights[i];
1339 }
1340 for (size_t i = 0; i < mu; ++i) {
1341 weights[i] /= sum_weights;
1342 }
1343
1344 float mu_eff = 1.0f / std::inner_product(weights.begin(), weights.end(), weights.begin(), 0.0f);
1345
1346 // Adaptation parameters
1347 float c_c = (4.0f + mu_eff / n) / (n + 4.0f + 2.0f * mu_eff / n);
1348 float c_s = (mu_eff + 2.0f) / (n + mu_eff + 5.0f);
1349 float c_1 = 2.0f / ((n + 1.3f) * (n + 1.3f) + mu_eff);
1350 float c_mu = std::min(1.0f - c_1, 2.0f * (mu_eff - 2.0f + 1.0f / mu_eff) / ((n + 2.0f) * (n + 2.0f) + mu_eff));
1351 float d_s = 1.0f + 2.0f * std::max(0.0f, sqrt((mu_eff - 1.0f) / (n + 1.0f)) - 1.0f) + c_s;
1352 float chi_n = sqrt(n) * (1.0f - 1.0f / (4.0f * n) + 1.0f / (21.0f * n * n));
1353
1354 // Initialize state
1355 std::vector<float> mean(n, 0.5f); // Start at center [0,1]
1356 float sigma = cmaes.sigma;
1357 std::vector<float> p_c(n, 0.0f);
1358 std::vector<float> p_s(n, 0.0f);
1359 std::vector<std::vector<float>> C(n, std::vector<float>(n, 0.0f));
1360 for (size_t i = 0; i < n; ++i)
1361 C[i][i] = 1.0f;
1362
1363 // Print starting message
1364 if (print_progress) {
1365 std::cout << "\n========================================\n";
1366 std::cout << "Beginning CMA-ES Optimization\n";
1367 std::cout << "========================================\n";
1368 std::cout << "Parameters to optimize: " << n << "\n";
1369 std::cout << "Max iterations: " << cmaes.max_evaluations << "\n";
1370 std::cout << "Population size (λ): " << lambda << "\n";
1371 std::cout << "Parents (μ): " << mu << "\n";
1372 std::cout << "Initial step-size (σ): " << sigma << "\n";
1373 std::cout << "========================================\n\n";
1374 }
1375
1376 ParameterOptimization::Result best;
1377 best.fitness = std::numeric_limits<float>::max();
1378 std::vector<float> fitness_history;
1379 std::normal_distribution<float> norm(0.0f, 1.0f);
1380
1381 size_t gen = 0;
1382 size_t total_evals = 0;
1383
1384 while (total_evals < cmaes.max_evaluations) {
1385 // Sample lambda offspring
1386 std::vector<std::vector<float>> offspring(lambda, std::vector<float>(n));
1387 std::vector<std::vector<float>> y_samples(lambda, std::vector<float>(n)); // Store y = L*z
1388 std::vector<float> fitnesses(lambda);
1389
1390 // Compute C^(1/2) using Cholesky decomposition: C = L * L^T, so C^(1/2) = L
1391 std::vector<std::vector<float>> L = C;
1392 bool chol_success = choleskyDecompose(L);
1393
1394 if (!chol_success) {
1395 // Add regularization to diagonal if Cholesky fails
1396 for (size_t i = 0; i < n; ++i) {
1397 C[i][i] += 1e-6f;
1398 }
1399 L = C;
1400 choleskyDecompose(L);
1401 }
1402
1403 for (size_t k = 0; k < lambda && total_evals < cmaes.max_evaluations; ++k) {
1404 // Sample z ~ N(0,I)
1405 std::vector<float> z(n);
1406 for (size_t i = 0; i < n; ++i) {
1407 z[i] = norm(rng);
1408 }
1409
1410 // y = C^(1/2) * z = L * z
1411 for (size_t i = 0; i < n; ++i) {
1412 y_samples[k][i] = 0.0f;
1413 for (size_t j = 0; j <= i; ++j) {
1414 y_samples[k][i] += L[i][j] * z[j];
1415 }
1416 }
1417
1418 // x = mean + sigma * y
1419 for (size_t i = 0; i < n; ++i) {
1420 offspring[k][i] = mean[i] + sigma * y_samples[k][i];
1421 offspring[k][i] = std::max(0.0f, std::min(1.0f, offspring[k][i]));
1422 }
1423
1424 // Evaluate
1425 ParametersToOptimize params = vectorToParams(offspring[k], parameters, param_names);
1426 fitnesses[k] = simulation(params);
1427 total_evals++;
1428
1429 if (fitnesses[k] < best.fitness) {
1430 best.fitness = fitnesses[k];
1431 best.parameters = params;
1432 }
1433 }
1434
1435 fitness_history.push_back(best.fitness);
1436
1437 // Display graph
1438 int display_freq = computeDisplayFrequency(gen + 1, cmaes.max_evaluations);
1439 if (print_progress && (gen == 0 || (gen + 1) % display_freq == 0 || total_evals >= cmaes.max_evaluations)) {
1440 double elapsed_time = timer.toc("mute");
1441 displayOptimizationGraph(fitness_history, total_evals, cmaes.max_evaluations, elapsed_time, gen == 0);
1442 }
1443
1444 // Write progress
1445 if (!write_progress_to_file.empty() && outProgress.is_open()) {
1446 for (const auto &[name, param]: best.parameters) {
1447 outProgress << gen << "," << best.fitness << "," << name << "," << param.value << "," << param.min << "," << param.max << "\n";
1448 }
1449 }
1450
1451 if (total_evals >= cmaes.max_evaluations)
1452 break;
1453
1454 // Sort by fitness
1455 std::vector<size_t> indices(lambda);
1456 for (size_t i = 0; i < lambda; ++i)
1457 indices[i] = i;
1458 std::partial_sort(indices.begin(), indices.begin() + mu, indices.end(), [&](size_t a, size_t b) { return fitnesses[a] < fitnesses[b]; });
1459
1460 // Compute effective displacement vectors from clamped offspring
1461 // This ensures consistency: the covariance adapts to the actual evaluated positions
1462 std::vector<std::vector<float>> y_effective(lambda, std::vector<float>(n));
1463 std::vector<float> old_mean = mean;
1464 for (size_t k = 0; k < lambda; ++k) {
1465 for (size_t i = 0; i < n; ++i) {
1466 y_effective[k][i] = (offspring[k][i] - old_mean[i]) / sigma;
1467 }
1468 }
1469
1470 // Update mean using clamped offspring (the positions we actually evaluated)
1471 for (size_t i = 0; i < n; ++i) {
1472 mean[i] = 0.0f;
1473 for (size_t k = 0; k < mu; ++k) {
1474 mean[i] += weights[k] * offspring[indices[k]][i];
1475 }
1476 }
1477
1478 // Compute mean step in search space: (mean_new - mean_old) / sigma
1479 std::vector<float> mean_step(n);
1480 for (size_t i = 0; i < n; ++i) {
1481 mean_step[i] = (mean[i] - old_mean[i]) / sigma;
1482 }
1483
1484 // Update evolution path for sigma (p_s)
1485 // C = L * L^T, so L^(-1) applied to the mean step gives the isotropic step
1486 std::vector<float> p_s_step = solveTriangular(L, mean_step, true);
1487
1488 float coeff_s = sqrt(c_s * (2.0f - c_s) * mu_eff);
1489 for (size_t i = 0; i < n; ++i) {
1490 p_s[i] = (1.0f - c_s) * p_s[i] + coeff_s * p_s_step[i];
1491 }
1492
1493 float norm_p_s = sqrt(std::inner_product(p_s.begin(), p_s.end(), p_s.begin(), 0.0f));
1494
1495 // Update evolution path for covariance (p_c)
1496 float h_s = (norm_p_s / sqrt(1.0f - pow(1.0f - c_s, 2.0f * (gen + 1.0f)))) < (1.4f + 2.0f / (n + 1.0f)) * chi_n ? 1.0f : 0.0f;
1497
1498 float coeff_c = sqrt(c_c * (2.0f - c_c) * mu_eff);
1499 for (size_t i = 0; i < n; ++i) {
1500 p_c[i] = (1.0f - c_c) * p_c[i] + h_s * coeff_c * mean_step[i];
1501 }
1502
1503 // Update C (rank-one + rank-mu)
1504 for (size_t i = 0; i < n; ++i) {
1505 for (size_t j = i; j < n; ++j) {
1506 float val = (1.0f - c_1 - c_mu) * C[i][j];
1507
1508 // Rank-one
1509 val += c_1 * p_c[i] * p_c[j];
1510
1511 // Rank-mu: use effective displacement vectors from clamped offspring
1512 for (size_t k = 0; k < mu; ++k) {
1513 val += c_mu * weights[k] * y_effective[indices[k]][i] * y_effective[indices[k]][j];
1514 }
1515
1516 C[i][j] = val;
1517 C[j][i] = val;
1518 }
1519 }
1520
1521 // Update sigma
1522 sigma *= exp((c_s / d_s) * (norm_p_s / chi_n - 1.0f));
1523 sigma = std::min(10.0f, std::max(1e-10f, sigma)); // Bounds
1524
1525 gen++;
1526 }
1527
1528 // Write final result to file if requested
1529 if (!write_result_to_file.empty()) {
1530 std::ofstream out(write_result_to_file);
1531 if (out.is_open()) {
1532 out << "parameter,value,min,max\n";
1533 for (const auto &[name, param]: best.parameters) {
1534 out << name << "," << param.value << "," << param.min << "," << param.max << "\n";
1535 }
1536 out.close();
1537 if (message_flag) {
1538 std::cout << "Best parameters written to: " << write_result_to_file << "\n";
1539 }
1540 } else {
1541 helios_runtime_error("ERROR (ParameterOptimization): Failed to write to result file '" + write_result_to_file + "'.");
1542 }
1543 }
1544
1545 // Print final summary
1546 if (print_progress) {
1547 double total_time = timer.toc("mute");
1548 std::cout << "\n========================================\n";
1549 std::cout << "CMA-ES Optimization Complete\n";
1550 std::cout << "========================================\n";
1551 std::cout << "Total time: " << std::fixed << std::setprecision(2) << total_time << " s\n";
1552 std::cout << "Generations: " << gen << "\n";
1553 std::cout << "Evaluations: " << total_evals << "\n";
1554 std::cout << "Best fitness: " << std::setprecision(6) << best.fitness << "\n";
1555 std::cout << "Final step-size (σ): " << std::setprecision(4) << sigma << "\n";
1556 std::cout << "\nBest parameters:\n";
1557 for (const auto &[name, param]: best.parameters) {
1558 std::cout << " " << name << ": " << param.value << "\n";
1559 }
1560 std::cout << "========================================\n";
1561 }
1562
1563 return best;
1564}
