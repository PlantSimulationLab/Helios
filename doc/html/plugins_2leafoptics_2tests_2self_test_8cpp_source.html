443DOCTEST_TEST_CASE("LeafOptics Physical Realism Checks") {
444 Context context_test;
445 LeafOptics leafoptics(&context_test);
446 leafoptics.disableMessages();
447
448 LeafOpticsProperties props;
449 props.numberlayers = 2.0f;
450 props.chlorophyllcontent = 60.0f;
451
452 std::vector<vec2> reflectivities, transmissivities;
453 leafoptics.getLeafSpectra(props, reflectivities, transmissivities);
454
455 // Check for physically realistic behavior
456 // 1. Red edge should show increased reflectance around 700-750 nm
457 bool found_red_edge = false;
458 for (size_t i = 1; i < reflectivities.size() - 1; ++i) {
459 float wavelength = reflectivities[i].x;
460 if (wavelength >= 700.0f && wavelength <= 750.0f) {
461 float slope = reflectivities[i + 1].y - reflectivities[i - 1].y;
462 if (slope > 0.001f) { // Positive slope indicating red edge
463 found_red_edge = true;
464 break;
465 }
466 }
467 }
468 DOCTEST_CHECK(found_red_edge);
469
470 // 2. NIR reflectance should generally be higher than visible
471 float visible_avg = 0.0f, nir_avg = 0.0f;
472 int visible_count = 0, nir_count = 0;
473
474 for (const auto &refl: reflectivities) {
475 if (refl.x >= 400.0f && refl.x <= 700.0f) {
476 visible_avg += refl.y;
477 visible_count++;
478 } else if (refl.x >= 800.0f && refl.x <= 1200.0f) {
479 nir_avg += refl.y;
480 nir_count++;
481 }
482 }
483
484 visible_avg /= visible_count;
485 nir_avg /= nir_count;
486
487 DOCTEST_CHECK(nir_avg > visible_avg); // NIR should be higher than visible
488}
