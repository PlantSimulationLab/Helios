192 
193  float fe = 0.5;
194  float Rxe = 200.f;
195 
196  const auto* coeffs = reinterpret_cast<const BBcoefficients*>(parameters);
197 
198  float pi_0 = coeffs->pi_0;
199  float pi_m = coeffs->pi_m;
200  float theta = coeffs->theta;
201  float sigma = coeffs->sigma;
202  float chi = coeffs->chi;
203 
204  float i = variables[0];
205  float D = variables[1]*1e-3F;
206  float Psix = variables[2];
207 
208  float pig = pi_0+(pi_m-pi_0)*i/(i+theta)+sigma*(Psix-Rxe*fe*gs*D);
209  float Psi_e = Psix-Rxe*fe*gs*D;
