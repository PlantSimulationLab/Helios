532float StomatalConductanceModel::evaluate_BWBmodel( float esurf, std::vector<float> &variables, const void* parameters ){
533
534 // We want to find the vapor pressure at the surface, esurf, that balances the equation gs(esurf)*(es-esurf) = gbw*(esurf-ea). This function returns the residual of this equation.
535
536 const auto* coeffs = reinterpret_cast<const BWBcoefficients*>(parameters);
537
538 float gs0 = coeffs->gs0;
539 float a1 = coeffs->a1;
540
541 float An = variables[0];
542 float Cs = variables[1];
543 float es = variables[2];
544 float ea = variables[3];
545 float gbw = variables[4];
546 float beta = variables[5];
547
548 float hs = esurf/es;
549 float gs = gs0+a1*An*beta*hs/Cs;
550
551 return gs*(es-esurf) - gbw*(esurf-ea);
552
553}

555float StomatalConductanceModel::evaluate_BBLmodel( float esurf, std::vector<float> &variables, const void* parameters ){
556
557 // We want to find the vapor pressure at the surface, esurf, that balances the equation gs(esurf)*(es-esurf) = gbw*(esurf-ea). This function returns the residual of this equation.
558
559 const auto* coeffs = reinterpret_cast<const BBLcoefficients*>(parameters);
560
561 float gs0 = coeffs->gs0;
562 float a1 = coeffs->a1;
563 float D0 = coeffs->D0;
564
565 float An = variables[0];
566 float Cs = variables[1];
567 float Gamma = variables[2];
568 float es = variables[3];
569 float ea = variables[4];
570 float gbw = variables[5];

581float StomatalConductanceModel::evaluate_MOPTmodel( float esurf, std::vector<float> &variables, const void* parameters ){
582
583 // We want to find the vapor pressure at the surface, esurf, that balances the equation gs(esurf)*(es-esurf) = gbw*(esurf-ea). This function returns the residual of this equation.
584
585 const auto* coeffs = reinterpret_cast<const MOPTcoefficients*>(parameters);
586
587 float gs0 = coeffs->gs0;
588 float g1 = coeffs->g1;
589
590 float An = variables[0];
591 float Cs = variables[1];
592 float es = variables[2];
593 float ea = variables[3];
594 float gbw = variables[4];
595 float beta = variables[5];
596
597 float Ds = max(0.f, (es - esurf)/1000.f ); //kPa
598 float gs = gs0+1.6f*(1.f+g1*sqrtf(beta/Ds))*An/Cs;
599
600 return gs*(es - esurf) - gbw*(esurf-ea);

604float StomatalConductanceModel::evaluate_BMFmodel( float esurf, std::vector<float> &variables, const void* parameters ){
605
606 // We want to find the vapor pressure at the surface, esurf, that balances the equation gs(esurf)*(es-esurf) = gbw*(esurf-ea). This function returns the residual of this equation.
607
608 const auto* coeffs = reinterpret_cast<const BMFcoefficients*>(parameters);
609
610 float Em = coeffs->Em;
611 float i0 = coeffs->i0;
612 float k = coeffs->k;
613 float b = coeffs->b;
614
615 float i = variables[0];
616 float es = variables[1];
617 float ea = variables[2];
618 float gbw = variables[3];
619 float press = variables[4];
620 float beta = variables[5];
621
622 float Ds = max(0.f, (es - esurf) / press * 1000.f); //mmol/mol
623 float gs = Em * beta * (i + i0) / (k + b * i + (i + i0) * Ds);
624
625 return gs*(es - esurf) - gbw*(esurf-ea);
626
627}

629float StomatalConductanceModel::evaluate_BBmodel( float gs, std::vector<float> &variables, const void* parameters ){
630
631 float fe = 0.5;
632 float Rxe = 200.f;
633
634 const auto* coeffs = reinterpret_cast<const BBcoefficients*>(parameters);
635
636 float pi_0 = coeffs->pi_0;
637 float pi_m = coeffs->pi_m;
638 float theta = coeffs->theta;
639 float sigma = coeffs->sigma;
640 float chi = coeffs->chi;
641
642 float i = variables[0];
643 float D = variables[1]*1e-3F;
644 float Psix = variables[2];
645
646 float pig = pi_0+(pi_m-pi_0)*i/(i+theta)+sigma*(Psix-Rxe*fe*gs*D);
647 float Psi_e = Psix-Rxe*fe*gs*D;
648
649 float Pg = max(0.f,pig+Psi_e);
650 float gsm = chi*Pg;
651
652 return gs-gsm;
653
654}
