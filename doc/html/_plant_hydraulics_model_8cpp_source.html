22std::vector<float> linspace(double begin, double end, int n) {
23 std::vector<float> result;
24 if (n <= 0)
25 return result;
26 if (n == 1) {
27 result.emplace_back(begin);
28 return result;
29 }
30
31 double step = (end - begin) / (n - 1);
32 for (int i = 0; i < n; i++) {
33 result.emplace_back(begin + i * step);
34 }
35 return result;
36}

59float computeTurgorPressure(const HydraulicCapacitance &coeffs, float relative_water_content) {
60 float pi_o = coeffs.osmotic_potential_at_full_turgor; // MPa
61 float w_tlp = coeffs.relative_water_content_at_turgor_loss; // -
62 float epsilon = coeffs.cell_wall_elasticity_exponent; // -
63 float w = relative_water_content; // -
64 float P = -pi_o * powf(max(0.f, (w - w_tlp) / (1.f - w_tlp)), epsilon);
65 return P;
66}

68float computeWaterPotential(const HydraulicCapacitance &coeffs, float relative_water_content) {
69 float pi_o = coeffs.osmotic_potential_at_full_turgor; // MPa
70 float w_tlp = coeffs.relative_water_content_at_turgor_loss; // -
71 float epsilon = coeffs.cell_wall_elasticity_exponent; // -
72 float w = relative_water_content; // -
73 float P = -pi_o * powf(max(0.f, (w - w_tlp) / (1.f - w_tlp)), epsilon);
74 float pi = pi_o / w;
75 float psi = P + pi;
76 return psi;
77}

83float computeConductance(const HydraulicConductance &coeffs, float water_potential, float temperature) {
84 float psi = water_potential; // MPa
85 float T = temperature; // K
86 float Ksat = coeffs.saturated_conductance; // mol/m2/s/MPa
87 float a = coeffs.potential_at_half_saturated; // MPa
88 float b = coeffs.sensitivity; // -
89 float K;
90 if (a == 0.f || b == 0.f) {
91 K = Ksat;
92 } else {
93 K = Ksat / (1.f + pow(abs(psi / a), b));
94 }
95
96 if (coeffs.temperature_dependence) {
97 K *= powf(T / 298.15f, 7.f);
98 }
99 return K;
100}

102float computeCapacitance(const HydraulicCapacitance &coeffs, float relative_water_content) {
103 float C = coeffs.fixed_constant_capacitance;
104 float Wsat = coeffs.saturated_specific_water_content;
105 if (C > 0.f) {
106 return C; // mol/m2/MPa
107 }
108 float w = relative_water_content; // -
109 float h = 0.001;
110 float f0 = computeWaterPotential(coeffs, w - h);
111 float f1 = computeWaterPotential(coeffs, w);
112 float f2 = computeWaterPotential(coeffs, w + h);
113 float dpsidw = (f0 - 2.f * f1 + f2) / (h * h);
114 C = Wsat / dpsidw; // mol/m2/MPa
115 return C;
116}

208int PlantHydraulicsModel::adjustTimestep(float time_step, float min_time_step, float max_time_step, float gradient, float gradient_upper_bound) {
209 // Adpative Timestepper
210 if (gradient > gradient_upper_bound) {
211 time_step *= gradient_upper_bound / gradient; // Reduce timestep if gradient too large
212 } else if (gradient < gradient_upper_bound * 0.1) {
213 time_step *= 1.1; // Increase timestep if gradient is small
214 }
215
216 time_step = std::max(min_time_step, std::min(time_step, max_time_step));
217 return int(time_step);
218}

220void PlantHydraulicsModel::updateRootAndStemWaterPotentialsOfPlant(const std::vector<uint> &UUIDs, int timespan, int timestep) {
221 float canopy_transpiration, K_root, K_stem, K_leaf, C_root, C_stem, leaf_water_potential, leaf_transpiration, leaf_temperature, J11, J12, J21, J22, F1, F2, det, delta_psi, delta_psi_root, delta_psi_stem;
222
223 int plantID = getPlantID(UUIDs);
224
225 if (soilWaterPotentialsByPlantID.find(plantID) == soilWaterPotentialsByPlantID.end()) {
226 soilWaterPotentialsByPlantID[plantID] = -0.001;
227 }
228
229 if (stemWaterPotentialsByPlantID.find(plantID) == stemWaterPotentialsByPlantID.end()) {
230 stemWaterPotentialsByPlantID[plantID] = -0.001;
231 }
232
233 if (rootWaterPotentialsByPlantID.find(plantID) == rootWaterPotentialsByPlantID.end()) {
234 rootWaterPotentialsByPlantID[plantID] = -0.001;
235 }
236
237
238 float soil_water_potential = soilWaterPotentialsByPlantID.at(plantID);
239 float stem_water_potential = stemWaterPotentialsByPlantID.at(plantID);
240 float root_water_potential = rootWaterPotentialsByPlantID.at(plantID);
241 bool steadystate = true;
242 if (timespan != 0) {
243 steadystate = false;
244 }
245 float stem_temperature = getOrInitializePrimitiveData(UUIDs.front(), "air_temperature", 298.15f, false);
246 float root_temperature = stem_temperature - 5.f;
247
248 context->calculatePrimitiveDataAreaWeightedSum(UUIDs, "latent_flux", canopy_transpiration); // Wm2 -> W
249 canopy_transpiration /= 44000.f; // W/m2 -> mol/m2/s
250
251 // Steady State Solution is assumed if time step is not specified/equal to 0.f
252 if (steadystate) {
253 K_root = computeConductance(modelcoeffs.RootHydraulicConductance, root_water_potential, root_temperature);
254 K_stem = computeConductance(modelcoeffs.StemHydraulicConductance, stem_water_potential, stem_temperature);
255 root_water_potential = soil_water_potential - canopy_transpiration / K_root;
256 stem_water_potential = root_water_potential - canopy_transpiration / K_stem;
257
258 rootWaterPotentialsByPlantID.at(plantID) = root_water_potential;
259 stemWaterPotentialsByPlantID.at(plantID) = stem_water_potential;
260 } else {
261 // Non-steady-state solution
262 float dt = float(timestep);
263 float dt_min = 10;
264 float dt_max = 500.0;
265 float delta_psi_max = 0.01;
266 int max_iter = 100;
267 float tol = 1e-5;
268 for (int t = 0; t < int(timespan); t += timestep) {
269 // Forward Euler
270 // Fin = K_root * (soil_water_potential - root_water_potential);
271 // Fout = K_stem * (root_water_potential - stem_water_potential);
272 // delta_psi_root = dt / C_root * (Fin - Fout);
273 // root_water_potential += delta_psi_root;
274 //
275 // Fin = K_stem * (root_water_potential - stem_water_potential);
276 // Fout = canopy_transpiration;
277 // delta_psi_stem = dt / C_stem * (Fin - Fout);
278 // stem_water_potential += delta_psi_stem;
279
280 // Backward Euler
281 float root_water_potential_new = root_water_potential;
282 float stem_water_potential_new = stem_water_potential;
283
284 for (int iter = 0; iter < max_iter; ++iter) {
285 K_root = computeConductance(modelcoeffs.RootHydraulicConductance, root_water_potential, root_temperature);
286 C_root = computeCapacitance(modelcoeffs.RootHydraulicCapacitance);
287 K_stem = computeConductance(modelcoeffs.StemHydraulicConductance, stem_water_potential, stem_temperature);
288 C_stem = computeCapacitance(modelcoeffs.StemHydraulicCapacitance);
289
290 // F : C psi/dt = Fin - Fout = Kpsi/dx - Kpsi/dx
291 // F1 : C_root dpsi_root/dt = K_root(psi_soil-psi_root) - K_stem(psi_root-psi_stem)
292 // F2: C_stem dpsi_stem/dt = K_stem(psi_root-psi_stem) - E
293
294 // Residuals (F)
295 F1 = C_root / dt * (root_water_potential_new - root_water_potential) - K_root * (soil_water_potential - root_water_potential_new) + K_stem * (root_water_potential_new - stem_water_potential_new);
296
297 F2 = C_stem / dt * (stem_water_potential_new - stem_water_potential) - K_stem * (root_water_potential_new - stem_water_potential_new) + canopy_transpiration;
298
299 // Jacobian (partial F / partial psi)
300 J11 = C_root / dt + K_root + K_stem;
301 J12 = -K_stem;
302 J21 = -K_stem;
303 J22 = C_stem / dt + K_stem;
304
305 // delta psi = inv(J) * -F
306 // inv(J) = 1/det(J) * [J22 -J12; -J21 J11]
307 // delta psi = 1/det(J) * [J22 -J12; -J21 J11] * [-F1 -F2]
308 det = J11 * J22 - J12 * J21;
309 delta_psi_root = (-F1 * J22 + -F2 * -J12) / det;
310 delta_psi_stem = (-F1 * -J21 + -F2 * J11) / det;
311
312 root_water_potential_new += delta_psi_root;
313 stem_water_potential_new += delta_psi_stem;
314
315 if (fabs(delta_psi_root) < tol && fabs(delta_psi_stem) < tol) {
316 break;
317 }
318 }
319
320 delta_psi = max(root_water_potential_new - root_water_potential, stem_water_potential_new - stem_water_potential);
321
322 root_water_potential = root_water_potential_new;
323 stem_water_potential = stem_water_potential_new;
324
325
326 // Adpative Timestepper: Reduce timestep if change is large, increase timestep if change is small
327 timestep = adjustTimestep(dt, dt_min, dt_max, delta_psi, delta_psi_max);
328 }
329
330 rootWaterPotentialsByPlantID.at(plantID) = root_water_potential;
331 stemWaterPotentialsByPlantID.at(plantID) = stem_water_potential;
332 }
333}

533std::vector<uint> PlantHydraulicsModel::getPrimitivesWithoutPlantID(std::vector<uint> UUIDs) {
534 auto objectIDs = context->getUniquePrimitiveParentObjectIDs(UUIDs);
535 std::vector<uint> allPrimitivesWithPlantID;
536 for (auto objectID: objectIDs) {
537 if (context->doesObjectDataExist(objectID, "plantID")) {
538 auto newleaves = context->getObjectPrimitiveUUIDs(objectID);
539 allPrimitivesWithPlantID.insert(allPrimitivesWithPlantID.end(), newleaves.begin(), newleaves.end());
540 }
541 }
542
543 std::unordered_set<uint> seen;
544 seen.reserve(allPrimitivesWithPlantID.size());
545 for (auto UUID: allPrimitivesWithPlantID) {
546 seen.insert(UUID);
547 }
548
549 std::vector<uint> PrimitivesWithoutPlantID;
550 PrimitivesWithoutPlantID.reserve(UUIDs.size());
551 for (auto UUID: UUIDs) {
552 if (seen.find(UUID) == seen.end()) {
553 PrimitivesWithoutPlantID.emplace_back(UUID);
554 }
555 }
556 return PrimitivesWithoutPlantID;
557}

619std::vector<std::vector<std::vector<float>>> PlantHydraulicsModel::meshgrid(const std::vector<float> &x, const std::vector<float> &y) {
620 std::vector<std::vector<float>> xGrid;
621 std::vector<std::vector<float>> yGrid;
622 size_t nx = x.size();
623 size_t ny = y.size();
624
625 xGrid.resize(ny, std::vector<float>(nx));
626 yGrid.resize(ny, std::vector<float>(nx));
627
628 for (size_t i = 0; i < ny; ++i) {
629 for (size_t j = 0; j < nx; ++j) {
630 xGrid[i][j] = x[j];
631 yGrid[i][j] = y[i];
632 }
633 }
634 return {xGrid, yGrid};
