24template<unsigned int N>
25static __host__ __device__ __inline__ unsigned int tea( unsigned int val0, unsigned int val1 )
26{
27 unsigned int v0 = val0;
28 unsigned int v1 = val1;
29 unsigned int s0 = 0;
30
31 for( unsigned int n = 0; n < N; n++ )
32 {
33 s0 += 0x9e3779b9;
34 v0 += ((v1<<4)+0xa341316c)^(v1+s0)^((v1>>5)+0xc8013ea4);
35 v1 += ((v0<<4)+0xad90777d)^(v0+s0)^((v0>>5)+0x7e95761e);
36 }
37
38 return v0;
39}

41// Generate random unsigned int in [0, 2^24)
42static __host__ __device__ __inline__ unsigned int lcg(unsigned int &prev)
43{
44 const unsigned int LCG_A = 1664525u;
45 const unsigned int LCG_C = 1013904223u;
46 prev = (LCG_A * prev + LCG_C);
47 return prev & 0x00FFFFFF;
48}

62// Multiply with carry
63static __host__ __inline__ unsigned int mwc()
64{
65 static unsigned long long r[4];
66 static unsigned long long carry;
67 static bool init = false;
68 if( !init ) {
69 init = true;
70 unsigned int seed = 7654321u, seed0, seed1, seed2, seed3;
71 r[0] = seed0 = lcg2(seed);
72 r[1] = seed1 = lcg2(seed0);
73 r[2] = seed2 = lcg2(seed1);
74 r[3] = seed3 = lcg2(seed2);
75 carry = lcg2(seed3);
76 }
77
78 unsigned long long sum = 2111111111ull * r[3] +
79 1492ull * r[2] +
80 1776ull * r[1] +
81 5115ull * r[0] +
82 1ull * carry;
83 r[3] = r[2];
84 r[2] = r[1];
85 r[1] = r[0];
86 r[0] = static_cast<unsigned int>(sum); // lower half
87 carry = static_cast<unsigned int>(sum >> 32); // upper half
88 return static_cast<unsigned int>(r[0]);
89}
