308void CollisionDetection::buildBVH(const std::vector<uint> &UUIDs) {
309
310 std::vector<uint> primitives_to_include;
311
312 if (UUIDs.empty()) {
313 // Include all primitives in context
314 primitives_to_include = context->getAllUUIDs();
315 } else {
316 primitives_to_include = UUIDs;
317 }
318
319
320 if (primitives_to_include.empty()) {
321 if (printmessages) {
322 std::cerr << "WARNING (CollisionDetection::buildBVH): No primitives found to build BVH" << std::endl;
323 }
324 return;
325 }
326
327 // Validate UUIDs - throw exception if any are invalid (only when specific UUIDs are provided)
328 std::vector<uint> valid_primitives;
329 if (!UUIDs.empty()) {
330 // When specific UUIDs are provided, they must all be valid
331 for (uint uuid : primitives_to_include) {
332 if (context->doesPrimitiveExist(uuid)) {
333 valid_primitives.push_back(uuid);
334 } else {
335 if (printmessages) {
336 std::cerr << "ERROR (CollisionDetection::buildBVH): Invalid UUID " << uuid << std::endl;
337 }
338 helios_runtime_error("ERROR (CollisionDetection::buildBVH): Invalid UUID " + std::to_string(uuid) + " provided");
339 }
340 }
341 } else {
342 // When no specific UUIDs provided (use all), filter out invalid ones
343 for (uint uuid : primitives_to_include) {
344 if (context->doesPrimitiveExist(uuid)) {
345 valid_primitives.push_back(uuid);
346 } else if (printmessages) {
347 std::cerr << "WARNING (CollisionDetection::buildBVH): Skipping invalid UUID " << uuid << std::endl;
348 }
349 }
350
351 if (valid_primitives.empty()) {
352 if (printmessages) {
353 std::cerr << "WARNING (CollisionDetection::buildBVH): No valid primitives found after filtering" << std::endl;
354 }
355 return;
356 }
357 }
358
359 primitives_to_include = valid_primitives;
360
361 // Check if the primitive set has actually changed before clearing cache
362 std::set<uint> new_primitive_set(primitives_to_include.begin(), primitives_to_include.end());
363 std::set<uint> old_primitive_set(primitive_indices.begin(), primitive_indices.end());
364
365 bool primitive_set_changed = (new_primitive_set != old_primitive_set);
366
367 if (primitive_set_changed) {
368 // Clear primitive cache only when primitive set changes - CRITICAL for performance
369 primitive_cache.clear();
370 if (printmessages) {
371 std::cout << "Primitive set changed, clearing cache (was " << old_primitive_set.size()
372 << " primitives, now " << new_primitive_set.size() << ")" << std::endl;
373 }
374 }
375
376 // Clear existing BVH
377 bvh_nodes.clear();
378 primitive_indices.clear();
379
380 // Copy primitives to indices array
381 primitive_indices = primitives_to_include;
382
383 // Pre-allocate BVH nodes to avoid excessive resizing
384 // For N primitives, we need at most 2*N-1 nodes for a complete binary tree
385 size_t max_nodes = std::max(size_t(1), 2 * primitives_to_include.size());
386 bvh_nodes.clear();
387 bvh_nodes.resize(max_nodes); // Pre-allocate ALL nodes at once to avoid any resizing
388 next_available_node_index = 1; // Track next available node (0 is root)
389
390 // OPTIMIZATION: Pre-cache bounding boxes with dirty flagging to avoid repeated expensive calculations
391 // Only clear cache for primitives that no longer exist or are dirty
392 std::unordered_set<uint> current_primitives(primitives_to_include.begin(), primitives_to_include.end());
393
394 // Remove cached entries for primitives that no longer exist
395 auto cache_it = primitive_aabbs_cache.begin();
396 while (cache_it != primitive_aabbs_cache.end()) {
397 if (current_primitives.find(cache_it->first) == current_primitives.end()) {
398 cache_it = primitive_aabbs_cache.erase(cache_it);
399 } else {
400 ++cache_it;
401 }
402 }
403
404 // Update only dirty or missing cache entries
405 for (uint UUID: primitives_to_include) {
406 if (!context->doesPrimitiveExist(UUID)) {
407 continue; // Skip invalid primitive
408 }
409
410 // Only update if not cached or marked as dirty
411 bool needs_update = (primitive_aabbs_cache.find(UUID) == primitive_aabbs_cache.end()) ||
412 (dirty_primitive_cache.find(UUID) != dirty_primitive_cache.end());
413
414 if (needs_update) {
415 vec3 aabb_min, aabb_max;
416 context->getPrimitiveBoundingBox(UUID, aabb_min, aabb_max);
417 primitive_aabbs_cache[UUID] = {aabb_min, aabb_max};
418 dirty_primitive_cache.erase(UUID); // Mark as clean
419 }
420 }
421
422 // Build BVH recursively starting from root
423 buildBVHRecursive(0, 0, primitive_indices.size(), 0);
424
425 // Resize to actual nodes used (much more efficient than shrink_to_fit)
426 bvh_nodes.resize(next_available_node_index);
427
428 // Clear any stale optimized structures since we just rebuilt the BVH
429 // They will be rebuilt on demand by ensureOptimizedBVH() when needed
430 bvh_nodes_soa.clear();
431 bvh_nodes_soa.node_count = 0;
432
433 // Transfer to GPU if acceleration is enabled
434#ifdef HELIOS_CUDA_AVAILABLE
435 if (gpu_acceleration_enabled) {
436 transferBVHToGPU();
437 }
438#endif
439
440 // Update internal tracking - record what we have processed
441 // This allows us to avoid redundant rebuilds until user marks geometry clean
442 std::vector<uint> context_deleted_uuids = context->getDeletedUUIDs();
443
444 // Track ALL primitives that are now in the BVH, not just dirty ones
445 // This ensures isBVHValid() works correctly when new geometry is added
446 last_processed_uuids.clear();
447 last_processed_uuids.insert(primitives_to_include.begin(), primitives_to_include.end());
448
449 last_processed_deleted_uuids.clear();
450 last_processed_deleted_uuids.insert(context_deleted_uuids.begin(), context_deleted_uuids.end());
451
452 // Update BVH geometry tracking for caching
453 last_bvh_geometry.clear();
454 last_bvh_geometry.insert(primitives_to_include.begin(), primitives_to_include.end());
455 bvh_dirty = false;
456
457}

565void CollisionDetection::updateBVH(const std::vector<uint> &UUIDs, bool force_rebuild) {
566 // Convert input to set for efficient comparison
567 std::set<uint> requested_geometry(UUIDs.begin(), UUIDs.end());
568
569 // Check if geometry has changed significantly
570 bool geometry_changed = (requested_geometry != last_bvh_geometry) || bvh_dirty;
571
572 if (!geometry_changed && !force_rebuild) {
573 return;
574 }
575
576 if (printmessages) {
577 std::cout << "BVH update: " << UUIDs.size() << " primitives" << std::endl;
578 }
579
580 // Use hierarchical BVH approach if enabled
581 if (hierarchical_bvh_enabled) {
582 updateHierarchicalBVH(requested_geometry, force_rebuild);
583 return;
584 }
585
586 // Determine if we need a full rebuild or can do incremental update
587 if (force_rebuild || bvh_nodes.empty()) {
588 // Full rebuild required
589 buildBVH(UUIDs);
590 } else {
591 // Check how much geometry has changed
592 std::set<uint> added_geometry, removed_geometry;
593
594 // Find added geometry (in requested but not in last_bvh_geometry)
595 std::set_difference(requested_geometry.begin(), requested_geometry.end(),
596 last_bvh_geometry.begin(), last_bvh_geometry.end(),
597 std::inserter(added_geometry, added_geometry.begin()));
598
599 // Find removed geometry (in last_bvh_geometry but not in requested)
600 std::set_difference(last_bvh_geometry.begin(), last_bvh_geometry.end(),
601 requested_geometry.begin(), requested_geometry.end(),
602 std::inserter(removed_geometry, removed_geometry.begin()));
603
604 // If more than 20% of geometry changed, do full rebuild, otherwise incremental
605 size_t total_change = added_geometry.size() + removed_geometry.size();
606 size_t current_size = std::max(last_bvh_geometry.size(), requested_geometry.size());
607
608 if (current_size == 0 || (float(total_change) / float(current_size)) > 0.2f) {
609 if (printmessages) {
610 std::cout << "Significant geometry change detected, performing full rebuild" << std::endl;
611 }
612 buildBVH(UUIDs);
613 } else {
614 if (printmessages) {
615 std::cout << "Minor geometry change detected, performing incremental update ("
616 << added_geometry.size() << " added, " << removed_geometry.size() << " removed)" << std::endl;
617 }
618 // Implement incremental update by selective insertion/removal
619 incrementalUpdateBVH(added_geometry, removed_geometry, requested_geometry);
620 }
621 }
622
623 // Update tracking
624 last_bvh_geometry = requested_geometry;
625 bvh_dirty = false;
626}

639 if (!automatic_bvh_rebuilds) {

652 // Use two-level dirty tracking pattern (similar to Visualizer plugin)
653 // Get dirty UUIDs from Context (but don't clear them - that's for the user)
654 std::vector<uint> context_dirty_uuids = context->getDirtyUUIDs(false); // Don't include deleted
655 std::vector<uint> context_deleted_uuids = context->getDeletedUUIDs();
656
657 // Convert to sets for efficient comparison
658 std::set<uint> current_dirty(context_dirty_uuids.begin(), context_dirty_uuids.end());
659 std::set<uint> current_deleted(context_deleted_uuids.begin(), context_deleted_uuids.end());
660
661 // Check if there are new changes since our last processing
662 bool has_new_dirty = false;
663 bool has_new_deleted = false;
664
665 // Check for new dirty UUIDs we haven't processed
666 for (uint uuid: current_dirty) {
667 if (last_processed_uuids.find(uuid) == last_processed_uuids.end()) {
668 has_new_dirty = true;
669 break;
670 }
671 }
672
673 // Check for new deleted UUIDs we haven't processed
674 for (uint uuid: current_deleted) {
675 if (last_processed_deleted_uuids.find(uuid) == last_processed_deleted_uuids.end()) {
676 has_new_deleted = true;
677 break;
678 }
679 }
680
681 // Only rebuild if there are genuinely new changes since our last processing
682 if (has_new_dirty || has_new_deleted) {
683 if (printmessages) {
684 std::cout << "Geometry has changed since last BVH build, rebuilding..." << std::endl;
685 }
686
687 buildBVH(); // This will update our internal tracking
688 }
689
690 // Note: We do NOT call context->markGeometryClean() here
691 // That should only be done by the user after all plugins have processed the changes
692}

694bool CollisionDetection::isBVHValid() const {
695 if (bvh_nodes.empty()) {
696 return false;
697 }
698
699 // Check if there are new primitives in the context that aren't in our BVH
700 std::vector<uint> all_context_uuids = context->getAllUUIDs();
701 for (uint uuid: all_context_uuids) {
702 if (last_processed_uuids.find(uuid) == last_processed_uuids.end()) {

792void CollisionDetection::calculateAABB(const std::vector<uint> &primitives, vec3 &aabb_min, vec3 &aabb_max) const {
793
794 if (primitives.empty()) {
795 aabb_min = make_vec3(0, 0, 0);
796 aabb_max = make_vec3(0, 0, 0);
797 return;
798 }
799
800 // Find first valid primitive for initialization
801 size_t first_valid = 0;
802 while (first_valid < primitives.size() && !context->doesPrimitiveExist(primitives[first_valid])) {
803 first_valid++;
804 }
805 if (first_valid >= primitives.size()) {
806 // No valid primitives found
807 aabb_min = make_vec3(0, 0, 0);
808 aabb_max = make_vec3(0, 0, 0);
809 return;
810 }
811
812 // Initialize with first valid primitive's bounding box
813 context->getPrimitiveBoundingBox(primitives[first_valid], aabb_min, aabb_max);
814
815 // Expand to include all remaining valid primitives
816 for (size_t i = first_valid + 1; i < primitives.size(); i++) {
817 if (!context->doesPrimitiveExist(primitives[i])) {
818 continue; // Skip invalid primitive
819 }
820 vec3 prim_min, prim_max;
821 context->getPrimitiveBoundingBox(primitives[i], prim_min, prim_max);
822
823 aabb_min.x = std::min(aabb_min.x, prim_min.x);
824 aabb_min.y = std::min(aabb_min.y, prim_min.y);
825 aabb_min.z = std::min(aabb_min.z, prim_min.z);
826
827 aabb_max.x = std::max(aabb_max.x, prim_max.x);
828 aabb_max.y = std::max(aabb_max.y, prim_max.y);
829 aabb_max.z = std::max(aabb_max.z, prim_max.z);
830 }

833void CollisionDetection::buildBVHRecursive(uint node_index, size_t primitive_start, size_t primitive_count, int depth) {
834
835 // Node should already be pre-allocated
836 if (node_index >= bvh_nodes.size()) {
837 throw std::runtime_error("CollisionDetection: BVH recursive access exceeded pre-allocated capacity");
838 }
839
840 // Bounds check for primitive_indices access
841 if (primitive_start + primitive_count > primitive_indices.size()) {
842 throw std::runtime_error("CollisionDetection: BVH primitive bounds check failed - primitive_start(" +
843 std::to_string(primitive_start) + ") + primitive_count(" +
844 std::to_string(primitive_count) + ") > primitive_indices.size(" +
845 std::to_string(primitive_indices.size()) + ")");
846 }
847
848 BVHNode &node = bvh_nodes[node_index];
849
850 // Calculate bounding box for this node using cached AABBs
851 if (primitive_count == 0) {
852 node.aabb_min = make_vec3(0, 0, 0);
853 node.aabb_max = make_vec3(0, 0, 0);
854 } else {
855 // Initialize with first primitive's cached AABB
856 uint first_uuid = primitive_indices[primitive_start];
857 auto it = primitive_aabbs_cache.find(first_uuid);
858 if (it == primitive_aabbs_cache.end()) {
859 // Handle missing primitive - use zero bounds
860 node.aabb_min = make_vec3(0, 0, 0);
861 node.aabb_max = make_vec3(0, 0, 0);
862 return;
863 }
864 const auto &first_cached_aabb = it->second;
865 node.aabb_min = first_cached_aabb.first;
866 node.aabb_max = first_cached_aabb.second;
867
868 // Expand to include all primitives in this node
869 for (size_t i = 1; i < primitive_count; i++) {
870 uint uuid = primitive_indices[primitive_start + i];
871 auto it = primitive_aabbs_cache.find(uuid);
872 if (it == primitive_aabbs_cache.end()) {
873 continue; // Skip missing primitive
874 }
875 const auto &cached_aabb = it->second;
876 node.aabb_min.x = std::min(node.aabb_min.x, cached_aabb.first.x);
877 node.aabb_min.y = std::min(node.aabb_min.y, cached_aabb.first.y);
878 node.aabb_min.z = std::min(node.aabb_min.z, cached_aabb.first.z);
879
880 node.aabb_max.x = std::max(node.aabb_max.x, cached_aabb.second.x);
881 node.aabb_max.y = std::max(node.aabb_max.y, cached_aabb.second.y);
882 node.aabb_max.z = std::max(node.aabb_max.z, cached_aabb.second.z);
883 }
884 }
885
886 // Stopping criteria - make this a leaf
887 // TEMPORARY: Very aggressive stopping for large scenes to prevent timeout
888 const int MAX_PRIMITIVES_PER_LEAF = (primitive_indices.size() > 500000) ? 500 : 100; // Much larger leaves for big scenes
889 const int MAX_DEPTH = (primitive_indices.size() > 500000) ? 6 : 10; // Shallower trees for big scenes
890
891 if (primitive_count <= MAX_PRIMITIVES_PER_LEAF || depth >= MAX_DEPTH) {
892 // Make leaf node
893 node.is_leaf = true;
894 node.primitive_start = primitive_start;
895 node.primitive_count = primitive_count;
896 node.left_child = 0xFFFFFFFF;
897 node.right_child = 0xFFFFFFFF;
898 return;
899 }
900
901 // Find best split using Surface Area Heuristic (simplified)
902 vec3 extent = node.aabb_max - node.aabb_min;
903 int split_axis = 0; // Split along longest axis
904 if (extent.y > extent.x)
905 split_axis = 1;
906 if (extent.z > (split_axis == 0 ? extent.x : extent.y))
907 split_axis = 2;
908
909 // In-place sort using pre-cached centroids to avoid temporary vector allocations
910 // This is critical for memory efficiency during recursive BVH construction
911 std::sort(primitive_indices.begin() + primitive_start,
912 primitive_indices.begin() + primitive_start + primitive_count,
913 [&](uint a, uint b) {
914 // Use cache lookup with bounds checking for safety
915 auto it_a = primitive_aabbs_cache.find(a);
916 auto it_b = primitive_aabbs_cache.find(b);
917 if (it_a == primitive_aabbs_cache.end() || it_b == primitive_aabbs_cache.end()) {
918 return a < b; // Fallback to UUID ordering for missing primitives
919 }
920
921 const auto &aabb_a = it_a->second;
922 const auto &aabb_b = it_b->second;
923
924 // Compute centroids inline to avoid vec3 temporaries
925 float centroid_a_coord, centroid_b_coord;
926 if (split_axis == 0) {
927 centroid_a_coord = 0.5f * (aabb_a.first.x + aabb_a.second.x);
928 centroid_b_coord = 0.5f * (aabb_b.first.x + aabb_b.second.x);
929 } else if (split_axis == 1) {
930 centroid_a_coord = 0.5f * (aabb_a.first.y + aabb_a.second.y);
931 centroid_b_coord = 0.5f * (aabb_b.first.y + aabb_b.second.y);
932 } else {
933 centroid_a_coord = 0.5f * (aabb_a.first.z + aabb_a.second.z);
934 centroid_b_coord = 0.5f * (aabb_b.first.z + aabb_b.second.z);
935 }
936
937 return centroid_a_coord < centroid_b_coord;
938 });
939
940 // Split in middle
941 size_t split_index = primitive_count / 2;
942
943 // Allocate child nodes from pre-allocated array (no resizing needed)
944 uint left_child_index = next_available_node_index++;
945 uint right_child_index = next_available_node_index++;
946
947 // Ensure we don't exceed pre-allocated capacity
948 if (right_child_index >= bvh_nodes.size()) {
949 throw std::runtime_error("CollisionDetection: BVH node allocation exceeded pre-calculated capacity");
950 }
951
952 // Re-get the node reference after potential reallocation
953 BVHNode &updated_node = bvh_nodes[node_index];
954 updated_node.left_child = left_child_index;
955 updated_node.right_child = right_child_index;
956 updated_node.is_leaf = false;
957 updated_node.primitive_start = 0;
958 updated_node.primitive_count = 0;
959
960 // Recursively build child nodes
961 buildBVHRecursive(left_child_index, primitive_start, split_index, depth + 1);
962 buildBVHRecursive(right_child_index, primitive_start + split_index, primitive_count - split_index, depth + 1);
963}

965std::vector<uint> CollisionDetection::traverseBVH_CPU(const vec3 &query_aabb_min, const vec3 &query_aabb_max) {
966
967 std::vector<uint> results;
968
969 if (bvh_nodes.empty()) {
970 return results;
971 }
972
973 // Stack-based traversal to avoid recursion
974 std::vector<uint> node_stack;
975 node_stack.push_back(0); // Start with root node
976
977 while (!node_stack.empty()) {
978 uint node_idx = node_stack.back();
979 node_stack.pop_back();
980
981 if (node_idx >= bvh_nodes.size())
982 continue;
983
984 const BVHNode &node = bvh_nodes[node_idx];
985
986 // Test if query AABB intersects node AABB
987 if (!aabbIntersect(query_aabb_min, query_aabb_max, node.aabb_min, node.aabb_max)) {
988 continue;
989 }
990
991 if (node.is_leaf) {
992 // Check each primitive in this leaf for intersection
993 for (uint i = 0; i < node.primitive_count; i++) {
994 uint primitive_id = primitive_indices[node.primitive_start + i];
995
996 // Get this primitive's AABB
997 if (!context->doesPrimitiveExist(primitive_id)) {
998 continue; // Skip invalid primitive
999 }
1000 vec3 prim_min, prim_max;
1001 context->getPrimitiveBoundingBox(primitive_id, prim_min, prim_max);
1002
1003 // Only add to results if AABBs actually intersect
1004 if (aabbIntersect(query_aabb_min, query_aabb_max, prim_min, prim_max)) {
1005 results.push_back(primitive_id);
1006 }
1007 }
1008 } else {
1009 // Add child nodes to stack for processing
1010 if (node.left_child != 0xFFFFFFFF) {
1011 node_stack.push_back(node.left_child);
1012 }
1013 if (node.right_child != 0xFFFFFFFF) {
1014 node_stack.push_back(node.right_child);
1015 }
1016 }
1017 }
1018
1019 return results;
1020}

1074bool CollisionDetection::rayAABBIntersect(const vec3 &origin, const vec3 &direction, const vec3 &aabb_min, const vec3 &aabb_max, float &t_min, float &t_max) const {
1075
1076 t_min = 0.0f;
1077 t_max = std::numeric_limits<float>::max();
1078
1079 // Check intersection with each pair of parallel planes (X, Y, Z)
1080 for (int i = 0; i < 3; i++) {
1081 float dir_component = (i == 0) ? direction.x : (i == 1) ? direction.y : direction.z;
1082 float orig_component = (i == 0) ? origin.x : (i == 1) ? origin.y : origin.z;
1083 float min_component = (i == 0) ? aabb_min.x : (i == 1) ? aabb_min.y : aabb_min.z;
1084 float max_component = (i == 0) ? aabb_max.x : (i == 1) ? aabb_max.y : aabb_max.z;
1085
1086 if (std::abs(dir_component) < 1e-9f) {
1087 // Ray is parallel to the slab
1088 if (orig_component < min_component || orig_component > max_component) {
1089 return false; // Ray is outside the slab and parallel to it
1090 }
1091 } else {
1092 // Compute intersection t values with the two planes
1093 float t1 = (min_component - orig_component) / dir_component;
1094 float t2 = (max_component - orig_component) / dir_component;
1095
1096 // Make sure t1 is the near intersection and t2 is the far intersection
1097 if (t1 > t2) {
1098 std::swap(t1, t2);
1099 }
1100
1101 // Update the overall near and far intersection parameters
1102 t_min = std::max(t_min, t1);
1103 t_max = std::min(t_max, t2);
1104
1105 // If the near intersection is farther than the far intersection, no intersection
1106 if (t_min > t_max) {
1107 return false;
1108 }
1109 }
1110 }
1111
1112 // Ray intersects AABB if t_min <= t_max and the intersection is in front of ray origin
1113 return t_max >= 0.0f;
1114}

1116bool CollisionDetection::coneAABBIntersect(const Cone &cone, const vec3 &aabb_min, const vec3 &aabb_max) {
1117 // Check if apex is inside AABB
1118 if (cone.apex.x >= aabb_min.x && cone.apex.x <= aabb_max.x && cone.apex.y >= aabb_min.y && cone.apex.y <= aabb_max.y && cone.apex.z >= aabb_min.z && cone.apex.z <= aabb_max.z) {
1119 return true; // Apex is inside AABB, definite intersection
1120 }
1121
1122 // Early rejection using bounding sphere around AABB
1123 vec3 box_center = 0.5f * (aabb_min + aabb_max);
1124 vec3 box_half_extents = 0.5f * (aabb_max - aabb_min);
1125 float box_radius = box_half_extents.magnitude();
1126
1127 // For infinite cone, check if any AABB corner is inside the cone
1128 if (cone.height <= 0.0f) {
1129 // Check all 8 corners of the AABB
1130 for (int i = 0; i < 8; i++) {
1131 vec3 corner = make_vec3((i & 1) ? aabb_max.x : aabb_min.x, (i & 2) ? aabb_max.y : aabb_min.y, (i & 4) ? aabb_max.z : aabb_min.z);
1132
1133 // Vector from apex to corner
1134 vec3 apex_to_corner = corner - cone.apex;
1135 float distance_along_axis = apex_to_corner * cone.axis;
1136
1137 // Point must be in front of apex
1138 if (distance_along_axis > 0) {
1139 // Check if point is within cone angle
1140 float cos_angle = distance_along_axis / apex_to_corner.magnitude();
1141 if (cos_angle >= cosf(cone.half_angle)) {
1142 return true; // This corner is inside the cone
1143 }
1144 }
1145 }
1146
1147 // Check if cone axis intersects AABB
1148 // Use ray-AABB intersection with cone axis as ray
1149 float t_min = 0.0f;
1150 float t_max = std::numeric_limits<float>::max();
1151
1152 for (int i = 0; i < 3; i++) {
1153 float axis_component = (i == 0) ? cone.axis.x : (i == 1) ? cone.axis.y : cone.axis.z;
1154 float apex_component = (i == 0) ? cone.apex.x : (i == 1) ? cone.apex.y : cone.apex.z;
1155 float min_component = (i == 0) ? aabb_min.x : (i == 1) ? aabb_min.y : aabb_min.z;
1156 float max_component = (i == 0) ? aabb_max.x : (i == 1) ? aabb_max.y : aabb_max.z;
1157
1158 if (std::abs(axis_component) < 1e-6f) {
1159 // Ray is parallel to slab
1160 if (apex_component < min_component || apex_component > max_component) {
1161 return false; // Ray is outside slab
1162 }
1163 } else {
1164 // Compute intersection t values
1165 float t1 = (min_component - apex_component) / axis_component;
1166 float t2 = (max_component - apex_component) / axis_component;
1167
1168 if (t1 > t2)
1169 std::swap(t1, t2);
1170
1171 t_min = std::max(t_min, t1);
1172 t_max = std::min(t_max, t2);
1173
1174 if (t_min > t_max) {
1175 return false; // No intersection
1176 }
1177 }
1178 }
1179
1180 // If we get here, the cone axis intersects the AABB
1181 // But for narrow cones, we need to check if the AABB is within the cone angle
1182 if (t_min >= 0 && t_max >= 0) {
1183 // Check if the intersection region on the axis is within the cone angle
1184 // Find the closest point on the axis within the intersection region
1185 float t_check = std::max(0.0f, t_min);
1186 vec3 axis_point = cone.apex + cone.axis * t_check;
1187
1188 // Find the closest point in the AABB to this axis point
1189 vec3 closest_in_box = make_vec3(std::max(aabb_min.x, std::min(axis_point.x, aabb_max.x)), std::max(aabb_min.y, std::min(axis_point.y, aabb_max.y)), std::max(aabb_min.z, std::min(axis_point.z, aabb_max.z)));
1190
1191 // Check if this closest point is within the cone
1192 vec3 apex_to_point = closest_in_box - cone.apex;
1193 float distance_along_axis = apex_to_point * cone.axis;
1194
1195 if (distance_along_axis > 0) {
1196 float distance_to_point = apex_to_point.magnitude();
1197 if (distance_to_point > 0) {
1198 float cos_angle = distance_along_axis / distance_to_point;
1199 if (cos_angle >= cosf(cone.half_angle)) {
1200 return true;
1201 }
1202 }
1203 }
1204 }
1205 } else {
1206 // Finite cone case
1207 // Check if any AABB corner is inside the finite cone
1208 for (int i = 0; i < 8; i++) {
1209 vec3 corner = make_vec3((i & 1) ? aabb_max.x : aabb_min.x, (i & 2) ? aabb_max.y : aabb_min.y, (i & 4) ? aabb_max.z : aabb_min.z);
1210
1211 // Vector from apex to corner
1212 vec3 apex_to_corner = corner - cone.apex;
1213 float distance_along_axis = apex_to_corner * cone.axis;
1214
1215 // Check if point is within cone height and in front of apex
1216 if (distance_along_axis > 0 && distance_along_axis <= cone.height) {
1217 // Check if point is within cone angle
1218 float cos_angle = distance_along_axis / apex_to_corner.magnitude();
1219 if (cos_angle >= cosf(cone.half_angle)) {
1220 return true; // This corner is inside the cone
1221 }
1222 }
1223 }
1224
1225 // Also need to check if cone base intersects AABB
1226 vec3 base_center = cone.apex + cone.axis * cone.height;
1227 float base_radius = cone.height * tanf(cone.half_angle);
1228
1229 // Simple sphere-AABB check for cone base
1230 vec3 closest_point = make_vec3(std::max(aabb_min.x, std::min(base_center.x, aabb_max.x)), std::max(aabb_min.y, std::min(base_center.y, aabb_max.y)), std::max(aabb_min.z, std::min(base_center.z, aabb_max.z)));
1231
1232 float dist_sq = (closest_point - base_center).magnitude();
1233 if (dist_sq <= base_radius) {
1234 return true; // Base circle intersects AABB
1235 }
1236 }
1237
1238 // More sophisticated tests could be added here for edge cases
1239 // For now, return false if no intersection found
1240 return false;
1241}

1422bool CollisionDetection::rayPrimitiveIntersection(const vec3 &origin, const vec3 &direction, uint primitive_UUID, float &distance) const {
1423 // Check if primitive exists first
1424 if (!context->doesPrimitiveExist(primitive_UUID)) {
1425 return false;
1426 }
1427
1428 try {
1429 // Get primitive type and vertices
1430 PrimitiveType type = context->getPrimitiveType(primitive_UUID);
1431 std::vector<vec3> vertices = context->getPrimitiveVertices(primitive_UUID);
1432
1433 if (vertices.empty()) {
1434 return false;
1435 }
1436
1437
1438 bool hit = false;
1439 float min_distance = std::numeric_limits<float>::max();
1440
1441 if (type == PRIMITIVE_TYPE_TRIANGLE) {
1442 // Triangle intersection using radiation model algorithm (proven to work)
1443 if (vertices.size() >= 3) {
1444 const vec3 &v0 = vertices[0];
1445 const vec3 &v1 = vertices[1];
1446 const vec3 &v2 = vertices[2];
1447
1448 // Use the same algorithm as radiation model's triangle_intersect
1449 float a = v0.x - v1.x, b = v0.x - v2.x, c = direction.x, d = v0.x - origin.x;
1450 float e = v0.y - v1.y, f = v0.y - v2.y, g = direction.y, h = v0.y - origin.y;
1451 float i = v0.z - v1.z, j = v0.z - v2.z, k = direction.z, l = v0.z - origin.z;
1452
1453 float m = f * k - g * j, n = h * k - g * l, p = f * l - h * j;
1454 float q = g * i - e * k, s = e * j - f * i;
1455
1456 float denom = a * m + b * q + c * s;
1457 if (std::abs(denom) < 1e-8f) {
1458 return false; // Ray is parallel to triangle
1459 }
1460
1461 float inv_denom = 1.0f / denom;
1462
1463 float e1 = d * m - b * n - c * p;
1464 float beta = e1 * inv_denom;
1465
1466 if (beta >= 0.0f) {
1467 float r = e * l - h * i;
1468 float e2 = a * n + d * q + c * r;
1469 float gamma = e2 * inv_denom;
1470
1471 if (gamma >= 0.0f && beta + gamma <= 1.0f) {
1472 float e3 = a * p - b * r + d * s;
1473 float t = e3 * inv_denom;
1474
1475 if (t > 1e-8f && t < min_distance) {
1476 min_distance = t;
1477 hit = true;
1478 }
1479 }
1480 }
1481 }
1482 } else if (type == PRIMITIVE_TYPE_PATCH) {
1483 // Patch (quadrilateral) intersection using radiation model algorithm
1484 if (vertices.size() >= 4) {
1485 const vec3 &v0 = vertices[0];
1486 const vec3 &v1 = vertices[1];
1487 const vec3 &v2 = vertices[2];
1488 const vec3 &v3 = vertices[3];
1489
1490 // Calculate patch vectors and normal (same as radiation model)
1491 vec3 anchor = v0;
1492 vec3 normal = cross(v1 - v0, v2 - v0);
1493 normal.normalize();
1494
1495 vec3 a = v1 - v0; // First edge vector
1496 vec3 b = v3 - v0; // Second edge vector
1497
1498 // Ray-plane intersection
1499 float denom = direction * normal;
1500 if (std::abs(denom) > 1e-8f) { // Not parallel to plane
1501 float t = (anchor - origin) * normal / denom;
1502
1503 if (t > 1e-8f && t < 1e8f) { // Valid intersection distance
1504 // Find intersection point
1505 vec3 p = origin + direction * t;
1506 vec3 d = p - anchor;
1507
1508 // Project onto patch coordinate system
1509 float ddota = d * a;
1510 float ddotb = d * b;
1511
1512 // Check if point is within patch bounds
1513 if (ddota >= 0.0f && ddota <= (a * a) &&
1514 ddotb >= 0.0f && ddotb <= (b * b)) {
1515
1516 if (t < min_distance) {
1517 min_distance = t;
1518 hit = true;
1519 }
1520 }
1521 }
1522 }
1523 }
1524 } else if (type == PRIMITIVE_TYPE_VOXEL) {
1525 // Voxel (AABB) intersection using slab method
1526 if (vertices.size() == 8) {
1527 // Calculate AABB from 8 vertices
1528 vec3 aabb_min = vertices[0];
1529 vec3 aabb_max = vertices[0];
1530
1531 for (int i = 1; i < 8; i++) {
1532 aabb_min.x = std::min(aabb_min.x, vertices[i].x);
1533 aabb_min.y = std::min(aabb_min.y, vertices[i].y);
1534 aabb_min.z = std::min(aabb_min.z, vertices[i].z);
1535 aabb_max.x = std::max(aabb_max.x, vertices[i].x);
1536 aabb_max.y = std::max(aabb_max.y, vertices[i].y);
1537 aabb_max.z = std::max(aabb_max.z, vertices[i].z);
1538 }
1539
1540 // Ray-AABB intersection using slab method
1541 float t_near = -std::numeric_limits<float>::max();
1542 float t_far = std::numeric_limits<float>::max();
1543
1544 // Check intersection with each slab (X, Y, Z)
1545 for (int i = 0; i < 3; i++) {
1546 float ray_dir_component = (i == 0) ? direction.x : (i == 1) ? direction.y : direction.z;
1547 float ray_orig_component = (i == 0) ? origin.x : (i == 1) ? origin.y : origin.z;
1548 float aabb_min_component = (i == 0) ? aabb_min.x : (i == 1) ? aabb_min.y : aabb_min.z;
1549 float aabb_max_component = (i == 0) ? aabb_max.x : (i == 1) ? aabb_max.y : aabb_max.z;
1550
1551 if (std::abs(ray_dir_component) < 1e-8f) {
1552 // Ray is parallel to slab
1553 if (ray_orig_component < aabb_min_component || ray_orig_component > aabb_max_component) {
1554 return false; // Ray is outside slab and parallel - no intersection
1555 }
1556 } else {
1557 // Calculate intersection distances for this slab
1558 float t1 = (aabb_min_component - ray_orig_component) / ray_dir_component;
1559 float t2 = (aabb_max_component - ray_orig_component) / ray_dir_component;
1560
1561 // Ensure t1 <= t2
1562 if (t1 > t2) {
1563 std::swap(t1, t2);
1564 }
1565
1566 // Update near and far intersection distances
1567 t_near = std::max(t_near, t1);
1568 t_far = std::min(t_far, t2);
1569
1570 // Early exit if no intersection possible
1571 if (t_near > t_far) {
1572 return false;
1573 }
1574 }
1575 }
1576
1577 // Check if intersection is in front of ray origin
1578 if (t_far >= 0.0f && t_near < min_distance) {
1579 // Use t_near if it's positive (ray starts outside box), otherwise t_far (ray starts inside box)
1580 float intersection_distance = (t_near >= 1e-8f) ? t_near : t_far;
1581 if (intersection_distance >= 1e-8f) {
1582 min_distance = intersection_distance;
1583 hit = true;
1584 }
1585 }
1586 }
1587 }
1588
1589 if (hit) {
1590 distance = min_distance;
1591 return true;
1592 }
1593
1594 return false;
1595 } catch (const std::exception& e) {
1596 // Primitive no longer exists or can't be accessed
1597 return false;
1598 }
1599}

2937std::vector<std::pair<helios::int3, float>> CollisionDetection::traverseVoxelGrid(const vec3 &ray_origin, const vec3 &ray_direction) const {
2938 std::vector<std::pair<helios::int3, float>> traversed_voxels;
2939
2940 // Grid bounds
2941 vec3 grid_min = voxel_grid_center - 0.5f * voxel_grid_size;
2942 vec3 grid_max = voxel_grid_center + 0.5f * voxel_grid_size;
2943 vec3 voxel_size = make_vec3(voxel_grid_size.x / static_cast<float>(voxel_grid_divisions.x),
2944 voxel_grid_size.y / static_cast<float>(voxel_grid_divisions.y),
2945 voxel_grid_size.z / static_cast<float>(voxel_grid_divisions.z));
2946
2947 // Test if ray intersects grid at all
2948 float t_grid_min, t_grid_max;
2949 if (!rayAABBIntersect(ray_origin, ray_direction, grid_min, grid_max, t_grid_min, t_grid_max)) {
2950 return traversed_voxels; // Empty - ray doesn't hit grid
2951 }
2952
2953 // Ensure intersection is in forward direction
2954 if (t_grid_max <= 1e-6) {
2955 return traversed_voxels; // Grid is behind ray
2956 }
2957
2958 // Clamp t_grid_min to 0 if ray starts inside grid
2959 t_grid_min = std::max(0.0f, t_grid_min);
2960
2961 // Fast path for single voxel grids
2962 if (voxel_grid_divisions.x == 1 && voxel_grid_divisions.y == 1 && voxel_grid_divisions.z == 1) {
2963 float path_length = t_grid_max - t_grid_min;
2964 if (path_length > 1e-6f) {
2965 traversed_voxels.emplace_back(helios::make_int3(0, 0, 0), path_length);
2966 }
2967 return traversed_voxels;
2968 }
2969
2970 // Starting position in grid space
2971 vec3 start_pos = ray_origin + t_grid_min * ray_direction;
2972
2973 // Convert to voxel indices (clamped to grid bounds)
2974 helios::int3 current_voxel;
2975 current_voxel.x = static_cast<int>(std::floor((start_pos.x - grid_min.x) / voxel_size.x));
2976 current_voxel.y = static_cast<int>(std::floor((start_pos.y - grid_min.y) / voxel_size.y));
2977 current_voxel.z = static_cast<int>(std::floor((start_pos.z - grid_min.z) / voxel_size.z));
