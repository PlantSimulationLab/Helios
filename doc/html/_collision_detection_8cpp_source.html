271void CollisionDetection::buildBVH(const std::vector<uint> &UUIDs) {
272
273 std::vector<uint> primitives_to_include;
274
275 if (UUIDs.empty()) {
276 // Include all primitives in context
277 primitives_to_include = context->getAllUUIDs();
278 } else {
279 primitives_to_include = UUIDs;
280 }
281
282 if (primitives_to_include.empty()) {
283 if (printmessages) {
284 std::cerr << "WARNING (CollisionDetection::buildBVH): No primitives found to build BVH" << std::endl;
285 }
286 return;
287 }
288
289 // Validate UUIDs
290 if (!validateUUIDs(primitives_to_include)) {
291 helios_runtime_error("ERROR (CollisionDetection::buildBVH): One or more invalid UUIDs provided");
292 }
293
294 if (printmessages) {
295 std::cout << "Building BVH with " << primitives_to_include.size() << " primitives..." << std::endl;
296 }
297
298 // Clear existing BVH
299 bvh_nodes.clear();
300 primitive_indices.clear();
301
302 // Copy primitives to indices array
303 primitive_indices = primitives_to_include;
304
305 // Pre-allocate BVH nodes to avoid excessive resizing
306 // For N primitives, we need at most 2*N-1 nodes for a complete binary tree
307 size_t max_nodes = std::max(size_t(1), 2 * primitives_to_include.size());
308 bvh_nodes.reserve(max_nodes);
309 bvh_nodes.resize(1); // Start with root node
310
311 // Pre-cache bounding boxes to avoid repeated expensive calculations
312 primitive_aabbs_cache.clear();
313 primitive_aabbs_cache.reserve(primitives_to_include.size());
314 for (uint UUID: primitives_to_include) {
315 vec3 aabb_min, aabb_max;
316 context->getPrimitiveBoundingBox(UUID, aabb_min, aabb_max);
317 primitive_aabbs_cache[UUID] = {aabb_min, aabb_max};
318 }
319
320 // Build BVH recursively starting from root
321 buildBVHRecursive(0, 0, primitive_indices.size(), 0);
322
323 // Shrink to actual size used
324 bvh_nodes.shrink_to_fit();
325
326 // Transfer to GPU if acceleration is enabled
327 if (gpu_acceleration_enabled) {
328 transferBVHToGPU();
329 }
330
331 // Update internal tracking - record what we have processed
332 // This allows us to avoid redundant rebuilds until user marks geometry clean
333 std::vector<uint> context_deleted_uuids = context->getDeletedUUIDs();
334
335 // Track ALL primitives that are now in the BVH, not just dirty ones
336 // This ensures isBVHValid() works correctly when new geometry is added
337 last_processed_uuids.clear();
338 last_processed_uuids.insert(primitives_to_include.begin(), primitives_to_include.end());
339
340 last_processed_deleted_uuids.clear();
341 last_processed_deleted_uuids.insert(context_deleted_uuids.begin(), context_deleted_uuids.end());
342
343 if (printmessages) {
344 size_t node_count, leaf_count, max_depth;
345 getBVHStatistics(node_count, leaf_count, max_depth);
346 std::cout << "BVH built successfully: " << node_count << " nodes, " << leaf_count << " leaves, max depth " << std::endl;
347 }
348}

355void CollisionDetection::ensureBVHCurrent() {
356 // If BVH is completely empty, build it with all geometry
357 if (bvh_nodes.empty()) {
358 if (printmessages) {
359 std::cout << "Building initial BVH..." << std::endl;
360 }
361 buildBVH();
362 return;
363 }
364
365 // Use two-level dirty tracking pattern (similar to Visualizer plugin)
366 // Get dirty UUIDs from Context (but don't clear them - that's for the user)
367 std::vector<uint> context_dirty_uuids = context->getDirtyUUIDs(false); // Don't include deleted
368 std::vector<uint> context_deleted_uuids = context->getDeletedUUIDs();
369
370 // Convert to sets for efficient comparison
371 std::set<uint> current_dirty(context_dirty_uuids.begin(), context_dirty_uuids.end());
372 std::set<uint> current_deleted(context_deleted_uuids.begin(), context_deleted_uuids.end());
373
374 // Check if there are new changes since our last processing
375 bool has_new_dirty = false;
376 bool has_new_deleted = false;
377
378 // Check for new dirty UUIDs we haven't processed
379 for (uint uuid: current_dirty) {
380 if (last_processed_uuids.find(uuid) == last_processed_uuids.end()) {
381 has_new_dirty = true;
382 break;
383 }
384 }
385
386 // Check for new deleted UUIDs we haven't processed
387 for (uint uuid: current_deleted) {
388 if (last_processed_deleted_uuids.find(uuid) == last_processed_deleted_uuids.end()) {
389 has_new_deleted = true;
390 break;
391 }
392 }
393
394 // Only rebuild if there are genuinely new changes since our last processing
395 if (has_new_dirty || has_new_deleted) {
396 if (printmessages) {
397 std::cout << "Geometry has changed since last BVH build, rebuilding..." << std::endl;
398 }
399
400 buildBVH(); // This will update our internal tracking
401 }
402
403 // Note: We do NOT call context->markGeometryClean() here
404 // That should only be done by the user after all plugins have processed the changes
405}

407bool CollisionDetection::isBVHValid() const {
408 if (bvh_nodes.empty()) {
409 return false;
410 }
411
412 // Check if there are new primitives in the context that aren't in our BVH
413 std::vector<uint> all_context_uuids = context->getAllUUIDs();
414 for (uint uuid: all_context_uuids) {
415 if (last_processed_uuids.find(uuid) == last_processed_uuids.end()) {
416 return false; // Found a primitive that's not in our BVH
417 }
418 }
419
420 // Check for deleted UUIDs we haven't processed
421 std::vector<uint> context_deleted_uuids = context->getDeletedUUIDs();
422 for (uint uuid: context_deleted_uuids) {
423 if (last_processed_deleted_uuids.find(uuid) == last_processed_deleted_uuids.end()) {
424 return false; // Found new deleted UUID we haven't processed
425 }
426 }
427
428 // Check if any primitives in our BVH have been deleted
429 for (uint uuid: primitive_indices) {
430 if (!context->doesPrimitiveExist(uuid)) {
431 return false; // A primitive in our BVH no longer exists
432 }
433 }
434
435 return true; // BVH is current with respect to geometry changes
436}

466void CollisionDetection::getBVHStatistics(size_t &node_count, size_t &leaf_count, size_t &max_depth) const {
467
468 node_count = bvh_nodes.size();
469 leaf_count = 0;
470 max_depth = 0;
471
472 // Simple traversal to count leaves and find max depth
473 std::function<void(uint, size_t)> traverse = [&](uint node_idx, size_t depth) {
474 if (node_idx >= bvh_nodes.size())
475 return;
476
477 const BVHNode &node = bvh_nodes[node_idx];
478 max_depth = std::max(max_depth, depth);
479
480 if (node.is_leaf) {
481 leaf_count++;
482 } else {
483 if (node.left_child != 0xFFFFFFFF) {
484 traverse(node.left_child, depth + 1);
485 }
486 if (node.right_child != 0xFFFFFFFF) {
487 traverse(node.right_child, depth + 1);
488 }
489 }
490 };
491
492 if (!bvh_nodes.empty()) {
493 traverse(0, 0);
494 }

523void CollisionDetection::buildBVHRecursive(uint node_index, size_t primitive_start, size_t primitive_count, int depth) {
524
525 // Ensure node exists (should already be pre-allocated)
526 if (node_index >= bvh_nodes.size()) {
527 bvh_nodes.resize(node_index + 1);
528 }
529
530 BVHNode &node = bvh_nodes[node_index];
531
532 // Calculate bounding box for this node using cached AABBs
533 if (primitive_count == 0) {
534 node.aabb_min = make_vec3(0, 0, 0);
535 node.aabb_max = make_vec3(0, 0, 0);
536 } else {
537 // Initialize with first primitive's cached AABB
538 uint first_uuid = primitive_indices[primitive_start];
539 const auto &first_cached_aabb = primitive_aabbs_cache.at(first_uuid);
540 node.aabb_min = first_cached_aabb.first;
541 node.aabb_max = first_cached_aabb.second;
542
543 // Expand to include all primitives in this node
544 for (size_t i = 1; i < primitive_count; i++) {
545 uint uuid = primitive_indices[primitive_start + i];
546 const auto &cached_aabb = primitive_aabbs_cache.at(uuid);
547 node.aabb_min.x = std::min(node.aabb_min.x, cached_aabb.first.x);
548 node.aabb_min.y = std::min(node.aabb_min.y, cached_aabb.first.y);
549 node.aabb_min.z = std::min(node.aabb_min.z, cached_aabb.first.z);
550
551 node.aabb_max.x = std::max(node.aabb_max.x, cached_aabb.second.x);
552 node.aabb_max.y = std::max(node.aabb_max.y, cached_aabb.second.y);
553 node.aabb_max.z = std::max(node.aabb_max.z, cached_aabb.second.z);
554 }
555 }
556
557 // Stopping criteria - make this a leaf
558 const int MAX_PRIMITIVES_PER_LEAF = 4;
559 const int MAX_DEPTH = 20;
560
561 if (primitive_count <= MAX_PRIMITIVES_PER_LEAF || depth >= MAX_DEPTH) {
562 // Make leaf node
563 node.is_leaf = true;
564 node.primitive_start = primitive_start;
565 node.primitive_count = primitive_count;
566 node.left_child = 0xFFFFFFFF;
567 node.right_child = 0xFFFFFFFF;
568 return;
569 }
570
571 // Find best split using Surface Area Heuristic (simplified)
572 vec3 extent = node.aabb_max - node.aabb_min;
573 int split_axis = 0; // Split along longest axis
574 if (extent.y > extent.x)
575 split_axis = 1;
576 if (extent.z > (split_axis == 0 ? extent.x : extent.y))
577 split_axis = 2;
578
579 // Sort primitives along split axis by their centroid using cached AABBs
580 std::sort(primitive_indices.begin() + primitive_start, primitive_indices.begin() + primitive_start + primitive_count, [&](uint a, uint b) {
581 const auto &cached_aabb_a = primitive_aabbs_cache.at(a);
582 const auto &cached_aabb_b = primitive_aabbs_cache.at(b);
583 vec3 centroid_a = 0.5f * (cached_aabb_a.first + cached_aabb_a.second);
584 vec3 centroid_b = 0.5f * (cached_aabb_b.first + cached_aabb_b.second);
585 float a_coord = (split_axis == 0) ? centroid_a.x : (split_axis == 1) ? centroid_a.y : centroid_a.z;
586 float b_coord = (split_axis == 0) ? centroid_b.x : (split_axis == 1) ? centroid_b.y : centroid_b.z;
587 return a_coord < b_coord;
588 });
589
590 // Split in middle
591 size_t split_index = primitive_count / 2;
592
593 // Create child nodes by allocating next two available indices
594 node.left_child = bvh_nodes.size();
595 node.right_child = bvh_nodes.size() + 1;
596
597 // Resize to accommodate both children at once (more efficient than incremental resizing)
598 bvh_nodes.resize(bvh_nodes.size() + 2);
599
600 node.is_leaf = false;
601 node.primitive_start = 0;
602 node.primitive_count = 0;
603
604 // Recursively build child nodes
605 buildBVHRecursive(node.left_child, primitive_start, split_index, depth + 1);
606 buildBVHRecursive(node.right_child, primitive_start + split_index, primitive_count - split_index, depth + 1);
607}

609std::vector<uint> CollisionDetection::traverseBVH_CPU(const vec3 &query_aabb_min, const vec3 &query_aabb_max) {
610
611 std::vector<uint> results;
612
613 if (bvh_nodes.empty()) {
614 return results;
615 }
616
617 // Stack-based traversal to avoid recursion
618 std::vector<uint> node_stack;
619 node_stack.push_back(0); // Start with root node
620
621 while (!node_stack.empty()) {
622 uint node_idx = node_stack.back();
623 node_stack.pop_back();
624
625 if (node_idx >= bvh_nodes.size())
626 continue;
627
628 const BVHNode &node = bvh_nodes[node_idx];
629
630 // Test if query AABB intersects node AABB
631 if (!aabbIntersect(query_aabb_min, query_aabb_max, node.aabb_min, node.aabb_max)) {
632 continue;
633 }
634
635 if (node.is_leaf) {
636 // Check each primitive in this leaf for intersection
637 for (uint i = 0; i < node.primitive_count; i++) {
638 uint primitive_id = primitive_indices[node.primitive_start + i];
639
640 // Get this primitive's AABB
641 vec3 prim_min, prim_max;
642 context->getPrimitiveBoundingBox(primitive_id, prim_min, prim_max);
643
644 // Only add to results if AABBs actually intersect
645 if (aabbIntersect(query_aabb_min, query_aabb_max, prim_min, prim_max)) {
646 results.push_back(primitive_id);
647 }
648 }
649 } else {
650 // Add child nodes to stack for processing
651 if (node.left_child != 0xFFFFFFFF) {
652 node_stack.push_back(node.left_child);
653 }
654 if (node.right_child != 0xFFFFFFFF) {
655 node_stack.push_back(node.right_child);
656 }
657 }

720bool CollisionDetection::rayAABBIntersect(const vec3 &origin, const vec3 &direction, const vec3 &aabb_min, const vec3 &aabb_max, float &t_min, float &t_max) {
721
722 t_min = 0.0f;
723 t_max = std::numeric_limits<float>::max();
724
725 // Check intersection with each pair of parallel planes (X, Y, Z)
726 for (int i = 0; i < 3; i++) {
727 float dir_component = (i == 0) ? direction.x : (i == 1) ? direction.y : direction.z;
728 float orig_component = (i == 0) ? origin.x : (i == 1) ? origin.y : origin.z;
729 float min_component = (i == 0) ? aabb_min.x : (i == 1) ? aabb_min.y : aabb_min.z;
730 float max_component = (i == 0) ? aabb_max.x : (i == 1) ? aabb_max.y : aabb_max.z;
731
732 if (std::abs(dir_component) < 1e-9f) {
733 // Ray is parallel to the slab
734 if (orig_component < min_component || orig_component > max_component) {
735 return false; // Ray is outside the slab and parallel to it
736 }
737 } else {
738 // Compute intersection t values with the two planes
739 float t1 = (min_component - orig_component) / dir_component;
740 float t2 = (max_component - orig_component) / dir_component;
741
742 // Make sure t1 is the near intersection and t2 is the far intersection
743 if (t1 > t2) {
744 std::swap(t1, t2);
745 }
746
747 // Update the overall near and far intersection parameters
748 t_min = std::max(t_min, t1);
749 t_max = std::min(t_max, t2);
750
751 // If the near intersection is farther than the far intersection, no intersection
752 if (t_min > t_max) {
753 return false;
754 }
755 }
756 }
757
758 // Ray intersects AABB if t_min <= t_max and the intersection is in front of ray origin
759 return t_max >= 0.0f;
760}

762bool CollisionDetection::coneAABBIntersect(const Cone &cone, const vec3 &aabb_min, const vec3 &aabb_max) {
763 // Check if apex is inside AABB
764 if (cone.apex.x >= aabb_min.x && cone.apex.x <= aabb_max.x && cone.apex.y >= aabb_min.y && cone.apex.y <= aabb_max.y && cone.apex.z >= aabb_min.z && cone.apex.z <= aabb_max.z) {
765 return true; // Apex is inside AABB, definite intersection
766 }
767
768 // Early rejection using bounding sphere around AABB
769 vec3 box_center = 0.5f * (aabb_min + aabb_max);
770 vec3 box_half_extents = 0.5f * (aabb_max - aabb_min);
771 float box_radius = box_half_extents.magnitude();
772
773 // For infinite cone, check if any AABB corner is inside the cone
774 if (cone.height <= 0.0f) {
775 // Check all 8 corners of the AABB
776 for (int i = 0; i < 8; i++) {
777 vec3 corner = make_vec3((i & 1) ? aabb_max.x : aabb_min.x, (i & 2) ? aabb_max.y : aabb_min.y, (i & 4) ? aabb_max.z : aabb_min.z);
778
779 // Vector from apex to corner
780 vec3 apex_to_corner = corner - cone.apex;
781 float distance_along_axis = apex_to_corner * cone.axis;
782
783 // Point must be in front of apex
784 if (distance_along_axis > 0) {
785 // Check if point is within cone angle
786 float cos_angle = distance_along_axis / apex_to_corner.magnitude();
787 if (cos_angle >= cosf(cone.half_angle)) {
788 return true; // This corner is inside the cone
789 }
790 }
791 }
792
793 // Check if cone axis intersects AABB
794 // Use ray-AABB intersection with cone axis as ray
795 float t_min = 0.0f;
796 float t_max = std::numeric_limits<float>::max();
797
798 for (int i = 0; i < 3; i++) {
799 float axis_component = (i == 0) ? cone.axis.x : (i == 1) ? cone.axis.y : cone.axis.z;
800 float apex_component = (i == 0) ? cone.apex.x : (i == 1) ? cone.apex.y : cone.apex.z;
801 float min_component = (i == 0) ? aabb_min.x : (i == 1) ? aabb_min.y : aabb_min.z;
802 float max_component = (i == 0) ? aabb_max.x : (i == 1) ? aabb_max.y : aabb_max.z;
803
804 if (std::abs(axis_component) < 1e-6f) {
805 // Ray is parallel to slab
806 if (apex_component < min_component || apex_component > max_component) {
807 return false; // Ray is outside slab
808 }
809 } else {
810 // Compute intersection t values
811 float t1 = (min_component - apex_component) / axis_component;
812 float t2 = (max_component - apex_component) / axis_component;
813
814 if (t1 > t2)
815 std::swap(t1, t2);
816
817 t_min = std::max(t_min, t1);
818 t_max = std::min(t_max, t2);
819
820 if (t_min > t_max) {
821 return false; // No intersection
822 }
823 }
824 }
825
826 // If we get here, the cone axis intersects the AABB
827 // But for narrow cones, we need to check if the AABB is within the cone angle
828 if (t_min >= 0 && t_max >= 0) {
829 // Check if the intersection region on the axis is within the cone angle
830 // Find the closest point on the axis within the intersection region
831 float t_check = std::max(0.0f, t_min);
832 vec3 axis_point = cone.apex + cone.axis * t_check;
833
834 // Find the closest point in the AABB to this axis point
835 vec3 closest_in_box = make_vec3(std::max(aabb_min.x, std::min(axis_point.x, aabb_max.x)), std::max(aabb_min.y, std::min(axis_point.y, aabb_max.y)), std::max(aabb_min.z, std::min(axis_point.z, aabb_max.z)));
836
837 // Check if this closest point is within the cone
838 vec3 apex_to_point = closest_in_box - cone.apex;
839 float distance_along_axis = apex_to_point * cone.axis;
840
841 if (distance_along_axis > 0) {
842 float distance_to_point = apex_to_point.magnitude();
843 if (distance_to_point > 0) {
844 float cos_angle = distance_along_axis / distance_to_point;
845 if (cos_angle >= cosf(cone.half_angle)) {
846 return true;
847 }
848 }
849 }
850 }
851 } else {
852 // Finite cone case
853 // Check if any AABB corner is inside the finite cone
854 for (int i = 0; i < 8; i++) {
855 vec3 corner = make_vec3((i & 1) ? aabb_max.x : aabb_min.x, (i & 2) ? aabb_max.y : aabb_min.y, (i & 4) ? aabb_max.z : aabb_min.z);
856
857 // Vector from apex to corner
858 vec3 apex_to_corner = corner - cone.apex;
859 float distance_along_axis = apex_to_corner * cone.axis;
860
861 // Check if point is within cone height and in front of apex
862 if (distance_along_axis > 0 && distance_along_axis <= cone.height) {
863 // Check if point is within cone angle
864 float cos_angle = distance_along_axis / apex_to_corner.magnitude();
865 if (cos_angle >= cosf(cone.half_angle)) {
866 return true; // This corner is inside the cone
867 }
868 }
869 }
870
871 // Also need to check if cone base intersects AABB
872 vec3 base_center = cone.apex + cone.axis * cone.height;
873 float base_radius = cone.height * tanf(cone.half_angle);
874
875 // Simple sphere-AABB check for cone base
876 vec3 closest_point = make_vec3(std::max(aabb_min.x, std::min(base_center.x, aabb_max.x)), std::max(aabb_min.y, std::min(base_center.y, aabb_max.y)), std::max(aabb_min.z, std::min(base_center.z, aabb_max.z)));
877
878 float dist_sq = (closest_point - base_center).magnitude();
879 if (dist_sq <= base_radius) {
880 return true; // Base circle intersects AABB
881 }
882 }
883
884 // More sophisticated tests could be added here for edge cases
885 // For now, return false if no intersection found
886 return false;
887}

967 return;

969
970 // Always reallocate GPU memory to handle size changes

972 freeGPUMemory();

977 if (!gpu_memory_allocated || d_bvh_nodes == nullptr || d_primitive_indices == nullptr) {
978 helios_runtime_error("ERROR: Failed to allocate GPU memory for BVH transfer");
979 }
980

982 std::vector<GPUBVHNode> gpu_nodes(bvh_nodes.size());
983 for (size_t i = 0; i < bvh_nodes.size(); i++) {

1008 if (printmessages) {
1009 std::cout << "Transferred BVH to GPU (" << bvh_nodes.size() << " nodes, " << primitive_indices.size() << " primitives)" << std::endl;
1010 }
1011#else
1012 // CUDA not available, disable GPU acceleration
1013 gpu_acceleration_enabled = false;

1067int CollisionDetection::countRayIntersections(const vec3 &origin, const vec3 &direction, float max_distance) {
1068
1069 int intersection_count = 0;
1070
1071 if (bvh_nodes.empty()) {
1072 return intersection_count;
1073 }
1074
1075 // Ensure the BVH is current before traversal
1076 const_cast<CollisionDetection *>(this)->ensureBVHCurrent();
1077
1078 // Stack-based traversal to avoid recursion
1079 std::vector<uint> node_stack;
1080 node_stack.push_back(0); // Start with root node
1081
1082 while (!node_stack.empty()) {
1083 uint node_idx = node_stack.back();
1084 node_stack.pop_back();
1085
1086 if (node_idx >= bvh_nodes.size())
1087 continue;
1088
1089 const BVHNode &node = bvh_nodes[node_idx];
1090
1091 // Test if ray intersects node AABB
1092 float t_min, t_max;
1093 if (!rayAABBIntersect(origin, direction, node.aabb_min, node.aabb_max, t_min, t_max)) {
1094 continue;
1095 }
1096
1097 // If max_distance is specified, check if intersection is within range
1098 if (max_distance > 0.0f && t_min > max_distance) {
1099 continue;
1100 }
1101
1102 if (node.is_leaf) {
1103 // Check each primitive in this leaf for ray intersection
1104 for (uint i = 0; i < node.primitive_count; i++) {
1105 uint primitive_id = primitive_indices[node.primitive_start + i];
1106
1107 // Get this primitive's AABB
1108 vec3 prim_min, prim_max;
1109 context->getPrimitiveBoundingBox(primitive_id, prim_min, prim_max);
1110
1111 // Test ray against primitive AABB
1112 float prim_t_min, prim_t_max;
1113 if (rayAABBIntersect(origin, direction, prim_min, prim_max, prim_t_min, prim_t_max)) {
1114 // Check distance constraint
1115 if (max_distance <= 0.0f || prim_t_min <= max_distance) {
1116 intersection_count++;
1117 }
1118 }
1119 }
1120 } else {
1121 // Add child nodes to stack for further traversal
1122 if (node.left_child != 0xFFFFFFFF) {
1123 node_stack.push_back(node.left_child);
1124 }
1125 if (node.right_child != 0xFFFFFFFF) {
1126 node_stack.push_back(node.right_child);
1127 }
1128 }
1129 }
1130
1131 return intersection_count;
1132}

1134std::vector<helios::vec3> CollisionDetection::sampleDirectionsInCone(const vec3 &apex, const vec3 &central_axis, float half_angle, int num_samples) {
1135
1136 std::vector<vec3> directions;
1137 directions.reserve(num_samples);
1138
1139 if (num_samples <= 0 || half_angle <= 0.0f) {
1140 return directions;
1141 }
1142
1143 // Normalize the central axis
1144 vec3 axis = central_axis;
1145 axis.normalize();
1146
1147 // Create an orthonormal basis with the central axis as the primary axis
1148 vec3 u, v;
1149 if (std::abs(axis.z) < 0.9f) {
1150 u = cross(axis, make_vec3(0, 0, 1));
1151 } else {
1152 u = cross(axis, make_vec3(1, 0, 0));
1153 }
1154 u.normalize();
1155 v = cross(axis, u);
1156 v.normalize();
1157
1158 // Generate uniform samples within the cone using rejection sampling on hemisphere
1159 std::random_device rd;
1160 std::mt19937 gen(rd());
1161 std::uniform_real_distribution<float> uniform_dist(0.0f, 1.0f);
1162
1163 int samples_generated = 0;
1164 int max_attempts = num_samples * 10; // Prevent infinite loops
1165 int attempts = 0;
1166
1167 while (samples_generated < num_samples && attempts < max_attempts) {
1168 attempts++;
1169
1170 // Generate uniform sample on unit hemisphere using spherical coordinates
1171 float u1 = uniform_dist(gen);
1172 float u2 = uniform_dist(gen);
1173
1174 // Use stratified sampling for better distribution
1175 if (samples_generated > 0) {
1176 float stratum_u1 = (float) samples_generated / (float) num_samples;
1177 float stratum_u2 = uniform_dist(gen);
1178 u1 = (stratum_u1 + u1 / (float) num_samples);
1179 if (u1 > 1.0f)
1180 u1 -= 1.0f;
1181 }
1182
1183 // Convert to spherical coordinates
1184 // For uniform sampling within cone, we need:
1185 // cos(theta) uniformly distributed between cos(half_angle) and 1
1186 float cos_half_angle = cosf(half_angle);
1187 float cos_theta = cos_half_angle + u1 * (1.0f - cos_half_angle);
1188 float sin_theta = sqrtf(1.0f - cos_theta * cos_theta);
1189 float phi = 2.0f * M_PI * u2;
1190
1191 // Convert to Cartesian coordinates in local coordinate system
1192 float x = sin_theta * cosf(phi);
1193 float y = sin_theta * sinf(phi);
1194 float z = cos_theta;
1195
1196 // Transform from local coordinates to world coordinates
1197 vec3 local_direction = make_vec3(x, y, z);
1198 vec3 world_direction = u * local_direction.x + v * local_direction.y + axis * local_direction.z;
1199 world_direction.normalize();
1200
1201 // Verify the direction is within the cone (numerical precision check)
1202 float dot_product = world_direction * axis;
1203 if (dot_product >= cos_half_angle - 1e-6f) {
1204 directions.push_back(world_direction);
1205 samples_generated++;
1206 }
1207 }
1208
1209 // If we couldn't generate enough samples, fill with the central axis
1210 while (directions.size() < (size_t) num_samples) {
1211 directions.push_back(axis);
1212 }
1213
1214 return directions;
1215}

1217CollisionDetection::OptimalPathResult CollisionDetection::findOptimalConePath(const vec3 &apex, const vec3 &centralAxis, float half_angle, float height, int initialSamples) {
1218
1219 OptimalPathResult result;
1220 result.direction = centralAxis;
1221 result.direction.normalize();
1222 result.collisionCount = 0;
1223 result.confidence = 0.0f;
1224
1225 // Validate input parameters
1226 if (initialSamples <= 0 || half_angle <= 0.0f || half_angle > M_PI) {
1227 if (printmessages) {
1228 std::cerr << "WARNING: Invalid parameters for findOptimalConePath" << std::endl;
1229 }
1230 return result;
1231 }
1232
1233 if (bvh_nodes.empty()) {
1234 // No geometry to collide with, central axis is optimal
1235 result.confidence = 1.0f;
1236 return result;
1237 }
1238
1239 // Use fish-eye camera gap detection algorithm
1240 std::vector<Gap> detected_gaps = detectGapsInCone(apex, centralAxis, half_angle, height, initialSamples);
1241
1242 if (detected_gaps.empty()) {
1243 // No gaps found, fall back to central axis
1244 if (printmessages) {
1245 std::cerr << "WARNING: No gaps detected in cone, using central axis" << std::endl;
1246 }
1247 result.confidence = 0.1f;
1248 return result;
1249 }
1250
1251 // Score gaps using fish-eye metric
1252 scoreGapsByFishEyeMetric(detected_gaps, centralAxis);
1253
1254 // Find optimal direction toward highest-scoring gap
1255 result.direction = findOptimalGapDirection(detected_gaps, centralAxis);
1256

1258 float max_distance = (height > 0.0f) ? height : -1.0f;

1260
1261 // Calculate confidence based on gap quality

1263 // Higher confidence for larger, well-defined gaps

1266 }

1388float CollisionDetection::calculateGapAngularSize(const std::vector<RaySample> &gap_samples, const vec3 &central_axis) {
1389

1392 }
1393
1394 // Find the angular extent of the gap by finding min/max angles
1395 float min_angle = M_PI;
1396 float max_angle = 0.0f;
1397
1398 for (const RaySample &sample: gap_samples) {
1399 float dot_product = sample.direction * central_axis;
1400 dot_product = std::max(-1.0f, std::min(1.0f, dot_product));
1401 float angle = acosf(dot_product);
1402
1403 min_angle = std::min(min_angle, angle);
1404 max_angle = std::max(max_angle, angle);
1405 }

1408 float angular_width = max_angle - min_angle;

1410 // Convert to approximate solid angle (steradians)
1411 // This is a rough approximation: solid_angle ≈ π * (angular_width)^2

1417void CollisionDetection::scoreGapsByFishEyeMetric(std::vector<Gap> &gaps, const vec3 &central_axis) {

1421
1422 // Gap size component (logarithmic scaling for larger gaps)
1423 float size_score = log(1.0f + gap.angular_size * 100.0f); // Scale up angular size
1424
1425 // Distance penalty (exponential penalty for gaps far from center)
1426 float distance_penalty = exp(gap.angular_distance * 2.0f);
1427
1428 // Combined score (higher is better)
1429 gap.score = size_score / distance_penalty;
1430 }
1431
1432 // Sort gaps by score (highest first)
1433 std::sort(gaps.begin(), gaps.end(), [](const Gap &a, const Gap &b) { return a.score > b.score; });
