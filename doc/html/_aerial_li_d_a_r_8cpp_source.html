481helios::int3 AerialLiDARcloud::gridindex2ijk( const int index ) const{
482
483 if( index<0 || index>=gridresolution.x*gridresolution.y*gridresolution.z ){
484 return make_int3(-1,-1,-1);
485 }else{
486
487 int k=floor(double(index)/double(gridresolution.x)/double(gridresolution.y));
488 int j=floor((double(index)-double(k)*double(gridresolution.x)*double(gridresolution.y))/double(gridresolution.x));
489 int i=index-k*gridresolution.x*gridresolution.y-j*gridresolution.x;
490
491 if( i>=0 && i<gridresolution.x && j>=0 && j<gridresolution.y && k>=0 && k<gridresolution.z ){
492 }else{
493 printf("resolution: %d x %d x %d\n",gridresolution.x,gridresolution.y,gridresolution.z);
494 printf("index: %d\n",index);
495 printf("ijk: (%d,%d,%d)\n",i,j,k);
496 }
497
498 assert( i>=0 && i<gridresolution.x && j>=0 && j<gridresolution.y && k>=0 && k<gridresolution.z );
499
500 return make_int3(i,j,k);
501
502 }

541
542 float x = -0.5*gridextent.x+(float(ijk.x)+0.5f)*gridextent.x/float(gridresolution.x);
543 float y = -0.5*gridextent.y+(float(ijk.y)+0.5f)*gridextent.y/float(gridresolution.y);

754helios::vec4 AerialLiDARcloud::RANSAC( const int maxIter, const float threshDist, const float inlierRatio, const std::vector<helios::vec3>& hits, std::vector<bool>& inliers ){
755
756 int N = hits.size();
757
758 //trying to find a model in the form of Ax+By+Cz+D=0
759 //bestModel.x = A, bestModel.y = B, bestModel.z = C, bestModel.w = D
760 helios::vec4 bestModel;
761 float bestError = 1e20;
762
763 for( int iter=0; iter<maxIter; iter++ ){
764
765 helios::vec4 maybeModel;
766
767 //generate 3 random points
768 int i0 = randu(0,N-1);
769 int i1 = randu(0,N-1);
770 int i2 = randu(0,N-1);
771
772 //get x,y,z coordinates for these points
773 helios::vec3 x0 = hits.at(i0);
774 helios::vec3 x1 = hits.at(i1);
775 helios::vec3 x2 = hits.at(i2);
776
777 //determine 'maybeModel' for these three points
778 helios::vec3 n = cross( x2-x0, x1-x0 );
779 maybeModel.x = n.x;
780 maybeModel.y = n.y;
781 maybeModel.z = n.z;
782
783 maybeModel.w = -(n*x2);
784
785 //loop over all other points
786 float errsum = 0.f;
787 int Ninlier = 0;
788 for( int p=0; p<N; p++ ){
789
790 if( p!=i0 && p!=i1 && p!=i2 ){
791
792 helios::vec3 point = hits.at(p);
793
794 float zplane = -(maybeModel.x*point.x+maybeModel.y*point.y+maybeModel.w)/maybeModel.z;
795 float zerr = fabs( zplane - point.z ); //NOTE: since we know the plane will be somewhat close to horizontal, we're only using the vertical distance to calculate the error of fit.
796
797 if( zerr<=threshDist ){
798 errsum += zerr;
799 Ninlier += 1;
800 }
801
802 }
803
804 }
805
806 //check if this model is better than the current 'best' model
807 if( errsum/float(Ninlier)<bestError && float(Ninlier)/float(N)>inlierRatio ){
808
809 //update the 'best' model
810 bestModel = maybeModel;
811 bestError = errsum/float(Ninlier);
812
813 }
814
815 }//end iterations loop (iter)
816
817 //separate inliers and outliers based on 'best' model
818 inliers.resize(N,0);
819 for( int p=0; p<N; p++ ){
820
821 helios::vec3 point = hits.at(p);
822
823 float zplane = -(bestModel.x*point.x+bestModel.y*point.y+bestModel.w)/bestModel.z;
824 float zerr = fabs( zplane - point.z );
825
826 if( zerr<=threshDist ){
827 inliers.at(p) = true;
828 }
829
830 }
831
832 return bestModel;
833
834}
835
836void AerialLiDARcloud::generateHeightModel( const int maxIter, const float threshDist_ground, const float inlierRatio_ground, const float threshDist_vegetation, const float inlierRatio_vegetation ){
837
838 if( printmessages ){
839 std::cout << "Calculating ground and vegetation heights..." << std::flush;
840 }
841
842 if( !hitgridcellcomputed ){
843 calculateHitGridCell();
844 }
845
846 std::srand(time(NULL));
847
848 const uint Nscans = getScanCount();
849
850 helios::int3 gsize3 = getGridResolution();
851 const uint Ncells = gsize3.x*gsize3.y*gsize3.z;
852 size_t Nhits = getHitCount();
853
854 if( Ncells==0 ){
855 std::cerr << "ERROR (generateHeightModel): There are no grid cells in the point cloud." << std::endl;
856 exit(EXIT_FAILURE);
857 }else if( Nhits==0 ){
858 std::cerr << "ERROR (generateHeightModel): There are no hits in the point cloud." << std::endl;
859 exit(EXIT_FAILURE);
860 }
861
862 //aggregate all points in a given x,y column
863
864 std::vector<std::vector<std::vector<int> > > first_hits_ind; //first index are hit indices, second index is grid cell in x-dir, third index is grid cell in y-dir
865 std::vector<std::vector<std::vector<int> > > last_hits_ind; //first index are hit indices, second index is grid cell in x-dir, third index is grid cell in y-dir
866 std::vector<std::vector<std::vector<float> > > max_height_xy;
867
868 first_hits_ind.resize(gsize3.y);
869 last_hits_ind.resize(gsize3.y);
870 for( int j=0; j<gsize3.y; j++ ){
871 first_hits_ind.at(j).resize(gsize3.x);
872 last_hits_ind.at(j).resize(gsize3.x);
873 }
874
875 max_height_xy.resize(3);
876 for( int k=0; k<max_height_xy.size(); k++ ){
877 max_height_xy.at(k).resize(gsize3.y);
878 for( int j=0; j<gsize3.y; j++ ){
879 max_height_xy.at(k).at(j).resize(gsize3.x);
880 for( int i=0; i<gsize3.x; i++ ){
881 max_height_xy.at(k).at(j).at(i) = -9999999;
882 }
883 }
884 }
885
886 //do an initial pass to see whether "target_index" starts at 0 or 1
887 float target_offset = 0;
888 for( size_t r=0; r<Nhits; r++ ){
889
890 if( doesHitDataExist(r,"target_index") ){
891 float target_index = getHitData(r,"target_index");
892 if( target_index==0 ){
893 target_offset=1;
894 break;
895 }
896 }
897
898 }
899
900 for( size_t r=0; r<Nhits; r++ ){
901
902 setHitData( r, "ground_flag", 0 );
903
904 if( !doesHitDataExist(r,"target_index") || !doesHitDataExist(r,"target_count") ){
905 continue;
906 }
907
908 helios::int3 index3 = getHitGridCell( r );
909
910 if( index3.x>=0 && index3.y>=0 ){
911
912 float target_index = getHitData(r,"target_index")+target_offset;
913 float target_count = getHitData(r,"target_count");
914
915 float height = getHitXYZ(r).z;
916
917 for( int k=max_height_xy.size()-1; k>=0; k-- ){
918 if( height>max_height_xy.at(k).at(index3.y).at(index3.x) ){//&& target_index==1 ){
919 max_height_xy.at(k).at(index3.y).at(index3.x) = height;
920 break;
921 }
922 }
923
924 if( target_index==target_count ){
925 last_hits_ind.at(index3.y).at(index3.x).push_back( r );
926 }else if( target_index==1 ){
927 first_hits_ind.at(index3.y).at(index3.x).push_back( r );
928 }
929
930 }
931
932 }
933
934 //apply RANSAC to first and last hits for each x,y colum to determine ground and vegetation height
935
936 std::vector<std::vector<float> > ground_height_xy, canopy_height_xy;
937
938 ground_height_xy.resize(gsize3.y);
939 canopy_height_xy.resize(gsize3.y);
940
941 for( int j=0; j<gsize3.y; j++ ){
942
943 ground_height_xy.at(j).resize(gsize3.x);
944 canopy_height_xy.at(j).resize(gsize3.x);
945
946 for( int i=0; i<gsize3.x; i++ ){
947
948 float canopy_height = nanf("");
949 float ground_height = nanf("");
950
951 ground_height_xy.at(j).at(i) = ground_height;
952 canopy_height_xy.at(j).at(i) = canopy_height;
953
954 if( last_hits_ind.at(j).at(i).size()==0 ){ //no hits in this column
955 continue;
956 }
957
958 helios::int3 index3 = getHitGridCell( last_hits_ind.at(j).at(i).front() );
959
960 helios::vec3 center = getCellCenter( index3 );
961
962 //compute ground height
963 if( last_hits_ind.at(j).at(i).size()>3 ){
964
965 std::vector<helios::vec3> last_hits_xyz;
966 last_hits_xyz.resize( last_hits_ind.at(j).at(i).size() );
967 for( int r=0; r<last_hits_xyz.size(); r++ ){
968 last_hits_xyz.at(r) = getHitXYZ( last_hits_ind.at(j).at(i).at(r) );
969 }
970
971 //ground planes
972 std::vector<bool> ground_inliers;
973 helios::vec4 groundModel = RANSAC( maxIter, threshDist_ground, inlierRatio_ground, last_hits_xyz, ground_inliers );
974
975 for( int r=0; r<ground_inliers.size(); r++ ){
976 if( ground_inliers.at(r) ){
977 setHitData( last_hits_ind.at(j).at(i).at(r), "ground_flag", float(1) );
978 }
979 }
980
981 ground_height = -(groundModel.x*center.x+groundModel.y*center.y+groundModel.w)/groundModel.z;
982
983 }
984
985 //compute canopy height
986 if( first_hits_ind.at(j).at(i).size()>=3 ){
987
988 std::vector<helios::vec3> first_hits_xyz;
989 first_hits_xyz.resize( first_hits_ind.at(j).at(i).size() );
990 for( int r=0; r<first_hits_xyz.size(); r++ ){
991 first_hits_xyz.at(r) = getHitXYZ( first_hits_ind.at(j).at(i).at(r) );
992 }
993
994 //canopy top planes
995 std::vector<bool> canopy_inliers;
996 helios::vec4 canopyModel = RANSAC( maxIter, threshDist_vegetation, inlierRatio_vegetation, first_hits_xyz, canopy_inliers );
997
998 canopy_height = -(canopyModel.x*center.x+canopyModel.y*center.y+canopyModel.w)/canopyModel.z;
999
1000 if( canopy_height<ground_height ){
1001 canopy_height = ground_height;
1002 }else if( canopy_height>max_height_xy.front().at(j).at(i) ){
1003 canopy_height = max_height_xy.front().at(j).at(i);
1004 }
1005
1006 }
1007
1008 ground_height_xy.at(j).at(i) = ground_height;
1009 canopy_height_xy.at(j).at(i) = canopy_height;
1010
1011 }//end i (x,y) loop
1012 }//end j (x,y) loop
1013
1014 //fill any holes in the ground height model
1015 std::vector<std::vector<float> > ground_height_filled;
1016 ground_height_filled.resize(gsize3.y);
1017 for( int j=0; j<gsize3.y; j++ ){
1018 ground_height_filled.at(j).insert(ground_height_filled.at(j).begin(), ground_height_xy.at(j).begin(), ground_height_xy.at(j).end() );
1019 }
1020
1021 for( int j=1; j<gsize3.y-1; j++ ){
1022 for( int i=1; i<gsize3.x-1; i++ ){
1023
1024 if( ground_height_xy.at(j).at(i)!=ground_height_xy.at(j).at(i) ){ //NaN
1025
1026 int count=0;
1027 float mean=0;
1028 for( int jj=-1; jj<2; jj+=2 ){
1029 for( int ii=-1; ii<2; ii+=2 ){
1030 if( ground_height_xy.at(j+jj).at(i+ii)==ground_height_xy.at(j+jj).at(i+ii) ){
1031 mean += ground_height_xy.at(j+jj).at(i+ii);
1032 count ++;
1033 }
1034 }
1035 }
1036
1037 if( count>0 ){
1038 ground_height_filled.at(j).at(i) = mean/float(count);
1039 }
1040
1041 }
1042
1043 }
1044 }
1045
1046 //remove outliers in the ground height model
1047 int stencil_half_size = 4;
1048 float std_fact = 0.5;
1049
1050 for( int j=0; j<gsize3.y; j++ ){
1051 for( int i=0; i<gsize3.x; i++ ){
1052 ground_height_xy.at(j).at(i) = ground_height_filled.at(j).at(i);
1053 }
1054 }
1055
1056 for( int j=stencil_half_size; j<gsize3.y-stencil_half_size; j++ ){
1057 for( int i=stencil_half_size; i<gsize3.x-stencil_half_size; i++ ){
1058
1059 if( ground_height_filled.at(j).at(i)!=ground_height_filled.at(j).at(i) ){
1060 continue;
1061 }
1062
1063 int count=0;
1064 float mean=0;
1065 float var=0;
1066 for( int jj=-stencil_half_size; jj<=stencil_half_size; jj++ ){
1067 for( int ii=-stencil_half_size; ii<=stencil_half_size; ii++ ){
1068 if( ii==0 || jj==0 ){
1069 continue;
1070 }
1071
1072 if( ground_height_filled.at(j+jj).at(i+ii)==ground_height_filled.at(j+jj).at(i+ii) ){
1073 mean += ground_height_filled.at(j+jj).at(i+ii);
1074 var += pow( ground_height_filled.at(j+jj).at(i+ii)-ground_height_filled.at(j).at(i), 2);
1075 count ++;
1076 }
1077 }
1078 }
1079
1080 float std = sqrt(var/float(count));
1081 mean = mean/float(count);
1082
1083 if( fabs(ground_height_filled.at(j).at(i)-mean)>std_fact*std ){
1084
1085 float mean_new=0;
1086 count = 0;
1087 for( int jj=-stencil_half_size; jj<=stencil_half_size; jj++ ){
1088 for( int ii=-stencil_half_size; ii<=stencil_half_size; ii++ ){
1089 if( ii==0 || jj==0 ){
1090 continue;
1091 }
1092 if( fabs(ground_height_filled.at(j+jj).at(i+ii)-mean)<std_fact*std ){
1093 mean_new += ground_height_filled.at(j+jj).at(i+ii);
1094 count ++;
1095 }
1096 }
1097 }
1098
1099 if( count>0 ){
1100 ground_height_xy.at(j).at(i) = mean_new/float(count);
1101 }
1102 }
1103 }
1104 }
1105
1106 //set the values for all grid cells
1107 for( int k=0; k<gsize3.z; k++ ){
1108 for( int j=0; j<gsize3.y; j++ ){
1109 for( int i=0; i<gsize3.x; i++ ){
1110
1111 helios::int2 cell(i,j);
1112
1113 setCellGroundHeight( ground_height_xy.at(j).at(i), cell );
1114 setCellVegetationHeight( canopy_height_xy.at(j).at(i), cell );
1115 if( max_height_xy.front().at(j).at(i)==-9999999 ){
1116 setCellMaximumHitHeight( nanf(""), cell );
1117 }else{
1118 setCellMaximumHitHeight( max_height_xy.front().at(j).at(i), cell );
1119 }
1120
1121 }
1122 }
1123 }
1124
1125 groundheightcomputed = true;
1126
1127 if( printmessages ){
1128 std::cout << "done." << std::endl;
1129 }
1130
1131}
